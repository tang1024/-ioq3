212 int AAS_PointAreaNum(vec3_t point)
213 {
214  int nodenum;
215  vec_t dist;
216  aas_node_t *node;
217  aas_plane_t *plane;
218 
219  if (!aasworld.loaded)
220  {
221  botimport.Print(PRT_ERROR, "AAS_PointAreaNum: aas not loaded\n");
222  return 0;
223  } //end if
224 
225  //start with node 1 because node zero is a dummy used for solid leafs
226  nodenum = 1;
227  while (nodenum > 0)
228  {
229 // botimport.Print(PRT_MESSAGE, "[%d]", nodenum);
230 #ifdef AAS_SAMPLE_DEBUG
231  if (nodenum >= aasworld.numnodes)
232  {
233  botimport.Print(PRT_ERROR, "nodenum = %d >= aasworld.numnodes = %d\n", nodenum, aasworld.numnodes);
234  return 0;
235  } //end if
236 #endif //AAS_SAMPLE_DEBUG
237  node = &aasworld.nodes[nodenum];
238 #ifdef AAS_SAMPLE_DEBUG
239  if (node->planenum < 0 || node->planenum >= aasworld.numplanes)
240  {
241  botimport.Print(PRT_ERROR, "node->planenum = %d >= aasworld.numplanes = %d\n", node->planenum, aasworld.numplanes);
242  return 0;
243  } //end if
244 #endif //AAS_SAMPLE_DEBUG
245  plane = &aasworld.planes[node->planenum];
246  dist = DotProduct(point, plane->normal) - plane->dist;
247  if (dist > 0) nodenum = node->children[0];
248  else nodenum = node->children[1];
249  } //end while
250  if (!nodenum)
251  {
252 #ifdef AAS_SAMPLE_DEBUG
253  botimport.Print(PRT_MESSAGE, "in solid\n");
254 #endif //AAS_SAMPLE_DEBUG
255  return 0;
256  } //end if
257  return -nodenum;
258 } //end of the function AAS_PointAreaNum

559  } //end if

581  } //end else

594  {
