28 /*
29 =============================================================================
30 
31  MODEL TESTING
32 
33 The viewthing and gun positioning tools from Q2 have been integrated and
34 enhanced into a single model testing facility.
35 
36 Model viewing can begin with either "testmodel <modelname>" or "testgun <modelname>".
37 
38 The names must be the full pathname after the basedir, like
39 "models/weapons/v_launch/tris.md3" or "players/male/tris.md3"
40 
41 Testmodel will create a fake entity 100 units in front of the current view
42 position, directly facing the viewer. It will remain immobile, so you can
43 move around it to view it from different angles.
44 
45 Testgun will cause the model to follow the player around and suppress the real
46 view weapon model. The default frame 0 of most guns is completely off screen,
47 so you will probably have to cycle a couple frames to see it.
48 
49 "nextframe", "prevframe", "nextskin", and "prevskin" commands will change the
50 frame or skin of the testmodel. These are bound to F5, F6, F7, and F8 in
51 q3default.cfg.
52 
53 If a gun is being tested, the "gun_x", "gun_y", and "gun_z" variables will let
54 you adjust the positioning.
55 
56 Note that none of the model testing features update while the game is paused, so
57 it may be convenient to test with deathmatch set to 1 so that bringing down the
58 console doesn't pause the game.
59 
60 =============================================================================
61 */

75  cg.testGun = qfalse;

85  cg.testModelEntity.backlerp = atof(CG_Argv(2));

92  }
93 
94  VectorMA(cg.refdef.vieworg, 100, cg.refdef.viewaxis[0], cg.testModelEntity.origin);

97  angles[YAW] = 180 + cg.refdefViewAngles[1];

101 }

194 static void CG_CalcVrect(void)
195 {

199  if (cg.snap->ps.pm_type == PM_INTERMISSION) {
200  size = 100;
201  } else {
202  // bound normal viewsize
203  if (cg_viewsize.integer < 30) {

205  size = 30;
206  } else if (cg_viewsize.integer > 100) {
207  trap_Cvar_Set("cg_viewsize", "100");
208  size = 100;
209  } else {
210  size = cg_viewsize.integer;
211  }
212 
213  }
214  cg.refdef.width = cgs.glconfig.vidWidth * size / 100;
215  cg.refdef.width &= ~1;
216 
217  cg.refdef.height = cgs.glconfig.vidHeight * size / 100;
218  cg.refdef.height &= ~1;
219 
220  cg.refdef.x = (cgs.glconfig.vidWidth - cg.refdef.width) / 2;
221  cg.refdef.y = (cgs.glconfig.vidHeight - cg.refdef.height) / 2;
222 }

228 ===============
229 CG_OffsetThirdPersonView
230 
231 ===============
232 */
233 #define FOCUS_DISTANCE 512
234 static void CG_OffsetThirdPersonView(void)
235 {
236  vec3_t forward, right, up;
237  vec3_t view;
238  vec3_t focusAngles;
239  trace_t trace;
240  static vec3_t mins = { -4, -4, -4 };
241  static vec3_t maxs = { 4, 4, 4 };
242  vec3_t focusPoint;
243  float focusDist;
244  float forwardScale, sideScale;
245 
246  cg.refdef.vieworg[2] += cg.predictedPlayerState.viewheight;
247 
248  VectorCopy(cg.refdefViewAngles, focusAngles);
249 
250  // if dead, look at killer
251  if (cg.predictedPlayerState.stats[STAT_HEALTH] <= 0) {
252  focusAngles[YAW] = cg.predictedPlayerState.stats[STAT_DEAD_YAW];
253  cg.refdefViewAngles[YAW] = cg.predictedPlayerState.stats[STAT_DEAD_YAW];
254  }
255 
256  if (focusAngles[PITCH] > 45) {
257  focusAngles[PITCH] = 45; // don't go too far overhead
258  }
259  AngleVectors(focusAngles, forward, NULL, NULL);
260 
261  VectorMA(cg.refdef.vieworg, FOCUS_DISTANCE, forward, focusPoint);
262 
263  VectorCopy(cg.refdef.vieworg, view);
264 
265  view[2] += 8;
266 
267  cg.refdefViewAngles[PITCH] *= 0.5;
268 
269  AngleVectors(cg.refdefViewAngles, forward, right, up);
270 
271  forwardScale = cos(cg_thirdPersonAngle.value / 180 * M_PI);
272  sideScale = sin(cg_thirdPersonAngle.value / 180 * M_PI);
273  VectorMA(view, -cg_thirdPersonRange.value * forwardScale, forward, view);
274  VectorMA(view, -cg_thirdPersonRange.value * sideScale, right, view);
275 
276  // trace a ray from the origin to the viewpoint to make sure the view isn't
277  // in a solid block. Use an 8 by 8 block to prevent the view from near clipping anything
278 
279  if (!cg_cameraMode.integer) {
280  CG_Trace(&trace, cg.refdef.vieworg, mins, maxs, view, cg.predictedPlayerState.clientNum, MASK_SOLID);
281 
282  if (trace.fraction != 1.0) {
283  VectorCopy(trace.endpos, view);
284  view[2] += (1.0 - trace.fraction) * 32;
285  // try another trace to this position, because a tunnel may have the ceiling
286  // close enough that this is poking out
287 
288  CG_Trace(&trace, cg.refdef.vieworg, mins, maxs, view, cg.predictedPlayerState.clientNum, MASK_SOLID);
289  VectorCopy(trace.endpos, view);
290  }
291  }
292 
293 
294  VectorCopy(view, cg.refdef.vieworg);
295 
296  // select pitch to look at focus point from vieword
297  VectorSubtract(focusPoint, cg.refdef.vieworg, focusPoint);
298  focusDist = sqrt(focusPoint[0] * focusPoint[0] + focusPoint[1] * focusPoint[1]);
299  if (focusDist < 1) {
300  focusDist = 1; // should never happen
301  }
302  cg.refdefViewAngles[PITCH] = -180 / M_PI * atan2(focusPoint[2], focusDist);
303  cg.refdefViewAngles[YAW] -= cg_thirdPersonAngle.value;
304 }

320 /*
321 ===============
322 CG_OffsetFirstPersonView
323 
324 ===============
325 */
326 static void CG_OffsetFirstPersonView(void)
327 {
328  float* origin;
329  float* angles;
330  float bob;
331  float ratio;
332  float delta;
333  float speed;
334  float f;
335  vec3_t predictedVelocity;
336  int timeDelta;
337 
338  if (cg.snap->ps.pm_type == PM_INTERMISSION) {
339  return;
340  }
341 
342  origin = cg.refdef.vieworg;
343  angles = cg.refdefViewAngles;
344 
345  // if dead, fix the angle and don't add any kick
346  if (cg.snap->ps.stats[STAT_HEALTH] <= 0) {
347  angles[ROLL] = 40;
348  angles[PITCH] = -15;
349  angles[YAW] = cg.snap->ps.stats[STAT_DEAD_YAW];
350  origin[2] += cg.predictedPlayerState.viewheight;
351  return;
352  }
353 
354  // add angles based on damage kick
355  if (cg.damageTime) {
356  ratio = cg.time - cg.damageTime;
357  if (ratio < DAMAGE_DEFLECT_TIME) {
358  ratio /= DAMAGE_DEFLECT_TIME;
359  angles[PITCH] += ratio * cg.v_dmg_pitch;
360  angles[ROLL] += ratio * cg.v_dmg_roll;
361  } else {
362  ratio = 1.0 - (ratio - DAMAGE_DEFLECT_TIME) / DAMAGE_RETURN_TIME;
363  if (ratio > 0) {
364  angles[PITCH] += ratio * cg.v_dmg_pitch;
365  angles[ROLL] += ratio * cg.v_dmg_roll;
366  }
367  }
368  }
369 
370  // add pitch based on fall kick
371 #if 0
372  ratio = (cg.time - cg.landTime) / FALL_TIME;
373  if (ratio < 0)
374  ratio = 0;
375  angles[PITCH] += ratio * cg.fall_value;
376 #endif
377 
378  // add angles based on velocity
379  VectorCopy(cg.predictedPlayerState.velocity, predictedVelocity);
380 
381  delta = DotProduct(predictedVelocity, cg.refdef.viewaxis[0]);
382  angles[PITCH] += delta * cg_runpitch.value;
383 
384  delta = DotProduct(predictedVelocity, cg.refdef.viewaxis[1]);
385  angles[ROLL] -= delta * cg_runroll.value;
386 
387  // add angles based on bob
388 
389  // make sure the bob is visible even at low speeds
390  speed = cg.xyspeed > 200 ? cg.xyspeed : 200;
391 
392  delta = cg.bobfracsin * cg_bobpitch.value * speed;
393  if (cg.predictedPlayerState.pm_flags & PMF_DUCKED)
394  delta *= 3; // crouching
395  angles[PITCH] += delta;
396  delta = cg.bobfracsin * cg_bobroll.value * speed;
397  if (cg.predictedPlayerState.pm_flags & PMF_DUCKED)
398  delta *= 3; // crouching accentuates roll
399  if (cg.bobcycle & 1)
400  delta = -delta;
401  angles[ROLL] += delta;
402 
403  //===================================
404 
405  // add view height
406  origin[2] += cg.predictedPlayerState.viewheight;
407 
408  // smooth out duck height changes
409  timeDelta = cg.time - cg.duckTime;
410  if (timeDelta < DUCK_TIME) {
411  cg.refdef.vieworg[2] -= cg.duckChange
412  * (DUCK_TIME - timeDelta) / DUCK_TIME;
413  }
414 
415  // add bob height
416  bob = cg.bobfracsin * cg.xyspeed * cg_bobup.value;
417  if (bob > 6) {
418  bob = 6;
419  }
420 
421  origin[2] += bob;
422 
423 
424  // add fall height
425  delta = cg.time - cg.landTime;
426  if (delta < LAND_DEFLECT_TIME) {
427  f = delta / LAND_DEFLECT_TIME;
428  cg.refdef.vieworg[2] += cg.landChange * f;
429  } else if (delta < LAND_DEFLECT_TIME + LAND_RETURN_TIME) {
430  delta -= LAND_DEFLECT_TIME;
431  f = 1.0 - (delta / LAND_RETURN_TIME);
432  cg.refdef.vieworg[2] += cg.landChange * f;
433  }
434 
435  // add step offset
436  CG_StepOffset();
437 
438  // pivot the eye based on a neck length
439 #if 0
440  {
441 #define NECK_LENGTH 8
442  vec3_t forward, up;
443 
444  cg.refdef.vieworg[2] -= NECK_LENGTH;
445  AngleVectors(cg.refdefViewAngles, forward, NULL, up);
446  VectorMA(cg.refdef.vieworg, 3, forward, cg.refdef.vieworg);
447  VectorMA(cg.refdef.vieworg, NECK_LENGTH, up, cg.refdef.vieworg);
448  }
449 #endif
450 }

473 /*
474 ====================
475 CG_CalcFov
476 
477 Fixed fov at intermissions, otherwise account for fov variable and zooms.
478 ====================
479 */
480 #define WAVE_AMPLITUDE 1
481 #define WAVE_FREQUENCY 0.4
482 
483 static int CG_CalcFov(void)
484 {
485  float x;
486  float phase;
487  float v;
488  int contents;
489  float fov_x, fov_y;
490  float zoomFov;
491  float f;
492  int inwater;
493 
494  if (cg.predictedPlayerState.pm_type == PM_INTERMISSION) {
495  // if in intermission, use a fixed value
496  fov_x = 90;
497  } else {
498  // user selectable
499  if (cgs.dmflags & DF_FIXED_FOV) {
500  // dmflag to prevent wide fov for all clients
501  fov_x = 90;
502  } else {
503  fov_x = cg_fov.value;
504  if (fov_x < 1) {
505  fov_x = 1;
506  } else if (fov_x > 160) {
507  fov_x = 160;
508  }
509  }
510 
511  // account for zooms
512  zoomFov = cg_zoomFov.value;
513  if (zoomFov < 1) {
514  zoomFov = 1;
515  } else if (zoomFov > 160) {
516  zoomFov = 160;
517  }
518 
519  if (cg.zoomed) {
520  f = (cg.time - cg.zoomTime) / (float)ZOOM_TIME;
521  if (f > 1.0) {
522  fov_x = zoomFov;
523  } else {
524  fov_x = fov_x + f * (zoomFov - fov_x);
525  }
526  } else {
527  f = (cg.time - cg.zoomTime) / (float)ZOOM_TIME;
528  if (f <= 1.0) {
529  fov_x = zoomFov + f * (fov_x - zoomFov);
530  }
531  }
532  }
533 
534  x = cg.refdef.width / tan(fov_x / 360 * M_PI);
535  fov_y = atan2(cg.refdef.height, x);
536  fov_y = fov_y * 360 / M_PI;
537 
538  // warp if underwater
539  contents = CG_PointContents(cg.refdef.vieworg, -1);
540  if (contents & (CONTENTS_WATER | CONTENTS_SLIME | CONTENTS_LAVA)) {
541  phase = cg.time / 1000.0 * WAVE_FREQUENCY * M_PI * 2;
542  v = WAVE_AMPLITUDE * sin(phase);
543  fov_x += v;
544  fov_y -= v;
545  inwater = qtrue;
546  } else {
547  inwater = qfalse;
548  }
549 
550 
551  // set it
552  cg.refdef.fov_x = fov_x;
553  cg.refdef.fov_y = fov_y;
554 
555  if (!cg.zoomed) {
556  cg.zoomSensitivity = 1;
557  } else {
558  cg.zoomSensitivity = cg.refdef.fov_y / 75.0;
559  }
560 
561  return inwater;
562 }

632 
633  // strings for in game rendering

635  // Q_strncpyz( cg.refdef.text[1], "19", sizeof(cg.refdef.text[1]) );
636

638  CG_CalcVrect();
639 
640  ps = &cg.predictedPlayerState;
641  /*
642  if (cg.cameraMode) {
643  vec3_t origin, angles;
644  if (trap_getCameraInfo(cg.time, &origin, &angles)) {
645  VectorCopy(origin, cg.refdef.vieworg);

659  AnglesToAxis(cg.refdefViewAngles, cg.refdef.viewaxis);
660  return CG_CalcFov();
661  }
662 
663  cg.bobcycle = (ps->bobCycle & 128) >> 7;
664  cg.bobfracsin = fabs(sin((ps->bobCycle & 127) / 127.0 * M_PI));
665  cg.xyspeed = sqrt(ps->velocity[0] * ps->velocity[0] +

675  cg_thirdPersonAngle.value += cg_cameraOrbit.value;
676  }
677  }
678  // add error decay
679  if (cg_errorDecay.value > 0) {
680  int t;
681  float f;
682 
683  t = cg.time - cg.predictedErrorTime;
684  f = (cg_errorDecay.value - t) / cg_errorDecay.value;
685  if (f > 0 && f < 1) {
686  VectorMA(cg.refdef.vieworg, f, cg.predictedError, cg.refdef.vieworg);
687  } else {
688  cg.predictedErrorTime = 0;
689  }
690  }

814  trap_SetUserCmdValue(cg.weaponSelect, cg.zoomSensitivity);
815 
816  // this counter will be bumped for every valid scene we generate
817  cg.clientFrame++;
818

820  CG_PredictPlayerState();
821

885  }
886  }
887

889  CG_DrawActive(stereoView);
890 
891  if (cg_stats.integer) {
892  CG_Printf("cg.clientFrame:%i\n", cg.clientFrame);
893  }
