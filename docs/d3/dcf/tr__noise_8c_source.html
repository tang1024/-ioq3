52 float R_NoiseGet4f( float x, float y, float z, double t )
53 {
54  int i;
55  int ix, iy, iz, it;
56  float fx, fy, fz, ft;
57  float front[4];
58  float back[4];
59  float fvalue, bvalue, value[2], finalvalue;
60 
61  ix = ( int ) floor( x );
62  fx = x - ix;
63  iy = ( int ) floor( y );
64  fy = y - iy;
65  iz = ( int ) floor( z );

71  {
72  front[0] = GetNoiseValue( ix, iy, iz, it + i );
73  front[1] = GetNoiseValue( ix+1, iy, iz, it + i );
74  front[2] = GetNoiseValue( ix, iy+1, iz, it + i );

76 
77  back[0] = GetNoiseValue( ix, iy, iz + 1, it + i );
78  back[1] = GetNoiseValue( ix+1, iy, iz + 1, it + i );

80  back[3] = GetNoiseValue( ix+1, iy+1, iz + 1, it + i );
