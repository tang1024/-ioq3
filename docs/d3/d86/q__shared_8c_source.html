200 short ShortSwap(short l)
201 {
202  byte b1, b2;
203 
204  b1 = l & 255;
205  b2 = (l >> 8) & 255;
206 
207  return (b1 << 8) + b2;

215 int LongSwap(int l)
216 {
217  byte b1, b2, b3, b4;

219  b1 = l & 255;

363 /*
364 ==============
365 COM_Parse
366 
367 Parse a token out of a string
368 Will never return NULL, just empty strings
369 
370 If "allowLineBreaks" is qtrue then an empty
371 string will be returned if the next token is
372 a newline.
373 ==============
374 */
375 static char* SkipWhitespace(char* data, qboolean* hasNewLines)
376 {
377  int c;
378 
379  while ((c = *data) <= ' ') {
380  if (!c) {
381  return NULL;
382  }
383  if (c == '\n') {
384  com_lines++;
385  *hasNewLines = qtrue;
386  }
387  data++;
388  }
389 
390  return data;
391 }

393 int COM_Compress(char* data_p)
394 {
395  char* in, * out;
396  int c;
397  qboolean newline = qfalse, whitespace = qfalse;
398 
399  in = out = data_p;
400  if (in) {
401  while ((c = *in) != 0) {
402  // skip double slash comments
403  if (c == '/' && in[1] == '/') {
404  while (*in && *in != '\n') {
405  in++;
406  }
407  // skip /* */ comments
408  } else if (c == '/' && in[1] == '*') {
409  while (*in && (*in != '*' || in[1] != '/'))
410  in++;
411  if (*in)
412  in += 2;
413  // record when we hit a newline
414  } else if (c == '\n' || c == '\r') {
415  newline = qtrue;
416  in++;
417  // record when we hit whitespace
418  } else if (c == ' ' || c == '\t') {
419  whitespace = qtrue;
420  in++;
421  // an actual token
422  } else {
423  // if we have a pending newline, emit it (and it counts as whitespace)
424  if (newline) {
425  *out++ = '\n';
426  newline = qfalse;
427  whitespace = qfalse;
428  } if (whitespace) {
429  *out++ = ' ';
430  whitespace = qfalse;
431  }
432 
433  // copy quoted strings unmolested
434  if (c == '"') {
435  *out++ = c;
436  in++;
437  while (1) {
438  c = *in;
439  if (c && c != '"') {
440  *out++ = c;
441  in++;
442  } else {
443  break;
444  }
445  }
446  if (c == '"') {
447  *out++ = c;
448  in++;
449  }
450  } else {
451  *out = c;
452  out++;
453  in++;
454  }
455  }
456  }
457 
458  *out = 0;
459  }
460  return out - data_p;
461 }

463 char* COM_ParseExt(char** data_p, qboolean allowLineBreaks)
464 {
465  int c = 0, len;
466  qboolean hasNewLines = qfalse;
467  char* data;
468 
469  data = *data_p;
470  len = 0;
471  com_token[0] = 0;
472  com_tokenline = 0;
473 
474  // make sure incoming data is valid
475  if (!data) {
476  *data_p = NULL;
477  return com_token;
478  }
479 
480  while (1) {
481  // skip whitespace
482  data = SkipWhitespace(data, &hasNewLines);
483  if (!data) {
484  *data_p = NULL;
485  return com_token;
486  }
487  if (hasNewLines && !allowLineBreaks) {
488  *data_p = data;
489  return com_token;
490  }
491 
492  c = *data;
493 
494  // skip double slash comments
495  if (c == '/' && data[1] == '/') {
496  data += 2;
497  while (*data && *data != '\n') {
498  data++;
499  }
500  }
501  // skip /* */ comments
502  else if (c == '/' && data[1] == '*') {
503  data += 2;
504  while (*data && (*data != '*' || data[1] != '/')) {
505  if (*data == '\n') {
506  com_lines++;
507  }
508  data++;
509  }
510  if (*data) {
511  data += 2;
512  }
513  } else {
514  break;
515  }
516  }
517 
518  // token starts on this line
519  com_tokenline = com_lines;
520 
521  // handle quoted strings
522  if (c == '\"') {
523  data++;
524  while (1) {
525  c = *data++;
526  if (c == '\"' || !c) {
527  com_token[len] = 0;
528  *data_p = (char*)data;
529  return com_token;
530  }
531  if (c == '\n') {
532  com_lines++;
533  }
534  if (len < MAX_TOKEN_CHARS - 1) {
535  com_token[len] = c;
536  len++;
537  }
538  }
539  }
540 
541  // parse a regular word
542  do {
543  if (len < MAX_TOKEN_CHARS - 1) {
544  com_token[len] = c;
545  len++;
546  }
547  data++;
548  c = *data;
549  } while (c > 32);
550 
551  com_token[len] = 0;
552 
553  *data_p = (char*)data;
554  return com_token;
555 }

573 /*
574 =================
575 SkipBracedSection
576 
577 The next token should be an open brace or set depth to 1 if already parsed it.
578 Skips until a matching close brace is found.
579 Internal brace depths are properly skipped.
580 =================
581 */
582 qboolean SkipBracedSection(char** program, int depth)
583 {
584  char* token;
585 
586  do {
587  token = COM_ParseExt(program, qtrue);
588  if (token[1] == 0) {
589  if (token[0] == '{') {
590  depth++;
591  } else if (token[0] == '}') {
592  depth--;
593  }
594  }
595  } while (depth && *program);
596 
597  return (depth == 0);

762 
763 Special wrapper function for Microsoft's broken _vsnprintf() function.
764 MinGW comes with its own vsnprintf() which is not broken. mingw-w64
765 however, uses Microsoft's broken _vsnprintf() function.

769 int Q_vsnprintf(char* str, size_t size, const char* format, va_list ap)
770 {

773  retval = _vsnprintf(str, size, format, ap);
774 
775  if (retval < 0 || retval == size) {
776  // Microsoft doesn't adhere to the C99 standard of vsnprintf,

814 int Q_stricmpn(const char* s1, const char* s2, int n)
815 {
816  int c1, c2;
817 
818  if (s1 == NULL) {
819  if (s2 == NULL)
820  return 0;
821  else
822  return -1;
823  } else if (s2 == NULL)
824  return 1;
825 
826 
827 
828  do {
829  c1 = *s1++;
830  c2 = *s2++;
831 
832  if (!n--) {
833  return 0; // strings are equal until end point
834  }
835 
836  if (c1 != c2) {
837  if (c1 >= 'a' && c1 <= 'z') {
838  c1 -= ('a' - 'A');
839  }
840  if (c2 >= 'a' && c2 <= 'z') {
841  c2 -= ('a' - 'A');
842  }
843  if (c1 != c2) {
844  return c1 < c2 ? -1 : 1;
845  }
846  }
847  } while (c1);
848 
849  return 0; // strings are equal
850 }
