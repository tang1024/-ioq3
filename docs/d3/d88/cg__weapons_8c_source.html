31 static void CG_MachineGunEjectBrass( centity_t *cent ) {
32  localEntity_t *le;
33  refEntity_t *re;
34  vec3_t velocity, xvelocity;
35  vec3_t offset, xoffset;
36  float waterScale = 1.0f;

38 
39  if ( cg_brassTime.integer <= 0 ) {
40  return;
41  }
42 
43  le = CG_AllocLocalEntity();
44  re = &le->refEntity;
45 
46  velocity[0] = 0;
47  velocity[1] = -50 + 40 * crandom();
48  velocity[2] = 100 + 50 * crandom();
49 
50  le->leType = LE_FRAGMENT;
51  le->startTime = cg.time;
52  le->endTime = le->startTime + cg_brassTime.integer + ( cg_brassTime.integer / 4 ) * random();
53 
54  le->pos.trType = TR_GRAVITY;
55  le->pos.trTime = cg.time - (rand()&15);
56 
57  AnglesToAxis( cent->lerpAngles, v );
58

61  offset[2] = 24;
62 
63  xoffset[0] = offset[0] * v[0][0] + offset[1] * v[1][0] + offset[2] * v[2][0];

67 
68  VectorCopy( re->origin, le->pos.trBase );
69 
70  if ( CG_PointContents( re->origin, -1 ) & CONTENTS_WATER ) {
71  waterScale = 0.10f;
72  }
73 
74  xvelocity[0] = velocity[0] * v[0][0] + velocity[1] * v[1][0] + velocity[2] * v[2][0];
75  xvelocity[1] = velocity[0] * v[0][1] + velocity[1] * v[1][1] + velocity[2] * v[2][1];

78 
79  AxisCopy( axisDefault, re->axis );
80  re->hModel = cgs.media.machinegunBrassModel;
81 
82  le->bounceFactor = 0.4 * waterScale;
83 
84  le->angles.trType = TR_LINEAR;
85  le->angles.trTime = cg.time;
86  le->angles.trBase[0] = rand()&31;
87  le->angles.trBase[1] = rand()&31;

215 void CG_RailTrail (clientInfo_t *ci, vec3_t start, vec3_t end) {
216  vec3_t axis[36], move, move2, vec, temp;
217  float len;
218  int i, j, skip;
219 
220  localEntity_t *le;
221  refEntity_t *re;
222 
223 #define RADIUS 4
224 #define ROTATION 1
225 #define SPACING 5
226 
227  start[2] -= 4;
228 
229  le = CG_AllocLocalEntity();
230  re = &le->refEntity;
231 
232  le->leType = LE_FADE_RGB;
233  le->startTime = cg.time;
234  le->endTime = cg.time + cg_railTrailTime.value;
235  le->lifeRate = 1.0 / (le->endTime - le->startTime);
236 
237  re->shaderTime = cg.time / 1000.0f;
238  re->reType = RT_RAIL_CORE;
239  re->customShader = cgs.media.railCoreShader;
240 
241  VectorCopy(start, re->origin);
242  VectorCopy(end, re->oldorigin);
243 
244  re->shaderRGBA[0] = ci->color1[0] * 255;
245  re->shaderRGBA[1] = ci->color1[1] * 255;
246  re->shaderRGBA[2] = ci->color1[2] * 255;
247  re->shaderRGBA[3] = 255;
248 
249  le->color[0] = ci->color1[0] * 0.75;
250  le->color[1] = ci->color1[1] * 0.75;
251  le->color[2] = ci->color1[2] * 0.75;
252  le->color[3] = 1.0f;
253 
254  AxisClear( re->axis );
255 
256  if (cg_oldRail.integer)
257  {
258  // nudge down a bit so it isn't exactly in center
259  re->origin[2] -= 8;
260  re->oldorigin[2] -= 8;
261  return;
262  }
263 
264  VectorCopy (start, move);
265  VectorSubtract (end, start, vec);
266  len = VectorNormalize (vec);
267  PerpendicularVector(temp, vec);
268  for (i = 0 ; i < 36; i++)
269  {
270  RotatePointAroundVector(axis[i], vec, temp, i * 10);//banshee 2.4 was 10
271  }
272 
273  VectorMA(move, 20, vec, move);
274  VectorScale (vec, SPACING, vec);
275 
276  skip = -1;
277 
278  j = 18;
279  for (i = 0; i < len; i += SPACING)
280  {
281  if (i != skip)
282  {
283  skip = i + SPACING;
284  le = CG_AllocLocalEntity();
285  re = &le->refEntity;
286  le->leFlags = LEF_PUFF_DONT_SCALE;
287  le->leType = LE_MOVE_SCALE_FADE;
288  le->startTime = cg.time;
289  le->endTime = cg.time + (i>>1) + 600;
290  le->lifeRate = 1.0 / (le->endTime - le->startTime);
291 
292  re->shaderTime = cg.time / 1000.0f;
293  re->reType = RT_SPRITE;
294  re->radius = 1.1f;
295  re->customShader = cgs.media.railRingsShader;
296 
297  re->shaderRGBA[0] = ci->color2[0] * 255;
298  re->shaderRGBA[1] = ci->color2[1] * 255;
299  re->shaderRGBA[2] = ci->color2[2] * 255;
300  re->shaderRGBA[3] = 255;
301 
302  le->color[0] = ci->color2[0] * 0.75;
303  le->color[1] = ci->color2[1] * 0.75;
304  le->color[2] = ci->color2[2] * 0.75;
305  le->color[3] = 1.0f;
306 
307  le->pos.trType = TR_LINEAR;
308  le->pos.trTime = cg.time;
309 
310  VectorCopy( move, move2);
311  VectorMA(move2, RADIUS , axis[j], move2);
312  VectorCopy(move2, le->pos.trBase);
313 
314  le->pos.trDelta[0] = axis[j][0]*6;
315  le->pos.trDelta[1] = axis[j][1]*6;
316  le->pos.trDelta[2] = axis[j][2]*6;
317  }
318 
319  VectorAdd (move, vec, move);
320 
321  j = (j + ROTATION) % 36;
322  }
323 }

328 ==========================

330 static void CG_RocketTrail( centity_t *ent, const weaponInfo_t *wi ) {
331  int step;
332  vec3_t origin, lastPos;
333  int t;
334  int startTime, contents;
335  int lastContents;
336  entityState_t *es;
337  vec3_t up;
338  localEntity_t *smoke;
339 
340  if ( cg_noProjectileTrail.integer ) {
341  return;
342  }
343 
344  up[0] = 0;
345  up[1] = 0;
346  up[2] = 0;
347 
348  step = 50;
349 
350  es = &ent->currentState;
351  startTime = ent->trailTime;
352  t = step * ( (startTime + step) / step );
353 
354  BG_EvaluateTrajectory( &es->pos, cg.time, origin );
355  contents = CG_PointContents( origin, -1 );
356 
357  // if object (e.g. grenade) is stationary, don't toss up smoke
358  if ( es->pos.trType == TR_STATIONARY ) {
359  ent->trailTime = cg.time;
360  return;
361  }
362 
363  BG_EvaluateTrajectory( &es->pos, ent->trailTime, lastPos );
364  lastContents = CG_PointContents( lastPos, -1 );
365 
366  ent->trailTime = cg.time;
367 
368  if ( contents & ( CONTENTS_WATER | CONTENTS_SLIME | CONTENTS_LAVA ) ) {
369  if ( contents & lastContents & CONTENTS_WATER ) {
370  CG_BubbleTrail( lastPos, origin, 8 );
371  }
372  return;
373  }
374 
375  for ( ; t <= ent->trailTime ; t += step ) {
376  BG_EvaluateTrajectory( &es->pos, t, lastPos );
377 
378  smoke = CG_SmokePuff( lastPos, up,
379  wi->trailRadius,
380  1, 1, 1, 0.33f,
381  wi->wiTrailTime,
382  t,
383  0,
384  0,
385  cgs.media.smokePuffShader );
386  // use the optimized local entity add
387  smoke->leType = LE_SCALE_FADE;
388  }
389 
390 }

606 void CG_RegisterWeapon( int weaponNum ) {
607  weaponInfo_t *weaponInfo;
608  gitem_t *item, *ammo;
609  char path[MAX_QPATH];
610  vec3_t mins, maxs;
611  int i;
612 
613  weaponInfo = &cg_weapons[weaponNum];
614 
615  if ( weaponNum == 0 ) {
616  return;
617  }
618 
619  if ( weaponInfo->registered ) {
620  return;
621  }
622 
623  memset( weaponInfo, 0, sizeof( *weaponInfo ) );
624  weaponInfo->registered = qtrue;
625 
626  for ( item = bg_itemlist + 1 ; item->classname ; item++ ) {
627  if ( item->giType == IT_WEAPON && item->giTag == weaponNum ) {
628  weaponInfo->item = item;
629  break;
630  }
631  }
632  if ( !item->classname ) {
633  CG_Error( "Couldn't find weapon %i", weaponNum );
634  }
635  CG_RegisterItemVisuals( item - bg_itemlist );
636 
637  // load cmodel before model so filecache works
638  weaponInfo->weaponModel = trap_R_RegisterModel( item->world_model[0] );
639 
640  // calc midpoint for rotation

911  float fracsin;
912 
913  VectorCopy( cg.refdef.vieworg, origin );

917  if ( cg.bobcycle & 1 ) {
918  scale = -cg.xyspeed;
919  } else {
920  scale = cg.xyspeed;
921  }
922 
923  // gun angles from bobbing
924  angles[ROLL] += scale * cg.bobfracsin * 0.005;
925  angles[YAW] += scale * cg.bobfracsin * 0.01;
926  angles[PITCH] += cg.xyspeed * cg.bobfracsin * 0.005;
927 
928  // drop the weapon when landing
929  delta = cg.time - cg.landTime;
930  if ( delta < LAND_DEFLECT_TIME ) {
931  origin[2] += cg.landChange*0.25 * delta / LAND_DEFLECT_TIME;
932  } else if ( delta < LAND_DEFLECT_TIME + LAND_RETURN_TIME ) {
933  origin[2] += cg.landChange*0.25 *
934  (LAND_DEFLECT_TIME + LAND_RETURN_TIME - delta) / LAND_RETURN_TIME;
935  }
936 
937 #if 0
938  // drop the weapon when stair climbing
939  delta = cg.time - cg.stepTime;
940  if ( delta < STEP_TIME/2 ) {
941  origin[2] -= cg.stepChange*0.25 * delta / (STEP_TIME/2);
942  } else if ( delta < STEP_TIME ) {
943  origin[2] -= cg.stepChange*0.25 * (STEP_TIME - delta) / (STEP_TIME/2);
944  }
945 #endif
946 
947  // idle drift
948  scale = cg.xyspeed + 40;
949  fracsin = sin( cg.time * 0.001 );
950  angles[ROLL] += scale * fracsin * 0.01;
