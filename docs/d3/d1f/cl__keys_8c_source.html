331 /*
332 ===================
333 Field_Draw
334 
335 Handles horizontal scrolling and cursor blinking
336 x, y, and width are in pixels
337 ===================
338 */
339 void Field_VariableSizeDraw( field_t *edit, int x, int y, int width, int size, qboolean showCursor,
340  qboolean noColorEscape ) {
341  int len;
342  int drawLen;
343  int prestep;
344  int cursorChar;
345  char str[MAX_STRING_CHARS];
346  int i;
347 
348  drawLen = edit->widthInChars - 1; // - 1 so there is always a space for the cursor
349  len = strlen( edit->buffer );
350 
351  // guarantee that cursor will be visible
352  if ( len <= drawLen ) {
353  prestep = 0;
354  } else {
355  if ( edit->scroll + drawLen > len ) {
356  edit->scroll = len - drawLen;
357  if ( edit->scroll < 0 ) {
358  edit->scroll = 0;
359  }
360  }
361  prestep = edit->scroll;
362  }
363 
364  if ( prestep + drawLen > len ) {
365  drawLen = len - prestep;
366  }
367 
368  // extract <drawLen> characters from the field at <prestep>
369  if ( drawLen >= MAX_STRING_CHARS ) {
370  Com_Error( ERR_DROP, "drawLen >= MAX_STRING_CHARS" );
371  }
372 
373  Com_Memcpy( str, edit->buffer + prestep, drawLen );
374  str[ drawLen ] = 0;
375 
376  // draw it
377  if ( size == SMALLCHAR_WIDTH ) {
378  float color[4];
379 
380  color[0] = color[1] = color[2] = color[3] = 1.0;
381  SCR_DrawSmallStringExt( x, y, str, color, qfalse, noColorEscape );
382  } else {
383  // draw big string with drop shadow
384  SCR_DrawBigString( x, y, str, 1.0, noColorEscape );
385  }
386 
387  // draw the cursor
388  if ( showCursor ) {
389  if ( (int)( cls.realtime >> 8 ) & 1 ) {
390  return; // off blink
391  }
392 
393  if ( key_overstrikeMode ) {
394  cursorChar = 11;
395  } else {
396  cursorChar = 10;
397  }
398 
399  i = drawLen - strlen( str );
400 
401  if ( size == SMALLCHAR_WIDTH ) {
402  SCR_DrawSmallChar( x + ( edit->cursor - prestep - i ) * size, y, cursorChar );
403  } else {
404  str[0] = cursorChar;
405  str[1] = 0;
406  SCR_DrawBigString( x + ( edit->cursor - prestep - i ) * size, y, str, 1.0, qfalse );
407 
408  }
409  }
410 }

446 /*
447 =================
448 Field_KeyDownEvent
449 
450 Performs the basic line editing functions for the console,
451 in-game talk, and menu fields
452 
453 Key events are used for non-printable characters, others are gotten from char events.
454 =================
455 */
456 void Field_KeyDownEvent( field_t *edit, int key ) {
457  int len;
458 
459  // shift-insert is paste
460  if ( ( ( key == K_INS ) || ( key == K_KP_INS ) ) && keys[K_SHIFT].down ) {
461  Field_Paste( edit );
462  return;
463  }
464 
465  key = tolower( key );
466  len = strlen( edit->buffer );
467 
468  switch ( key ) {
469  case K_DEL:
470  if ( edit->cursor < len ) {
471  memmove( edit->buffer + edit->cursor,
472  edit->buffer + edit->cursor + 1, len - edit->cursor );
473  }
474  break;
475 
476  case K_RIGHTARROW:
477  if ( edit->cursor < len ) {
478  edit->cursor++;
479  }
480  break;
481 
482  case K_LEFTARROW:
483  if ( edit->cursor > 0 ) {
484  edit->cursor--;
485  }
486  break;
487 
488  case K_HOME:
489  edit->cursor = 0;
490  break;
491 
492  case K_END:
493  edit->cursor = len;
494  break;
495 
496  case K_INS:
497  key_overstrikeMode = !key_overstrikeMode;
498  break;
499 
500  default:
501  break;
502  }
503 
504  // Change scroll if cursor is no longer visible
505  if ( edit->cursor < edit->scroll ) {
506  edit->scroll = edit->cursor;
507  } else if ( edit->cursor >= edit->scroll + edit->widthInChars && edit->cursor <= len ) {
508  edit->scroll = edit->cursor - edit->widthInChars + 1;
509  }
510 }

512 /*
513 ==================
514 Field_CharEvent
515 ==================
516 */
517 void Field_CharEvent( field_t *edit, int ch ) {
518  int len;
519 
520  if ( ch == 'v' - 'a' + 1 ) { // ctrl-v is paste
521  Field_Paste( edit );
522  return;
523  }
524 
525  if ( ch == 'c' - 'a' + 1 ) { // ctrl-c clears the field
526  Field_Clear( edit );
527  return;
528  }
529 
530  len = strlen( edit->buffer );
531 
532  if ( ch == 'h' - 'a' + 1 ) { // ctrl-h is backspace
533  if ( edit->cursor > 0 ) {
534  memmove( edit->buffer + edit->cursor - 1,
535  edit->buffer + edit->cursor, len + 1 - edit->cursor );
536  edit->cursor--;
537  if ( edit->cursor < edit->scroll )
538  {
539  edit->scroll--;
540  }
541  }
542  return;
543  }
544 
545  if ( ch == 'a' - 'a' + 1 ) { // ctrl-a is home
546  edit->cursor = 0;
547  edit->scroll = 0;
548  return;
549  }
550 
551  if ( ch == 'e' - 'a' + 1 ) { // ctrl-e is end
552  edit->cursor = len;
553  edit->scroll = edit->cursor - edit->widthInChars;
554  return;
555  }
556 
557  //
558  // ignore any other non printable chars
559  //
560  if ( ch < 32 ) {
561  return;
562  }
563 
564  if ( key_overstrikeMode ) {
565  // - 2 to leave room for the leading slash and trailing \0
566  if ( edit->cursor == MAX_EDIT_LINE - 2 )
567  return;
568  edit->buffer[edit->cursor] = ch;
569  edit->cursor++;
570  } else { // insert mode
571  // - 2 to leave room for the leading slash and trailing \0
572  if ( len == MAX_EDIT_LINE - 2 ) {
573  return; // all full
574  }
575  memmove( edit->buffer + edit->cursor + 1,
576  edit->buffer + edit->cursor, len + 1 - edit->cursor );
577  edit->buffer[edit->cursor] = ch;
578  edit->cursor++;
579  }
580 
581 
582  if ( edit->cursor >= edit->widthInChars ) {
583  edit->scroll++;
584  }
585 
586  if ( edit->cursor == len + 1) {
587  edit->buffer[edit->cursor] = 0;
588  }
589 }

599 /*
600 ====================
601 Console_Key
602 
603 Handles history and console scrollback
604 ====================
605 */
606 void Console_Key (int key) {
607  // ctrl-L clears screen
608  if ( key == 'l' && keys[K_CTRL].down ) {
609  Cbuf_AddText ("clear\n");
610  return;
611  }
612 
613  // enter finishes the line
614  if ( key == K_ENTER || key == K_KP_ENTER ) {
615  // if not in the game explicitly prepend a slash if needed
616  if ( clc.state != CA_ACTIVE && con_autochat->integer &&
617  g_consoleField.buffer[0] &&
618  g_consoleField.buffer[0] != '\\' &&
619  g_consoleField.buffer[0] != '/' ) {
620  char temp[MAX_EDIT_LINE-1];
621 
622  Q_strncpyz( temp, g_consoleField.buffer, sizeof( temp ) );
623  Com_sprintf( g_consoleField.buffer, sizeof( g_consoleField.buffer ), "\\%s", temp );
624  g_consoleField.cursor++;
625  }
626 
627  Com_Printf ( "]%s\n", g_consoleField.buffer );
628 
629  // leading slash is an explicit command
630  if ( g_consoleField.buffer[0] == '\\' || g_consoleField.buffer[0] == '/' ) {
631  Cbuf_AddText( g_consoleField.buffer+1 ); // valid command
632  Cbuf_AddText ("\n");
633  } else {
634  // other text will be chat messages
635  if ( !g_consoleField.buffer[0] ) {
636  return; // empty lines just scroll the console without adding to history
637  } else {
638  if ( con_autochat->integer ) {
639  Cbuf_AddText ("cmd say ");
640  }
641 
642  Cbuf_AddText( g_consoleField.buffer );
643  Cbuf_AddText ("\n");
644  }
645  }
646 
647  // copy line to history buffer
648  historyEditLines[nextHistoryLine % COMMAND_HISTORY] = g_consoleField;
649  nextHistoryLine++;
650  historyLine = nextHistoryLine;
651 
652  Field_Clear( &g_consoleField );
653 
654  g_consoleField.widthInChars = g_console_field_width;
655 
656  CL_SaveConsoleHistory( );
657 
658  if ( clc.state == CA_DISCONNECTED ) {
659  SCR_UpdateScreen (); // force an update, because the command
660  } // may take some time
661  return;
662  }
663 
664  // command completion
665 
666  if (key == K_TAB) {
667  Field_AutoComplete(&g_consoleField);
668  return;
669  }
670 
671  // command history (ctrl-p ctrl-n for unix style)
672

813 /*
814 ===================
815 Key_StringToKeynum
816 
817 Returns a key number to be used to index keys[] by looking at
818 the given string. Single ascii characters return themselves, while
819 the K_* names are matched up.
820 
821 0x11 will be interpreted as raw hex, which will allow new controlers
822 
823 to be configured even if they don't have defined names.
824 ===================
825 */
826 int Key_StringToKeynum( char *str ) {
827  keyname_t *kn;
828  int n;
829 
830  if ( !str || !str[0] ) {
831  return -1;
832  }
833  if ( !str[1] ) {
834  return tolower( str[0] );
835  }
836 
837  // check for hex code
838  n = Com_HexStrToInt( str );
839  if ( n >= 0 && n < MAX_KEYS ) {
840  return n;
841  }
842 
843  // scan for a text match
844  for ( kn=keynames ; kn->name ; kn++ ) {
845  if ( !Q_stricmp( str,kn->name ) )
846  return kn->keynum;
847  }
848 
849  return -1;
850 }

852 /*
853 ===================
854 Key_KeynumToString
855 
856 Returns a string (either a single ascii char, a K_* name, or a 0x11 hex string) for the
857 given keynum.
858 ===================
859 */
860 char *Key_KeynumToString( int keynum ) {
861  keyname_t *kn;
862  static char tinystr[5];
863  int i, j;
864 
865  if ( keynum == -1 ) {
866  return "<KEY NOT FOUND>";
867  }
868 
869  if ( keynum < 0 || keynum >= MAX_KEYS ) {
870  return "<OUT OF RANGE>";
871  }
872 
873  // check for printable ascii (don't use quote)
874  if ( keynum > 32 && keynum < 127 && keynum != '"' && keynum != ';' ) {
875  tinystr[0] = keynum;
876  tinystr[1] = 0;
877  return tinystr;
878  }
879 
880  // check for a key string
881  for ( kn=keynames ; kn->name ; kn++ ) {
882  if (keynum == kn->keynum) {
883  return kn->name;
884  }
885  }
886 
887  // make a hex string
888  i = keynum >> 4;
889  j = keynum & 15;
890 
891  tinystr[0] = '0';
892  tinystr[1] = 'x';
893  tinystr[2] = i > 9 ? i - 10 + 'a' : i + '0';
894  tinystr[3] = j > 9 ? j - 10 + 'a' : j + '0';
895  tinystr[4] = 0;
896 
897  return tinystr;
898 }

1170 /*

1172 CL_ParseBinding
1173 
1174 Execute the commands in the bind string
1175 ===================
1176 */
1177 void CL_ParseBinding( int key, qboolean down, unsigned time )
1178 {

1180  qboolean allCommands, allowUpCmds;
1181 
1182  if( clc.state == CA_DISCONNECTED && Key_GetCatcher( ) == 0 )
1183  return;
1184  if( !keys[key].binding || !keys[key].binding[0] )
1185  return;
1186  Q_strncpyz( buf, keys[key].binding, sizeof( buf ) );
1187

1190 
1191  // allow button up commands if in game even if key catcher is set
1192  allowUpCmds = ( clc.state != CA_DISCONNECTED );
1193 
1194  while( 1 )
1195  {
1196  while( isspace( *p ) )
1197  p++;
1198  end = strchr( p, ';' );
1199  if( end )
1200  *end = '\0';

1205  // subframe corrected

1207  char cmd[1024];

1211  }
1212  }
1213  else if( down )
1214  {

1216  if ( allCommands || CL_BindUICommand( p ) ) {
1217  Cbuf_AddText( p );

1219  }
1220  }

1223  p = end + 1;
1224  }
1225 }
