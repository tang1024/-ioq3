42 #define RF_CROSSHAIR 0x0010 // This item is a cross hair and will draw over everything similar to
43  // DEPTHHACK in stereo rendering mode, with the difference that the
44  // projection matrix won't be hacked to reduce the stereo separation as
45  // is done for the gun.

47 #define RF_NOSHADOW 0x0040 // don't add stencil shadows

49 #define RF_LIGHTING_ORIGIN 0x0080 // use refEntity->lightingOrigin instead of refEntity->origin
