30 /*
31 ==============
32 CG_CheckAmmo
33 
34 If the ammo has gone low enough to generate the warning, play a sound
35 ==============
36 */
37 void CG_CheckAmmo(void)
38 {
39  int i;
40  int total;
41  int previous;
42  int weapons;
43 
44  // see about how many seconds of ammo we have remaining
45  weapons = cg.snap->ps.stats[STAT_WEAPONS];
46  total = 0;
47  for (i = WP_MACHINEGUN; i < WP_NUM_WEAPONS; i++) {
48  if (!(weapons & (1 << i))) {
49  continue;
50  }
51  if (cg.snap->ps.ammo[i] < 0) {
52  continue;
53  }
54  switch (i) {
55  case WP_ROCKET_LAUNCHER:
56  case WP_GRENADE_LAUNCHER:
57  case WP_RAILGUN:
58  case WP_SHOTGUN:
59 #ifdef MISSIONPACK
60  case WP_PROX_LAUNCHER:
61 #endif
62  total += cg.snap->ps.ammo[i] * 1000;
63  break;
64  default:
65  total += cg.snap->ps.ammo[i] * 200;
66  break;
67  }
68  if (total >= 5000) {
69  cg.lowAmmoWarning = 0;
70  return;
71  }
72  }
73 
74  previous = cg.lowAmmoWarning;
75 
76  if (total == 0) {
77  cg.lowAmmoWarning = 2;
78  } else {
79  cg.lowAmmoWarning = 1;
80  }
81 
82  // play a sound on transitions
83  if (cg.lowAmmoWarning != previous) {
84  trap_S_StartLocalSound(cgs.media.noAmmoSound, CHAN_LOCAL_SOUND);
85  }
86 }

88 /*
89 ==============
90 CG_DamageFeedback
91 ==============
92 */
93 void CG_DamageFeedback(int yawByte, int pitchByte, int damage)
94 {
95  float left, front, up;
96  float kick;
97  int health;
98  float scale;
99  vec3_t dir;
100  vec3_t angles;
101  float dist;
102  float yaw, pitch;
103 
104  // show the attacking player's head and name in corner
105  cg.attackerTime = cg.time;
106 
107  // the lower on health you are, the greater the view kick will be
108  health = cg.snap->ps.stats[STAT_HEALTH];
109  if (health < 40) {
110  scale = 1;
111  } else {
112  scale = 40.0 / health;
113  }
114  kick = damage * scale;
115 
116  if (kick < 5)
117  kick = 5;
118  if (kick > 10)
119  kick = 10;
120 
121  // if yaw and pitch are both 255, make the damage always centered (falling, etc)
122  if (yawByte == 255 && pitchByte == 255) {
123  cg.damageX = 0;
124  cg.damageY = 0;
125  cg.v_dmg_roll = 0;
126  cg.v_dmg_pitch = -kick;
127  } else {
128  // positional
129  pitch = pitchByte / 255.0 * 360;
130  yaw = yawByte / 255.0 * 360;
131 
132  angles[PITCH] = pitch;
133  angles[YAW] = yaw;
134  angles[ROLL] = 0;
135 
136  AngleVectors(angles, dir, NULL, NULL);
137  VectorSubtract(vec3_origin, dir, dir);
138 
139  front = DotProduct(dir, cg.refdef.viewaxis[0]);
140  left = DotProduct(dir, cg.refdef.viewaxis[1]);
141  up = DotProduct(dir, cg.refdef.viewaxis[2]);
142 
143  dir[0] = front;
144  dir[1] = left;
145  dir[2] = 0;
146  dist = VectorLength(dir);
147  if (dist < 0.1) {
148  dist = 0.1f;
149  }
150 
151  cg.v_dmg_roll = kick * left;
152 
153  cg.v_dmg_pitch = -kick * front;
154 
155  if (front <= 0.1) {
156  front = 0.1f;
157  }
158  cg.damageX = -left / front;
159  cg.damageY = up / dist;
160  }
161 
162  // clamp the position
163  if (cg.damageX > 1.0) {
164  cg.damageX = 1.0;
165  }
166  if (cg.damageX < -1.0) {
167  cg.damageX = -1.0;
168  }
169 
170  if (cg.damageY > 1.0) {
171  cg.damageY = 1.0;
172  }
173  if (cg.damageY < -1.0) {
174  cg.damageY = -1.0;
175  }
176 
177  // don't let the screen flashes vary as much
178  if (kick > 10) {
179  kick = 10;
180  }
181  cg.damageValue = kick;
182  cg.v_dmg_time = cg.time + DAMAGE_TIME;
183  cg.damageTime = cg.snap->serverTime;
184 }

215 void CG_CheckPlayerstateEvents(playerState_t* ps, playerState_t* ops)
216 {
217  int i;
218  int event;
219  centity_t* cent;
220 
221  if (ps->externalEvent && ps->externalEvent != ops->externalEvent) {
222  cent = &cg_entities[ps->clientNum];
223  cent->currentState.event = ps->externalEvent;
224  cent->currentState.eventParm = ps->externalEventParm;
225  CG_EntityEvent(cent, cent->lerpOrigin);
226  }
227 
228  cent = &cg.predictedPlayerEntity; // cg_entities[ ps->clientNum ];
229  // go through the predictable events buffer
230  for (i = ps->eventSequence - MAX_PS_EVENTS; i < ps->eventSequence; i++) {
231  // if we have a new predictable event
232  if (i >= ops->eventSequence
233  // or the server told us to play another event instead of a predicted event we already issued
234  // or something the server told us changed our prediction causing a different event
235  || (i > ops->eventSequence - MAX_PS_EVENTS && ps->events[i & (MAX_PS_EVENTS - 1)] != ops->events[i & (MAX_PS_EVENTS - 1)])) {
236 
237  event = ps->events[i & (MAX_PS_EVENTS - 1)];
238  cent->currentState.event = event;
239  cent->currentState.eventParm = ps->eventParms[i & (MAX_PS_EVENTS - 1)];
240  CG_EntityEvent(cent, cent->lerpOrigin);
241 
242  cg.predictableEvents[i & (MAX_PREDICTED_EVENTS - 1)] = event;
243 
244  cg.eventSequence++;
245  }
246  }
247 }

254 void CG_CheckChangedPredictableEvents(playerState_t* ps)
255 {
256  int i;
257  int event;

259 
260  cent = &cg.predictedPlayerEntity;
261  for (i = ps->eventSequence - MAX_PS_EVENTS; i < ps->eventSequence; i++) {
262  //
263  if (i >= cg.eventSequence) {
264  continue;
265  }
266  // if this event is not further back in than the maximum predictable events we remember
267  if (i > cg.eventSequence - MAX_PREDICTED_EVENTS) {
268  // if the new playerstate event is different from a previously predicted one
269  if (ps->events[i & (MAX_PS_EVENTS - 1)] != cg.predictableEvents[i & (MAX_PREDICTED_EVENTS - 1)]) {
270 
271  event = ps->events[i & (MAX_PS_EVENTS - 1)];
272  cent->currentState.event = event;
273  cent->currentState.eventParm = ps->eventParms[i & (MAX_PS_EVENTS - 1)];
274  CG_EntityEvent(cent, cent->lerpOrigin);
275 
276  cg.predictableEvents[i & (MAX_PREDICTED_EVENTS - 1)] = event;
277 
278  if (cg_showmiss.integer) {
279  CG_Printf("WARNING: changed predicted event\n");
280  }
281  }
282  }

302 ==================
303 CG_CheckLocalSounds

305 */
306 void CG_CheckLocalSounds(playerState_t* ps, playerState_t* ops)
307 {
308  int highScore, reward;
309 #ifdef MISSIONPACK
310  int health, armor;
311 #endif
312  sfxHandle_t sfx;
313 
314  // don't play the sounds if the player just changed teams
315  if (ps->persistant[PERS_TEAM] != ops->persistant[PERS_TEAM]) {
316  return;
317  }
318 
319  // hit changes
320  if (ps->persistant[PERS_HITS] > ops->persistant[PERS_HITS]) {
321 #ifdef MISSIONPACK
322  armor = ps->persistant[PERS_ATTACKEE_ARMOR] & 0xff;
323  health = ps->persistant[PERS_ATTACKEE_ARMOR] >> 8;
324  if (armor > 50) {
325  trap_S_StartLocalSound(cgs.media.hitSoundHighArmor, CHAN_LOCAL_SOUND);
326  } else if (armor || health > 100) {
327  trap_S_StartLocalSound(cgs.media.hitSoundLowArmor, CHAN_LOCAL_SOUND);
328  } else {
329  trap_S_StartLocalSound(cgs.media.hitSound, CHAN_LOCAL_SOUND);
330  }
331 #else
332  trap_S_StartLocalSound(cgs.media.hitSound, CHAN_LOCAL_SOUND);
333 #endif
334  } else if (ps->persistant[PERS_HITS] < ops->persistant[PERS_HITS]) {
335  trap_S_StartLocalSound(cgs.media.hitTeamSound, CHAN_LOCAL_SOUND);
336  }
337 
338  // health changes of more than -1 should make pain sounds
339  if (ps->stats[STAT_HEALTH] < ops->stats[STAT_HEALTH] - 1) {
340  if (ps->stats[STAT_HEALTH] > 0) {
341  CG_PainEvent(&cg.predictedPlayerEntity, ps->stats[STAT_HEALTH]);
342  }
343  }
344 
345 
346  // if we are going into the intermission, don't start any voices
347  if (cg.intermissionStarted) {
348  return;
349  }
350 
351  // reward sounds
352  reward = qfalse;
353  if (ps->persistant[PERS_CAPTURES] != ops->persistant[PERS_CAPTURES]) {
354  pushReward(cgs.media.captureAwardSound, cgs.media.medalCapture, ps->persistant[PERS_CAPTURES]);
355  reward = qtrue;
356  //Com_Printf("capture\n");
357  }
358  if (ps->persistant[PERS_IMPRESSIVE_COUNT] != ops->persistant[PERS_IMPRESSIVE_COUNT]) {
359 #ifdef MISSIONPACK
360  if (ps->persistant[PERS_IMPRESSIVE_COUNT] == 1) {
361  sfx = cgs.media.firstImpressiveSound;
362  } else {
363  sfx = cgs.media.impressiveSound;
364  }
365 #else
366  sfx = cgs.media.impressiveSound;
367 #endif
368  pushReward(sfx, cgs.media.medalImpressive, ps->persistant[PERS_IMPRESSIVE_COUNT]);
369  reward = qtrue;
370  //Com_Printf("impressive\n");
371  }
372  if (ps->persistant[PERS_EXCELLENT_COUNT] != ops->persistant[PERS_EXCELLENT_COUNT]) {
373 #ifdef MISSIONPACK
374  if (ps->persistant[PERS_EXCELLENT_COUNT] == 1) {
375  sfx = cgs.media.firstExcellentSound;
376  } else {
377  sfx = cgs.media.excellentSound;
378  }
379 #else
380  sfx = cgs.media.excellentSound;
381 #endif
382  pushReward(sfx, cgs.media.medalExcellent, ps->persistant[PERS_EXCELLENT_COUNT]);
383  reward = qtrue;
384  //Com_Printf("excellent\n");
385  }
386  if (ps->persistant[PERS_GAUNTLET_FRAG_COUNT] != ops->persistant[PERS_GAUNTLET_FRAG_COUNT]) {
387 #ifdef MISSIONPACK
388  if (ps->persistant[PERS_GAUNTLET_FRAG_COUNT] == 1) {
389  sfx = cgs.media.firstHumiliationSound;
390  } else {
391  sfx = cgs.media.humiliationSound;
392  }
393 #else
394  sfx = cgs.media.humiliationSound;
395 #endif
396  pushReward(sfx, cgs.media.medalGauntlet, ps->persistant[PERS_GAUNTLET_FRAG_COUNT]);
397  reward = qtrue;
398  //Com_Printf("gauntlet frag\n");
399  }
400  if (ps->persistant[PERS_DEFEND_COUNT] != ops->persistant[PERS_DEFEND_COUNT]) {
401  pushReward(cgs.media.defendSound, cgs.media.medalDefend, ps->persistant[PERS_DEFEND_COUNT]);
402  reward = qtrue;
403  //Com_Printf("defend\n");
404  }
405  if (ps->persistant[PERS_ASSIST_COUNT] != ops->persistant[PERS_ASSIST_COUNT]) {
406  pushReward(cgs.media.assistSound, cgs.media.medalAssist, ps->persistant[PERS_ASSIST_COUNT]);
407  reward = qtrue;
408  //Com_Printf("assist\n");
409  }
410  // if any of the player event bits changed
411  if (ps->persistant[PERS_PLAYEREVENTS] != ops->persistant[PERS_PLAYEREVENTS]) {
412  if ((ps->persistant[PERS_PLAYEREVENTS] & PLAYEREVENT_DENIEDREWARD) !=
413  (ops->persistant[PERS_PLAYEREVENTS] & PLAYEREVENT_DENIEDREWARD)) {
414  trap_S_StartLocalSound(cgs.media.deniedSound, CHAN_ANNOUNCER);
415  } else if ((ps->persistant[PERS_PLAYEREVENTS] & PLAYEREVENT_GAUNTLETREWARD) !=
416  (ops->persistant[PERS_PLAYEREVENTS] & PLAYEREVENT_GAUNTLETREWARD)) {
417  trap_S_StartLocalSound(cgs.media.humiliationSound, CHAN_ANNOUNCER);
418  } else if ((ps->persistant[PERS_PLAYEREVENTS] & PLAYEREVENT_HOLYSHIT) !=
419  (ops->persistant[PERS_PLAYEREVENTS] & PLAYEREVENT_HOLYSHIT)) {
420  trap_S_StartLocalSound(cgs.media.holyShitSound, CHAN_ANNOUNCER);
421  }
422  reward = qtrue;
423  }
424 
425  // check for flag pickup
426  if (cgs.gametype > GT_TEAM) {
427  if ((ps->powerups[PW_REDFLAG] != ops->powerups[PW_REDFLAG] && ps->powerups[PW_REDFLAG]) ||
428  (ps->powerups[PW_BLUEFLAG] != ops->powerups[PW_BLUEFLAG] && ps->powerups[PW_BLUEFLAG]) ||
429  (ps->powerups[PW_NEUTRALFLAG] != ops->powerups[PW_NEUTRALFLAG] && ps->powerups[PW_NEUTRALFLAG])) {
430  trap_S_StartLocalSound(cgs.media.youHaveFlagSound, CHAN_ANNOUNCER);
431  }
432  }
433 
434  // lead changes
435  if (!reward) {
436  //
437  if (!cg.warmup) {
438  // never play lead changes during warmup
439  if (ps->persistant[PERS_RANK] != ops->persistant[PERS_RANK]) {
440  if (cgs.gametype < GT_TEAM) {
441  if (ps->persistant[PERS_RANK] == 0) {
442  CG_AddBufferedSound(cgs.media.takenLeadSound);
443  } else if (ps->persistant[PERS_RANK] == RANK_TIED_FLAG) {
444  CG_AddBufferedSound(cgs.media.tiedLeadSound);
445  } else if ((ops->persistant[PERS_RANK] & ~RANK_TIED_FLAG) == 0) {
446  CG_AddBufferedSound(cgs.media.lostLeadSound);
447  }
448  }
449  }
450  }
451  }
452 
453  // timelimit warnings
454  if (cgs.timelimit > 0) {
455  int msec;
456 
457  msec = cg.time - cgs.levelStartTime;
458  if (!(cg.timelimitWarnings & 4) && msec > (cgs.timelimit * 60 + 2) * 1000) {
459  cg.timelimitWarnings |= 1 | 2 | 4;
460  trap_S_StartLocalSound(cgs.media.suddenDeathSound, CHAN_ANNOUNCER);
461  } else if (!(cg.timelimitWarnings & 2) && msec > (cgs.timelimit - 1) * 60 * 1000) {
462  cg.timelimitWarnings |= 1 | 2;
463  trap_S_StartLocalSound(cgs.media.oneMinuteSound, CHAN_ANNOUNCER);
464  } else if (cgs.timelimit > 5 && !(cg.timelimitWarnings & 1) && msec > (cgs.timelimit - 5) * 60 * 1000) {
465  cg.timelimitWarnings |= 1;
466  trap_S_StartLocalSound(cgs.media.fiveMinuteSound, CHAN_ANNOUNCER);
467  }
468  }
469 
470  // fraglimit warnings
471  if (cgs.fraglimit > 0 && cgs.gametype < GT_CTF) {
472  highScore = cgs.scores1;
473 
474  if (cgs.gametype == GT_TEAM && cgs.scores2 > highScore) {
475  highScore = cgs.scores2;
476  }
477 
478  if (!(cg.fraglimitWarnings & 4) && highScore == (cgs.fraglimit - 1)) {
479  cg.fraglimitWarnings |= 1 | 2 | 4;
480  CG_AddBufferedSound(cgs.media.oneFragSound);
481  } else if (cgs.fraglimit > 2 && !(cg.fraglimitWarnings & 2) && highScore == (cgs.fraglimit - 2)) {
482  cg.fraglimitWarnings |= 1 | 2;
483  CG_AddBufferedSound(cgs.media.twoFragSound);
484  } else if (cgs.fraglimit > 3 && !(cg.fraglimitWarnings & 1) && highScore == (cgs.fraglimit - 3)) {
485  cg.fraglimitWarnings |= 1;
486  CG_AddBufferedSound(cgs.media.threeFragSound);
487  }
488  }
489 }

491 /*
492 ===============
493 CG_TransitionPlayerState
494 
495 ===============
496 */
497 void CG_TransitionPlayerState(playerState_t* ps, playerState_t* ops)
498 {
499  // check for changing follow mode
500  if (ps->clientNum != ops->clientNum) {
501  cg.thisFrameTeleport = qtrue;
502  // make sure we don't get any unwanted transition effects
503  *ops = *ps;
504  }
505 
506  // damage events (player is getting wounded)
507  if (ps->damageEvent != ops->damageEvent && ps->damageCount) {
508  CG_DamageFeedback(ps->damageYaw, ps->damagePitch, ps->damageCount);
509  }
510 
511  // respawning
512  if (ps->persistant[PERS_SPAWN_COUNT] != ops->persistant[PERS_SPAWN_COUNT]) {
513  CG_Respawn();
514  }
515 
516  if (cg.mapRestart) {
517  CG_Respawn();
518  cg.mapRestart = qfalse;
519  }
520 
521  if (cg.snap->ps.pm_type != PM_INTERMISSION
522  && ps->persistant[PERS_TEAM] != TEAM_SPECTATOR) {
523  CG_CheckLocalSounds(ps, ops);
524  }
525 
526  // check for going low on ammo
527  CG_CheckAmmo();
528 
529  // run events
530  CG_CheckPlayerstateEvents(ps, ops);
531 
532  // smooth the ducking viewheight change
533  if (ps->viewheight != ops->viewheight) {
534  cg.duckChange = ps->viewheight - ops->viewheight;
535  cg.duckTime = cg.time;
536  }
537 }
