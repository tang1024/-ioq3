104 // negative bit values include signs
105 void MSG_WriteBits( msg_t *msg, int value, int bits ) {
106  int i;
107 
108  oldsize += bits;
109 
110  if ( msg->overflowed ) {
111  return;
112  }
113 
114  if ( bits == 0 || bits < -31 || bits > 32 ) {
115  Com_Error( ERR_DROP, "MSG_WriteBits: bad bits %i", bits );
116  }
117 
118  if ( bits < 0 ) {
119  bits = -bits;
120  }
121 
122  if ( msg->oob ) {
123  if ( msg->cursize + ( bits >> 3 ) > msg->maxsize ) {
124  msg->overflowed = qtrue;
125  return;
126  }
127 
128  if ( bits == 8 ) {
129  msg->data[msg->cursize] = value;
130  msg->cursize += 1;
131  msg->bit += 8;
132  } else if ( bits == 16 ) {
133  short temp = value;
134 
135  CopyLittleShort( &msg->data[msg->cursize], &temp );
136  msg->cursize += 2;
137  msg->bit += 16;
138  } else if ( bits==32 ) {
139  CopyLittleLong( &msg->data[msg->cursize], &value );
140  msg->cursize += 4;
141  msg->bit += 32;
142  } else {
143  Com_Error( ERR_DROP, "can't write %d bits", bits );
144  }
145  } else {
146  value &= (0xffffffff >> (32 - bits));
147  if ( bits&7 ) {
148  int nbits;
149  nbits = bits&7;
150  if ( msg->bit + nbits > msg->maxsize << 3 ) {
151  msg->overflowed = qtrue;
152  return;
153  }
154  for( i = 0; i < nbits; i++ ) {
155  Huff_putBit( (value & 1), msg->data, &msg->bit );
156  value = (value >> 1);
157  }
158  bits = bits - nbits;
159  }
160  if ( bits ) {
161  for( i = 0; i < bits; i += 8 ) {
162  Huff_offsetTransmit( &msgHuff.compressor, (value & 0xff), msg->data, &msg->bit, msg->maxsize << 3 );
163  value = (value >> 8);
164 
165  if ( msg->bit > msg->maxsize << 3 ) {
166  msg->overflowed = qtrue;
167  return;
168  }
169  }
170  }
171  msg->cursize = (msg->bit >> 3) + 1;
172  }
173 }

175 int MSG_ReadBits( msg_t *msg, int bits ) {
176  int value;
177  int get;
178  qboolean sgn;
179  int i, nbits;
180 // FILE* fp;
181 
182  if ( msg->readcount > msg->cursize ) {
183  return 0;
184  }
185 
186  value = 0;
187 
188  if ( bits < 0 ) {
189  bits = -bits;
190  sgn = qtrue;
191  } else {
192  sgn = qfalse;
193  }
194 
195  if (msg->oob) {
196  if (msg->readcount + (bits>>3) > msg->cursize) {
197  msg->readcount = msg->cursize + 1;
198  return 0;
199  }
200 
201  if(bits==8)
202  {
203  value = msg->data[msg->readcount];
204  msg->readcount += 1;
205  msg->bit += 8;
206  }
207  else if(bits==16)
208  {
209  short temp;
210 
211  CopyLittleShort(&temp, &msg->data[msg->readcount]);
212  value = temp;
213  msg->readcount += 2;
214  msg->bit += 16;
215  }
216  else if(bits==32)
217  {
218  CopyLittleLong(&value, &msg->data[msg->readcount]);
219  msg->readcount += 4;
220  msg->bit += 32;
221  }
222  else
223  Com_Error(ERR_DROP, "can't read %d bits", bits);
224  } else {
225  nbits = 0;
226  if (bits&7) {
227  nbits = bits&7;
228  if (msg->bit + nbits > msg->cursize << 3) {
229  msg->readcount = msg->cursize + 1;
230  return 0;
231  }
232  for(i=0;i<nbits;i++) {
233  value |= (Huff_getBit(msg->data, &msg->bit)<<i);
234  }
235  bits = bits - nbits;
236  }
237  if (bits) {
238 // fp = fopen("c:\\netchan.bin", "a");
239  for(i=0;i<bits;i+=8) {
240  Huff_offsetReceive (msgHuff.decompressor.tree, &get, msg->data, &msg->bit, msg->cursize<<3);
241 // fwrite(&get, 1, 1, fp);
242  value = (unsigned int)value | ((unsigned int)get<<(i+nbits));
243 
244  if (msg->bit > msg->cursize<<3) {
245  msg->readcount = msg->cursize + 1;
246  return 0;
247  }
248  }
249 // fclose(fp);
250  }
251  msg->readcount = (msg->bit>>3)+1;
252  }
253  if ( sgn && bits > 0 && bits < 32 ) {
254  if ( value & ( 1 << ( bits - 1 ) ) ) {
255  value |= -1 ^ ( ( 1 << bits ) - 1 );
256  }
257  }
258 
259  return value;
260 }

549 // a string hasher which gives the same hash value even if the
550 // string is later modified via the legacy MSG read/write code
551 int MSG_HashKey(const char *string, int maxlen) {
552  int hash, i;
553 
554  hash = 0;
555  for (i = 0; i < maxlen && string[i] != '\0'; i++) {
556  if (string[i] & 0x80 || string[i] == '%')
557  hash += '.' * (119 + i);
558  else
559  hash += string[i] * (119 + i);
560  }
561  hash = (hash ^ (hash >> 10) ^ (hash >> 20));
562  return hash;
563 }

830  // struct without updating the message fields
831  assert( numFields + 1 == sizeof( *from )/4 );

835  if ( from == NULL ) {

837  }
