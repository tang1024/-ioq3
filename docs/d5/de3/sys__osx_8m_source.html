44 dialogResult_t Sys_Dialog( dialogType_t type, const char *message, const char *title )
45 {
46  dialogResult_t result = DR_OK;
47  NSAlert *alert = [NSAlert new];
48 
49  [alert setMessageText: [NSString stringWithUTF8String: title]];
50  [alert setInformativeText: [NSString stringWithUTF8String: message]];
51 
52  if( type == DT_ERROR )
53  [alert setAlertStyle: NSCriticalAlertStyle];
54  else
55  [alert setAlertStyle: NSWarningAlertStyle];
56 
57  switch( type )
58  {
59  default:
60  [alert runModal];
61  result = DR_OK;
62  break;
63 
64  case DT_YES_NO:
65  [alert addButtonWithTitle: @"Yes"];
66  [alert addButtonWithTitle: @"No"];
67  switch( [alert runModal] )
68  {
69  default:
70  case NSAlertFirstButtonReturn: result = DR_YES; break;
71  case NSAlertSecondButtonReturn: result = DR_NO; break;
72  }
73  break;
74 
75  case DT_OK_CANCEL:
76  [alert addButtonWithTitle: @"OK"];
77  [alert addButtonWithTitle: @"Cancel"];
78 
79  switch( [alert runModal] )
80  {
81  default:
82  case NSAlertFirstButtonReturn: result = DR_OK; break;
83  case NSAlertSecondButtonReturn: result = DR_CANCEL; break;
84  }
85  break;
86  }
87 
88  [alert release];
89 
90  return result;
91 }

93 /*
94 =================
95 Sys_StripAppBundle
96 
97 Discovers if passed dir is suffixed with the directory structure of a Mac OS X
98 .app bundle. If it is, the .app directory structure is stripped off the end and
99 the result is returned. If not, dir is returned untouched.
100 =================
101 */
102 char *Sys_StripAppBundle( char *dir )
103 {
104  static char cwd[MAX_OSPATH];
105 
106  Q_strncpyz(cwd, dir, sizeof(cwd));
107  if(strcmp(Sys_Basename(cwd), "MacOS"))
108  return dir;
109  Q_strncpyz(cwd, Sys_Dirname(cwd), sizeof(cwd));
110  if(strcmp(Sys_Basename(cwd), "Contents"))
111  return dir;
112  Q_strncpyz(cwd, Sys_Dirname(cwd), sizeof(cwd));
113  if(!strstr(Sys_Basename(cwd), ".app"))
114  return dir;
115  Q_strncpyz(cwd, Sys_Dirname(cwd), sizeof(cwd));
116  return cwd;
117 }
