123 /*
124 ================
125 Sys_Milliseconds
126 ================
127 */
128 /* base time in seconds, that's our origin
129  timeval:tv_sec is an int:
130  assuming this wraps every 0x7fffffff - ~68 years since the Epoch (1970) - we're safe till 2038 */
131 unsigned long sys_timeBase = 0;
132 /* current time in ms, using sys_timeBase as origin
133  NOTE: sys_timeBase*1000 + curtime -> ms since the Epoch
134  0x7fffffff ms - ~24 days
135  although timeval:tv_usec is an int, I'm not sure wether it is actually used as an unsigned int
136  (which would affect the wrap period) */
137 int curtime;
138 int Sys_Milliseconds (void)
139 {
140  struct timeval tp;
141 
142  gettimeofday(&tp, NULL);
143 
144  if (!sys_timeBase)
145  {
146  sys_timeBase = tp.tv_sec;
147  return tp.tv_usec/1000;
148  }
149 
150  curtime = (tp.tv_sec - sys_timeBase)*1000 + tp.tv_usec/1000;
151 
152  return curtime;
153 }

317 /*
318 ==================
319 Sys_ListFilteredFiles
320 ==================
321 */
322 void Sys_ListFilteredFiles( const char *basedir, char *subdirs, char *filter, char **list, int *numfiles )
323 {
324  char search[MAX_OSPATH], newsubdirs[MAX_OSPATH];
325  char filename[MAX_OSPATH];
326  DIR *fdir;
327  struct dirent *d;
328  struct stat st;
329 
330  if ( *numfiles >= MAX_FOUND_FILES - 1 ) {
331  return;
332  }
333 
334  if (strlen(subdirs)) {
335  Com_sprintf( search, sizeof(search), "%s/%s", basedir, subdirs );
336  }
337  else {
338  Com_sprintf( search, sizeof(search), "%s", basedir );
339  }
340 
341  if ((fdir = opendir(search)) == NULL) {
342  return;
343  }
344 
345  while ((d = readdir(fdir)) != NULL) {
346  Com_sprintf(filename, sizeof(filename), "%s/%s", search, d->d_name);
347  if (stat(filename, &st) == -1)
348  continue;
349 
350  if (st.st_mode & S_IFDIR) {
351  if (Q_stricmp(d->d_name, ".") && Q_stricmp(d->d_name, "..")) {
352  if (strlen(subdirs)) {
353  Com_sprintf( newsubdirs, sizeof(newsubdirs), "%s/%s", subdirs, d->d_name);
354  }
355  else {
356  Com_sprintf( newsubdirs, sizeof(newsubdirs), "%s", d->d_name);
357  }
358  Sys_ListFilteredFiles( basedir, newsubdirs, filter, list, numfiles );
359  }
360  }
361  if ( *numfiles >= MAX_FOUND_FILES - 1 ) {
362  break;
363  }
364  Com_sprintf( filename, sizeof(filename), "%s/%s", subdirs, d->d_name );
365  if (!Com_FilterPath( filter, filename, qfalse ))
366  continue;
367  list[ *numfiles ] = CopyString( filename );
368  (*numfiles)++;
369  }
370 
371  closedir(fdir);
372 }

374 /*
375 ==================
376 Sys_ListFiles
377 ==================
378 */
379 char **Sys_ListFiles( const char *directory, const char *extension, char *filter, int *numfiles, qboolean wantsubs )
380 {
381  struct dirent *d;
382  DIR *fdir;
383  qboolean dironly = wantsubs;
384  char search[MAX_OSPATH];
385  int nfiles;
386  char **listCopy;
387  char *list[MAX_FOUND_FILES];
388  int i;
389  struct stat st;
390 
391  int extLen;
392 
393  if (filter) {
394 
395  nfiles = 0;
396  Sys_ListFilteredFiles( directory, "", filter, list, &nfiles );
397 
398  list[ nfiles ] = NULL;
399  *numfiles = nfiles;
400 
401  if (!nfiles)
402  return NULL;
403 
404  listCopy = Z_Malloc( ( nfiles + 1 ) * sizeof( *listCopy ) );
405  for ( i = 0 ; i < nfiles ; i++ ) {
406  listCopy[i] = list[i];
407  }
408  listCopy[i] = NULL;
409 
410  return listCopy;
411  }
412 
413  if ( !extension)
414  extension = "";
415 
416  if ( extension[0] == '/' && extension[1] == 0 ) {
417  extension = "";
418  dironly = qtrue;
419  }
420 
421  extLen = strlen( extension );
422 
423  // search
424  nfiles = 0;
425 
426  if ((fdir = opendir(directory)) == NULL) {
427  *numfiles = 0;
428  return NULL;
429  }
430 
431  while ((d = readdir(fdir)) != NULL) {
432  Com_sprintf(search, sizeof(search), "%s/%s", directory, d->d_name);
433  if (stat(search, &st) == -1)
434  continue;
435  if ((dironly && !(st.st_mode & S_IFDIR)) ||
436  (!dironly && (st.st_mode & S_IFDIR)))
437  continue;
438 
439  if (*extension) {
440  if ( strlen( d->d_name ) < extLen ||
441  Q_stricmp(
442  d->d_name + strlen( d->d_name ) - extLen,
443  extension ) ) {
444  continue; // didn't match
445  }
446  }
447 
448  if ( nfiles == MAX_FOUND_FILES - 1 )
449  break;
450  list[ nfiles ] = CopyString( d->d_name );
451  nfiles++;
452  }
453 
454  list[ nfiles ] = NULL;
455 
456  closedir(fdir);
457 
458  // return a copy of the list
459  *numfiles = nfiles;
460 
461  if ( !nfiles ) {
462  return NULL;
463  }
464 
465  listCopy = Z_Malloc( ( nfiles + 1 ) * sizeof( *listCopy ) );
466  for ( i = 0 ; i < nfiles ; i++ ) {
467  listCopy[i] = list[i];
468  }
469  listCopy[i] = NULL;
470 
471  return listCopy;
472 }

494 /*
495 ==================
496 Sys_Sleep
497 
498 Block execution for msec or until input is received.
499 ==================
500 */
501 void Sys_Sleep( int msec )
502 {
503  if( msec == 0 )
504  return;
505 
506  if( stdinIsATTY )
507  {
508  fd_set fdset;
509 
510  FD_ZERO(&fdset);
511  FD_SET(STDIN_FILENO, &fdset);
512  if( msec < 0 )
513  {
514  select(STDIN_FILENO + 1, &fdset, NULL, NULL, NULL);
515  }
516  else
517  {
518  struct timeval timeout;
519 
520  timeout.tv_sec = msec/1000;
521  timeout.tv_usec = (msec%1000)*1000;
522  select(STDIN_FILENO + 1, &fdset, NULL, NULL, &timeout);
523  }
524  }
525  else
526  {
527  // With nothing to select() on, we can't wait indefinitely
528  if( msec < 0 )
529  msec = 10;
530 
531  usleep( msec * 1000 );
532  }
533 }

536 ==============
537 Sys_ErrorDialog
538 
539 Display an error message
540 ==============
541 */
542 void Sys_ErrorDialog( const char *error )
543 {
544  char buffer[ 1024 ];
545  unsigned int size;
546  int f = -1;
547  const char *homepath = Cvar_VariableString( "fs_homepath" );
548  const char *gamedir = Cvar_VariableString( "fs_game" );
549  const char *fileName = "crashlog.txt";
550  char *dirpath = FS_BuildOSPath( homepath, gamedir, "");
551  char *ospath = FS_BuildOSPath( homepath, gamedir, fileName );
552 
553  Sys_Print( va( "%s\n", error ) );
554 
555 #ifndef DEDICATED
556  Sys_Dialog( DT_ERROR, va( "%s. See \"%s\" for details.", error, ospath ), "Error" );
557 #endif
558 
559  // Make sure the write path for the crashlog exists...
560 
561  if(!Sys_Mkdir(homepath))
562  {
563  Com_Printf("ERROR: couldn't create path '%s' for crash log.\n", homepath);
564  return;
565  }
566 
567  if(!Sys_Mkdir(dirpath))
568  {
569  Com_Printf("ERROR: couldn't create path '%s' for crash log.\n", dirpath);
570  return;
571  }
572 
573  // We might be crashing because we maxed out the Quake MAX_FILE_HANDLES,
574  // which will come through here, so we don't want to recurse forever by
575  // calling FS_FOpenFileWrite()...use the Unix system APIs instead.
576  f = open( ospath, O_CREAT | O_TRUNC | O_WRONLY, 0640 );
577  if( f == -1 )
578  {
579  Com_Printf( "ERROR: couldn't open %s\n", fileName );
580  return;
581  }
582 
583  // We're crashing, so we don't care much if write() or close() fails.
584  while( ( size = CON_LogRead( buffer, sizeof( buffer ) ) ) > 0 ) {
585  if( write( f, buffer, size ) != size ) {
586  Com_Printf( "ERROR: couldn't fully write to %s\n", fileName );
587  break;
588  }
589  }
590 
591  close( f );
592 }

631 /*
632 ==============
633 Sys_Exec
634 ==============
635 */
636 static int Sys_Exec( void )
637 {
638  pid_t pid = fork( );
639 
640  if( pid < 0 )
641  return -1;
642 
643  if( pid )
644  {
645  // Parent
646  int exitCode;
647 
648  wait( &exitCode );
649 
650  return WEXITSTATUS( exitCode );
651  }
652  else
653  {
654  // Child
655  execvp( execArgv[ 0 ], execArgv );
656 
657  // Failed to execute
658  exit( -1 );
659 
660  return -1;
661  }
662 }

746 
747 Display a *nix dialog box
748 ==============
749 */
750 dialogResult_t Sys_Dialog( dialogType_t type, const char *message, const char *title )
751 {
752  typedef enum
753  {
754  NONE = 0,
755  ZENITY,
756  KDIALOG,
757  XMESSAGE,
758  NUM_DIALOG_PROGRAMS
759  } dialogCommandType_t;
760  typedef void (*dialogCommandBuilder_t)( dialogType_t, const char *, const char * );
761 
762  const char *session = getenv( "DESKTOP_SESSION" );
763  qboolean tried[ NUM_DIALOG_PROGRAMS ] = { qfalse };
764  dialogCommandBuilder_t commands[ NUM_DIALOG_PROGRAMS ] = { NULL };
765  dialogCommandType_t preferredCommandType = NONE;
766  int i;
767 
768  commands[ ZENITY ] = &Sys_ZenityCommand;
769  commands[ KDIALOG ] = &Sys_KdialogCommand;
770  commands[ XMESSAGE ] = &Sys_XmessageCommand;
771 
772  // This may not be the best way
773  if( !Q_stricmp( session, "gnome" ) )
774  preferredCommandType = ZENITY;
775  else if( !Q_stricmp( session, "kde" ) )
776  preferredCommandType = KDIALOG;
777 
778  for( i = NONE + 1; i < NUM_DIALOG_PROGRAMS; i++ )
779  {
780  if( preferredCommandType != NONE && preferredCommandType != i )
781  continue;
782 
783  if( !tried[ i ] )
784  {
785  int exitCode;
786 
787  commands[ i ]( type, message, title );
788  exitCode = Sys_Exec( );
789 
790  if( exitCode >= 0 )
791  {
792  switch( type )
793  {
794  case DT_YES_NO: return exitCode ? DR_NO : DR_YES;
795  case DT_OK_CANCEL: return exitCode ? DR_CANCEL : DR_OK;
796  default: return DR_OK;
797  }
798  }
799 
800  tried[ i ] = qtrue;
801 
802  // The preference failed, so start again in order
803  if( preferredCommandType != NONE )
804  {
805  preferredCommandType = NONE;
806  i = NONE + 1;
807  }
808  }
809  }
810 
811  Com_DPrintf( S_COLOR_YELLOW "WARNING: failed to show a dialog\n" );
812  return DR_OK;
813 }
