85 {

87 #ifdef MISSIONPACK
88  CG_BuildSpectatorString();
89 #endif
90  if (cg.scoresRequestTime + 2000 < cg.time) {
91  // the scores are more than two seconds out of data,
92  // so request new ones
93  cg.scoresRequestTime = cg.time;
94  trap_SendClientCommand("score");
95 
96  // leave the current scores up if they were already
97  // displayed, but if this is the first hit, clear them out
98  if (!cg.showScores) {

101  }
102  } else {
103  // show the cached contents even if they just pressed if it
104  // is within two seconds
105  cg.showScores = qtrue;
106  }

262 // ASS U ME's enumeration order as far as task specific orders, OFFENSE is zero, CAMP is last
263 //
264 static void CG_NextOrder_f(void)
265 {
266  clientInfo_t* ci = cgs.clientinfo + cg.snap->ps.clientNum;
267  if (ci) {
268  if (!ci->teamLeader && sortedTeamPlayers[cg_currentSelectedPlayer.integer] != cg.snap->ps.clientNum) {
269  return;
270  }
271  }
272  if (cgs.currentOrder < TEAMTASK_CAMP) {
273  cgs.currentOrder++;
274 
275  if (cgs.currentOrder == TEAMTASK_RETRIEVE) {
276  if (!CG_OtherTeamHasFlag()) {
277  cgs.currentOrder++;
278  }
279  }
280 
281  if (cgs.currentOrder == TEAMTASK_ESCORT) {
282  if (!CG_YourTeamHasFlag()) {
283  cgs.currentOrder++;
284  }
285  }
286 
287  } else {
288  cgs.currentOrder = TEAMTASK_OFFENSE;
289  }
290  cgs.orderPending = qtrue;
291  cgs.orderTime = cg.time + 3000;
292 }

538 
539 The string has been tokenized and can be retrieved with
540 Cmd_Argc() / Cmd_Argv()
541 =================
542 */
543 qboolean CG_ConsoleCommand(void)
544 {
545  const char* cmd;
546  int i;
547 
548  cmd = CG_Argv(0);
549 
550  for (i = 0; i < ARRAY_LEN(commands); i++) {
551  if (!Q_stricmp(cmd, commands[i].cmd)) {
552  commands[i].function();
553  return qtrue;
554  }
555  }
556 
557  return qfalse;
558 }
