29 #define MAX_LOCAL_ENTITIES 512
30 localEntity_t cg_localEntities[MAX_LOCAL_ENTITIES];

32 localEntity_t* cg_freeLocalEntities; // single linked list

34 /*
35 ===================
36 CG_InitLocalEntities
37 
38 This is called at startup and for tournement restarts
39 ===================
40 */
41 void CG_InitLocalEntities(void)
42 {
43  int i;
44 
45  memset(cg_localEntities, 0, sizeof(cg_localEntities));
46  cg_activeLocalEntities.next = &cg_activeLocalEntities;
47  cg_activeLocalEntities.prev = &cg_activeLocalEntities;
48  cg_freeLocalEntities = cg_localEntities;
49  for (i = 0; i < MAX_LOCAL_ENTITIES - 1; i++) {
50  cg_localEntities[i].next = &cg_localEntities[i + 1];
51  }
52 }

55 /*
56 ==================
57 CG_FreeLocalEntity
58 ==================
59 */
60 void CG_FreeLocalEntity(localEntity_t* le)
61 {
62  if (!le->prev) {
63  CG_Error("CG_FreeLocalEntity: not active");
64  }
65 
66  // remove from the doubly linked active list
67  le->prev->next = le->next;
68  le->next->prev = le->prev;
69 
70  // the free list is only singly linked
71  le->next = cg_freeLocalEntities;
72  cg_freeLocalEntities = le;
73 }
74 
75 /*
76 ===================
77 CG_AllocLocalEntity
78 
79 Will always succeed, even if it requires freeing an old active entity
80 ===================
81 */
82 localEntity_t* CG_AllocLocalEntity(void)
83 {
84  localEntity_t* le;
85 
86  if (!cg_freeLocalEntities) {
87  // no free entities, so free the one at the end of the chain
88  // remove the oldest active entity
89  CG_FreeLocalEntity(cg_activeLocalEntities.prev);
90  }
91 
92  le = cg_freeLocalEntities;
93  cg_freeLocalEntities = cg_freeLocalEntities->next;
94 
95  memset(le, 0, sizeof(*le));
96 
97  // link into the active list
98  le->next = cg_activeLocalEntities.next;
99  le->prev = &cg_activeLocalEntities;
100  cg_activeLocalEntities.next->prev = le;
101  cg_activeLocalEntities.next = le;
102  return le;
103 }

117 /*
118 ================
119 CG_BloodTrail
120 
121 Leave expanding blood puffs behind gibs
122 ================
123 */
124 void CG_BloodTrail(localEntity_t* le)
125 {
126  int t;
127  int t2;
128  int step;
129  vec3_t newOrigin;
130  localEntity_t* blood;
131 
132  step = 150;
133  t = step * ((cg.time - cg.frametime + step) / step);
134  t2 = step * (cg.time / step);
135 
136  for (; t <= t2; t += step) {
137  BG_EvaluateTrajectory(&le->pos, t, newOrigin);
138 
139  blood = CG_SmokePuff(newOrigin, vec3_origin,
140  20, // radius
141  1, 1, 1, 1, // color
142  2000, // trailTime
143  t, // startTime
144  0, // fadeInTime
145  0, // flags
146  cgs.media.bloodTrailShader);
147  // use the optimized version
148  blood->leType = LE_FALL_SCALE_FADE;
149  // drop a total of 40 units over its lifetime
150  blood->pos.trDelta[2] = 40;
151  }
152 }

214 /*
215 ================
216 CG_ReflectVelocity
217 ================
218 */
219 void CG_ReflectVelocity(localEntity_t* le, trace_t* trace)
220 {
221  vec3_t velocity;
222  float dot;
223  int hitTime;
224 
225  // reflect the velocity on the trace plane
226  hitTime = cg.time - cg.frametime + cg.frametime * trace->fraction;
227  BG_EvaluateTrajectoryDelta(&le->pos, hitTime, velocity);
228  dot = DotProduct(velocity, trace->plane.normal);
229  VectorMA(velocity, -2 * dot, trace->plane.normal, le->pos.trDelta);
230 
231  VectorScale(le->pos.trDelta, le->bounceFactor, le->pos.trDelta);
232 
233  VectorCopy(trace->endpos, le->pos.trBase);
234  le->pos.trTime = cg.time;
235 
236 
237  // check for stop, making sure that even on low FPS systems it doesn't bobble
238  if (trace->allsolid ||
239  (trace->plane.normal[2] > 0 &&
240  (le->pos.trDelta[2] < 40 || le->pos.trDelta[2] < -cg.frametime * le->pos.trDelta[2]))) {
241  le->pos.trType = TR_STATIONARY;
242  } else {
243 
244  }
245 }

247 /*
248 ================
249 CG_AddFragment
250 ================
251 */
252 void CG_AddFragment(localEntity_t* le)
253 {
254  vec3_t newOrigin;
255  trace_t trace;
256 
257  if (le->pos.trType == TR_STATIONARY) {
258  // sink into the ground if near the removal time
259  int t;
260  float oldZ;
261 
262  t = le->endTime - cg.time;
263  if (t < SINK_TIME) {
264  // we must use an explicit lighting origin, otherwise the
265  // lighting would be lost as soon as the origin went
266  // into the ground
267  VectorCopy(le->refEntity.origin, le->refEntity.lightingOrigin);
268  le->refEntity.renderfx |= RF_LIGHTING_ORIGIN;
269  oldZ = le->refEntity.origin[2];
270  le->refEntity.origin[2] -= 16 * (1.0 - (float)t / SINK_TIME);
271  trap_R_AddRefEntityToScene(&le->refEntity);
272  le->refEntity.origin[2] = oldZ;
273  } else {
274  trap_R_AddRefEntityToScene(&le->refEntity);
275  }
276 
277  return;
278  }
279 
280  // calculate new position
281  BG_EvaluateTrajectory(&le->pos, cg.time, newOrigin);
282 
283  // trace a line from previous position to new position
284  CG_Trace(&trace, le->refEntity.origin, NULL, NULL, newOrigin, -1, CONTENTS_SOLID);
285  if (trace.fraction == 1.0) {
286  // still in free fall
287  VectorCopy(newOrigin, le->refEntity.origin);
288 
289  if (le->leFlags & LEF_TUMBLE) {
290  vec3_t angles;
291 
292  BG_EvaluateTrajectory(&le->angles, cg.time, angles);
293  AnglesToAxis(angles, le->refEntity.axis);
294  }
295 
296  trap_R_AddRefEntityToScene(&le->refEntity);
297 
298  // add a blood trail
299  if (le->leBounceSoundType == LEBS_BLOOD) {
300  CG_BloodTrail(le);
301  }
302 
303  return;
304  }
305 
306  // if it is in a nodrop zone, remove it
307  // this keeps gibs from waiting at the bottom of pits of death
308  // and floating levels
309  if (CG_PointContents(trace.endpos, 0) & CONTENTS_NODROP) {
310  CG_FreeLocalEntity(le);
311  return;
312  }
313 
314  // leave a mark
315  CG_FragmentBounceMark(le, &trace);
316 
317  // do a bouncy sound
318  CG_FragmentBounceSound(le, &trace);
319 
320  // reflect the velocity on the trace plane
321  CG_ReflectVelocity(le, &trace);
322 
323  trap_R_AddRefEntityToScene(&le->refEntity);
324 }

358 /*
359 ==================
360 CG_AddMoveScaleFade
361 ==================
362 */
363 static void CG_AddMoveScaleFade(localEntity_t* le)
364 {
365  refEntity_t* re;
366  float c;
367  vec3_t delta;
368  float len;
369 
370  re = &le->refEntity;
371 
372  if (le->fadeInTime > le->startTime&& cg.time < le->fadeInTime) {
373  // fade / grow time
374  c = 1.0 - (float)(le->fadeInTime - cg.time) / (le->fadeInTime - le->startTime);
375  } else {
376  // fade / grow time
377  c = (le->endTime - cg.time) * le->lifeRate;
378  }
379 
380  re->shaderRGBA[3] = 0xff * c * le->color[3];
381 
382  if (!(le->leFlags & LEF_PUFF_DONT_SCALE)) {
383  re->radius = le->radius * (1.0 - c) + 8;
384  }
385 
386  BG_EvaluateTrajectory(&le->pos, cg.time, re->origin);
387 
388  // if the view would be "inside" the sprite, kill the sprite
389  // so it doesn't add too much overdraw
390  VectorSubtract(re->origin, cg.refdef.vieworg, delta);
391  len = VectorLength(delta);
392  if (len < le->radius) {
393  CG_FreeLocalEntity(le);
394  return;
395  }
396 
397  trap_R_AddRefEntityToScene(re);
398 }

731 void CG_AddScorePlum(localEntity_t* le)
732 {
733  refEntity_t* re;
734  vec3_t origin, delta, dir, vec, up = { 0, 0, 1 };
735  float c, len;
736  int i, score, digits[10], numdigits, negative;
737 
738  re = &le->refEntity;
739 
740  c = (le->endTime - cg.time) * le->lifeRate;
741 
742  score = le->radius;
743  if (score < 0) {
744  re->shaderRGBA[0] = 0xff;
745  re->shaderRGBA[1] = 0x11;
746  re->shaderRGBA[2] = 0x11;
747  } else {
748  re->shaderRGBA[0] = 0xff;
749  re->shaderRGBA[1] = 0xff;
750  re->shaderRGBA[2] = 0xff;
751  if (score >= 50) {
752  re->shaderRGBA[1] = 0;
753  } else if (score >= 20) {
754  re->shaderRGBA[0] = re->shaderRGBA[1] = 0;
755  } else if (score >= 10) {
756  re->shaderRGBA[2] = 0;
757  } else if (score >= 2) {
758  re->shaderRGBA[0] = re->shaderRGBA[2] = 0;
759  }
760 
761  }
762  if (c < 0.25)
763  re->shaderRGBA[3] = 0xff * 4 * c;
764  else
765  re->shaderRGBA[3] = 0xff;
766 
767  re->radius = NUMBER_SIZE / 2;
768 
769  VectorCopy(le->pos.trBase, origin);
770  origin[2] += 110 - c * 100;
771 
772  VectorSubtract(cg.refdef.vieworg, origin, dir);
773  CrossProduct(dir, up, vec);
774  VectorNormalize(vec);
775 
776  VectorMA(origin, -10 + 20 * sin(c * 2 * M_PI), vec, origin);
777 
778  // if the view would be "inside" the sprite, kill the sprite
779  // so it doesn't add too much overdraw
780  VectorSubtract(origin, cg.refdef.vieworg, delta);
781  len = VectorLength(delta);
782  if (len < 20) {
783  CG_FreeLocalEntity(le);
784  return;
785  }
786 
787  negative = qfalse;
788  if (score < 0) {
789  negative = qtrue;
790  score = -score;
791  }
792 
793  for (numdigits = 0; !(numdigits && !score); numdigits++) {
794  digits[numdigits] = score % 10;
795  score = score / 10;
796  }
797 
798  if (negative) {
799  digits[numdigits] = 10;
800  numdigits++;
801  }
802 
803  for (i = 0; i < numdigits; i++) {
804  VectorMA(origin, (float)(((float)numdigits / 2) - i) * NUMBER_SIZE, vec, re->origin);
805  re->customShader = cgs.media.numberShaders[digits[numdigits - 1 - i]];
806  trap_R_AddRefEntityToScene(re);
807  }
808 }
