37 /*
38 ====================
39 CG_BuildSolidList
40 
41 When a new cg.snap has been set, this function builds a sublist
42 of the entities that are actually solid, to make for more
43 efficient collision detection
44 ====================
45 */
46 void CG_BuildSolidList(void)
47 {
48  int i;
49  centity_t* cent;
50  snapshot_t* snap;
51  entityState_t* ent;
52 
53  cg_numSolidEntities = 0;
54  cg_numTriggerEntities = 0;
55 
56  if (cg.nextSnap && !cg.nextFrameTeleport && !cg.thisFrameTeleport) {
57  snap = cg.nextSnap;
58  } else {
59  snap = cg.snap;
60  }
61 
62  for (i = 0; i < snap->numEntities; i++) {
63  cent = &cg_entities[snap->entities[i].number];
64  ent = &cent->currentState;
65 
66  if (ent->eType == ET_ITEM || ent->eType == ET_PUSH_TRIGGER || ent->eType == ET_TELEPORT_TRIGGER) {
67  cg_triggerEntities[cg_numTriggerEntities] = cent;
68  cg_numTriggerEntities++;
69  continue;
70  }
71 
72  if (cent->nextState.solid) {
73  cg_solidEntities[cg_numSolidEntities] = cent;
74  cg_numSolidEntities++;
75  continue;
76  }
77  }
78 }

86 static void CG_ClipMoveToEntities(const vec3_t start, const vec3_t mins, const vec3_t maxs, const vec3_t end,
87  int skipNumber, int mask, trace_t* tr)
88 {
89  int i, x, zd, zu;
90  trace_t trace;
91  entityState_t* ent;
92  clipHandle_t cmodel;
93  vec3_t bmins, bmaxs;
94  vec3_t origin, angles;
95  centity_t* cent;
96 
97  for (i = 0; i < cg_numSolidEntities; i++) {
98  cent = cg_solidEntities[i];
99  ent = &cent->currentState;
100 
101  if (ent->number == skipNumber) {
102  continue;
103  }
104 
105  if (ent->solid == SOLID_BMODEL) {
106  // special value for bmodel
107  cmodel = trap_CM_InlineModel(ent->modelindex);
108  VectorCopy(cent->lerpAngles, angles);
109  BG_EvaluateTrajectory(&cent->currentState.pos, cg.physicsTime, origin);
110  } else {
111  // encoded bbox
112  x = (ent->solid & 255);
113  zd = ((ent->solid >> 8) & 255);
114  zu = ((ent->solid >> 16) & 255) - 32;
115 
116  bmins[0] = bmins[1] = -x;
117  bmaxs[0] = bmaxs[1] = x;
118  bmins[2] = -zd;
119  bmaxs[2] = zu;
120 
121  cmodel = trap_CM_TempBoxModel(bmins, bmaxs);
122  VectorCopy(vec3_origin, angles);
123  VectorCopy(cent->lerpOrigin, origin);
124  }
125 
126 
127  trap_CM_TransformedBoxTrace(&trace, start, end,
128  mins, maxs, cmodel, mask, origin, angles);
129 
130  if (trace.allsolid || trace.fraction < tr->fraction) {
131  trace.entityNum = ent->number;
132  *tr = trace;
133  } else if (trace.startsolid) {
134  tr->startsolid = qtrue;
135  }
136  if (tr->allsolid) {
137  return;
138  }
139  }
140 }

147 void CG_Trace(trace_t* result, const vec3_t start, const vec3_t mins, const vec3_t maxs, const vec3_t end,
148  int skipNumber, int mask)
149 {
150  trace_t t;
151 
152  trap_CM_BoxTrace(&t, start, end, mins, maxs, 0, mask);
153  t.entityNum = t.fraction != 1.0 ? ENTITYNUM_WORLD : ENTITYNUM_NONE;
154  // check all other solid models
155  CG_ClipMoveToEntities(start, mins, maxs, end, skipNumber, mask, &t);
156 
157  *result = t;

200 /*
201 ========================
202 CG_InterpolatePlayerState
203 
204 Generates cg.predictedPlayerState by interpolating between
205 cg.snap->player_state and cg.nextFrame->player_state
206 ========================
207 */
208 static void CG_InterpolatePlayerState(qboolean grabAngles)
209 {
210  float f;
211  int i;
212  playerState_t* out;
213  snapshot_t* prev, * next;
214 
215  out = &cg.predictedPlayerState;
216  prev = cg.snap;
217  next = cg.nextSnap;
218 
219  *out = cg.snap->ps;
220 
221  // if we are still allowing local input, short circuit the view angles
222  if (grabAngles) {
223  usercmd_t cmd;
224  int cmdNum;
225 
226  cmdNum = trap_GetCurrentCmdNumber();
227  trap_GetUserCmd(cmdNum, &cmd);
228 
229  PM_UpdateViewAngles(out, &cmd);
230  }
231 
232  // if the next frame is a teleport, we can't lerp to it
233  if (cg.nextFrameTeleport) {
234  return;
235  }
236 
237  if (!next || next->serverTime <= prev->serverTime) {
238  return;
239  }
240 
241  f = (float)(cg.time - prev->serverTime) / (next->serverTime - prev->serverTime);
242 
243  i = next->ps.bobCycle;
244  if (i < prev->ps.bobCycle) {
245  i += 256; // handle wraparound
246  }
247  out->bobCycle = prev->ps.bobCycle + f * (i - prev->ps.bobCycle);
248 
249  for (i = 0; i < 3; i++) {
250  out->origin[i] = prev->ps.origin[i] + f * (next->ps.origin[i] - prev->ps.origin[i]);
251  if (!grabAngles) {
252  out->viewangles[i] = LerpAngle(
253  prev->ps.viewangles[i], next->ps.viewangles[i], f);
254  }
255  out->velocity[i] = prev->ps.velocity[i] +
256  f * (next->ps.velocity[i] - prev->ps.velocity[i]);
257  }
258 
259 }

261 /*
262 ===================
263 CG_TouchItem
264 ===================
265 */
266 static void CG_TouchItem(centity_t* cent)
267 {
268  gitem_t* item;
269 
270  if (!cg_predictItems.integer) {
271  return;
272  }
273  if (!BG_PlayerTouchesItem(&cg.predictedPlayerState, &cent->currentState, cg.time)) {
274  return;
275  }
276 
277  // never pick an item up twice in a prediction
278  if (cent->miscTime == cg.time) {
279  return;
280  }
281 
282  if (!BG_CanItemBeGrabbed(cgs.gametype, &cent->currentState, &cg.predictedPlayerState)) {
283  return; // can't hold it
284  }
285 
286  item = &bg_itemlist[cent->currentState.modelindex];
287 
288  // Special case for flags.
289  // We don't predict touching our own flag
290 #ifdef MISSIONPACK
291  if (cgs.gametype == GT_1FCTF) {
292  if (item->giType == IT_TEAM && item->giTag != PW_NEUTRALFLAG) {
293  return;
294  }
295  }
296 #endif
297  if (cgs.gametype == GT_CTF) {
298  if (cg.predictedPlayerState.persistant[PERS_TEAM] == TEAM_RED &&
299  item->giType == IT_TEAM && item->giTag == PW_REDFLAG)
300  return;
301  if (cg.predictedPlayerState.persistant[PERS_TEAM] == TEAM_BLUE &&
302  item->giType == IT_TEAM && item->giTag == PW_BLUEFLAG)
303  return;
304  }
305 
306  // grab it
307  BG_AddPredictableEventToPlayerstate(EV_ITEM_PICKUP, cent->currentState.modelindex, &cg.predictedPlayerState);
308 
309  // remove it from the frame so it won't be drawn
310  cent->currentState.eFlags |= EF_NODRAW;
311 
312  // don't touch it again this prediction
313  cent->miscTime = cg.time;
314 
315  // if it's a weapon, give them some predicted ammo so the autoswitch will work
316  if (item->giType == IT_WEAPON) {
317  cg.predictedPlayerState.stats[STAT_WEAPONS] |= 1 << item->giTag;
318  if (!cg.predictedPlayerState.ammo[item->giTag]) {
319  cg.predictedPlayerState.ammo[item->giTag] = 1;
320  }
321  }
322 }

325 /*
326 =========================
327 CG_TouchTriggerPrediction
328 
329 Predict push triggers and items
330 =========================
331 */
332 static void CG_TouchTriggerPrediction(void)
333 {
334  int i;
335  trace_t trace;
336  entityState_t* ent;
337  clipHandle_t cmodel;
338  centity_t* cent;
339  qboolean spectator;
340 
341  // dead clients don't activate triggers
342  if (cg.predictedPlayerState.stats[STAT_HEALTH] <= 0) {
343  return;
344  }
345 
346  spectator = (cg.predictedPlayerState.pm_type == PM_SPECTATOR);
347 
348  if (cg.predictedPlayerState.pm_type != PM_NORMAL && !spectator) {
349  return;
350  }
351 
352  for (i = 0; i < cg_numTriggerEntities; i++) {
353  cent = cg_triggerEntities[i];
354  ent = &cent->currentState;
355 
356  if (ent->eType == ET_ITEM && !spectator) {
357  CG_TouchItem(cent);
358  continue;
359  }
360 
361  if (ent->solid != SOLID_BMODEL) {
362  continue;
363  }
364 
365  cmodel = trap_CM_InlineModel(ent->modelindex);
366  if (!cmodel) {
367  continue;
368  }
369 
370  trap_CM_BoxTrace(&trace, cg.predictedPlayerState.origin, cg.predictedPlayerState.origin,
371  cg_pmove.mins, cg_pmove.maxs, cmodel, -1);
372 
373  if (!trace.startsolid) {
374  continue;
375  }
376 
377  if (ent->eType == ET_TELEPORT_TRIGGER) {
378  cg.hyperspace = qtrue;
379  } else if (ent->eType == ET_PUSH_TRIGGER) {
380  BG_TouchJumpPad(&cg.predictedPlayerState, ent);
381  }
382  }
383 
384  // if we didn't touch a jump pad this pmove frame
385  if (cg.predictedPlayerState.jumppad_frame != cg.predictedPlayerState.pmove_framecount) {
386  cg.predictedPlayerState.jumppad_frame = 0;
387  cg.predictedPlayerState.jumppad_ent = 0;
388  }
389 }

393 /*
394 =================
395 CG_PredictPlayerState
396 
397 Generates cg.predictedPlayerState for the current cg.time
398 cg.predictedPlayerState is guaranteed to be valid after exiting.
399 
400 For demo playback, this will be an interpolation between two valid
401 playerState_t.
402 
403 For normal gameplay, it will be the result of predicted usercmd_t on
404 top of the most recent playerState_t received from the server.
405 
406 Each new snapshot will usually have one or more new usercmd over the last,
407 but we simulate all unacknowledged commands each time, not just the new ones.
408 This means that on an internet connection, quite a few pmoves may be issued
409 each frame.
410 
411 OPTIMIZE: don't re-simulate unless the newly arrived snapshot playerState_t
412 differs from the predicted one. Would require saving all intermediate
413 playerState_t during prediction.
414 
415 We detect prediction errors and allow them to be decayed off over several frames
416 to ease the jerk.
417 =================
418 */
419 void CG_PredictPlayerState(void)
420 {
421  int cmdNum, current;
422  playerState_t oldPlayerState;
423  qboolean moved;
424  usercmd_t oldestCmd;
425  usercmd_t latestCmd;
426 
427  cg.hyperspace = qfalse; // will be set if touching a trigger_teleport
428 
429  // if this is the first frame we must guarantee
430  // predictedPlayerState is valid even if there is some
431  // other error condition
432  if (!cg.validPPS) {
433  cg.validPPS = qtrue;
434  cg.predictedPlayerState = cg.snap->ps;
435  }
436 
437 
438  // demo playback just copies the moves
439  if (cg.demoPlayback || (cg.snap->ps.pm_flags & PMF_FOLLOW)) {
440  CG_InterpolatePlayerState(qfalse);
441  return;
442  }
443 
444  // non-predicting local movement will grab the latest angles
445  if (cg_nopredict.integer || cg_synchronousClients.integer) {
446  CG_InterpolatePlayerState(qtrue);
447  return;
448  }
449 
450  // prepare for pmove
451  cg_pmove.ps = &cg.predictedPlayerState;
452  cg_pmove.trace = CG_Trace;
453  cg_pmove.pointcontents = CG_PointContents;
454  if (cg_pmove.ps->pm_type == PM_DEAD) {
455  cg_pmove.tracemask = MASK_PLAYERSOLID & ~CONTENTS_BODY;
456  } else {
457  cg_pmove.tracemask = MASK_PLAYERSOLID;
458  }
459  if (cg.snap->ps.persistant[PERS_TEAM] == TEAM_SPECTATOR) {
460  cg_pmove.tracemask &= ~CONTENTS_BODY; // spectators can fly through bodies
461  }
462  cg_pmove.noFootsteps = (cgs.dmflags & DF_NO_FOOTSTEPS) > 0;
463 
464  // save the state before the pmove so we can detect transitions
465  oldPlayerState = cg.predictedPlayerState;
466 
467  current = trap_GetCurrentCmdNumber();
468 
469  // if we don't have the commands right after the snapshot, we
470  // can't accurately predict a current position, so just freeze at
471  // the last good position we had
472  cmdNum = current - CMD_BACKUP + 1;
473  trap_GetUserCmd(cmdNum, &oldestCmd);
474  if (oldestCmd.serverTime > cg.snap->ps.commandTime
475  && oldestCmd.serverTime < cg.time) { // special check for map_restart
476  if (cg_showmiss.integer) {
477  CG_Printf("exceeded PACKET_BACKUP on commands\n");
478  }
479  return;
480  }
481 
482  // get the latest command so we can know which commands are from previous map_restarts
483  trap_GetUserCmd(current, &latestCmd);
484 
485  // get the most recent information we have, even if
486  // the server time is beyond our current cg.time,
487  // because predicted player positions are going to
488  // be ahead of everything else anyway
489  if (cg.nextSnap && !cg.nextFrameTeleport && !cg.thisFrameTeleport) {
490  cg.predictedPlayerState = cg.nextSnap->ps;
491  cg.physicsTime = cg.nextSnap->serverTime;
492  } else {
493  cg.predictedPlayerState = cg.snap->ps;
494  cg.physicsTime = cg.snap->serverTime;
495  }
496 
497  if (pmove_msec.integer < 8) {
498  trap_Cvar_Set("pmove_msec", "8");
499  trap_Cvar_Update(&pmove_msec);
500  } else if (pmove_msec.integer > 33) {
501  trap_Cvar_Set("pmove_msec", "33");
502  trap_Cvar_Update(&pmove_msec);
503  }
504 
505  cg_pmove.pmove_fixed = pmove_fixed.integer;// | cg_pmove_fixed.integer;
506  cg_pmove.pmove_msec = pmove_msec.integer;
507 
508  // run cmds
509  moved = qfalse;
510  for (cmdNum = current - CMD_BACKUP + 1; cmdNum <= current; cmdNum++) {
511  // get the command
512  trap_GetUserCmd(cmdNum, &cg_pmove.cmd);
513 
514  if (cg_pmove.pmove_fixed) {
515  PM_UpdateViewAngles(cg_pmove.ps, &cg_pmove.cmd);
516  }
517 
518  // don't do anything if the time is before the snapshot player time
519  if (cg_pmove.cmd.serverTime <= cg.predictedPlayerState.commandTime) {
520  continue;
521  }
522 
523  // don't do anything if the command was from a previous map_restart
524  if (cg_pmove.cmd.serverTime > latestCmd.serverTime) {
525  continue;
526  }
527 
528  // check for a prediction error from last frame
529  // on a lan, this will often be the exact value
530  // from the snapshot, but on a wan we will have
531  // to predict several commands to get to the point
532  // we want to compare
533  if (cg.predictedPlayerState.commandTime == oldPlayerState.commandTime) {
534  vec3_t delta;
535  float len;
536 
537  if (cg.thisFrameTeleport) {
538  // a teleport will not cause an error decay
539  VectorClear(cg.predictedError);
540  if (cg_showmiss.integer) {
541  CG_Printf("PredictionTeleport\n");
542  }
543  cg.thisFrameTeleport = qfalse;
544  } else {
545  vec3_t adjusted, new_angles;
546  CG_AdjustPositionForMover(cg.predictedPlayerState.origin,
547  cg.predictedPlayerState.groundEntityNum, cg.physicsTime, cg.oldTime, adjusted, cg.predictedPlayerState.viewangles, new_angles);
548 
549  if (cg_showmiss.integer) {
550  if (!VectorCompare(oldPlayerState.origin, adjusted)) {
551  CG_Printf("prediction error\n");
552  }
553  }
554  VectorSubtract(oldPlayerState.origin, adjusted, delta);
555  len = VectorLength(delta);
556  if (len > 0.1) {
557  if (cg_showmiss.integer) {
558  CG_Printf("Prediction miss: %f\n", len);
559  }
560  if (cg_errorDecay.integer) {
561  int t;
562  float f;
563 
564  t = cg.time - cg.predictedErrorTime;
565  f = (cg_errorDecay.value - t) / cg_errorDecay.value;
566  if (f < 0) {
567  f = 0;
568  }
569  if (f > 0 && cg_showmiss.integer) {
570  CG_Printf("Double prediction decay: %f\n", f);
571  }
572  VectorScale(cg.predictedError, f, cg.predictedError);
573  } else {
574  VectorClear(cg.predictedError);
575  }
576  VectorAdd(delta, cg.predictedError, cg.predictedError);
577  cg.predictedErrorTime = cg.oldTime;
578  }
579  }
580  }
581 
582  // don't predict gauntlet firing, which is only supposed to happen
583  // when it actually inflicts damage
584  cg_pmove.gauntletHit = qfalse;
585 
586  if (cg_pmove.pmove_fixed) {
587  cg_pmove.cmd.serverTime = ((cg_pmove.cmd.serverTime + pmove_msec.integer - 1) / pmove_msec.integer) * pmove_msec.integer;
588  }
589 
590  Pmove(&cg_pmove);
591 
592  moved = qtrue;
593 
594  // add push trigger movement effects
595  CG_TouchTriggerPrediction();
596 
597  // check for predictable events that changed from previous predictions
598  //CG_CheckChangedPredictableEvents(&cg.predictedPlayerState);
599  }
600 
601  if (cg_showmiss.integer > 1) {
602  CG_Printf("[%i : %i] ", cg_pmove.cmd.serverTime, cg.time);
603  }
604 
605  if (!moved) {
606  if (cg_showmiss.integer) {
607  CG_Printf("not moved\n");
608  }
609  return;
610  }
611 
612  // adjust for the movement of the groundentity
613  CG_AdjustPositionForMover(cg.predictedPlayerState.origin,
614  cg.predictedPlayerState.groundEntityNum,
615  cg.physicsTime, cg.time, cg.predictedPlayerState.origin, cg.predictedPlayerState.viewangles, cg.predictedPlayerState.viewangles);
616 
617  if (cg_showmiss.integer) {
618  if (cg.predictedPlayerState.eventSequence > oldPlayerState.eventSequence + MAX_PS_EVENTS) {
619  CG_Printf("WARNING: dropped event\n");
620  }
621  }
622 
623  // fire events and other transition triggered things
624  CG_TransitionPlayerState(&cg.predictedPlayerState, &oldPlayerState);
625 
626  if (cg_showmiss.integer) {
627  if (cg.eventSequence > cg.predictedPlayerState.eventSequence) {
628  CG_Printf("WARNING: double event\n");
629  cg.eventSequence = cg.predictedPlayerState.eventSequence;
630  }
631  }
632 }
