228  st[1] = 0;

269 /*
270 ================
271 Team_FragBonuses
272 
273 Calculate the bonuses for flag defense, flag carrier defense, etc.
274 Note that bonuses are not cumulative. You get one, they are in importance
275 order.
276 ================
277 */
278 void Team_FragBonuses(gentity_t* targ, gentity_t* inflictor, gentity_t* attacker)
279 {
280  int i;
281  gentity_t* ent;
282  int flag_pw, enemy_flag_pw;
283  int otherteam;
284  int tokens;
285  gentity_t* flag, * carrier = NULL;
286  char* c;
287  vec3_t v1, v2;
288  int team;
289 
290  // no bonus for fragging yourself or team mates
291  if (!targ->client || !attacker->client || targ == attacker || OnSameTeam(targ, attacker))
292  return;
293 
294  team = targ->client->sess.sessionTeam;
295  otherteam = OtherTeam(targ->client->sess.sessionTeam);
296  if (otherteam < 0)
297  return; // whoever died isn't on a team
298 
299  // same team, if the flag at base, check to he has the enemy flag
300  if (team == TEAM_RED) {
301  flag_pw = PW_REDFLAG;
302  enemy_flag_pw = PW_BLUEFLAG;
303  } else {
304  flag_pw = PW_BLUEFLAG;
305  enemy_flag_pw = PW_REDFLAG;
306  }
307 
308 #ifdef MISSIONPACK
309  if (g_gametype.integer == GT_1FCTF) {
310  flag_pw = PW_NEUTRALFLAG;
311  enemy_flag_pw = PW_NEUTRALFLAG;
312  }
313 #endif
314 
315  // did the attacker frag the flag carrier?
316  tokens = 0;
317 #ifdef MISSIONPACK
318  if (g_gametype.integer == GT_HARVESTER) {
319  tokens = targ->client->ps.generic1;
320  }
321 #endif
322  if (targ->client->ps.powerups[enemy_flag_pw]) {
323  attacker->client->pers.teamState.lastfraggedcarrier = level.time;
324  AddScore(attacker, targ->r.currentOrigin, CTF_FRAG_CARRIER_BONUS);
325  attacker->client->pers.teamState.fragcarrier++;
326  PrintMsg(NULL, "%s" S_COLOR_WHITE " fragged %s's flag carrier!\n",
327  attacker->client->pers.netname, TeamName(team));
328 
329  // the target had the flag, clear the hurt carrier
330  // field on the other team
331  for (i = 0; i < g_maxclients.integer; i++) {
332  ent = g_entities + i;
333  if (ent->inuse && ent->client->sess.sessionTeam == otherteam)
334  ent->client->pers.teamState.lasthurtcarrier = 0;
335  }
336  return;
337  }
338 
339  // did the attacker frag a head carrier? other->client->ps.generic1
340  if (tokens) {
341  attacker->client->pers.teamState.lastfraggedcarrier = level.time;
342  AddScore(attacker, targ->r.currentOrigin, CTF_FRAG_CARRIER_BONUS * tokens * tokens);
343  attacker->client->pers.teamState.fragcarrier++;
344  PrintMsg(NULL, "%s" S_COLOR_WHITE " fragged %s's skull carrier!\n",
345  attacker->client->pers.netname, TeamName(team));
346 
347  // the target had the flag, clear the hurt carrier
348  // field on the other team
349  for (i = 0; i < g_maxclients.integer; i++) {
350  ent = g_entities + i;
351  if (ent->inuse && ent->client->sess.sessionTeam == otherteam)
352  ent->client->pers.teamState.lasthurtcarrier = 0;
353  }
354  return;
355  }
356 
357  if (targ->client->pers.teamState.lasthurtcarrier &&
358  level.time - targ->client->pers.teamState.lasthurtcarrier < CTF_CARRIER_DANGER_PROTECT_TIMEOUT &&
359  !attacker->client->ps.powerups[flag_pw]) {
360  // attacker is on the same team as the flag carrier and
361  // fragged a guy who hurt our flag carrier
362  AddScore(attacker, targ->r.currentOrigin, CTF_CARRIER_DANGER_PROTECT_BONUS);
363 
364  attacker->client->pers.teamState.carrierdefense++;
365  targ->client->pers.teamState.lasthurtcarrier = 0;
366 
367  attacker->client->ps.persistant[PERS_DEFEND_COUNT]++;
368  // add the sprite over the player's head
369  attacker->client->ps.eFlags &= ~(EF_AWARD_IMPRESSIVE | EF_AWARD_EXCELLENT | EF_AWARD_GAUNTLET | EF_AWARD_ASSIST | EF_AWARD_DEFEND | EF_AWARD_CAP);
370  attacker->client->ps.eFlags |= EF_AWARD_DEFEND;
371  attacker->client->rewardTime = level.time + REWARD_SPRITE_TIME;
372 
373  return;
374  }
375 
376  // flag and flag carrier area defense bonuses
377 
378  // we have to find the flag and carrier entities
379 
380 #ifdef MISSIONPACK
381  if (g_gametype.integer == GT_OBELISK) {
382  // find the team obelisk
383  switch (attacker->client->sess.sessionTeam) {
384  case TEAM_RED:
385  c = "team_redobelisk";
386  break;
387  case TEAM_BLUE:
388  c = "team_blueobelisk";
389  break;
390  default:
391  return;
392  }
393 
394  } else if (g_gametype.integer == GT_HARVESTER) {
395  // find the center obelisk
396  c = "team_neutralobelisk";
397  } else {
398 #endif
399  // find the flag
400  switch (attacker->client->sess.sessionTeam) {
401  case TEAM_RED:
402  c = "team_CTF_redflag";
403  break;
404  case TEAM_BLUE:
405  c = "team_CTF_blueflag";
406  break;
407  default:
408  return;
409  }
410  // find attacker's team's flag carrier
411  for (i = 0; i < g_maxclients.integer; i++) {
412  carrier = g_entities + i;
413  if (carrier->inuse && carrier->client->ps.powerups[flag_pw])
414  break;
415  carrier = NULL;
416  }
417 #ifdef MISSIONPACK
418  }
419 #endif
420  flag = NULL;
421  while ((flag = G_Find(flag, FOFS(classname), c)) != NULL) {
422  if (!(flag->flags & FL_DROPPED_ITEM))
423  break;
424  }
425 
426  if (!flag)
427  return; // can't find attacker's flag
428 
429  // ok we have the attackers flag and a pointer to the carrier
430 
431  // check to see if we are defending the base's flag
432  VectorSubtract(targ->r.currentOrigin, flag->r.currentOrigin, v1);
433  VectorSubtract(attacker->r.currentOrigin, flag->r.currentOrigin, v2);
434 
435  if (((VectorLength(v1) < CTF_TARGET_PROTECT_RADIUS &&
436  trap_InPVS(flag->r.currentOrigin, targ->r.currentOrigin)) ||
437  (VectorLength(v2) < CTF_TARGET_PROTECT_RADIUS &&
438  trap_InPVS(flag->r.currentOrigin, attacker->r.currentOrigin))) &&
439  attacker->client->sess.sessionTeam != targ->client->sess.sessionTeam) {
440 
441  // we defended the base flag
442  AddScore(attacker, targ->r.currentOrigin, CTF_FLAG_DEFENSE_BONUS);
443  attacker->client->pers.teamState.basedefense++;
444 
445  attacker->client->ps.persistant[PERS_DEFEND_COUNT]++;
446  // add the sprite over the player's head
447  attacker->client->ps.eFlags &= ~(EF_AWARD_IMPRESSIVE | EF_AWARD_EXCELLENT | EF_AWARD_GAUNTLET | EF_AWARD_ASSIST | EF_AWARD_DEFEND | EF_AWARD_CAP);
448  attacker->client->ps.eFlags |= EF_AWARD_DEFEND;
449  attacker->client->rewardTime = level.time + REWARD_SPRITE_TIME;
450 
451  return;
452  }
453 
454  if (carrier && carrier != attacker) {
455  VectorSubtract(targ->r.currentOrigin, carrier->r.currentOrigin, v1);
456  VectorSubtract(attacker->r.currentOrigin, carrier->r.currentOrigin, v2);
457 
458  if (((VectorLength(v1) < CTF_ATTACKER_PROTECT_RADIUS &&
459  trap_InPVS(carrier->r.currentOrigin, targ->r.currentOrigin)) ||
460  (VectorLength(v2) < CTF_ATTACKER_PROTECT_RADIUS &&
461  trap_InPVS(carrier->r.currentOrigin, attacker->r.currentOrigin))) &&
462  attacker->client->sess.sessionTeam != targ->client->sess.sessionTeam) {
463  AddScore(attacker, targ->r.currentOrigin, CTF_CARRIER_PROTECT_BONUS);
464  attacker->client->pers.teamState.carrierdefense++;
465 
466  attacker->client->ps.persistant[PERS_DEFEND_COUNT]++;
467  // add the sprite over the player's head
468  attacker->client->ps.eFlags &= ~(EF_AWARD_IMPRESSIVE | EF_AWARD_EXCELLENT | EF_AWARD_GAUNTLET | EF_AWARD_ASSIST | EF_AWARD_DEFEND | EF_AWARD_CAP);
469  attacker->client->ps.eFlags |= EF_AWARD_DEFEND;
470  attacker->client->rewardTime = level.time + REWARD_SPRITE_TIME;
471 
472  return;
473  }
474  }
475 }
