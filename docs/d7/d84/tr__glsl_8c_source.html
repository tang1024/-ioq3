163 static void GLSL_PrintLog(GLuint programOrShader, glslPrintLog_t type, qboolean developerOnly)
164 {
165  char *msg;
166  static char msgPart[1024];
167  int maxLength = 0;
168  int i;
169  int printLevel = developerOnly ? PRINT_DEVELOPER : PRINT_ALL;
170 
171  switch (type)
172  {
173  case GLSL_PRINTLOG_PROGRAM_INFO:
174  ri.Printf(printLevel, "Program info log:\n");
175  qglGetProgramiv(programOrShader, GL_INFO_LOG_LENGTH, &maxLength);
176  break;
177 
178  case GLSL_PRINTLOG_SHADER_INFO:
179  ri.Printf(printLevel, "Shader info log:\n");
180  qglGetShaderiv(programOrShader, GL_INFO_LOG_LENGTH, &maxLength);
181  break;
182 
183  case GLSL_PRINTLOG_SHADER_SOURCE:
184  ri.Printf(printLevel, "Shader source:\n");
185  qglGetShaderiv(programOrShader, GL_SHADER_SOURCE_LENGTH, &maxLength);
186  break;
187  }
188 
189  if (maxLength <= 0)
190  {
191  ri.Printf(printLevel, "None.\n");
192  return;
193  }
194 
195  if (maxLength < 1023)
196  msg = msgPart;
197  else
198  msg = ri.Malloc(maxLength);
199 
200  switch (type)
201  {
202  case GLSL_PRINTLOG_PROGRAM_INFO:
203  qglGetProgramInfoLog(programOrShader, maxLength, &maxLength, msg);
204  break;
205 
206  case GLSL_PRINTLOG_SHADER_INFO:
207  qglGetShaderInfoLog(programOrShader, maxLength, &maxLength, msg);
208  break;
209 
210  case GLSL_PRINTLOG_SHADER_SOURCE:
211  qglGetShaderSource(programOrShader, maxLength, &maxLength, msg);
212  break;
213  }
214 
215  if (maxLength < 1023)
216  {
217  msgPart[maxLength + 1] = '\0';
218 
219  ri.Printf(printLevel, "%s\n", msgPart);
220  }
221  else
222  {
223  for(i = 0; i < maxLength; i += 1023)
224  {
225  Q_strncpyz(msgPart, msg + i, sizeof(msgPart));
226 
227  ri.Printf(printLevel, "%s", msgPart);
228  }
229 
230  ri.Printf(printLevel, "\n");
231 
232  ri.Free(msg);
233  }

368 static int GLSL_CompileGPUShader(GLuint program, GLuint *prevShader, const GLchar *buffer, int size, GLenum shaderType)
369 {
370  GLint compiled;
371  GLuint shader;
372 
373  shader = qglCreateShader(shaderType);
374 
375  qglShaderSource(shader, 1, (const GLchar **)&buffer, &size);
376 
377  // compile shader
378  qglCompileShader(shader);
379 
380  // check if shader compiled
381  qglGetShaderiv(shader, GL_COMPILE_STATUS, &compiled);
382  if(!compiled)
383  {
384  GLSL_PrintLog(shader, GLSL_PRINTLOG_SHADER_SOURCE, qfalse);
385  GLSL_PrintLog(shader, GLSL_PRINTLOG_SHADER_INFO, qfalse);
386  ri.Error(ERR_DROP, "Couldn't compile shader");
387  return 0;
388  }
389 
390  if (*prevShader)
391  {
392  qglDetachShader(program, *prevShader);
393  qglDeleteShader(*prevShader);
394  }
395 
396  // attach shader to program
397  qglAttachShader(program, shader);
398 
399  *prevShader = shader;
400 
401  return 1;
402 }

404 static int GLSL_LoadGPUShaderText(const char *name, const char *fallback,
405  GLenum shaderType, char *dest, int destSize)
406 {
407  char filename[MAX_QPATH];
408  GLchar *buffer = NULL;

411  int result;
412 
413  if(shaderType == GL_VERTEX_SHADER)
414  {
415  Com_sprintf(filename, sizeof(filename), "glsl/%s_vp.glsl", name);
416  }
417  else
418  {
419  Com_sprintf(filename, sizeof(filename), "glsl/%s_fp.glsl", name);
420  }
421 
422  if ( r_externalGLSL->integer ) {
423  size = ri.FS_ReadFile(filename, (void **)&buffer);
424  } else {
425  size = 0;
426  buffer = NULL;
427  }
428 
429  if(!buffer)
430  {
431  if (fallback)
432  {
433  ri.Printf(PRINT_DEVELOPER, "...loading built-in '%s'\n", filename);
434  shaderText = fallback;
435  size = strlen(shaderText);
436  }
437  else
438  {
439  ri.Printf(PRINT_DEVELOPER, "couldn't load '%s'\n", filename);
440  return 0;
441  }
442  }
443  else
444  {
445  ri.Printf(PRINT_DEVELOPER, "...loading '%s'\n", filename);
446  shaderText = buffer;
447  }
448 
449  if (size > destSize)
450  {
451  result = 0;
452  }
453  else
454  {
455  Q_strncpyz(dest, shaderText, size + 1);
456  result = 1;
457  }
458 
459  if (buffer)
460  {
461  ri.FS_FreeFile(buffer);
462  }
463 
464  return result;
465 }
