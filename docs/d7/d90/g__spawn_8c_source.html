264 ===============
265 G_CallSpawn
266 
267 Finds the spawn function for the entity and calls it,
268 returning qfalse if not found
269 ===============
270 */
271 qboolean G_CallSpawn(gentity_t* ent)
272 {
273  spawn_t* s;
274  gitem_t* item;
275 
276  if (!ent->classname) {
277  G_Printf("G_CallSpawn: NULL classname\n");
278  return qfalse;
279  }
280 
281  // check item spawn functions
282  for (item = bg_itemlist + 1; item->classname; item++) {
283  if (!strcmp(item->classname, ent->classname)) {
284  G_SpawnItem(ent, item);
285  return qtrue;
286  }
287  }
288 
289  // check normal spawn functions
290  for (s = spawns; s->name; s++) {
291  if (!strcmp(s->name, ent->classname)) {
292  // found it
293  s->spawn(ent);
294  return qtrue;
295  }
296  }
297  G_Printf("%s doesn't have a spawn function\n", ent->classname);
298  return qfalse;
299 }

301 /*
302 =============
303 G_NewString
304 
305 Builds a copy of the string, translating \n to real linefeeds
306 so message texts can be multi-line
307 =============
308 */
309 char* G_NewString(const char* string)
310 {
311  char* newb, * new_p;
312  int i, l;
313 
314  l = strlen(string) + 1;
315 
316  newb = G_Alloc(l);
317 
318  new_p = newb;
319 
320  // turn \n into a real linefeed
321  for (i = 0; i < l; i++) {
322  if (string[i] == '\\' && i < l - 1) {
323  i++;
324  if (string[i] == 'n') {
325  *new_p++ = '\n';
326  } else {
327  *new_p++ = '\\';
328  }
329  } else {
330  *new_p++ = string[i];
331  }
332  }
333 
334  return newb;
335 }

340 /*
341 ===============
342 G_ParseField
343 
344 Takes a key/value pair and sets the binary values
345 in a gentity
346 ===============
347 */
348 void G_ParseField(const char* key, const char* value, gentity_t* ent)
349 {
350  field_t* f;
351  byte* b;
352  float v;
353  vec3_t vec;
354 
355  for (f = fields; f->name; f++) {
356  if (!Q_stricmp(f->name, key)) {
357  // found it
358  b = (byte*)ent;
359 
360  switch (f->type) {
361  case F_STRING:
362  *(char**)(b + f->ofs) = G_NewString(value);
363  break;
364  case F_VECTOR:
365  sscanf(value, "%f %f %f", &vec[0], &vec[1], &vec[2]);
366  ((float*)(b + f->ofs))[0] = vec[0];
367  ((float*)(b + f->ofs))[1] = vec[1];
368  ((float*)(b + f->ofs))[2] = vec[2];
369  break;
370  case F_INT:
371  *(int*)(b + f->ofs) = atoi(value);
372  break;
373  case F_FLOAT:
374  *(float*)(b + f->ofs) = atof(value);
375  break;
376  case F_ANGLEHACK:
377  v = atof(value);
378  ((float*)(b + f->ofs))[0] = 0;
379  ((float*)(b + f->ofs))[1] = v;
380  ((float*)(b + f->ofs))[2] = 0;
381  break;
382  }
383  return;
384  }
385  }
386 }

507 /*
508 ====================
509 G_ParseSpawnVars
510 
511 Parses a brace bounded set of key / value pairs out of the
512 level's entity strings into level.spawnVars[]
513 
514 This does not actually spawn an entity.
515 ====================
516 */
517 qboolean G_ParseSpawnVars(void)
518 {
519  char keyname[MAX_TOKEN_CHARS];
520  char com_token[MAX_TOKEN_CHARS];
521 
522  level.numSpawnVars = 0;
523  level.numSpawnVarChars = 0;
524 
525  // parse the opening brace
526  if (!trap_GetEntityToken(com_token, sizeof(com_token))) {
527  // end of spawn string
528  return qfalse;
529  }
530  if (com_token[0] != '{') {
531  G_Error("G_ParseSpawnVars: found %s when expecting {", com_token);
532  }
533 
534  // go through all the key / value pairs
535  while (1) {
536  // parse key
537  if (!trap_GetEntityToken(keyname, sizeof(keyname))) {
538  G_Error("G_ParseSpawnVars: EOF without closing brace");
539  }
540 
541  if (keyname[0] == '}') {
542  break;
543  }
544 
545  // parse value
546  if (!trap_GetEntityToken(com_token, sizeof(com_token))) {
547  G_Error("G_ParseSpawnVars: EOF without closing brace");
548  }
549 
550  if (com_token[0] == '}') {
551  G_Error("G_ParseSpawnVars: closing brace without data");
552  }
553  if (level.numSpawnVars == MAX_SPAWN_VARS) {
554  G_Error("G_ParseSpawnVars: MAX_SPAWN_VARS");
555  }
556  level.spawnVars[level.numSpawnVars][0] = G_AddSpawnVarToken(keyname);
557  level.spawnVars[level.numSpawnVars][1] = G_AddSpawnVarToken(com_token);
558  level.numSpawnVars++;
559  }
560 
561  return qtrue;
562 }
