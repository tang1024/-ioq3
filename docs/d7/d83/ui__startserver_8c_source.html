108 /*
109 =================
110 GametypeBits
111 =================
112 */
113 static int GametypeBits( char *string ) {
114  int bits;
115  char *p;
116  char *token;
117 
118  bits = 0;
119  p = string;
120  while( 1 ) {
121  token = COM_ParseExt( &p, qfalse );
122  if ( !token[0] ) {
123  break;
124  }
125 
126  if( Q_stricmp( token, "ffa" ) == 0 ) {
127  bits |= 1 << GT_FFA;
128  continue;
129  }
130 
131  if( Q_stricmp( token, "tourney" ) == 0 ) {
132  bits |= 1 << GT_TOURNAMENT;
133  continue;
134  }
135 
136  if( Q_stricmp( token, "single" ) == 0 ) {
137  bits |= 1 << GT_SINGLE_PLAYER;
138  continue;
139  }
140 
141  if( Q_stricmp( token, "team" ) == 0 ) {
142  bits |= 1 << GT_TEAM;
143  continue;
144  }
145 
146  if( Q_stricmp( token, "ctf" ) == 0 ) {
147  bits |= 1 << GT_CTF;
148  continue;
149  }
150  }
151 
152  return bits;
153 }

158 StartServer_Update

161 static void StartServer_Update( void ) {
162  int i;
163  int top;
164  static char picname[MAX_MAPSPERPAGE][64];
165  const char *info;
166  char mapname[MAX_NAMELENGTH];
167 
168  top = s_startserver.page*MAX_MAPSPERPAGE;
169 
170  for (i=0; i<MAX_MAPSPERPAGE; i++)
171  {
172  if (top+i >= s_startserver.nummaps)
173  break;
174 
175  info = UI_GetArenaInfoByNumber( s_startserver.maplist[ top + i ]);
176  Q_strncpyz( mapname, Info_ValueForKey( info, "map"), MAX_NAMELENGTH );
177  Q_strupr( mapname );
178 
179  Com_sprintf( picname[i], sizeof(picname[i]), "levelshots/%s", mapname );
180 
181  s_startserver.mappics[i].generic.flags &= ~QMF_HIGHLIGHT;
182  s_startserver.mappics[i].generic.name = picname[i];
183  s_startserver.mappics[i].shader = 0;
184 
185  // reset
186  s_startserver.mapbuttons[i].generic.flags |= QMF_PULSEIFFOCUS;
187  s_startserver.mapbuttons[i].generic.flags &= ~QMF_INACTIVE;
188  }
189 
190  for (; i<MAX_MAPSPERPAGE; i++)
191  {
192  s_startserver.mappics[i].generic.flags &= ~QMF_HIGHLIGHT;
193  s_startserver.mappics[i].generic.name = NULL;
194  s_startserver.mappics[i].shader = 0;
195 
196  // disable
197  s_startserver.mapbuttons[i].generic.flags &= ~QMF_PULSEIFFOCUS;
198  s_startserver.mapbuttons[i].generic.flags |= QMF_INACTIVE;
199  }
200 
201 
202  // no servers to start
203  if( !s_startserver.nummaps ) {
204  s_startserver.next.generic.flags |= QMF_INACTIVE;
205 
206  // set the map name
207  strcpy( s_startserver.mapname.string, "NO MAPS FOUND" );
208  }
209  else {
210  // set the highlight
211  s_startserver.next.generic.flags &= ~QMF_INACTIVE;
212  i = s_startserver.currentmap - top;
213  if ( i >=0 && i < MAX_MAPSPERPAGE )
214  {
215  s_startserver.mappics[i].generic.flags |= QMF_HIGHLIGHT;
216  s_startserver.mapbuttons[i].generic.flags &= ~QMF_PULSEIFFOCUS;
217  }
218 
219  // set the map name
220  info = UI_GetArenaInfoByNumber( s_startserver.maplist[ s_startserver.currentmap ]);
221  Q_strncpyz( s_startserver.mapname.string, Info_ValueForKey( info, "map" ), MAX_NAMELENGTH);
222  }
223 
224  Q_strupr( s_startserver.mapname.string );
225 }

243 /*
244 =================
245 StartServer_GametypeEvent
246 =================
247 */
248 static void StartServer_GametypeEvent( void* ptr, int event ) {
249  int i;
250  int count;
251  int gamebits;
252  int matchbits;
253  const char *info;
254 
255  if( event != QM_ACTIVATED) {
256  return;
257  }
258 
259  count = UI_GetNumArenas();
260  s_startserver.nummaps = 0;
261  matchbits = 1 << gametype_remap[s_startserver.gametype.curvalue];
262  if( gametype_remap[s_startserver.gametype.curvalue] == GT_FFA ) {
263  matchbits |= ( 1 << GT_SINGLE_PLAYER );
264  }
265  for( i = 0; i < count; i++ ) {
266  info = UI_GetArenaInfoByNumber( i );
267 
268  gamebits = GametypeBits( Info_ValueForKey( info, "type") );
269  if( !( gamebits & matchbits ) ) {
270  continue;
271  }
272 
273  s_startserver.maplist[ s_startserver.nummaps ] = i;
274  s_startserver.nummaps++;
275  }
276  s_startserver.maxpages = (s_startserver.nummaps + MAX_MAPSPERPAGE-1)/MAX_MAPSPERPAGE;
277  s_startserver.page = 0;
278  s_startserver.currentmap = 0;
279 
280  StartServer_Update();
281 }

301 
302  case ID_NEXTPAGE:
303  if( s_startserver.page < s_startserver.maxpages - 1 ) {
304  s_startserver.page++;

306  }

310  trap_Cvar_SetValue( "g_gameType", gametype_remap[s_startserver.gametype.curvalue] );
311  UI_ServerOptionsMenu( s_startserver.multiplayer );
312  break;
313 
314  case ID_STARTSERVERBACK:
315  UI_PopMenu();
316  break;
317  }
