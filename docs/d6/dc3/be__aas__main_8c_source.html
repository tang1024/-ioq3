110 void AAS_ContinueInit(float time)
111 {
112  //if no AAS file loaded
113  if (!aasworld.loaded) return;
114  //if AAS is already initialized
115  if (aasworld.initialized) return;
116  //calculate reachability, if not finished return
117  if (AAS_ContinueInitReachability(time)) return;
118  //initialize clustering for the new map
119  AAS_InitClustering();
120  //if reachability has been calculated and an AAS file should be written
121  //or there is a forced data optimization
122  if (aasworld.savefile || ((int)LibVarGetValue("forcewrite")))
123  {
124  //optimize the AAS data
125  if ((int)LibVarValue("aasoptimize", "0")) AAS_Optimize();
126  //save the AAS file
127  if (AAS_WriteAASFile(aasworld.filename))
128  {
129  botimport.Print(PRT_MESSAGE, "%s written successfully\n", aasworld.filename);
130  } //end if
131  else
132  {
133  botimport.Print(PRT_ERROR, "couldn't write %s\n", aasworld.filename);
134  } //end else
135  } //end if
136  //initialize the routing
137  AAS_InitRouting();
138  //at this point AAS is initialized
139  AAS_SetInitialized();
140 } //end of the function AAS_ContinueInit

204 void AAS_ProjectPointOntoVector( vec3_t point, vec3_t vStart, vec3_t vEnd, vec3_t vProj )
205 {
206  vec3_t pVec, vec;
207 
208  VectorSubtract( point, vStart, pVec );
209  VectorSubtract( vEnd, vStart, vec );
210  VectorNormalize( vec );
211  // project onto the directional vector for this segment
212  VectorMA( vStart, DotProduct( pVec, vec ), vec, vProj );
213 } //end of the function AAS_ProjectPointOntoVector

220 int AAS_LoadFiles(const char *mapname)
221 {
222  int errnum;
223  char aasfile[MAX_QPATH];
224 
225  Q_strncpyz(aasworld.mapname, mapname, sizeof(aasworld.mapname));
226  //NOTE: first reset the entity links into the AAS areas and BSP leaves
227  // the AAS link heap and BSP link heap are reset after respectively the
228  // AAS file and BSP file are loaded
229  AAS_ResetEntityLinks();
230  // load bsp info
231  AAS_LoadBSPFile();
232 
233  //load the aas file
234  Com_sprintf(aasfile, sizeof(aasfile), "maps/%s.aas", mapname);
235  errnum = AAS_LoadAASFile(aasfile);
236  if (errnum != BLERR_NOERROR)
237  return errnum;
238 
239  botimport.Print(PRT_MESSAGE, "loaded %s\n", aasfile);
240  Q_strncpyz(aasworld.filename, aasfile, sizeof(aasworld.filename));
241  return BLERR_NOERROR;
242 } //end of the function AAS_LoadFiles

250 int AAS_LoadMap(const char *mapname)
251 {
252  int errnum;
253 
254  //if no mapname is provided then the string indexes are updated
255  if (!mapname)
256  {
257  return 0;
258  } //end if
259  //
260  aasworld.initialized = qfalse;
261  //NOTE: free the routing caches before loading a new map because
262  // to free the caches the old number of areas, number of clusters
263  // and number of areas in a clusters must be available
264  AAS_FreeRoutingCaches();
265  //load the map
266  errnum = AAS_LoadFiles(mapname);
267  if (errnum != BLERR_NOERROR)
268  {
269  aasworld.loaded = qfalse;
270  return errnum;
271  } //end if
272  //
273  AAS_InitSettings();
274  //initialize the AAS link heap for the new map
275  AAS_InitAASLinkHeap();
276  //initialize the AAS linked entities for the new map
277  AAS_InitAASLinkedEntities();
278  //initialize reachability for the new map
279  AAS_InitReachability();
280  //initialize the alternative routing
281  AAS_InitAlternativeRouting();
282  //everything went ok
283  return 0;
284 } //end of the function AAS_LoadMap
