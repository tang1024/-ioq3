26 /*
27 =============================================================================
28 
29 LIGHT FLARES
30 
31 A light flare is an effect that takes place inside the eye when bright light
32 sources are visible. The size of the flare reletive to the screen is nearly
33 constant, irrespective of distance, but the intensity should be proportional to the
34 projected area of the light source.
35 
36 A surface that has been flagged as having a light flare will calculate the depth
37 buffer value that its midpoint should have when the surface is added.
38 
39 After all opaque surfaces have been rendered, the depth buffer is read back for
40 each flare in view. If the point has not been obscured by a closer surface, the
41 flare should be drawn.
42 
43 Surfaces that have a repeated texture should never be flagged as flaring, because
44 there will only be a single flare added at the midpoint of the polygon.
45 
46 To prevent abrupt popping, the intensity of the flare is interpolated up and
47 down as it changes visibility. This involves scene to scene state, unlike almost
48 all other aspects of the renderer, and is complicated by the fact that a single
49 frame may have multiple scenes.
50 
51 RB_RenderFlares() will be called once per view (twice in a mirrored scene, potentially
52 up to five or more times in a frame with 3D status bar icons).
53 
54 =============================================================================
55 */

125 RB_AddFlare
126 
127 This is called at surface tesselation time
128 ==================
129 */
130 void RB_AddFlare( void *surface, int fogNum, vec3_t point, vec3_t color, vec3_t normal ) {
131  int i;
132  flare_t *f;
133  vec3_t local;
134  float d = 1;
135  vec4_t eye, clip, normalized, window;
136 
137  backEnd.pc.c_flareAdds++;
138 
139  if(normal && (normal[0] || normal[1] || normal[2]))
140  {
141  VectorSubtract( backEnd.viewParms.or.origin, point, local );
142  VectorNormalizeFast(local);
143  d = DotProduct(local, normal);
144 
145  // If the viewer is behind the flare don't add it.
146  if(d < 0)
147  return;
148  }
149 
150  // if the point is off the screen, don't bother adding it
151  // calculate screen coordinates and depth
152  R_TransformModelToClip( point, backEnd.or.modelMatrix,
153  backEnd.viewParms.projectionMatrix, eye, clip );
154 
155  // check to see if the point is completely off screen
156  for ( i = 0 ; i < 3 ; i++ ) {
157  if ( clip[i] >= clip[3] || clip[i] <= -clip[3] ) {
158  return;
159  }
160  }
161 
162  R_TransformClipToWindow( clip, &backEnd.viewParms, normalized, window );
163 
164  if ( window[0] < 0 || window[0] >= backEnd.viewParms.viewportWidth
165  || window[1] < 0 || window[1] >= backEnd.viewParms.viewportHeight ) {
166  return; // shouldn't happen, since we check the clip[] above, except for FP rounding
167  }
168 
169  // see if a flare with a matching surface, scene, and view exists
170  for ( f = r_activeFlares ; f ; f = f->next ) {
171  if ( f->surface == surface && f->frameSceneNum == backEnd.viewParms.frameSceneNum
172  && f->inPortal == backEnd.viewParms.isPortal ) {
173  break;
174  }
175  }
176 
177  // allocate a new one
178  if (!f ) {
179  if ( !r_inactiveFlares ) {
180  // the list is completely full
181  return;
182  }
183  f = r_inactiveFlares;
184  r_inactiveFlares = r_inactiveFlares->next;
185  f->next = r_activeFlares;
186  r_activeFlares = f;
187 
188  f->surface = surface;
189  f->frameSceneNum = backEnd.viewParms.frameSceneNum;
190  f->inPortal = backEnd.viewParms.isPortal;
191  f->addedFrame = -1;
192  }
193 
194  if ( f->addedFrame != backEnd.viewParms.frameCount - 1 ) {
195  f->visible = qfalse;
196  f->fadeTime = backEnd.refdef.time - 2000;
197  }
198 
199  f->addedFrame = backEnd.viewParms.frameCount;
200  f->fogNum = fogNum;
201 
202  VectorCopy(point, f->origin);
203  VectorCopy( color, f->color );
204 
205  // fade the intensity of the flare down as the
206  // light surface turns away from the viewer
207  VectorScale( f->color, d, f->color );
208 
209  // save info needed to test
210  f->windowX = backEnd.viewParms.viewportX + window[0];
211  f->windowY = backEnd.viewParms.viewportY + window[1];
212 
213  f->eyeZ = eye[2];
214 }

271 ==================
272 RB_TestFlare
273 ==================
274 */
275 void RB_TestFlare( flare_t *f ) {
276  float depth;
277  qboolean visible;
278  float fade;
279  float screenZ;
280  FBO_t *oldFbo;
281 
282  backEnd.pc.c_flareTests++;
283 
284  // doing a readpixels is as good as doing a glFinish(), so
285  // don't bother with another sync
286  glState.finishCalled = qfalse;
287 
288  // if we're doing multisample rendering, read from the correct FBO
289  oldFbo = glState.currentFBO;
290  if (tr.msaaResolveFbo)
291  {
292  FBO_Bind(tr.msaaResolveFbo);
293  }
294 
295  // read back the z buffer contents
296  qglReadPixels( f->windowX, f->windowY, 1, 1, GL_DEPTH_COMPONENT, GL_FLOAT, &depth );
297 
298  // if we're doing multisample rendering, switch to the old FBO
299  if (tr.msaaResolveFbo)
300  {
301  FBO_Bind(oldFbo);
302  }
303 
304  screenZ = backEnd.viewParms.projectionMatrix[14] /
305  ( ( 2*depth - 1 ) * backEnd.viewParms.projectionMatrix[11] - backEnd.viewParms.projectionMatrix[10] );
306 
307  visible = ( -f->eyeZ - -screenZ ) < 24;
308 
309  if ( visible ) {
310  if ( !f->visible ) {
311  f->visible = qtrue;
312  f->fadeTime = backEnd.refdef.time - 1;
313  }
314  fade = ( ( backEnd.refdef.time - f->fadeTime ) /1000.0f ) * r_flareFade->value;
315  } else {
316  if ( f->visible ) {
317  f->visible = qfalse;
318  f->fadeTime = backEnd.refdef.time - 1;
319  }
320  fade = 1.0f - ( ( backEnd.refdef.time - f->fadeTime ) / 1000.0f ) * r_flareFade->value;
321  }
322 
323  if ( fade < 0 ) {
324  fade = 0;
325  }
326  if ( fade > 1 ) {
327  fade = 1;
328  }
329 
330  f->drawIntensity = fade;
331 }

334 /*
335 ==================
336 RB_RenderFlare
337 ==================
338 */
339 void RB_RenderFlare( flare_t *f ) {
340  float size;
341  vec3_t color;
342  int iColor[3];
343  float distance, intensity, factor;
344  byte fogFactors[3] = {255, 255, 255};
345 
346  backEnd.pc.c_flareRenders++;
347 
348  // We don't want too big values anyways when dividing by distance.
349  if(f->eyeZ > -1.0f)
350  distance = 1.0f;
351  else
352  distance = -f->eyeZ;
353 
354  // calculate the flare size..
355  size = backEnd.viewParms.viewportWidth * ( r_flareSize->value/640.0f + 8 / distance );
356 
357 /*
358  * This is an alternative to intensity scaling. It changes the size of the flare on screen instead
359  * with growing distance. See in the description at the top why this is not the way to go.
360  // size will change ~ 1/r.
361  size = backEnd.viewParms.viewportWidth * (r_flareSize->value / (distance * -2.0f));
362 */
363 
364 /*
365  * As flare sizes stay nearly constant with increasing distance we must decrease the intensity
366  * to achieve a reasonable visual result. The intensity is ~ (size^2 / distance^2) which can be
367  * got by considering the ratio of
368  * (flaresurface on screen) : (Surface of sphere defined by flare origin and distance from flare)
369  * An important requirement is:
370  * intensity <= 1 for all distances.
371  *
372  * The formula used here to compute the intensity is as follows:
373  * intensity = flareCoeff * size^2 / (distance + size*sqrt(flareCoeff))^2
374  * As you can see, the intensity will have a max. of 1 when the distance is 0.
375  * The coefficient flareCoeff will determine the falloff speed with increasing distance.
376  */
377 
378  factor = distance + size * sqrt(flareCoeff);
379 
380  intensity = flareCoeff * size * size / (factor * factor);
381 
382  VectorScale(f->color, f->drawIntensity * intensity, color);
383 
384  // Calculations for fogging
385  if(tr.world && f->fogNum > 0 && f->fogNum < tr.world->numfogs)
386  {
387  tess.numVertexes = 1;
388  VectorCopy(f->origin, tess.xyz[0]);
389  tess.fogNum = f->fogNum;
390 
391  RB_CalcModulateColorsByFog(fogFactors);
392 
393  // We don't need to render the flare if colors are 0 anyways.
394  if(!(fogFactors[0] || fogFactors[1] || fogFactors[2]))
395  return;
396  }
397 
398  iColor[0] = color[0] * fogFactors[0] * 257;
399  iColor[1] = color[1] * fogFactors[1] * 257;
400  iColor[2] = color[2] * fogFactors[2] * 257;
401 
402  RB_BeginSurface( tr.flareShader, f->fogNum, 0 );
403 
404  // FIXME: use quadstamp?
405  tess.xyz[tess.numVertexes][0] = f->windowX - size;
406  tess.xyz[tess.numVertexes][1] = f->windowY - size;
407  tess.texCoords[tess.numVertexes][0] = 0;
408  tess.texCoords[tess.numVertexes][1] = 0;
409  tess.color[tess.numVertexes][0] = iColor[0];
410  tess.color[tess.numVertexes][1] = iColor[1];
411  tess.color[tess.numVertexes][2] = iColor[2];
412  tess.color[tess.numVertexes][3] = 65535;
413  tess.numVertexes++;
414 
415  tess.xyz[tess.numVertexes][0] = f->windowX - size;
416  tess.xyz[tess.numVertexes][1] = f->windowY + size;
417  tess.texCoords[tess.numVertexes][0] = 0;
418  tess.texCoords[tess.numVertexes][1] = 1;
419  tess.color[tess.numVertexes][0] = iColor[0];
420  tess.color[tess.numVertexes][1] = iColor[1];
421  tess.color[tess.numVertexes][2] = iColor[2];
422  tess.color[tess.numVertexes][3] = 65535;
423  tess.numVertexes++;
424 
425  tess.xyz[tess.numVertexes][0] = f->windowX + size;
426  tess.xyz[tess.numVertexes][1] = f->windowY + size;
427  tess.texCoords[tess.numVertexes][0] = 1;
428  tess.texCoords[tess.numVertexes][1] = 1;
429  tess.color[tess.numVertexes][0] = iColor[0];
430  tess.color[tess.numVertexes][1] = iColor[1];
431  tess.color[tess.numVertexes][2] = iColor[2];
432  tess.color[tess.numVertexes][3] = 65535;
433  tess.numVertexes++;
434 
435  tess.xyz[tess.numVertexes][0] = f->windowX + size;
436  tess.xyz[tess.numVertexes][1] = f->windowY - size;
437  tess.texCoords[tess.numVertexes][0] = 1;
438  tess.texCoords[tess.numVertexes][1] = 0;
439  tess.color[tess.numVertexes][0] = iColor[0];
440  tess.color[tess.numVertexes][1] = iColor[1];
441  tess.color[tess.numVertexes][2] = iColor[2];
442  tess.color[tess.numVertexes][3] = 65535;
443  tess.numVertexes++;
444 
445  tess.indexes[tess.numIndexes++] = 0;
446  tess.indexes[tess.numIndexes++] = 1;
447  tess.indexes[tess.numIndexes++] = 2;
448  tess.indexes[tess.numIndexes++] = 0;
449  tess.indexes[tess.numIndexes++] = 2;
450  tess.indexes[tess.numIndexes++] = 3;
451 
452  RB_EndSurface();
453 }

455 /*
456 ==================
457 RB_RenderFlares
458 
459 Because flares are simulating an occular effect, they should be drawn after
460 everything (all views) in the entire frame has been drawn.
461 
462 Because of the way portals use the depth buffer to mark off areas, the
463 needed information would be lost after each view, so we are forced to draw
464 flares after each view.
465 
466 The resulting artifact is that flares in mirrors or portals don't dim properly
467 when occluded by something in the main view, and portal flares that should
468 extend past the portal edge will be overwritten.
469 ==================
470 */
471 void RB_RenderFlares (void) {
472  flare_t *f;
473  flare_t **prev;
474  qboolean draw;
475  mat4_t oldmodelview, oldprojection, matrix;
476 
477  if ( !r_flares->integer ) {
478  return;
479  }
480 
481  if(r_flareCoeff->modified)
482  {
483  R_SetFlareCoeff();
484  r_flareCoeff->modified = qfalse;
485  }
486 
487  // Reset currentEntity to world so that any previously referenced entities
488  // don't have influence on the rendering of these flares (i.e. RF_ renderer flags).
489  backEnd.currentEntity = &tr.worldEntity;
490  backEnd.or = backEnd.viewParms.world;
491 
492 // RB_AddDlightFlares();
493 
494  // perform z buffer readback on each flare in this view
495  draw = qfalse;
496  prev = &r_activeFlares;
497  while ( ( f = *prev ) != NULL ) {
498  // throw out any flares that weren't added last frame
499  if ( f->addedFrame < backEnd.viewParms.frameCount - 1 ) {
500  *prev = f->next;
501  f->next = r_inactiveFlares;
502  r_inactiveFlares = f;
503  continue;
504  }
505 
506  // don't draw any here that aren't from this scene / portal
507  f->drawIntensity = 0;
508  if ( f->frameSceneNum == backEnd.viewParms.frameSceneNum
509  && f->inPortal == backEnd.viewParms.isPortal ) {
510  RB_TestFlare( f );
511  if ( f->drawIntensity ) {
512  draw = qtrue;
513  } else {
514  // this flare has completely faded out, so remove it from the chain
515  *prev = f->next;
516  f->next = r_inactiveFlares;
517  r_inactiveFlares = f;
518  continue;
519  }
520  }
521 
522  prev = &f->next;
523  }
524 
525  if ( !draw ) {
526  return; // none visible
527  }
528 
529  Mat4Copy(glState.projection, oldprojection);
530  Mat4Copy(glState.modelview, oldmodelview);
531  Mat4Identity(matrix);
532  GL_SetModelviewMatrix(matrix);
533  Mat4Ortho( backEnd.viewParms.viewportX, backEnd.viewParms.viewportX + backEnd.viewParms.viewportWidth,
534  backEnd.viewParms.viewportY, backEnd.viewParms.viewportY + backEnd.viewParms.viewportHeight,
535  -99999, 99999, matrix );
536  GL_SetProjectionMatrix(matrix);
537 
538  for ( f = r_activeFlares ; f ; f = f->next ) {
539  if ( f->frameSceneNum == backEnd.viewParms.frameSceneNum
540  && f->inPortal == backEnd.viewParms.isPortal
541  && f->drawIntensity ) {
542  RB_RenderFlare( f );
543  }
544  }
545 
546  GL_SetProjectionMatrix(oldprojection);
547  GL_SetModelviewMatrix(oldmodelview);
548 }
