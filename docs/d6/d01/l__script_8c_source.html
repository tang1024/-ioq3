173 void PS_CreatePunctuationTable(script_t *script, punctuation_t *punctuations)
174 {
175  int i;
176  punctuation_t *p, *lastp, *newp;
177 
178  //get memory for the table
179  if (!script->punctuationtable) script->punctuationtable = (punctuation_t **)
180  GetMemory(256 * sizeof(punctuation_t *));
181  Com_Memset(script->punctuationtable, 0, 256 * sizeof(punctuation_t *));
182  //add the punctuations in the list to the punctuation table
183  for (i = 0; punctuations[i].p; i++)
184  {
185  newp = &punctuations[i];
186  lastp = NULL;
187  //sort the punctuations in this table entry on length (longer punctuations first)
188  for (p = script->punctuationtable[(unsigned int) newp->p[0]]; p; p = p->next)
189  {
190  if (strlen(p->p) < strlen(newp->p))
191  {
192  newp->next = p;
193  if (lastp) lastp->next = newp;
194  else script->punctuationtable[(unsigned int) newp->p[0]] = newp;
195  break;
196  } //end if
197  lastp = p;
198  } //end for
199  if (!p)
200  {
201  newp->next = NULL;
202  if (lastp) lastp->next = newp;
203  else script->punctuationtable[(unsigned int) newp->p[0]] = newp;
204  } //end if
205  } //end for
206 } //end of the function PS_CreatePunctuationTable

290 //============================================================================
291 // Reads spaces, tabs, C-like comments etc.
292 // When a newline character is found the scripts line counter is increased.
293 //
294 // Parameter: -
295 // Returns: -
296 // Changes Globals: -
297 //============================================================================
298 int PS_ReadWhiteSpace(script_t *script)
299 {
300  while(1)
301  {
302  //skip white space
303  while(*script->script_p <= ' ')
304  {
305  if (!*script->script_p) return 0;
306  if (*script->script_p == '\n') script->line++;
307  script->script_p++;
308  } //end while
309  //skip comments
310  if (*script->script_p == '/')
311  {
312  //comments //
313  if (*(script->script_p+1) == '/')
314  {
315  script->script_p++;
316  do
317  {
318  script->script_p++;
319  if (!*script->script_p) return 0;
320  } //end do
321  while(*script->script_p != '\n');
322  script->line++;
323  script->script_p++;
324  if (!*script->script_p) return 0;
325  continue;
326  } //end if
327  //comments /* */
328  else if (*(script->script_p+1) == '*')
329  {
330  script->script_p++;
331  do
332  {
333  script->script_p++;
334  if (!*script->script_p) return 0;
335  if (*script->script_p == '\n') script->line++;
336  } //end do
337  while(!(*script->script_p == '*' && *(script->script_p+1) == '/'));
338  script->script_p++;
339  if (!*script->script_p) return 0;
340  script->script_p++;
341  if (!*script->script_p) return 0;
342  continue;
343  } //end if
344  } //end if
345  break;
346  } //end while
347  return 1;
348 } //end of the function PS_ReadWhiteSpace
349 //============================================================================
350 // Reads an escape character.
351 //
352 // Parameter: script : script to read from
353 // ch : place to store the read escape character
354 // Returns: -
355 // Changes Globals: -
356 //============================================================================
357 int PS_ReadEscapeCharacter(script_t *script, char *ch)
358 {
359  int c, val, i;
360 
361  //step over the leading '\\'
362  script->script_p++;
363  //determine the escape character
364  switch(*script->script_p)
365  {
366  case '\\': c = '\\'; break;
367  case 'n': c = '\n'; break;
368  case 'r': c = '\r'; break;
369  case 't': c = '\t'; break;
370  case 'v': c = '\v'; break;
371  case 'b': c = '\b'; break;
372  case 'f': c = '\f'; break;
373  case 'a': c = '\a'; break;
374  case '\'': c = '\''; break;
375  case '\"': c = '\"'; break;
376  case '\?': c = '\?'; break;
377  case 'x':
378  {
379  script->script_p++;
380  for (i = 0, val = 0; ; i++, script->script_p++)
381  {
382  c = *script->script_p;
383  if (c >= '0' && c <= '9') c = c - '0';
384  else if (c >= 'A' && c <= 'Z') c = c - 'A' + 10;
385  else if (c >= 'a' && c <= 'z') c = c - 'a' + 10;
386  else break;
387  val = (val << 4) + c;
388  } //end for
389  script->script_p--;
390  if (val > 0xFF)
391  {
392  ScriptWarning(script, "too large value in escape character");
393  val = 0xFF;
394  } //end if
395  c = val;
396  break;
397  } //end case
398  default: //NOTE: decimal ASCII code, NOT octal
399  {
400  if (*script->script_p < '0' || *script->script_p > '9') ScriptError(script, "unknown escape char");
401  for (i = 0, val = 0; ; i++, script->script_p++)
402  {
403  c = *script->script_p;
404  if (c >= '0' && c <= '9') c = c - '0';
405  else break;
406  val = val * 10 + c;
407  } //end for
408  script->script_p--;
409  if (val > 0xFF)
410  {
411  ScriptWarning(script, "too large value in escape character");
412  val = 0xFF;
413  } //end if
414  c = val;
415  break;
416  } //end default
417  } //end switch
418  //step over the escape character or the last digit of the number
419  script->script_p++;
420  //store the escape character
421  *ch = c;
422  //successfully read escape character
423  return 1;
424 } //end of the function PS_ReadEscapeCharacter
425 //============================================================================
426 // Reads C-like string. Escape characters are interpretted.
427 // Quotes are included with the string.
428 // Reads two strings with a white space between them as one string.
429 //
430 // Parameter: script : script to read from
431 // token : buffer to store the string
432 // Returns: qtrue when a string was read successfully
433 // Changes Globals: -
434 //============================================================================
435 int PS_ReadString(script_t *script, token_t *token, int quote)
436 {
437  int len, tmpline;
438  char *tmpscript_p;
439 
440  if (quote == '\"') token->type = TT_STRING;
441  else token->type = TT_LITERAL;
442 
443  len = 0;
444  //leading quote
445  token->string[len++] = *script->script_p++;
446  //
447  while(1)
448  {
449  //minus 2 because trailing double quote and zero have to be appended
450  if (len >= MAX_TOKEN - 2)
451  {
452  ScriptError(script, "string longer than MAX_TOKEN = %d", MAX_TOKEN);
453  return 0;
454  } //end if
455  //if there is an escape character and
456  //if escape characters inside a string are allowed
457  if (*script->script_p == '\\' && !(script->flags & SCFL_NOSTRINGESCAPECHARS))
458  {
459  if (!PS_ReadEscapeCharacter(script, &token->string[len]))
460  {
461  token->string[len] = 0;
462  return 0;
463  } //end if
464  len++;
465  } //end if
466  //if a trailing quote
467  else if (*script->script_p == quote)
468  {
469  //step over the double quote
470  script->script_p++;
471  //if white spaces in a string are not allowed
472  if (script->flags & SCFL_NOSTRINGWHITESPACES) break;
473  //
474  tmpscript_p = script->script_p;
475  tmpline = script->line;
476  //read unusefull stuff between possible two following strings
477  if (!PS_ReadWhiteSpace(script))
478  {
479  script->script_p = tmpscript_p;
480  script->line = tmpline;
481  break;
482  } //end if
483  //if there's no leading double qoute
484  if (*script->script_p != quote)
485  {
486  script->script_p = tmpscript_p;
487  script->line = tmpline;
488  break;
489  } //end if
490  //step over the new leading double quote
491  script->script_p++;
492  } //end if
493  else
494  {
495  if (*script->script_p == '\0')
496  {
497  token->string[len] = 0;
498  ScriptError(script, "missing trailing quote");
499  return 0;
500  } //end if
501  if (*script->script_p == '\n')
502  {
503  token->string[len] = 0;
504  ScriptError(script, "newline inside string %s", token->string);
505  return 0;
506  } //end if
507  token->string[len++] = *script->script_p++;
508  } //end else
509  } //end while
510  //trailing quote
511  token->string[len++] = quote;
512  //end string with a zero
513  token->string[len] = '\0';
514  //the sub type is the length of the string
515  token->subtype = len;
516  return 1;
517 } //end of the function PS_ReadString

554 void NumberValue(char *string, int subtype, unsigned long int *intvalue,
555  float *floatvalue)
556 {
557  unsigned long int dotfound = 0;
558 
559  *intvalue = 0;
560  *floatvalue = 0;
561  //floating point number
562  if (subtype & TT_FLOAT)
563  {
564  while(*string)
565  {
566  if (*string == '.')
567  {
568  if (dotfound) return;
569  dotfound = 10;
570  string++;
571  } //end if
572  if (dotfound)
573  {
574  *floatvalue = *floatvalue + (float) (*string - '0') /
575  (float) dotfound;
576  dotfound *= 10;
577  } //end if
578  else
579  {
580  *floatvalue = *floatvalue * 10.0 + (float) (*string - '0');
581  } //end else
582  string++;
583  } //end while
584  *intvalue = (unsigned long) *floatvalue;
585  } //end if
586  else if (subtype & TT_DECIMAL)
587  {
588  while(*string) *intvalue = *intvalue * 10 + (*string++ - '0');
589  *floatvalue = *intvalue;
590  } //end else if
591  else if (subtype & TT_HEX)
592  {
593  //step over the leading 0x or 0X
594  string += 2;
595  while(*string)
596  {
597  *intvalue <<= 4;
598  if (*string >= 'a' && *string <= 'f') *intvalue += *string - 'a' + 10;
599  else if (*string >= 'A' && *string <= 'F') *intvalue += *string - 'A' + 10;
600  else *intvalue += *string - '0';
601  string++;
602  } //end while
603  *floatvalue = *intvalue;
604  } //end else if
605  else if (subtype & TT_OCTAL)
606  {
607  //step over the first zero
608  string += 1;
609  while(*string) *intvalue = (*intvalue << 3) + (*string++ - '0');
610  *floatvalue = *intvalue;
611  } //end else if
612  else if (subtype & TT_BINARY)
613  {
614  //step over the leading 0b or 0B
615  string += 2;
616  while(*string) *intvalue = (*intvalue << 1) + (*string++ - '0');
617  *floatvalue = *intvalue;
618  } //end else if
619 } //end of the function NumberValue

626 int PS_ReadNumber(script_t *script, token_t *token)
627 {
628  int len = 0, i;
629  int octal, dot;
630  char c;
631 // unsigned long int intvalue = 0;
632 // double floatvalue = 0;
633 
634  token->type = TT_NUMBER;
635  //check for a hexadecimal number
636  if (*script->script_p == '0' &&
637  (*(script->script_p + 1) == 'x' ||
638  *(script->script_p + 1) == 'X'))
639  {
640  token->string[len++] = *script->script_p++;
641  token->string[len++] = *script->script_p++;
642  c = *script->script_p;
643  //hexadecimal
644  while((c >= '0' && c <= '9') ||
645  (c >= 'a' && c <= 'f') ||
646  (c >= 'A' && c <= 'A'))
647  {
648  token->string[len++] = *script->script_p++;
649  if (len >= MAX_TOKEN)
650  {
651  ScriptError(script, "hexadecimal number longer than MAX_TOKEN = %d", MAX_TOKEN);
652  return 0;
653  } //end if
654  c = *script->script_p;
655  } //end while
656  token->subtype |= TT_HEX;
657  } //end if
658 #ifdef BINARYNUMBERS
659  //check for a binary number
660  else if (*script->script_p == '0' &&
661  (*(script->script_p + 1) == 'b' ||
662  *(script->script_p + 1) == 'B'))
663  {
664  token->string[len++] = *script->script_p++;
665  token->string[len++] = *script->script_p++;
666  c = *script->script_p;
667  //binary
668  while(c == '0' || c == '1')
669  {
670  token->string[len++] = *script->script_p++;
671  if (len >= MAX_TOKEN)
672  {
673  ScriptError(script, "binary number longer than MAX_TOKEN = %d", MAX_TOKEN);
674  return 0;
675  } //end if
676  c = *script->script_p;
677  } //end while
678  token->subtype |= TT_BINARY;
679  } //end if
680 #endif //BINARYNUMBERS
681  else //decimal or octal integer or floating point number
682  {
683  octal = qfalse;
684  dot = qfalse;
685  if (*script->script_p == '0') octal = qtrue;
686  while(1)
687  {
688  c = *script->script_p;
689  if (c == '.') dot = qtrue;
690  else if (c == '8' || c == '9') octal = qfalse;
691  else if (c < '0' || c > '9') break;
692  token->string[len++] = *script->script_p++;
693  if (len >= MAX_TOKEN - 1)
694  {
695  ScriptError(script, "number longer than MAX_TOKEN = %d", MAX_TOKEN);
696  return 0;
697  } //end if
698  } //end while
699  if (octal) token->subtype |= TT_OCTAL;
700  else token->subtype |= TT_DECIMAL;
701  if (dot) token->subtype |= TT_FLOAT;
702  } //end else
703  for (i = 0; i < 2; i++)
704  {
705  c = *script->script_p;
706  //check for a LONG number
707  if ( (c == 'l' || c == 'L')
708  && !(token->subtype & TT_LONG))
709  {
710  script->script_p++;
711  token->subtype |= TT_LONG;
712  } //end if
713  //check for an UNSIGNED number
714  else if ( (c == 'u' || c == 'U')
715  && !(token->subtype & (TT_UNSIGNED | TT_FLOAT)))
716  {
717  script->script_p++;
718  token->subtype |= TT_UNSIGNED;
719  } //end if
720  } //end for
721  token->string[len] = '\0';
722 #ifdef NUMBERVALUE
723  NumberValue(token->string, token->subtype, &token->intvalue, &token->floatvalue);
724 #endif //NUMBERVALUE
725  if (!(token->subtype & TT_FLOAT)) token->subtype |= TT_INTEGER;
726  return 1;
727 } //end of the function PS_ReadNumber

843 //
844 // Parameter: -
845 // Returns: -
846 // Changes Globals: -
847 //============================================================================
848 int PS_ReadToken(script_t *script, token_t *token)
849 {
850  //if there is a token available (from UnreadToken)
851  if (script->tokenavailable)
852  {
853  script->tokenavailable = 0;
854  Com_Memcpy(token, &script->token, sizeof(token_t));
855  return 1;
856  } //end if
857  //save script pointer
858  script->lastscript_p = script->script_p;
859  //save line counter
860  script->lastline = script->line;
861  //clear the token stuff
862  Com_Memset(token, 0, sizeof(token_t));
863  //start of the white space
864  script->whitespace_p = script->script_p;
865  token->whitespace_p = script->script_p;
866  //read unusefull stuff
867  if (!PS_ReadWhiteSpace(script)) return 0;
868  //end of the white space
869  script->endwhitespace_p = script->script_p;
870  token->endwhitespace_p = script->script_p;
871  //line the token is on
872  token->line = script->line;
873  //number of lines crossed before token
874  token->linescrossed = script->line - script->lastline;
875  //if there is a leading double quote
876  if (*script->script_p == '\"')
877  {
878  if (!PS_ReadString(script, token, '\"')) return 0;
879  } //end if
880  //if a literal
881  else if (*script->script_p == '\'')
882  {
883  //if (!PS_ReadLiteral(script, token)) return 0;
884  if (!PS_ReadString(script, token, '\'')) return 0;
885  } //end if
886  //if there is a number
887  else if ((*script->script_p >= '0' && *script->script_p <= '9') ||
888  (*script->script_p == '.' &&
889  (*(script->script_p + 1) >= '0' && *(script->script_p + 1) <= '9')))
890  {
891  if (!PS_ReadNumber(script, token)) return 0;
892  } //end if
893  //if this is a primitive script
894  else if (script->flags & SCFL_PRIMITIVE)
895  {
896  return PS_ReadPrimitive(script, token);
897  } //end else if
898  //if there is a name
899  else if ((*script->script_p >= 'a' && *script->script_p <= 'z') ||
900  (*script->script_p >= 'A' && *script->script_p <= 'Z') ||
901  *script->script_p == '_')
902  {
903  if (!PS_ReadName(script, token)) return 0;
904  } //end if
905  //check for punctuations
906  else if (!PS_ReadPunctuation(script, token))
907  {
908  ScriptError(script, "can't read token");
909  return 0;
910  } //end if
911  //copy the token into the script structure
912  Com_Memcpy(&script->token, token, sizeof(token_t));
913  //successfully read a token
914  return 1;
915 } //end of the function PS_ReadToken
