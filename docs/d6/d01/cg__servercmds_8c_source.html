238 void CG_ShaderStateChanged(void) {
239  char originalShader[MAX_QPATH];
240  char newShader[MAX_QPATH];
241  char timeOffset[16];
242  const char *o;
243  char *n,*t;
244 
245  o = CG_ConfigString( CS_SHADERSTATE );
246  while (o && *o) {
247  n = strstr(o, "=");
248  if (n && *n) {
249  strncpy(originalShader, o, n-o);
250  originalShader[n-o] = 0;
251  n++;
252  t = strstr(n, ":");
253  if (t && *t) {
254  strncpy(newShader, n, t-n);
255  newShader[t-n] = 0;
256  } else {
257  break;
258  }
259  t++;
260  o = strstr(t, "@");
261  if (o) {
262  strncpy(timeOffset, t, o-t);
263  timeOffset[o-t] = 0;
264  o++;
265  trap_R_RemapShader( originalShader, newShader, timeOffset );
266  }
267  } else {
268  break;
269  }
270  }

369 static void CG_AddToTeamChat( const char *str ) {
370  int len;
371  char *p, *ls;
372  int lastcolor;
373  int chatHeight;
374 
375  if (cg_teamChatHeight.integer < TEAMCHAT_HEIGHT) {
376  chatHeight = cg_teamChatHeight.integer;
377  } else {
378  chatHeight = TEAMCHAT_HEIGHT;
379  }
380 
381  if (chatHeight <= 0 || cg_teamChatTime.integer <= 0) {
382  // team chat disabled, dump into normal chat
383  cgs.teamChatPos = cgs.teamLastChatPos = 0;
384  return;
385  }
386 
387  len = 0;
388 
389  p = cgs.teamChatMsgs[cgs.teamChatPos % chatHeight];
390  *p = 0;
391 
392  lastcolor = '7';
393 
394  ls = NULL;
395  while (*str) {
396  if (len > TEAMCHAT_WIDTH - 1) {
397  if (ls) {
398  str -= (p - ls);
399  str++;
400  p -= (p - ls);
401  }
402  *p = 0;
403 
404  cgs.teamChatMsgTimes[cgs.teamChatPos % chatHeight] = cg.time;
405 
406  cgs.teamChatPos++;
407  p = cgs.teamChatMsgs[cgs.teamChatPos % chatHeight];
408  *p = 0;
409  *p++ = Q_COLOR_ESCAPE;
410  *p++ = lastcolor;
411  len = 0;
412  ls = NULL;
413  }
414 
415  if ( Q_IsColorString( str ) ) {
416  *p++ = *str++;
417  lastcolor = *str;
418  *p++ = *str++;
419  continue;
420  }
421  if (*str == ' ') {
422  ls = p;
423  }
424  *p++ = *str++;
425  len++;
426  }
427  *p = 0;
428 
429  cgs.teamChatMsgTimes[cgs.teamChatPos % chatHeight] = cg.time;
430  cgs.teamChatPos++;
431 
432  if (cgs.teamChatPos - cgs.teamLastChatPos > chatHeight)
433  cgs.teamLastChatPos = cgs.teamChatPos - chatHeight;
434 }

525 /*
526 =================
527 CG_ParseVoiceChats
528 =================
529 */
530 int CG_ParseVoiceChats( const char *filename, voiceChatList_t *voiceChatList, int maxVoiceChats ) {
531  int len, i;
532  fileHandle_t f;
533  char buf[MAX_VOICEFILESIZE];
534  char **p, *ptr;
535  char *token;
536  voiceChat_t *voiceChats;
537  qboolean compress;
538  sfxHandle_t sound;
539 
540  compress = qtrue;
541  if (cg_buildScript.integer) {
542  compress = qfalse;
543  }
544 
545  len = trap_FS_FOpenFile( filename, &f, FS_READ );
546  if ( !f ) {
547  trap_Print( va( S_COLOR_RED "voice chat file not found: %s\n", filename ) );
548  return qfalse;
549  }
550  if ( len >= MAX_VOICEFILESIZE ) {
551  trap_Print( va( S_COLOR_RED "voice chat file too large: %s is %i, max allowed is %i\n", filename, len, MAX_VOICEFILESIZE ) );
552  trap_FS_FCloseFile( f );
553  return qfalse;
554  }
555 
556  trap_FS_Read( buf, len, f );
557  buf[len] = 0;
558  trap_FS_FCloseFile( f );
559 
560  ptr = buf;
561  p = &ptr;
562 
563  Com_sprintf(voiceChatList->name, sizeof(voiceChatList->name), "%s", filename);
564  voiceChats = voiceChatList->voiceChats;
565  for ( i = 0; i < maxVoiceChats; i++ ) {
566  voiceChats[i].id[0] = 0;
567  }
568  token = COM_ParseExt(p, qtrue);
569  if (!token[0]) {
570  return qtrue;
571  }
572  if (!Q_stricmp(token, "female")) {
573  voiceChatList->gender = GENDER_FEMALE;
574  }
575  else if (!Q_stricmp(token, "male")) {
576  voiceChatList->gender = GENDER_MALE;
577  }
578  else if (!Q_stricmp(token, "neuter")) {
579  voiceChatList->gender = GENDER_NEUTER;
580  }
581  else {
582  trap_Print( va( S_COLOR_RED "expected gender not found in voice chat file: %s\n", filename ) );
583  return qfalse;
584  }
585 
586  voiceChatList->numVoiceChats = 0;
587  while ( 1 ) {
588  token = COM_ParseExt(p, qtrue);
589  if (!token[0]) {
590  return qtrue;
591  }
592  Com_sprintf(voiceChats[voiceChatList->numVoiceChats].id, sizeof( voiceChats[voiceChatList->numVoiceChats].id ), "%s", token);
593  token = COM_ParseExt(p, qtrue);
594  if (Q_stricmp(token, "{")) {
595  trap_Print( va( S_COLOR_RED "expected { found %s in voice chat file: %s\n", token, filename ) );
596  return qfalse;
597  }
598  voiceChats[voiceChatList->numVoiceChats].numSounds = 0;
599  while(1) {
600  token = COM_ParseExt(p, qtrue);
601  if (!token[0]) {
602  return qtrue;
603  }
604  if (!Q_stricmp(token, "}"))
605  break;
606  sound = trap_S_RegisterSound( token, compress );
607  voiceChats[voiceChatList->numVoiceChats].sounds[voiceChats[voiceChatList->numVoiceChats].numSounds] = sound;
608  token = COM_ParseExt(p, qtrue);
609  if (!token[0]) {
610  return qtrue;
611  }
612  Com_sprintf(voiceChats[voiceChatList->numVoiceChats].chats[
613  voiceChats[voiceChatList->numVoiceChats].numSounds], MAX_CHATSIZE, "%s", token);
614  if (sound)
615  voiceChats[voiceChatList->numVoiceChats].numSounds++;
616  if (voiceChats[voiceChatList->numVoiceChats].numSounds >= MAX_VOICESOUNDS)
617  break;
618  }
619  voiceChatList->numVoiceChats++;
620  if (voiceChatList->numVoiceChats >= maxVoiceChats)
621  return qtrue;
622  }
623  return qtrue;
624 }
