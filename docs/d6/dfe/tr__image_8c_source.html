319 /*
320 ================
321 ResampleTexture
322 
323 Used to resample images in a more general than quartering fashion.
324 
325 This will only be filtered properly if the resampled size
326 is greater than half the original size.
327 
328 If a larger shrinking is needed, use the mipmap function
329 before or after.
330 ================
331 */
332 static void ResampleTexture( byte *in, int inwidth, int inheight, byte *out,
333  int outwidth, int outheight ) {
334  int i, j;
335  byte *inrow, *inrow2;
336  int frac, fracstep;
337  int p1[2048], p2[2048];
338  byte *pix1, *pix2, *pix3, *pix4;
339 
340  if (outwidth>2048)
341  ri.Error(ERR_DROP, "ResampleTexture: max width");
342 
343  fracstep = inwidth*0x10000/outwidth;
344 
345  frac = fracstep>>2;
346  for ( i=0 ; i<outwidth ; i++ ) {
347  p1[i] = 4*(frac>>16);
348  frac += fracstep;
349  }
350  frac = 3*(fracstep>>2);
351  for ( i=0 ; i<outwidth ; i++ ) {
352  p2[i] = 4*(frac>>16);
353  frac += fracstep;
354  }
355 
356  for (i=0 ; i<outheight ; i++) {
357  inrow = in + 4*inwidth*(int)((i+0.25)*inheight/outheight);
358  inrow2 = in + 4*inwidth*(int)((i+0.75)*inheight/outheight);
359  for (j=0 ; j<outwidth ; j++) {
360  pix1 = inrow + p1[j];
361  pix2 = inrow + p2[j];
362  pix3 = inrow2 + p1[j];
363  pix4 = inrow2 + p2[j];
364  *out++ = (pix1[0] + pix2[0] + pix3[0] + pix4[0])>>2;
365  *out++ = (pix1[1] + pix2[1] + pix3[1] + pix4[1])>>2;
366  *out++ = (pix1[2] + pix2[2] + pix3[2] + pix4[2])>>2;
367  *out++ = (pix1[3] + pix2[3] + pix3[3] + pix4[3])>>2;
368  }
369  }
370 }

372 static void RGBAtoYCoCgA(const byte *in, byte *out, int width, int height)
373 {
374  int x, y;
375 
376  for (y = 0; y < height; y++)
377  {
378  const byte *inbyte = in + y * width * 4;
379  byte *outbyte = out + y * width * 4;
380 
381  for (x = 0; x < width; x++)
382  {
383  byte r, g, b, a, rb2;
384 
385  r = *inbyte++;
386  g = *inbyte++;
387  b = *inbyte++;
388  a = *inbyte++;
389  rb2 = (r + b) >> 1;
390 
391  *outbyte++ = (g + rb2) >> 1; // Y = R/4 + G/2 + B/4
392  *outbyte++ = (r - b + 256) >> 1; // Co = R/2 - B/2
393  *outbyte++ = (g - rb2 + 256) >> 1; // Cg = -R/4 + G/2 - B/4
394  *outbyte++ = a;
395  }
396  }

426 // uses a sobel filter to change a texture to a normal map
427 static void RGBAtoNormal(const byte *in, byte *out, int width, int height, qboolean clampToEdge)
428 {
429  int x, y, max;
430 
431  // convert to heightmap, storing in alpha
432  // same as converting to Y in YCoCg
433  max = 1;
434  for (y = 0; y < height; y++)
435  {
436  const byte *inbyte = in + y * width * 4;
437  byte *outbyte = out + y * width * 4 + 3;
438 
439  for (x = 0; x < width; x++)
440  {
441  byte result = (inbyte[0] >> 2) + (inbyte[1] >> 1) + (inbyte[2] >> 2);
442  result = result * result / 255; // Make linear
443  *outbyte = result;
444  max = MAX(max, *outbyte);
445  outbyte += 4;
446  inbyte += 4;
447  }
448  }
449 
450  // level out heights
451  if (max < 255)
452  {
453  for (y = 0; y < height; y++)
454  {
455  byte *outbyte = out + y * width * 4 + 3;
456 
457  for (x = 0; x < width; x++)
458  {
459  *outbyte = *outbyte + (255 - max);
460  outbyte += 4;
461  }
462  }
463  }
464 
465 
466  // now run sobel filter over height values to generate X and Y
467  // then normalize
468  for (y = 0; y < height; y++)
469  {
470  byte *outbyte = out + y * width * 4;
471 
472  for (x = 0; x < width; x++)
473  {
474  // 0 1 2
475  // 3 4 5
476  // 6 7 8
477 
478  byte s[9];
479  int x2, y2, i;
480  vec3_t normal;
481 
482  i = 0;
483  for (y2 = -1; y2 <= 1; y2++)
484  {
485  int src_y = y + y2;
486 
487  if (clampToEdge)
488  {
489  src_y = CLAMP(src_y, 0, height - 1);
490  }
491  else
492  {
493  src_y = (src_y + height) % height;
494  }
495 
496 
497  for (x2 = -1; x2 <= 1; x2++)
498  {
499  int src_x = x + x2;
500 
501  if (clampToEdge)
502  {
503  src_x = CLAMP(src_x, 0, width - 1);
504  }
505  else
506  {
507  src_x = (src_x + width) % width;
508  }
509 
510  s[i++] = *(out + (src_y * width + src_x) * 4 + 3);
511  }
512  }
513 
514  normal[0] = s[0] - s[2]
515  + 2 * s[3] - 2 * s[5]
516  + s[6] - s[8];
517 
518  normal[1] = s[0] + 2 * s[1] + s[2]
519 
520  - s[6] - 2 * s[7] - s[8];
521 
522  normal[2] = s[4] * 4;
523 
524  if (!VectorNormalize2(normal, normal))
525  {
526  VectorSet(normal, 0, 0, 1);
527  }
528 
529  *outbyte++ = FloatToOffsetByte(normal[0]);
530  *outbyte++ = FloatToOffsetByte(normal[1]);
531  *outbyte++ = FloatToOffsetByte(normal[2]);
532  outbyte++;
533  }
534  }

539 // based on Fast Curve Based Interpolation
540 // from Fast Artifacts-Free Image Interpolation (http://www.andreagiachetti.it/icbi/)
541 // assumes data has a 2 pixel thick border of clamped or wrapped data
542 // expects data to be a grid with even (0, 0), (2, 0), (0, 2), (2, 2) etc pixels filled
543 // only performs FCBI on specified component
544 static void DoFCBI(byte *in, byte *out, int width, int height, int component)
545 {
546  int x, y;
547  byte *outbyte, *inbyte;
548 
549  // copy in to out
550  for (y = 2; y < height - 2; y += 2)
551  {
552  inbyte = in + (y * width + 2) * 4 + component;
553  outbyte = out + (y * width + 2) * 4 + component;
554 
555  for (x = 2; x < width - 2; x += 2)
556  {
557  *outbyte = *inbyte;
558  outbyte += 8;
559  inbyte += 8;
560  }
561  }
562 
563  for (y = 3; y < height - 3; y += 2)
564  {
565  // diagonals
566  //
567  // NWp - northwest interpolated pixel
568  // NEp - northeast interpolated pixel
569  // NWd - northwest first derivative
570  // NEd - northeast first derivative
571  // NWdd - northwest second derivative
572  // NEdd - northeast second derivative
573  //
574  // Uses these samples:
575  //
576  // 0
577  // - - a - b - -
578  // - - - - - - -
579  // c - d - e - f
580  // 0 - - - - - - -
581  // g - h - i - j
582  // - - - - - - -
583  // - - k - l - -
584  //
585  // x+2 uses these samples:
586  //
587  // 0
588  // - - - - a - b - -
589  // - - - - - - - - -
590  // - - c - d - e - f
591  // 0 - - - - - - - - -
592  // - - g - h - i - j
593  // - - - - - - - - -
594  // - - - - k - l - -
595  //
596  // so we can reuse 8 of them on next iteration
597  //
598  // a=b, c=d, d=e, e=f, g=h, h=i, i=j, k=l
599  //
600  // only b, f, j, and l need to be sampled on next iteration
601 
602  byte sa, sb, sc, sd, se, sf, sg, sh, si, sj, sk, sl;
603  byte *line1, *line2, *line3, *line4;
604 
605  x = 3;
606 
607  // optimization one
608  // SAMPLE2(sa, x-1, y-3);
609  //SAMPLE2(sc, x-3, y-1); SAMPLE2(sd, x-1, y-1); SAMPLE2(se, x+1, y-1);
610  //SAMPLE2(sg, x-3, y+1); SAMPLE2(sh, x-1, y+1); SAMPLE2(si, x+1, y+1);
611  // SAMPLE2(sk, x-1, y+3);
612 
613  // optimization two
614  line1 = in + ((y - 3) * width + (x - 1)) * 4 + component;
615  line2 = in + ((y - 1) * width + (x - 3)) * 4 + component;
616  line3 = in + ((y + 1) * width + (x - 3)) * 4 + component;
617  line4 = in + ((y + 3) * width + (x - 1)) * 4 + component;
618 
619  // COPYSAMPLE(sa, line1); line1 += 8;
620  //COPYSAMPLE(sc, line2); line2 += 8; COPYSAMPLE(sd, line2); line2 += 8; COPYSAMPLE(se, line2); line2 += 8;
621  //COPYSAMPLE(sg, line3); line3 += 8; COPYSAMPLE(sh, line3); line3 += 8; COPYSAMPLE(si, line3); line3 += 8;
622  // COPYSAMPLE(sk, line4); line4 += 8;
623 
624  sa = *line1; line1 += 8;
625  sc = *line2; line2 += 8; sd = *line2; line2 += 8; se = *line2; line2 += 8;
626  sg = *line3; line3 += 8; sh = *line3; line3 += 8; si = *line3; line3 += 8;
627  sk = *line4; line4 += 8;
628 
629  outbyte = out + (y * width + x) * 4 + component;
630 
631  for ( ; x < width - 3; x += 2)
632  {
633  int NWd, NEd, NWp, NEp;
634 
635  // original
636  // SAMPLE2(sa, x-1, y-3); SAMPLE2(sb, x+1, y-3);
637  //SAMPLE2(sc, x-3, y-1); SAMPLE2(sd, x-1, y-1); SAMPLE2(se, x+1, y-1); SAMPLE2(sf, x+3, y-1);
638  //SAMPLE2(sg, x-3, y+1); SAMPLE2(sh, x-1, y+1); SAMPLE2(si, x+1, y+1); SAMPLE2(sj, x+3, y+1);
639  // SAMPLE2(sk, x-1, y+3); SAMPLE2(sl, x+1, y+3);
640 
641  // optimization one
642  //SAMPLE2(sb, x+1, y-3);
643  //SAMPLE2(sf, x+3, y-1);
644  //SAMPLE2(sj, x+3, y+1);
645  //SAMPLE2(sl, x+1, y+3);
646 
647  // optimization two
648  //COPYSAMPLE(sb, line1); line1 += 8;
649  //COPYSAMPLE(sf, line2); line2 += 8;
650  //COPYSAMPLE(sj, line3); line3 += 8;
651  //COPYSAMPLE(sl, line4); line4 += 8;
652 
653  sb = *line1; line1 += 8;
654  sf = *line2; line2 += 8;
655  sj = *line3; line3 += 8;
656  sl = *line4; line4 += 8;
657 
658  NWp = sd + si;
659  NEp = se + sh;
660  NWd = abs(sd - si);
661  NEd = abs(se - sh);
662 
663  if (NWd > 100 || NEd > 100 || abs(NWp-NEp) > 200)
664  {
665  if (NWd < NEd)
666  *outbyte = NWp >> 1;
667  else
668  *outbyte = NEp >> 1;
669  }
670  else
671  {
672  int NWdd, NEdd;
673 
674  //NEdd = abs(sg + sd + sb - 3 * (se + sh) + sk + si + sf);
675  //NWdd = abs(sa + se + sj - 3 * (sd + si) + sc + sh + sl);
676  NEdd = abs(sg + sb - 3 * NEp + sk + sf + NWp);
