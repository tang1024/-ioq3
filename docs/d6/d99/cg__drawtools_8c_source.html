161 /*
162 ==================
163 CG_DrawStringExt
164 
165 Draws a multi-colored string with a drop shadow, optionally forcing
166 to a fixed color.
167 
168 Coordinates are at 640 by 480 virtual resolution
169 ==================
170 */
171 void CG_DrawStringExt(int x, int y, const char* string, const float* setColor,
172  qboolean forceColor, qboolean shadow, int charWidth, int charHeight, int maxChars)
173 {
174  vec4_t color;
175  const char* s;
176  int xx;
177  int cnt;
178 
179  if (maxChars <= 0)
180  maxChars = 32767; // do them all!
181 
182  // draw the drop shadow
183  if (shadow) {
184  color[0] = color[1] = color[2] = 0;
185  color[3] = setColor[3];
186  trap_R_SetColor(color);
187  s = string;
188  xx = x;
189  cnt = 0;
190  while (*s && cnt < maxChars) {
191  if (Q_IsColorString(s)) {
192  s += 2;
193  continue;
194  }
195  CG_DrawChar(xx + 2, y + 2, charWidth, charHeight, *s);
196  cnt++;
197  xx += charWidth;
198  s++;
199  }
200  }
201 
202  // draw the colored text
203  s = string;
204  xx = x;
205  cnt = 0;
206  trap_R_SetColor(setColor);
207  while (*s && cnt < maxChars) {
208  if (Q_IsColorString(s)) {
209  if (!forceColor) {
210  memcpy(color, g_color_table[ColorIndex(*(s + 1))], sizeof(color));
211  color[3] = setColor[3];
212  trap_R_SetColor(color);
213  }
214  s += 2;
215  continue;
216  }
217  CG_DrawChar(xx, y, charWidth, charHeight, *s);
218  xx += charWidth;
219  cnt++;
220  s++;
221  }
222  trap_R_SetColor(NULL);
223 }

340 CG_FadeColor
341 ================
342 */
343 float* CG_FadeColor(int startMsec, int totalMsec)
344 {
345  static vec4_t color;
346  int t;
347 
348  if (startMsec == 0) {
349  return NULL;
350  }
351 
352  t = cg.time - startMsec;
353 
354  if (t >= totalMsec) {
355  return NULL;
356  }
357 
358  // fade out
359  if (totalMsec - t < FADE_TIME) {
360  color[3] = (totalMsec - t) * 1.0 / FADE_TIME;
361  } else {
362  color[3] = 1.0;
363  }
364  color[0] = color[1] = color[2] = 1;
365 
366  return color;
367 }

396 /*
397 =================
398 CG_GetColorForHealth
399 =================
400 */
401 void CG_GetColorForHealth(int health, int armor, vec4_t hcolor)
402 {
403  int count;
404  int max;
405 
406  // calculate the total points of damage that can
407  // be sustained at the current health / armor level
408  if (health <= 0) {
409  VectorClear(hcolor); // black
410  hcolor[3] = 1;
411  return;
412  }
413  count = armor;
414  max = health * ARMOR_PROTECTION / (1.0 - ARMOR_PROTECTION);
415  if (max < count) {
416  count = max;
417  }
418  health += count;
419 
420  // set the color based on health
421  hcolor[0] = 1.0;
422  hcolor[3] = 1.0;
423  if (health >= 100) {
424  hcolor[2] = 1.0;
425  } else if (health < 66) {
426  hcolor[2] = 0;
427  } else {
428  hcolor[2] = (health - 66) / 33.0;
429  }
430 
431  if (health > 60) {
432  hcolor[1] = 1.0;
433  } else if (health < 30) {
434  hcolor[1] = 0;
435  } else {
436  hcolor[1] = (health - 30) / 30.0;
437  }
438 }

609 static void UI_DrawBannerString2(int x, int y, const char* str, vec4_t color)
610 {
611  const char* s;
612  unsigned char ch;
613  float ax;
614  float ay;
615  float aw;
616  float ah;
617  float frow;
618  float fcol;
619  float fwidth;
620  float fheight;
621 
622  // draw the colored text
623  trap_R_SetColor(color);
624 
625  ax = x * cgs.screenXScale + cgs.screenXBias;
626  ay = y * cgs.screenYScale;
627 
628  s = str;
629  while (*s) {
630  ch = *s & 127;
631  if (ch == ' ') {
632  ax += ((float)PROPB_SPACE_WIDTH + (float)PROPB_GAP_WIDTH) * cgs.screenXScale;
633  } else if (ch >= 'A' && ch <= 'Z') {
634  ch -= 'A';
635  fcol = (float)propMapB[ch][0] / 256.0f;
636  frow = (float)propMapB[ch][1] / 256.0f;
637  fwidth = (float)propMapB[ch][2] / 256.0f;
638  fheight = (float)PROPB_HEIGHT / 256.0f;
639  aw = (float)propMapB[ch][2] * cgs.screenXScale;
640  ah = (float)PROPB_HEIGHT * cgs.screenYScale;
641  trap_R_DrawStretchPic(ax, ay, aw, ah, fcol, frow, fcol + fwidth, frow + fheight, cgs.media.charsetPropB);
642  ax += (aw + (float)PROPB_GAP_WIDTH * cgs.screenXScale);
643  }
644  s++;
645  }
646 
647  trap_R_SetColor(NULL);
648 }
