35 ================
36 G_BounceProjectile
37 ================
38 */
39 void G_BounceProjectile(vec3_t start, vec3_t impact, vec3_t dir, vec3_t endout)
40 {
41  vec3_t v, newv;
42  float dot;
43 
44  VectorSubtract(impact, start, v);
45  dot = DotProduct(v, dir);
46  VectorMA(v, -2 * dot, dir, newv);
47 
48  VectorNormalize(newv);
49  VectorMA(impact, 8192, newv, endout);
50 }

71 qboolean CheckGauntletAttack(gentity_t* ent)
72 {
73  trace_t tr;
74  vec3_t end;
75  gentity_t* tent;
76  gentity_t* traceEnt;
77  int damage;
78 
79  // set aiming directions
80  AngleVectors(ent->client->ps.viewangles, forward, right, up);

82  CalcMuzzlePoint(ent, forward, right, up, muzzle);
83 
84  VectorMA(muzzle, 32, forward, end);
85 
86  trap_Trace(&tr, muzzle, NULL, NULL, end, ent->s.number, MASK_SHOT);
87  if (tr.surfaceFlags & SURF_NOIMPACT) {
88  return qfalse;
89  }
90 
91  if (ent->client->noclip) {
92  return qfalse;
93  }
94 
95  traceEnt = &g_entities[tr.entityNum];
96 
97  // send blood impact
98  if (traceEnt->takedamage && traceEnt->client) {
99  tent = G_TempEntity(tr.endpos, EV_MISSILE_HIT);
100  tent->s.otherEntityNum = traceEnt->s.number;
101  tent->s.eventParm = DirToByte(tr.plane.normal);
102  tent->s.weapon = ent->s.weapon;
103  }
104 
105  if (!traceEnt->takedamage) {
106  return qfalse;
107  }
108 
109  if (ent->client->ps.powerups[PW_QUAD]) {
110  G_AddEvent(ent, EV_POWERUP_QUAD, 0);
111  s_quadFactor = g_quadfactor.value;
112  } else {
113  s_quadFactor = 1;
114  }
115 #ifdef MISSIONPACK
116  if (ent->client->persistantPowerup && ent->client->persistantPowerup->item && ent->client->persistantPowerup->item->giTag == PW_DOUBLER) {
117  s_quadFactor *= 2;
118  }
119 #endif
120 
121  damage = 50 * s_quadFactor;
122  G_Damage(traceEnt, ent, ent, forward, tr.endpos,
123  damage, 0, MOD_GAUNTLET);
124 
125  return qtrue;
126 }

137 /*
138 ======================
139 SnapVectorTowards
140 
141 Round a vector to integers for more efficient network
142 transmission, but make sure that it rounds towards a given point
143 rather than blindly truncating. This prevents it from truncating
144 into a wall.
145 ======================
146 */
147 void SnapVectorTowards(vec3_t v, vec3_t to)
148 {
149  int i;
150 
151  for (i = 0; i < 3; i++) {
152  if (to[i] <= v[i]) {
153  v[i] = floor(v[i]);
154  } else {
155  v[i] = ceil(v[i]);
156  }
157  }
158 }

168 void Bullet_Fire(gentity_t* ent, float spread, int damage, int mod)
169 {
170  trace_t tr;
171  vec3_t end;
172 #ifdef MISSIONPACK
173  vec3_t impactpoint, bouncedir;
174 #endif
175  float r;
176  float u;
177  gentity_t* tent;
178  gentity_t* traceEnt;
179  int i, passent;
180 
181  damage *= s_quadFactor;
182 
183  r = random() * M_PI * 2.0f;
184  u = sin(r) * crandom() * spread * 16;
185  r = cos(r) * crandom() * spread * 16;
186  VectorMA(muzzle, 8192 * 16, forward, end);
187  VectorMA(end, r, right, end);
188  VectorMA(end, u, up, end);
189 
190  passent = ent->s.number;
191  for (i = 0; i < 10; i++) {
192 
193  trap_Trace(&tr, muzzle, NULL, NULL, end, passent, MASK_SHOT);
194  if (tr.surfaceFlags & SURF_NOIMPACT) {
195  return;
196  }
197 
198  traceEnt = &g_entities[tr.entityNum];
199 
200  // snap the endpos to integers, but nudged towards the line
201  SnapVectorTowards(tr.endpos, muzzle);
202 
203  // send bullet impact
204  if (traceEnt->takedamage && traceEnt->client) {
205  tent = G_TempEntity(tr.endpos, EV_BULLET_HIT_FLESH);
206  tent->s.eventParm = traceEnt->s.number;
207  if (LogAccuracyHit(traceEnt, ent)) {
208  ent->client->accuracy_hits++;
209  }
210  } else {
211  tent = G_TempEntity(tr.endpos, EV_BULLET_HIT_WALL);
212  tent->s.eventParm = DirToByte(tr.plane.normal);
213  }
214  tent->s.otherEntityNum = ent->s.number;
215 
216  if (traceEnt->takedamage) {
217 #ifdef MISSIONPACK
218  if (traceEnt->client && traceEnt->client->invulnerabilityTime > level.time) {
219  if (G_InvulnerabilityEffect(traceEnt, forward, tr.endpos, impactpoint, bouncedir)) {
220  G_BounceProjectile(muzzle, impactpoint, bouncedir, end);
221  VectorCopy(impactpoint, muzzle);
222  // the player can hit him/herself with the bounced rail
223  passent = ENTITYNUM_NONE;
224  } else {
225  VectorCopy(tr.endpos, muzzle);
226  passent = traceEnt->s.number;
227  }
228  continue;
229  } else {
230 #endif
231  G_Damage(traceEnt, ent, ent, forward, tr.endpos,
232  damage, 0, mod);
233 #ifdef MISSIONPACK
234  }
235 #endif
236  }
237  break;
238  }
239 }

283 {

285  int damage, i, passent;
286  gentity_t* traceEnt;
287 #ifdef MISSIONPACK
288  vec3_t impactpoint, bouncedir;
289 #endif
290  vec3_t tr_start, tr_end;
291  qboolean hitClient = qfalse;
292 
293  passent = ent->s.number;
294  VectorCopy(start, tr_start);
295  VectorCopy(end, tr_end);
296  for (i = 0; i < 10; i++) {
297  trap_Trace(&tr, tr_start, NULL, NULL, tr_end, passent, MASK_SHOT);
298  traceEnt = &g_entities[tr.entityNum];

300  // send bullet impact

302  return qfalse;
303  }
304

306  damage = DEFAULT_SHOTGUN_DAMAGE * s_quadFactor;
307 #ifdef MISSIONPACK
308  if (traceEnt->client && traceEnt->client->invulnerabilityTime > level.time) {
309  if (G_InvulnerabilityEffect(traceEnt, forward, tr.endpos, impactpoint, bouncedir)) {
310  G_BounceProjectile(tr_start, impactpoint, bouncedir, tr_end);
311  VectorCopy(impactpoint, tr_start);
312  // the player can hit him/herself with the bounced rail
313  passent = ENTITYNUM_NONE;
314  } else {
315  VectorCopy(tr.endpos, tr_start);
316  passent = traceEnt->s.number;
317  }
318  continue;
319  }
320 #endif
321  if (LogAccuracyHit(traceEnt, ent)) {
322  hitClient = qtrue;
323  }
324  G_Damage(traceEnt, ent, ent, forward, tr.endpos, damage, 0, MOD_SHOTGUN);
325  return hitClient;
326  }
327  return qfalse;
328  }

341 void ShotgunPattern(vec3_t origin, vec3_t origin2, int seed, gentity_t* ent)
342 {
343  int i;
344  float r, u;
345  vec3_t end;
346  vec3_t forward, right, up;
347  qboolean hitClient = qfalse;
348 
349  // derive the right and up vectors from the forward vector, because
350  // the client won't have any other information
351  VectorNormalize2(origin2, forward);
352  PerpendicularVector(right, forward);
353  CrossProduct(forward, right, up);
354 
355  // generate the "random" spread pattern
356  for (i = 0; i < DEFAULT_SHOTGUN_COUNT; i++) {
357  r = Q_crandom(&seed) * DEFAULT_SHOTGUN_SPREAD * 16;
358  u = Q_crandom(&seed) * DEFAULT_SHOTGUN_SPREAD * 16;
359  VectorMA(origin, 8192 * 16, forward, end);
360  VectorMA(end, r, right, end);
361  VectorMA(end, u, up, end);
362  if (ShotgunPellet(origin, end, ent) && !hitClient) {
363  hitClient = qtrue;
364  ent->client->accuracy_hits++;
365  }
366  }
367 }

459 /*
460 =================
461 weapon_railgun_fire
462 =================
463 */
464 #define MAX_RAIL_HITS 4
465 void weapon_railgun_fire(gentity_t* ent)
466 {
467  vec3_t end;
468 #ifdef MISSIONPACK
469  vec3_t impactpoint, bouncedir;
470 #endif
471  trace_t trace;
472  gentity_t* tent;
473  gentity_t* traceEnt;
474  int damage;
475  int i;
476  int hits;
477  int unlinked;
478  int passent;
479  gentity_t* unlinkedEntities[MAX_RAIL_HITS];
480 
481  damage = 100 * s_quadFactor;
482 
483  VectorMA(muzzle, 8192, forward, end);
484 
485  // trace only against the solids, so the railgun will go through people
486  unlinked = 0;
487  hits = 0;
488  passent = ent->s.number;
489  do {
490  trap_Trace(&trace, muzzle, NULL, NULL, end, passent, MASK_SHOT);
491  if (trace.entityNum >= ENTITYNUM_MAX_NORMAL) {
492  break;
493  }
494  traceEnt = &g_entities[trace.entityNum];
495  if (traceEnt->takedamage) {
496 #ifdef MISSIONPACK
497  if (traceEnt->client && traceEnt->client->invulnerabilityTime > level.time) {
498  if (G_InvulnerabilityEffect(traceEnt, forward, trace.endpos, impactpoint, bouncedir)) {
499  G_BounceProjectile(muzzle, impactpoint, bouncedir, end);
500  // snap the endpos to integers to save net bandwidth, but nudged towards the line
501  SnapVectorTowards(trace.endpos, muzzle);
502  // send railgun beam effect
503  tent = G_TempEntity(trace.endpos, EV_RAILTRAIL);
504  // set player number for custom colors on the railtrail
505  tent->s.clientNum = ent->s.clientNum;
506  VectorCopy(muzzle, tent->s.origin2);
507  // move origin a bit to come closer to the drawn gun muzzle
508  VectorMA(tent->s.origin2, 4, right, tent->s.origin2);
509  VectorMA(tent->s.origin2, -1, up, tent->s.origin2);
510  tent->s.eventParm = 255; // don't make the explosion at the end
511  //
512  VectorCopy(impactpoint, muzzle);
513  // the player can hit him/herself with the bounced rail
514  passent = ENTITYNUM_NONE;
515  }
516  } else {
517  if (LogAccuracyHit(traceEnt, ent)) {
518  hits++;
519  }
520  G_Damage(traceEnt, ent, ent, forward, trace.endpos, damage, 0, MOD_RAILGUN);
521  }
522 #else
523  if (LogAccuracyHit(traceEnt, ent)) {
524  hits++;
525  }
526  G_Damage(traceEnt, ent, ent, forward, trace.endpos, damage, 0, MOD_RAILGUN);
527 #endif
528  }
529  if (trace.contents & CONTENTS_SOLID) {
530  break; // we hit something solid enough to stop the beam
531  }
532  // unlink this entity, so the next trace will go past it
533  trap_UnlinkEntity(traceEnt);
534  unlinkedEntities[unlinked] = traceEnt;
535  unlinked++;
536  } while (unlinked < MAX_RAIL_HITS);
537 
538  // link back in any entities we unlinked
539  for (i = 0; i < unlinked; i++) {
540  trap_LinkEntity(unlinkedEntities[i]);
541  }
542 
543  // the final trace endpos will be the terminal point of the rail trail
544 
545  // snap the endpos to integers to save net bandwidth, but nudged towards the line
546  SnapVectorTowards(trace.endpos, muzzle);
547 
548  // send railgun beam effect
549  tent = G_TempEntity(trace.endpos, EV_RAILTRAIL);
550 
551  // set player number for custom colors on the railtrail
552  tent->s.clientNum = ent->s.clientNum;
553 
554  VectorCopy(muzzle, tent->s.origin2);
555  // move origin a bit to come closer to the drawn gun muzzle
556  VectorMA(tent->s.origin2, 4, right, tent->s.origin2);
557  VectorMA(tent->s.origin2, -1, up, tent->s.origin2);
558 
559  // no explosion at end if SURF_NOIMPACT, but still make the trail
560  if (trace.surfaceFlags & SURF_NOIMPACT) {
561  tent->s.eventParm = 255; // don't make the explosion at the end
562  } else {
563  tent->s.eventParm = DirToByte(trace.plane.normal);
564  }
565  tent->s.clientNum = ent->s.clientNum;
566 
567  // give the shooter a reward sound if they have made two railgun hits in a row
568  if (hits == 0) {
569  // complete miss
570  ent->client->accurateCount = 0;
571  } else {
572  // check for "impressive" reward sound
573  ent->client->accurateCount += hits;
574  if (ent->client->accurateCount >= 2) {
575  ent->client->accurateCount -= 2;
576  ent->client->ps.persistant[PERS_IMPRESSIVE_COUNT]++;
577  // add the sprite over the player's head
578  ent->client->ps.eFlags &= ~(EF_AWARD_IMPRESSIVE | EF_AWARD_EXCELLENT | EF_AWARD_GAUNTLET | EF_AWARD_ASSIST | EF_AWARD_DEFEND | EF_AWARD_CAP);
579  ent->client->ps.eFlags |= EF_AWARD_IMPRESSIVE;
580  ent->client->rewardTime = level.time + REWARD_SPRITE_TIME;
581  }
582  ent->client->accuracy_hits++;
583  }
584 
585 }

637 #ifdef MISSIONPACK

639 #endif

641  int damage, i, passent;
642 
643  damage = 8 * s_quadFactor;
644 
645  passent = ent->s.number;
646  for (i = 0; i < 10; i++) {
647  VectorMA(muzzle, LIGHTNING_RANGE, forward, end);
648 
649  trap_Trace(&tr, muzzle, NULL, NULL, end, passent, MASK_SHOT);
650 
651 #ifdef MISSIONPACK
652  // if not the first trace (the lightning bounced of an invulnerability sphere)
653  if (i) {

655  // the first lightning bolt is a cgame only visual
656  //
657  tent = G_TempEntity(muzzle, EV_LIGHTNINGBOLT);
658  VectorCopy(tr.endpos, end);
659  SnapVector(end);
660  VectorCopy(end, tent->s.origin2);
661  }
662 #endif
663  if (tr.entityNum == ENTITYNUM_NONE)
664  return;
665

672  G_BounceProjectile(muzzle, impactpoint, bouncedir, end);

674  VectorSubtract(end, impactpoint, forward);
675  VectorNormalize(forward);
676  // the player can hit him/herself with the bounced lightning
677  passent = ENTITYNUM_NONE;
678  } else {
679  VectorCopy(tr.endpos, muzzle);
680  passent = traceEnt->s.number;
681  }
682  continue;
683  }
684 #endif
685  if (LogAccuracyHit(traceEnt, ent))
686  ent->client->accuracy_hits++;
687 
688  G_Damage(traceEnt, ent, ent, forward, tr.endpos, damage, 0, MOD_LIGHTNING);
689  }
690 
691  if (traceEnt->takedamage && traceEnt->client) {
692  tent = G_TempEntity(tr.endpos, EV_MISSILE_HIT);
693  tent->s.otherEntityNum = traceEnt->s.number;
694  tent->s.eventParm = DirToByte(tr.plane.normal);
695  tent->s.weapon = ent->s.weapon;
696  } else if (!(tr.surfaceFlags & SURF_NOIMPACT)) {
697  tent = G_TempEntity(tr.endpos, EV_MISSILE_MISS);
698  tent->s.eventParm = DirToByte(tr.plane.normal);
699  }
700 
701  break;
702  }
