48 void RB_CheckOverflow( int verts, int indexes ) {
49  if (tess.numVertexes + verts < SHADER_MAX_VERTEXES
50  && tess.numIndexes + indexes < SHADER_MAX_INDEXES) {
51  return;
52  }
53 
54  RB_EndSurface();
55 
56  if ( verts >= SHADER_MAX_VERTEXES ) {
57  ri.Error(ERR_DROP, "RB_CheckOverflow: verts > MAX (%d > %d)", verts, SHADER_MAX_VERTEXES );
58  }
59  if ( indexes >= SHADER_MAX_INDEXES ) {
60  ri.Error(ERR_DROP, "RB_CheckOverflow: indices > MAX (%d > %d)", indexes, SHADER_MAX_INDEXES );
61  }
62 
63  RB_BeginSurface(tess.shader, tess.fogNum, tess.cubemapIndex );
64 }

86 void RB_AddQuadStampExt( vec3_t origin, vec3_t left, vec3_t up, float color[4], float s1, float t1, float s2, float t2 ) {
87  vec3_t normal;
88  int16_t iNormal[4];
89  uint16_t iColor[4];
90  int ndx;
91 
92  RB_CheckVao(tess.vao);
93 
94  RB_CHECKOVERFLOW( 4, 6 );
95 
96  ndx = tess.numVertexes;
97 
98  // triangle indexes for a simple quad
99  tess.indexes[ tess.numIndexes ] = ndx;
100  tess.indexes[ tess.numIndexes + 1 ] = ndx + 1;
101  tess.indexes[ tess.numIndexes + 2 ] = ndx + 3;
102 
103  tess.indexes[ tess.numIndexes + 3 ] = ndx + 3;
104  tess.indexes[ tess.numIndexes + 4 ] = ndx + 1;
105  tess.indexes[ tess.numIndexes + 5 ] = ndx + 2;
106 
107  tess.xyz[ndx][0] = origin[0] + left[0] + up[0];
108  tess.xyz[ndx][1] = origin[1] + left[1] + up[1];
109  tess.xyz[ndx][2] = origin[2] + left[2] + up[2];
110 
111  tess.xyz[ndx+1][0] = origin[0] - left[0] + up[0];
112  tess.xyz[ndx+1][1] = origin[1] - left[1] + up[1];
113  tess.xyz[ndx+1][2] = origin[2] - left[2] + up[2];
114 
115  tess.xyz[ndx+2][0] = origin[0] - left[0] - up[0];
116  tess.xyz[ndx+2][1] = origin[1] - left[1] - up[1];
117  tess.xyz[ndx+2][2] = origin[2] - left[2] - up[2];
118 
119  tess.xyz[ndx+3][0] = origin[0] + left[0] - up[0];
120  tess.xyz[ndx+3][1] = origin[1] + left[1] - up[1];
121  tess.xyz[ndx+3][2] = origin[2] + left[2] - up[2];
122 
123 
124  // constant normal all the way around
125  VectorSubtract( vec3_origin, backEnd.viewParms.or.axis[0], normal );
126 
127  R_VaoPackNormal(iNormal, normal);
128 
129  VectorCopy4(iNormal, tess.normal[ndx]);
130  VectorCopy4(iNormal, tess.normal[ndx + 1]);
131  VectorCopy4(iNormal, tess.normal[ndx + 2]);
132  VectorCopy4(iNormal, tess.normal[ndx + 3]);
133 
134  // standard square texture coordinates
135  VectorSet2(tess.texCoords[ndx], s1, t1);
136  VectorSet2(tess.lightCoords[ndx], s1, t1);
137 
138  VectorSet2(tess.texCoords[ndx+1], s2, t1);
139  VectorSet2(tess.lightCoords[ndx+1], s2, t1);
140 
141  VectorSet2(tess.texCoords[ndx+2], s2, t2);
142  VectorSet2(tess.lightCoords[ndx+2], s2, t2);
143 
144  VectorSet2(tess.texCoords[ndx+3], s1, t2);
145  VectorSet2(tess.lightCoords[ndx+3], s1, t2);
146 
147  // constant color all the way around
148  // should this be identity and let the shader specify from entity?
149 
150  R_VaoPackColor(iColor, color);
151 
152  VectorCopy4(iColor, tess.color[ndx]);
153  VectorCopy4(iColor, tess.color[ndx + 1]);
154  VectorCopy4(iColor, tess.color[ndx + 2]);
155  VectorCopy4(iColor, tess.color[ndx + 3]);
156 
157  tess.numVertexes += 4;
158  tess.numIndexes += 6;
159 }

242 static void RB_SurfaceSprite( void ) {
243  vec3_t left, up;
244  float radius;
245  float colors[4];
246  trRefEntity_t *ent = backEnd.currentEntity;
247 
248  // calculate the xyz locations for the four corners
249  radius = ent->e.radius;
250  if ( ent->e.rotation == 0 ) {
251  VectorScale( backEnd.viewParms.or.axis[1], radius, left );
252  VectorScale( backEnd.viewParms.or.axis[2], radius, up );
253  } else {
254  float s, c;
255  float ang;
256 
257  ang = M_PI * ent->e.rotation / 180;
258  s = sin( ang );
259  c = cos( ang );
260 
261  VectorScale( backEnd.viewParms.or.axis[1], c * radius, left );
262  VectorMA( left, -s * radius, backEnd.viewParms.or.axis[2], left );
263 
264  VectorScale( backEnd.viewParms.or.axis[2], c * radius, up );
265  VectorMA( up, s * radius, backEnd.viewParms.or.axis[1], up );
266  }
267  if ( backEnd.viewParms.isMirror ) {
268  VectorSubtract( vec3_origin, left, left );
269  }
270 
271  VectorScale4(ent->e.shaderRGBA, 1.0f / 255.0f, colors);
272 
273  RB_AddQuadStamp( ent->e.origin, left, up, colors );
274 }

282 static void RB_SurfacePolychain( srfPoly_t *p ) {
283  int i;
284  int numv;
285 
286  RB_CheckVao(tess.vao);
287 
288  RB_CHECKOVERFLOW( p->numVerts, 3*(p->numVerts - 2) );
289 
290  // fan triangles into the tess array
291  numv = tess.numVertexes;
292  for ( i = 0; i < p->numVerts; i++ ) {
293  VectorCopy( p->verts[i].xyz, tess.xyz[numv] );
294  tess.texCoords[numv][0] = p->verts[i].st[0];
295  tess.texCoords[numv][1] = p->verts[i].st[1];
296  tess.color[numv][0] = (int)p->verts[i].modulate[0] * 257;
297  tess.color[numv][1] = (int)p->verts[i].modulate[1] * 257;
298  tess.color[numv][2] = (int)p->verts[i].modulate[2] * 257;
299  tess.color[numv][3] = (int)p->verts[i].modulate[3] * 257;
300 
301  numv++;
302  }
303 
304  // generate fan indexes into the tess array
305  for ( i = 0; i < p->numVerts-2; i++ ) {
306  tess.indexes[tess.numIndexes + 0] = tess.numVertexes;
307  tess.indexes[tess.numIndexes + 1] = tess.numVertexes + i + 1;
308  tess.indexes[tess.numIndexes + 2] = tess.numVertexes + i + 2;
309  tess.numIndexes += 3;
310  }
311 
312  tess.numVertexes = numv;
313 }

315 static void RB_SurfaceVertsAndIndexes( int numVerts, srfVert_t *verts, int numIndexes, glIndex_t *indexes, int dlightBits, int pshadowBits)
316 {
317  int i;
318  glIndex_t *inIndex;
319  srfVert_t *dv;
320  float *xyz, *texCoords, *lightCoords;
321  int16_t *lightdir;
322  int16_t *normal;
323  int16_t *tangent;
324  glIndex_t *outIndex;
325  uint16_t *color;
326 
327  RB_CheckVao(tess.vao);
328 
329  RB_CHECKOVERFLOW( numVerts, numIndexes );
330 
331  inIndex = indexes;
332  outIndex = &tess.indexes[ tess.numIndexes ];
333  for ( i = 0 ; i < numIndexes ; i++ ) {
334  *outIndex++ = tess.numVertexes + *inIndex++;
335  }
336  tess.numIndexes += numIndexes;
337 
338  if ( tess.shader->vertexAttribs & ATTR_POSITION )
339  {
340  dv = verts;
341  xyz = tess.xyz[ tess.numVertexes ];
342  for ( i = 0 ; i < numVerts ; i++, dv++, xyz+=4 )
343  VectorCopy(dv->xyz, xyz);
344  }
345 
346  if ( tess.shader->vertexAttribs & ATTR_NORMAL )
347  {
348  dv = verts;
349  normal = tess.normal[ tess.numVertexes ];
350  for ( i = 0 ; i < numVerts ; i++, dv++, normal+=4 )
351  VectorCopy4(dv->normal, normal);
352  }
353 
354  if ( tess.shader->vertexAttribs & ATTR_TANGENT )
355  {
356  dv = verts;
357  tangent = tess.tangent[ tess.numVertexes ];
358  for ( i = 0 ; i < numVerts ; i++, dv++, tangent+=4 )
359  VectorCopy4(dv->tangent, tangent);
360  }
361 
362  if ( tess.shader->vertexAttribs & ATTR_TEXCOORD )
363  {
364  dv = verts;
365  texCoords = tess.texCoords[tess.numVertexes];
366  for ( i = 0 ; i < numVerts ; i++, dv++, texCoords+=2 )
367  VectorCopy2(dv->st, texCoords);
368  }
369 
370  if ( tess.shader->vertexAttribs & ATTR_LIGHTCOORD )
371  {
372  dv = verts;
373  lightCoords = tess.lightCoords[ tess.numVertexes ];
374  for ( i = 0 ; i < numVerts ; i++, dv++, lightCoords+=2 )
375  VectorCopy2(dv->lightmap, lightCoords);
376  }
377 
378  if ( tess.shader->vertexAttribs & ATTR_COLOR )
379  {
380  dv = verts;
381  color = tess.color[ tess.numVertexes ];
382  for ( i = 0 ; i < numVerts ; i++, dv++, color+=4 )
383  VectorCopy4(dv->color, color);
384  }
385 
386  if ( tess.shader->vertexAttribs & ATTR_LIGHTDIRECTION )
387  {
388  dv = verts;
389  lightdir = tess.lightdir[ tess.numVertexes ];
390  for ( i = 0 ; i < numVerts ; i++, dv++, lightdir+=4 )
391  VectorCopy4(dv->lightdir, lightdir);
392  }
393 
394 #if 0 // nothing even uses vertex dlightbits
395  for ( i = 0 ; i < numVerts ; i++ ) {
396  tess.vertexDlightBits[ tess.numVertexes + i ] = dlightBits;
397  }
398 #endif
399 
400  tess.dlightBits |= dlightBits;
401  tess.pshadowBits |= pshadowBits;
402 
403  tess.numVertexes += numVerts;
404 }

406 static qboolean RB_SurfaceVaoCached(int numVerts, srfVert_t *verts, int numIndexes, glIndex_t *indexes, int dlightBits, int pshadowBits)
407 {
408  qboolean recycleVertexBuffer = qfalse;
409  qboolean recycleIndexBuffer = qfalse;
410  qboolean endSurface = qfalse;
411 
412  if (!(!ShaderRequiresCPUDeforms(tess.shader) && !tess.shader->isSky && !tess.shader->isPortal))
413  return qfalse;
414 
415  if (!numIndexes || !numVerts)
416  return qfalse;
417 
418  VaoCache_BindVao();
419 
420  tess.dlightBits |= dlightBits;
421  tess.pshadowBits |= pshadowBits;
422 
423  VaoCache_CheckAdd(&endSurface, &recycleVertexBuffer, &recycleIndexBuffer, numVerts, numIndexes);
424 
425  if (endSurface)
426  {
427  RB_EndSurface();
428  RB_BeginSurface(tess.shader, tess.fogNum, tess.cubemapIndex);
429  }
430 
431  if (recycleVertexBuffer)
432  VaoCache_RecycleVertexBuffer();
433 
434  if (recycleIndexBuffer)
435  VaoCache_RecycleIndexBuffer();
436 
437  if (!tess.numVertexes)
438  VaoCache_InitQueue();
439 
440  VaoCache_AddSurface(verts, numVerts, indexes, numIndexes);
441 
442  tess.numIndexes += numIndexes;
443  tess.numVertexes += numVerts;
444  tess.useInternalVao = qfalse;
445  tess.useCacheVao = qtrue;
446 
447  return qtrue;
448 }

470 ==============
471 RB_SurfaceBeam
472 ==============
473 */
474 static void RB_SurfaceBeam( void )
475 {
476 #define NUM_BEAM_SEGS 6
477  refEntity_t *e;
478  shaderProgram_t *sp = &tr.textureColorShader;
479  int i;
480  vec3_t perpvec;
481  vec3_t direction, normalized_direction;
482  vec3_t start_points[NUM_BEAM_SEGS], end_points[NUM_BEAM_SEGS];
483  vec3_t oldorigin, origin;
484 
485  e = &backEnd.currentEntity->e;
486 
487  oldorigin[0] = e->oldorigin[0];
488  oldorigin[1] = e->oldorigin[1];
489  oldorigin[2] = e->oldorigin[2];
490 
491  origin[0] = e->origin[0];
492  origin[1] = e->origin[1];
493  origin[2] = e->origin[2];
494 
495  normalized_direction[0] = direction[0] = oldorigin[0] - origin[0];
496  normalized_direction[1] = direction[1] = oldorigin[1] - origin[1];
497  normalized_direction[2] = direction[2] = oldorigin[2] - origin[2];
498 
499  if ( VectorNormalize( normalized_direction ) == 0 )
500  return;
501 
502  PerpendicularVector( perpvec, normalized_direction );
503 
504  VectorScale( perpvec, 4, perpvec );
505 
506  for ( i = 0; i < NUM_BEAM_SEGS ; i++ )
507  {
508  RotatePointAroundVector( start_points[i], normalized_direction, perpvec, (360.0/NUM_BEAM_SEGS)*i );
509 // VectorAdd( start_points[i], origin, start_points[i] );
510  VectorAdd( start_points[i], direction, end_points[i] );
511  }
512 
513  GL_BindToTMU( tr.whiteImage, TB_COLORMAP );
514 
515  GL_State( GLS_SRCBLEND_ONE | GLS_DSTBLEND_ONE );
516 
517  // FIXME: Quake3 doesn't use this, so I never tested it
518  tess.numVertexes = 0;
519  tess.numIndexes = 0;
520  tess.firstIndex = 0;
521 
522  for ( i = 0; i <= NUM_BEAM_SEGS; i++ ) {
523  VectorCopy(start_points[ i % NUM_BEAM_SEGS ], tess.xyz[tess.numVertexes++]);
524  VectorCopy(end_points [ i % NUM_BEAM_SEGS ], tess.xyz[tess.numVertexes++]);
525  }
526 
527  for ( i = 0; i < NUM_BEAM_SEGS; i++ ) {
528  tess.indexes[tess.numIndexes++] = i * 2;
529  tess.indexes[tess.numIndexes++] = (i + 1) * 2;
530  tess.indexes[tess.numIndexes++] = 1 + i * 2;
531 
532  tess.indexes[tess.numIndexes++] = 1 + i * 2;
533  tess.indexes[tess.numIndexes++] = (i + 1) * 2;
534  tess.indexes[tess.numIndexes++] = 1 + (i + 1) * 2;
535  }
536 
537  // FIXME: A lot of this can probably be removed for speed, and refactored into a more convenient function
538  RB_UpdateTessVao(ATTR_POSITION);
539 
540  GLSL_BindProgram(sp);
541 
542  GLSL_SetUniformMat4(sp, UNIFORM_MODELVIEWPROJECTIONMATRIX, glState.modelviewProjection);
543 
544  GLSL_SetUniformVec4(sp, UNIFORM_COLOR, colorRed);
545 
546  GLSL_SetUniformInt(sp, UNIFORM_ALPHATEST, 0);
547 
548  R_DrawElements(tess.numIndexes, tess.firstIndex);
549 
550  tess.numIndexes = 0;
551  tess.numVertexes = 0;
552  tess.firstIndex = 0;
553 }

903 static float LodErrorForVolume( vec3_t local, float radius ) {
904  vec3_t world;
905  float d;
906 
907  // never let it go negative
908  if ( r_lodCurveError->value < 0 ) {
909  return 0;
910  }
911 
912  world[0] = local[0] * backEnd.or.axis[0][0] + local[1] * backEnd.or.axis[1][0] +
913  local[2] * backEnd.or.axis[2][0] + backEnd.or.origin[0];
914  world[1] = local[0] * backEnd.or.axis[0][1] + local[1] * backEnd.or.axis[1][1] +
915  local[2] * backEnd.or.axis[2][1] + backEnd.or.origin[1];
916  world[2] = local[0] * backEnd.or.axis[0][2] + local[1] * backEnd.or.axis[1][2] +
917  local[2] * backEnd.or.axis[2][2] + backEnd.or.origin[2];
918 
919  VectorSubtract( world, backEnd.viewParms.or.origin, world );
920  d = DotProduct( world, backEnd.viewParms.or.axis[0] );
921 
922  if ( d < 0 ) {
923  d = -d;
924  }
925  d -= radius;
926  if ( d < 1 ) {
927  d = 1;
928  }
929 
930  return r_lodCurveError->value / d;
