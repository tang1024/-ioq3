28 /*
29 =================
30 SV_GetChallenge
31 
32 A "getchallenge" OOB command has been received
33 Returns a challenge number that can be used
34 in a subsequent connectResponse command.
35 We do this to prevent denial of service attacks that
36 flood the server with invalid connection IPs. With a
37 challenge, they must give a valid IP address.
38 
39 If we are authorizing, a challenge request will cause a packet
40 to be sent to the authorize server.
41 
42 When an authorizeip is returned, a challenge response will be
43 sent to that ip.
44 
45 ioquake3: we added a possibility for clients to add a challenge
46 to their packets, to make it more difficult for malicious servers
47 to hi-jack client connections.
48 Also, the auth stuff is completely disabled for com_standalone games
49 as well as IPv6 connections, since there is no way to use the
50 v4-only auth server for these new types of connections.
51 =================
52 */
53 void SV_GetChallenge(netadr_t from)
54 {
55  int i;
56  int oldest;
57  int oldestTime;
58  int oldestClientTime;
59  int clientChallenge;
60  challenge_t *challenge;
61  qboolean wasfound = qfalse;
62  char *gameName;
63  qboolean gameMismatch;
64 
65  // ignore if we are in single player
66  if ( Cvar_VariableValue( "g_gametype" ) == GT_SINGLE_PLAYER || Cvar_VariableValue("ui_singlePlayerActive")) {
67  return;
68  }
69 
70  // Prevent using getchallenge as an amplifier
71  if ( SVC_RateLimitAddress( from, 10, 1000 ) ) {
72  Com_DPrintf( "SV_GetChallenge: rate limit from %s exceeded, dropping request\n",
73  NET_AdrToString( from ) );
74  return;
75  }
76 
77  // Allow getchallenge to be DoSed relatively easily, but prevent
78  // excess outbound bandwidth usage when being flooded inbound
79  if ( SVC_RateLimit( &outboundLeakyBucket, 10, 100 ) ) {
80  Com_DPrintf( "SV_GetChallenge: rate limit exceeded, dropping request\n" );
81  return;
82  }
83 
84  gameName = Cmd_Argv(2);
85 
86 #ifdef LEGACY_PROTOCOL
87  // gamename is optional for legacy protocol
88  if (com_legacyprotocol->integer && !*gameName)
89  gameMismatch = qfalse;
90  else
91 #endif
92  gameMismatch = !*gameName || strcmp(gameName, com_gamename->string) != 0;
93 
94  // reject client if the gamename string sent by the client doesn't match ours
95  if (gameMismatch)
96  {
97  NET_OutOfBandPrint(NS_SERVER, from, "print\nGame mismatch: This is a %s server\n",
98  com_gamename->string);
99  return;
100  }
101 
102  oldest = 0;
103  oldestClientTime = oldestTime = 0x7fffffff;
104 
105  // see if we already have a challenge for this ip
106  challenge = &svs.challenges[0];
107  clientChallenge = atoi(Cmd_Argv(1));
108 
109  for(i = 0 ; i < MAX_CHALLENGES ; i++, challenge++)
110  {
111  if(!challenge->connected && NET_CompareAdr(from, challenge->adr))
112  {
113  wasfound = qtrue;
114 
115  if(challenge->time < oldestClientTime)
116  oldestClientTime = challenge->time;
117  }
118 
119  if(wasfound && i >= MAX_CHALLENGES_MULTI)
120  {
121  i = MAX_CHALLENGES;
122  break;
123  }
124 
125  if(challenge->time < oldestTime)
126  {
127  oldestTime = challenge->time;
128  oldest = i;
129  }
130  }
131 
132  if (i == MAX_CHALLENGES)
133  {
134  // this is the first time this client has asked for a challenge
135  challenge = &svs.challenges[oldest];
136  challenge->clientChallenge = clientChallenge;
137  challenge->adr = from;
138  challenge->firstTime = svs.time;
139  challenge->connected = qfalse;
140  }
141 
142  // always generate a new challenge number, so the client cannot circumvent sv_maxping
143  challenge->challenge = ( ((unsigned int)rand() << 16) ^ (unsigned int)rand() ) ^ svs.time;
144  challenge->wasrefused = qfalse;
145  challenge->time = svs.time;
146 
147 #ifndef STANDALONE
148  // Drop the authorize stuff if this client is coming in via v6 as the auth server does not support ipv6.
149  // Drop also for addresses coming in on local LAN and for stand-alone games independent from id's assets.
150  if(challenge->adr.type == NA_IP && !com_standalone->integer && !Sys_IsLANAddress(from))
151  {
152  // look up the authorize server's IP
153  if (svs.authorizeAddress.type == NA_BAD)
154  {
155  Com_Printf( "Resolving %s\n", AUTHORIZE_SERVER_NAME );
156 
157  if (NET_StringToAdr(AUTHORIZE_SERVER_NAME, &svs.authorizeAddress, NA_IP))
158  {
159  svs.authorizeAddress.port = BigShort( PORT_AUTHORIZE );
160  Com_Printf( "%s resolved to %i.%i.%i.%i:%i\n", AUTHORIZE_SERVER_NAME,
161  svs.authorizeAddress.ip[0], svs.authorizeAddress.ip[1],
162  svs.authorizeAddress.ip[2], svs.authorizeAddress.ip[3],
163  BigShort( svs.authorizeAddress.port ) );
164  }
165  }
166 
167  // we couldn't contact the auth server, let them in.
168  if(svs.authorizeAddress.type == NA_BAD)
169  Com_Printf("Couldn't resolve auth server address\n");
170 
171  // if they have been challenging for a long time and we
172  // haven't heard anything from the authorize server, go ahead and
173  // let them in, assuming the id server is down
174  else if(svs.time - oldestClientTime > AUTHORIZE_TIMEOUT)
175  Com_DPrintf( "authorize server timed out\n" );
176  else
177  {
178  // otherwise send their ip to the authorize server
179  const char *game;
180 
181  Com_DPrintf( "sending getIpAuthorize for %s\n", NET_AdrToString( from ));
182 
183  game = Cvar_VariableString( "fs_game" );
184  if (game[0] == 0) {
185  game = BASEGAME;
186  }
187 
188  // the 0 is for backwards compatibility with obsolete sv_allowanonymous flags
189  // getIpAuthorize <challenge> <IP> <game> 0 <auth-flag>
190  NET_OutOfBandPrint( NS_SERVER, svs.authorizeAddress,
191  "getIpAuthorize %i %i.%i.%i.%i %s 0 %s", challenge->challenge,
192  from.ip[0], from.ip[1], from.ip[2], from.ip[3], game, sv_strictAuth->string );
193 
194  return;
195  }
196  }
197 #endif
198 
199  challenge->pingTime = svs.time;
200  NET_OutOfBandPrint(NS_SERVER, challenge->adr, "challengeResponse %d %d %d",
201  challenge->challenge, clientChallenge, com_protocol->integer);
202 }

261  NET_OutOfBandPrint( NS_SERVER, challengeptr->adr, "print\n%s\n", r);
262  }
263  // clear the challenge record so it won't timeout and let them through
264  Com_Memset( challengeptr, 0, sizeof( *challengeptr ) );
265  return;
266  }
267 
268  // authorization failed
269  if (!r) {
270  NET_OutOfBandPrint( NS_SERVER, challengeptr->adr, "print\nSomeone is using this CD Key\n" );
271  } else {
272  NET_OutOfBandPrint( NS_SERVER, challengeptr->adr, "print\n%s\n", r );
273  }
274 
275  // clear the challenge record so it won't timeout and let them through
276  Com_Memset( challengeptr, 0, sizeof(*challengeptr) );
277 }

322 void SV_DirectConnect( netadr_t from ) {
323  char userinfo[MAX_INFO_STRING];
324  int i;
325  client_t *cl, *newcl;
326  client_t temp;
327  sharedEntity_t *ent;
328  int clientNum;
329  int version;
330  int qport;
331  int challenge;
332  char *password;
333  int startIndex;
334  intptr_t denied;
335  int count;
336  char *ip;
337 #ifdef LEGACY_PROTOCOL
338  qboolean compat = qfalse;
339 #endif
340 
341  Com_DPrintf ("SVC_DirectConnect ()\n");
342 
343  // Check whether this client is banned.
344  if(SV_IsBanned(&from, qfalse))
345  {
346  NET_OutOfBandPrint(NS_SERVER, from, "print\nYou are banned from this server.\n");
347  return;
348  }
349 
350  Q_strncpyz( userinfo, Cmd_Argv(1), sizeof(userinfo) );
351 
352  version = atoi(Info_ValueForKey(userinfo, "protocol"));
353 
354 #ifdef LEGACY_PROTOCOL
355  if(version > 0 && com_legacyprotocol->integer == version)
356  compat = qtrue;
357  else
358 #endif
359  {
360  if(version != com_protocol->integer)
361  {
362  NET_OutOfBandPrint(NS_SERVER, from, "print\nServer uses protocol version %i "
363  "(yours is %i).\n", com_protocol->integer, version);
364  Com_DPrintf(" rejected connect from version %i\n", version);
365  return;
366  }
367  }
368 
369  challenge = atoi( Info_ValueForKey( userinfo, "challenge" ) );
370  qport = atoi( Info_ValueForKey( userinfo, "qport" ) );
371 
372  // quick reject
373  for (i=0,cl=svs.clients ; i < sv_maxclients->integer ; i++,cl++) {
374  if ( cl->state == CS_FREE ) {
375  continue;
376  }
377  if ( NET_CompareBaseAdr( from, cl->netchan.remoteAddress )
378  && ( cl->netchan.qport == qport
379  || from.port == cl->netchan.remoteAddress.port ) ) {
380  if (( svs.time - cl->lastConnectTime)
381  < (sv_reconnectlimit->integer * 1000)) {
382  Com_DPrintf ("%s:reconnect rejected : too soon\n", NET_AdrToString (from));
383  return;
384  }
385  break;
386  }
387  }
388 
389  // don't let "ip" overflow userinfo string
390  if ( NET_IsLocalAddress (from) )
391  ip = "localhost";
392  else
393  ip = (char *)NET_AdrToString( from );
394  if( ( strlen( ip ) + strlen( userinfo ) + 4 ) >= MAX_INFO_STRING ) {
395  NET_OutOfBandPrint( NS_SERVER, from,
396  "print\nUserinfo string length exceeded. "
397  "Try removing setu cvars from your config.\n" );
398  return;
399  }
400  Info_SetValueForKey( userinfo, "ip", ip );
401 
402  // see if the challenge is valid (LAN clients don't need to challenge)
403  if (!NET_IsLocalAddress(from))
404  {
405  int ping;
406  challenge_t *challengeptr;
407 
408  for (i=0; i<MAX_CHALLENGES; i++)
409  {
410  if (NET_CompareAdr(from, svs.challenges[i].adr))
411  {
412  if(challenge == svs.challenges[i].challenge)
413  break;
414  }
415  }
416 
417  if (i == MAX_CHALLENGES)
418  {
419  NET_OutOfBandPrint( NS_SERVER, from, "print\nNo or bad challenge for your address.\n" );
420  return;
421  }
422 
423  challengeptr = &svs.challenges[i];
424 
425  if(challengeptr->wasrefused)
426  {
427  // Return silently, so that error messages written by the server keep being displayed.
428  return;
429  }
430 
431  ping = svs.time - challengeptr->pingTime;
432 
433  // never reject a LAN client based on ping
434  if ( !Sys_IsLANAddress( from ) ) {
435  if ( sv_minPing->value && ping < sv_minPing->value ) {
436  NET_OutOfBandPrint( NS_SERVER, from, "print\nServer is for high pings only\n" );
437  Com_DPrintf ("Client %i rejected on a too low ping\n", i);
438  challengeptr->wasrefused = qtrue;
439  return;
440  }
441  if ( sv_maxPing->value && ping > sv_maxPing->value ) {
442  NET_OutOfBandPrint( NS_SERVER, from, "print\nServer is for low pings only\n" );
443  Com_DPrintf ("Client %i rejected on a too high ping\n", i);
444  challengeptr->wasrefused = qtrue;
445  return;
446  }
447  }
448 
449  Com_Printf("Client %i connecting with %i challenge ping\n", i, ping);
450  challengeptr->connected = qtrue;
451  }
452 
453  newcl = &temp;
454  Com_Memset (newcl, 0, sizeof(client_t));
455 
456  // if there is already a slot for this ip, reuse it
457  for (i=0,cl=svs.clients ; i < sv_maxclients->integer ; i++,cl++) {
458  if ( cl->state == CS_FREE ) {
459  continue;
460  }
461  if ( NET_CompareBaseAdr( from, cl->netchan.remoteAddress )
462  && ( cl->netchan.qport == qport
463  || from.port == cl->netchan.remoteAddress.port ) ) {
464  Com_Printf ("%s:reconnect\n", NET_AdrToString (from));
465  newcl = cl;
466 
467  // this doesn't work because it nukes the players userinfo
468 
469 // // disconnect the client from the game first so any flags the
470 // // player might have are dropped
471 // VM_Call( gvm, GAME_CLIENT_DISCONNECT, newcl - svs.clients );
472  //
473  goto gotnewcl;
474  }
475  }
476 
477  // find a client slot
478  // if "sv_privateClients" is set > 0, then that number
479  // of client slots will be reserved for connections that
480  // have "password" set to the value of "sv_privatePassword"
481  // Info requests will report the maxclients as if the private
482  // slots didn't exist, to prevent people from trying to connect
483  // to a full server.
484  // This is to allow us to reserve a couple slots here on our
485  // servers so we can play without having to kick people.
486 
487  // check for privateClient password
488  password = Info_ValueForKey( userinfo, "password" );
489  if ( *password && !strcmp( password, sv_privatePassword->string ) ) {
490  startIndex = 0;
491  } else {
492  // skip past the reserved slots
493  startIndex = sv_privateClients->integer;
494  }
495 
496  newcl = NULL;
497  for ( i = startIndex; i < sv_maxclients->integer ; i++ ) {
498  cl = &svs.clients[i];
499  if (cl->state == CS_FREE) {
500  newcl = cl;
501  break;
502  }
503  }
504 
505  if ( !newcl ) {
506  if ( NET_IsLocalAddress( from ) ) {
507  count = 0;
508  for ( i = startIndex; i < sv_maxclients->integer ; i++ ) {
509  cl = &svs.clients[i];
510  if (cl->netchan.remoteAddress.type == NA_BOT) {
511  count++;
512  }
513  }
514  // if they're all bots
515  if (count >= sv_maxclients->integer - startIndex) {
516  SV_DropClient(&svs.clients[sv_maxclients->integer - 1], "only bots on server");
517  newcl = &svs.clients[sv_maxclients->integer - 1];
518  }
519  else {
520  Com_Error( ERR_FATAL, "server is full on local connect" );
521  return;
522  }
523  }
524  else {
525  NET_OutOfBandPrint( NS_SERVER, from, "print\nServer is full.\n" );
526  Com_DPrintf ("Rejected a connection.\n");
527  return;
528  }
529  }
530 
531  // we got a newcl, so reset the reliableSequence and reliableAcknowledge
532  cl->reliableAcknowledge = 0;
533  cl->reliableSequence = 0;
534 
535 gotnewcl:
536  // build a new connection
537  // accept the new client
538  // this is the only place a client_t is ever initialized
539  *newcl = temp;
540  clientNum = newcl - svs.clients;
541  ent = SV_GentityNum( clientNum );
542  newcl->gentity = ent;
543 
544  // save the challenge
545  newcl->challenge = challenge;
546 
547  // save the address
548 #ifdef LEGACY_PROTOCOL
549  newcl->compat = compat;
550  Netchan_Setup(NS_SERVER, &newcl->netchan, from, qport, challenge, compat);
551 #else
552  Netchan_Setup(NS_SERVER, &newcl->netchan, from, qport, challenge, qfalse);
553 #endif
554  // init the netchan queue
555  newcl->netchan_end_queue = &newcl->netchan_start_queue;
556 
557  // save the userinfo
558  Q_strncpyz( newcl->userinfo, userinfo, sizeof(newcl->userinfo) );
559 
560  // get the game a chance to reject this connection or modify the userinfo
561  denied = VM_Call( gvm, GAME_CLIENT_CONNECT, clientNum, qtrue, qfalse ); // firstTime = qtrue
562  if ( denied ) {
563  // we can't just use VM_ArgPtr, because that is only valid inside a VM_Call
564  char *str = VM_ExplicitArgPtr( gvm, denied );
565 
566  NET_OutOfBandPrint( NS_SERVER, from, "print\n%s\n", str );
567  Com_DPrintf ("Game rejected a connection: %s.\n", str);
568  return;
569  }
570 
571  SV_UserinfoChanged( newcl );
572 
573  // send the connect packet to the client
574  NET_OutOfBandPrint(NS_SERVER, from, "connectResponse %d", challenge);
575 
576  Com_DPrintf( "Going from CS_FREE to CS_CONNECTED for %s\n", newcl->name );
577 
578  newcl->state = CS_CONNECTED;
579  newcl->lastSnapshotTime = 0;
580  newcl->lastPacketTime = svs.time;
581  newcl->lastConnectTime = svs.time;
582 
583  // when we receive the first packet from the client, we will
584  // notice that it is from a different serverid and that the
585  // gamestate message was not just sent, forcing a retransmit
586  newcl->gamestateMessageNum = -1;
587 
588  // if this was the first client on the server, or the last client
589  // the server can hold, send a heartbeat to the master.
590  count = 0;
591  for (i=0,cl=svs.clients ; i < sv_maxclients->integer ; i++,cl++) {
592  if ( svs.clients[i].state >= CS_CONNECTED ) {
593  count++;
594  }
595  }
596  if ( count == 1 || count == sv_maxclients->integer ) {
597  SV_Heartbeat_f();
598  }
599 }

627 /*
628 =====================
629 SV_DropClient
630 
631 Called when the player is totally leaving the server, either willingly
632 or unwillingly. This is NOT called if the entire server is quiting
633 or crashing -- SV_FinalMessage() will handle that
634 =====================
635 */
636 void SV_DropClient( client_t *drop, const char *reason ) {
637  int i;
638  challenge_t *challenge;
639  const qboolean isBot = drop->netchan.remoteAddress.type == NA_BOT;
640 
641  if ( drop->state == CS_ZOMBIE ) {
642  return; // already dropped
643  }
644 
645  if ( !isBot ) {
646  // see if we already have a challenge for this ip
647  challenge = &svs.challenges[0];
648 
649  for (i = 0 ; i < MAX_CHALLENGES ; i++, challenge++)
650  {
651  if(NET_CompareAdr(drop->netchan.remoteAddress, challenge->adr))
652  {
653  Com_Memset(challenge, 0, sizeof(*challenge));
654  break;
655  }
656  }
657  }
658 
659  // Free all allocated data on the client structure
660  SV_FreeClient(drop);
661 
662  // tell everyone why they got dropped
663  SV_SendServerCommand( NULL, "print \"%s" S_COLOR_WHITE " %s\n\"", drop->name, reason );
664 
665  // call the prog function for removing a client
666  // this will remove the body, among other things
667  VM_Call( gvm, GAME_CLIENT_DISCONNECT, drop - svs.clients );
668 
669  // add the disconnect command
670  SV_SendServerCommand( drop, "disconnect \"%s\"", reason);
671 
672  if ( isBot ) {
673  SV_BotFreeClient( drop - svs.clients );
674 
675  // bots shouldn't go zombie, as there's no real net connection.
676  drop->state = CS_FREE;
677  } else {
678  Com_DPrintf( "Going to CS_ZOMBIE for %s\n", drop->name );
679  drop->state = CS_ZOMBIE; // become free in a few seconds
680  }
681 
682  // nuke user info
683  SV_SetUserinfo( drop - svs.clients, "" );
684 
685  // if this was the last client on the server, send a heartbeat
686  // to the master so it is known the server is empty
687  // send a heartbeat now so the master will get up to date info
688  // if there is already a slot for this ip, reuse it

924 
925 Check to see if the client wants a file, open it if needed and start pumping the client

927 ==================
928 */
929 int SV_WriteDownloadToClient(client_t *cl, msg_t *msg)
930 {

932  int unreferenced = 1;
933  char errorMessage[1024];

935  int numRefPaks;
936 
937  if (!*cl->downloadName)
938  return 0; // Nothing being downloaded
939 
940  if(!cl->download)
941  {
942  qboolean idPack = qfalse;
943  #ifndef STANDALONE

946

950 
951  if(pakptr)
952  {
953  *pakptr = '\0';
954 
955  // Check for pk3 filename extension
956  if(!Q_stricmp(pakptr + 1, "pk3"))
957  {
958  const char *referencedPaks = FS_ReferencedPakNames();
959

961  // paks to prevent downloading of arbitrary files.
962  Cmd_TokenizeStringIgnoreQuotes(referencedPaks);

965  for(curindex = 0; curindex < numRefPaks; curindex++)
966  {
967  if(!FS_FilenameCompare(Cmd_Argv(curindex), pakbuf))
968  {
969  unreferenced = 0;
970

972  // check whether it's legal to download it.

975  idPack = missionPack;
976 #endif

978 
979  break;
980  }
981  }
982  }
983  }
984 
985  cl->download = 0;
986 
987  // We open the file here
988  if ( !(sv_allowDownload->integer & DLF_ENABLE) ||
989  (sv_allowDownload->integer & DLF_NO_UDP) ||
990  idPack || unreferenced ||
991  ( cl->downloadSize = FS_SV_FOpenFileRead( cl->downloadName, &cl->download ) ) < 0 ) {
992  // cannot auto-download file
993  if(unreferenced)
994  {
995  Com_Printf("clientDownload: %d : \"%s\" is not referenced and cannot be downloaded.\n", (int) (cl - svs.clients), cl->downloadName);

997  }
998  else if (idPack) {
999  Com_Printf("clientDownload: %d : \"%s\" cannot download id pk3 files\n", (int) (cl - svs.clients), cl->downloadName);
1000 #ifndef STANDALONE
1001  if(missionPack)
1002  {
1003  Com_sprintf(errorMessage, sizeof(errorMessage), "Cannot autodownload Team Arena file \"%s\"\n"
1004  "The Team Arena mission pack can be found in your local game store.", cl->downloadName);
1005  }
1006  else
1007 #endif
1008  {
1009  Com_sprintf(errorMessage, sizeof(errorMessage), "Cannot autodownload id pk3 file \"%s\"", cl->downloadName);
1010  }
1011  }
1012  else if ( !(sv_allowDownload->integer & DLF_ENABLE) ||
1013  (sv_allowDownload->integer & DLF_NO_UDP) ) {
1014 
1015  Com_Printf("clientDownload: %d : \"%s\" download disabled\n", (int) (cl - svs.clients), cl->downloadName);
1016  if (sv_pure->integer) {
1017  Com_sprintf(errorMessage, sizeof(errorMessage), "Could not download \"%s\" because autodownloading is disabled on the server.\n\n"
1018  "You will need to get this file elsewhere before you "

1020  } else {

1026  } else {
1027  // NOTE TTimo this is NOT supposed to happen unless bug in our filesystem scheme?
1028  // if the pk3 is referenced, it must have been found somewhere in the filesystem
1029  Com_Printf("clientDownload: %d : \"%s\" file not found on server\n", (int) (cl - svs.clients), cl->downloadName);
1030  Com_sprintf(errorMessage, sizeof(errorMessage), "File \"%s\" not found on server for autodownloading.\n", cl->downloadName);
1031  }
1032  MSG_WriteByte( msg, svc_download );
1033  MSG_WriteShort( msg, 0 ); // client is expecting block zero
1034  MSG_WriteLong( msg, -1 ); // illegal file size
1035  MSG_WriteString( msg, errorMessage );
1036 
1037  *cl->downloadName = 0;
1038 
1039  if(cl->download)
1040  FS_FCloseFile(cl->download);
1041 
1042  return 1;
1043  }
1044 
1045  Com_Printf( "clientDownload: %d : beginning \"%s\"\n", (int) (cl - svs.clients), cl->downloadName );
1046 
1047  // Init
1048  cl->downloadCurrentBlock = cl->downloadClientBlock = cl->downloadXmitBlock = 0;
1049  cl->downloadCount = 0;
1050  cl->downloadEOF = qfalse;
1051  }
1052 
1053  // Perform any reads that we need to
1054  while (cl->downloadCurrentBlock - cl->downloadClientBlock < MAX_DOWNLOAD_WINDOW &&
1055  cl->downloadSize != cl->downloadCount) {
1056 
1057  curindex = (cl->downloadCurrentBlock % MAX_DOWNLOAD_WINDOW);
1058 
1059  if (!cl->downloadBlocks[curindex])
1060  cl->downloadBlocks[curindex] = Z_Malloc(MAX_DOWNLOAD_BLKSIZE);
1061 
1062  cl->downloadBlockSize[curindex] = FS_Read( cl->downloadBlocks[curindex], MAX_DOWNLOAD_BLKSIZE, cl->download );
1063 
1064  if (cl->downloadBlockSize[curindex] < 0) {
1065  // EOF right now
1066  cl->downloadCount = cl->downloadSize;
1067  break;
1068  }
1069 
1070  cl->downloadCount += cl->downloadBlockSize[curindex];
1071 
1072  // Load in next block
1073  cl->downloadCurrentBlock++;
1074  }
1075 
1076  // Check to see if we have eof condition and add the EOF block
1077  if (cl->downloadCount == cl->downloadSize &&
1078  !cl->downloadEOF &&
1079  cl->downloadCurrentBlock - cl->downloadClientBlock < MAX_DOWNLOAD_WINDOW) {
1080 
1081  cl->downloadBlockSize[cl->downloadCurrentBlock % MAX_DOWNLOAD_WINDOW] = 0;
1082  cl->downloadCurrentBlock++;
1083 
1084  cl->downloadEOF = qtrue; // We have added the EOF block
1085  }
1086 
1087  if (cl->downloadClientBlock == cl->downloadCurrentBlock)
1088  return 0; // Nothing to transmit
1089 
1090  // Write out the next section of the file, if we have already reached our window,
1091  // automatically start retransmitting
1092  if (cl->downloadXmitBlock == cl->downloadCurrentBlock)
1093  {
1094  // We have transmitted the complete window, should we start resending?
1095  if (svs.time - cl->downloadSendTime > 1000)
1096  cl->downloadXmitBlock = cl->downloadClientBlock;
1097  else
1098  return 0;
1099  }
1100 
1101  // Send current block
1102  curindex = (cl->downloadXmitBlock % MAX_DOWNLOAD_WINDOW);
1103 
1104  MSG_WriteByte( msg, svc_download );
1105  MSG_WriteShort( msg, cl->downloadXmitBlock );
1106 
1107  // block zero is special, contains file size
1108  if ( cl->downloadXmitBlock == 0 )
1109  MSG_WriteLong( msg, cl->downloadSize );
1110 
1111  MSG_WriteShort( msg, cl->downloadBlockSize[curindex] );
1112 
1113  // Write the block
1114  if(cl->downloadBlockSize[curindex])
1115  MSG_WriteData(msg, cl->downloadBlocks[curindex], cl->downloadBlockSize[curindex]);
1116 
1117  Com_DPrintf( "clientDownload: %d : writing block %d\n", (int) (cl - svs.clients), cl->downloadXmitBlock );
1118 
1119  // Move on to the next block
1120  // It will get sent with next snap shot. The rate will keep us in line.
1121  cl->downloadXmitBlock++;
1122  cl->downloadSendTime = svs.time;
1123 
1124  return 1;
1125 }

1212 SV_VerifyPaks_f

1215

1217 2. there are no any additional checksums that we do not have
1218 
1219 This routine would be a bit simpler with a goto but i abstained
1220 
1221 =================
1222 */
1223 static void SV_VerifyPaks_f( client_t *cl ) {
1224  int nChkSum1, nChkSum2, nClientPaks, nServerPaks, i, j, nCurArg;
1225  int nClientChkSum[1024];
1226  int nServerChkSum[1024];
1227  const char *pPaks, *pArg;
1228  qboolean bGood = qtrue;
1229 
1230  // if we are pure, we "expect" the client to load certain things from
1231  // certain pk3 files, namely we want the client to have loaded the
1232  // ui and cgame that we think should be loaded based on the pure setting
1233  //
1234  if ( sv_pure->integer != 0 ) {
1235 
1236  nChkSum1 = nChkSum2 = 0;
1237  // we run the game, so determine which cgame and ui the client "should" be running
1238  bGood = (FS_FileIsInPAK("vm/cgame.qvm", &nChkSum1) == 1);
1239  if (bGood)
1240  bGood = (FS_FileIsInPAK("vm/ui.qvm", &nChkSum2) == 1);
1241 
1242  nClientPaks = Cmd_Argc();
1243 
1244  // start at arg 2 ( skip serverId cl_paks )
1245  nCurArg = 1;
1246 
1247  pArg = Cmd_Argv(nCurArg++);
1248  if(!pArg) {
1249  bGood = qfalse;
1250  }
1251  else
1252  {
1253  // https://zerowing.idsoftware.com/bugzilla/show_bug.cgi?id=475
1254  // we may get incoming cp sequences from a previous checksumFeed, which we need to ignore
1255  // since serverId is a frame count, it always goes up
1256  if (atoi(pArg) < sv.checksumFeedServerId)
1257  {
1258  Com_DPrintf("ignoring outdated cp command from client %s\n", cl->name);
1259  return;
1260  }
1261  }
1262 
1263  // we basically use this while loop to avoid using 'goto' :)
1264  while (bGood) {
1265 
1266  // must be at least 6: "cl_paks cgame ui @ firstref ... numChecksums"
1267  // numChecksums is encoded
1268  if (nClientPaks < 6) {
1269  bGood = qfalse;
1270  break;
1271  }
1272  // verify first to be the cgame checksum
1273  pArg = Cmd_Argv(nCurArg++);
1274  if (!pArg || *pArg == '@' || atoi(pArg) != nChkSum1 ) {
1275  bGood = qfalse;
1276  break;
1277  }
1278  // verify the second to be the ui checksum
1279  pArg = Cmd_Argv(nCurArg++);
1280  if (!pArg || *pArg == '@' || atoi(pArg) != nChkSum2 ) {
1281  bGood = qfalse;
1282  break;
1283  }
1284  // should be sitting at the delimeter now
1285  pArg = Cmd_Argv(nCurArg++);
1286  if (*pArg != '@') {
1287  bGood = qfalse;
1288  break;
1289  }
1290  // store checksums since tokenization is not re-entrant
1291  for (i = 0; nCurArg < nClientPaks; i++) {
1292  nClientChkSum[i] = atoi(Cmd_Argv(nCurArg++));
1293  }
1294 
1295  // store number to compare against (minus one cause the last is the number of checksums)
1296  nClientPaks = i - 1;
1297 
1298  // make sure none of the client check sums are the same
1299  // so the client can't send 5 the same checksums
1300  for (i = 0; i < nClientPaks; i++) {
1301  for (j = 0; j < nClientPaks; j++) {
1302  if (i == j)
1303  continue;
1304  if (nClientChkSum[i] == nClientChkSum[j]) {
1305  bGood = qfalse;
1306  break;
1307  }
1308  }
1309  if (bGood == qfalse)
1310  break;
1311  }
1312  if (bGood == qfalse)
1313  break;
1314 
1315  // get the pure checksums of the pk3 files loaded by the server
1316  pPaks = FS_LoadedPakPureChecksums();
1317  Cmd_TokenizeString( pPaks );
1318  nServerPaks = Cmd_Argc();
1319  if (nServerPaks > 1024)
1320  nServerPaks = 1024;
1321 
1322  for (i = 0; i < nServerPaks; i++) {
1323  nServerChkSum[i] = atoi(Cmd_Argv(i));
1324  }
1325 
1326  // check if the client has provided any pure checksums of pk3 files not loaded by the server
1327  for (i = 0; i < nClientPaks; i++) {
1328  for (j = 0; j < nServerPaks; j++) {
1329  if (nClientChkSum[i] == nServerChkSum[j]) {
1330  break;
1331  }
1332  }
1333  if (j >= nServerPaks) {
1334  bGood = qfalse;
1335  break;
1336  }
1337  }
1338  if ( bGood == qfalse ) {
1339  break;
1340  }
1341 
1342  // check if the number of checksums was correct
1343  nChkSum1 = sv.checksumFeed;
1344  for (i = 0; i < nClientPaks; i++) {
1345  nChkSum1 ^= nClientChkSum[i];
1346  }
1347  nChkSum1 ^= nClientPaks;
1348  if (nChkSum1 != nClientChkSum[nClientPaks]) {
1349  bGood = qfalse;
1350  break;
1351  }
