368

372 
373  // restore current value
374  trap_Cvar_SetValue( cvarptr->name, cvarptr->value );
375  }

553 /*
554 =================
555 Controls_Update
556 =================
557 */
558 static void Controls_Update( void ) {
559  int i;
560  int j;
561  int y;
562  menucommon_s **controls;
563  menucommon_s *control;
564 
565  // disable all controls in all groups
566  for( i = 0; i < C_MAX; i++ ) {
567  controls = g_controls[i];
568  for( j = 0; (control = controls[j]) ; j++ ) {
569  control->flags |= (QMF_HIDDEN|QMF_INACTIVE);
570  }
571  }
572 
573  controls = g_controls[s_controls.section];
574 
575  // enable controls in active group (and count number of items for vertical centering)
576  for( j = 0; (control = controls[j]) ; j++ ) {
577  control->flags &= ~(QMF_GRAYED|QMF_HIDDEN|QMF_INACTIVE);
578  }
579 
580  // position controls
581  y = ( SCREEN_HEIGHT - j * SMALLCHAR_HEIGHT ) / 2;
582  for( j = 0; (control = controls[j]) ; j++, y += SMALLCHAR_HEIGHT ) {
583  control->x = 320;
584  control->y = y;
585  control->left = 320 - 19*SMALLCHAR_WIDTH;
586  control->right = 320 + 21*SMALLCHAR_WIDTH;
587  control->top = y;
588  control->bottom = y + SMALLCHAR_HEIGHT;
589  }
590 
591  if( s_controls.waitingforkey ) {
592  // disable everybody
593  for( i = 0; i < s_controls.menu.nitems; i++ ) {
594  ((menucommon_s*)(s_controls.menu.items[i]))->flags |= QMF_GRAYED;
595  }
596 
597  // enable action item
598  ((menucommon_s*)(s_controls.menu.items[s_controls.menu.cursor]))->flags &= ~QMF_GRAYED;
599 
600  // don't gray out player's name
601  s_controls.name.generic.flags &= ~QMF_GRAYED;
602 
603  return;
604  }
605 
606  // enable everybody
607  for( i = 0; i < s_controls.menu.nitems; i++ ) {
608  ((menucommon_s*)(s_controls.menu.items[i]))->flags &= ~QMF_GRAYED;
609  }
610 
611  // makes sure flags are right on the group selection controls
612  s_controls.looking.generic.flags &= ~(QMF_GRAYED|QMF_HIGHLIGHT|QMF_HIGHLIGHT_IF_FOCUS);
613  s_controls.movement.generic.flags &= ~(QMF_GRAYED|QMF_HIGHLIGHT|QMF_HIGHLIGHT_IF_FOCUS);
614  s_controls.weapons.generic.flags &= ~(QMF_GRAYED|QMF_HIGHLIGHT|QMF_HIGHLIGHT_IF_FOCUS);
615  s_controls.misc.generic.flags &= ~(QMF_GRAYED|QMF_HIGHLIGHT|QMF_HIGHLIGHT_IF_FOCUS);
616 
617  s_controls.looking.generic.flags |= QMF_PULSEIFFOCUS;
618  s_controls.movement.generic.flags |= QMF_PULSEIFFOCUS;
619  s_controls.weapons.generic.flags |= QMF_PULSEIFFOCUS;
620  s_controls.misc.generic.flags |= QMF_PULSEIFFOCUS;
621 
622  // set buttons
623  switch( s_controls.section ) {
624  case C_MOVEMENT:

901 static sfxHandle_t Controls_MenuKey( int key )
902 {
903  int id;
904  int i;
905  qboolean found;
906  bind_t* bindptr;
907  found = qfalse;
908 
909  if (!s_controls.waitingforkey)
910  {
911  switch (key)
912  {
913  case K_BACKSPACE:
914  case K_DEL:
915  case K_KP_DEL:
916  key = -1;
917  break;
918 
919  case K_MOUSE2:
920  case K_ESCAPE:
921  if (s_controls.changesmade)
922  Controls_SetConfig();
923  goto ignorekey;
924 
925  default:
926  goto ignorekey;
927  }
928  }
929  else
930  {
931  if (key & K_CHAR_FLAG)
932  goto ignorekey;
933 
934  switch (key)
935  {
936  case K_ESCAPE:
937  s_controls.waitingforkey = qfalse;
938  Controls_Update();
939  return (menu_out_sound);
940 
941  case '`':
942  goto ignorekey;
943  }
944  }
945 
946  s_controls.changesmade = qtrue;
947 
948  if (key != -1)
949  {
950  // remove from any other bind
951  bindptr = g_bindings;
952  for (i=0; ;i++,bindptr++)
953  {
954  if (!bindptr->label)
955  break;
956 
957  if (bindptr->bind2 == key)
958  bindptr->bind2 = -1;
959 
960  if (bindptr->bind1 == key)
961  {
962  bindptr->bind1 = bindptr->bind2;
963  bindptr->bind2 = -1;
964  }
965  }
966  }
967 
968  // assign key to local store
969  id = ((menucommon_s*)(s_controls.menu.items[s_controls.menu.cursor]))->id;
970  bindptr = g_bindings;
971  for (i=0; ;i++,bindptr++)
972  {
973  if (!bindptr->label)
974  break;
975 
976  if (bindptr->id == id)
977  {
978  found = qtrue;
979  if (key == -1)
980  {
981  if( bindptr->bind1 != -1 ) {
982  trap_Key_SetBinding( bindptr->bind1, "" );
983  bindptr->bind1 = -1;
984  }
985  if( bindptr->bind2 != -1 ) {
986  trap_Key_SetBinding( bindptr->bind2, "" );
987  bindptr->bind2 = -1;
988  }
989  }
990  else if (bindptr->bind1 == -1) {
991  bindptr->bind1 = key;
992  }
993  else if (bindptr->bind1 != key && bindptr->bind2 == -1) {
994  bindptr->bind2 = key;
995  }
996  else
997  {
998  trap_Key_SetBinding( bindptr->bind1, "" );
999  trap_Key_SetBinding( bindptr->bind2, "" );
1000  bindptr->bind1 = key;
1001  bindptr->bind2 = -1;
1002  }
1003  break;
1004  }
1005  }
1006 
1007  s_controls.waitingforkey = qfalse;
1008 
1009  if (found)
1010  {
1011  Controls_Update();
1012  return (menu_out_sound);
1013  }
1014 
1015 ignorekey:
1016  return Menu_DefaultKey( &s_controls.menu, key );
1017 }
