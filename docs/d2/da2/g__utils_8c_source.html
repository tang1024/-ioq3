38 void AddRemap(const char* oldShader, const char* newShader, float timeOffset)
39 {
40  int i;
41 
42  for (i = 0; i < remapCount; i++) {
43  if (Q_stricmp(oldShader, remappedShaders[i].oldShader) == 0) {
44  // found it, just update this one
45  strcpy(remappedShaders[i].newShader, newShader);
46  remappedShaders[i].timeOffset = timeOffset;
47  return;
48  }
49  }
50  if (remapCount < MAX_SHADER_REMAPS) {
51  strcpy(remappedShaders[remapCount].newShader, newShader);
52  strcpy(remappedShaders[remapCount].oldShader, oldShader);
53  remappedShaders[remapCount].timeOffset = timeOffset;
54  remapCount++;
55  }
56 }

87 {
88  int i;
89  char s[MAX_STRING_CHARS];

91  if (!name || !name[0]) {
92  return 0;
93  }
94 
95  for (i = 1; i < max; i++) {
96  trap_GetConfigstring(start + i, s, sizeof(s));
97  if (!s[0]) {
98  break;
99  }
100  if (!strcmp(s, name)) {
101  return i;
102  }
103  }
104 
105  if (!create) {
106  return 0;
107  }
108 
109  if (i == max) {
110  G_Error("G_FindConfigstringIndex: overflow");
111  }
112 
113  trap_SetConfigstring(start + i, name);
114 
115  return i;
116 }

156 
157 Searches all active entities for the next one that holds
158 the matching string at fieldofs (use the FOFS() macro) in the structure.
159 
160 Searches beginning at the entity after from, or the beginning if NULL
161 NULL will be returned if the end of the list is reached.
162 
163 =============
164 */
165 gentity_t* G_Find(gentity_t* from, int fieldofs, const char* match)
166 {
167  char* s;
168 
169  if (!from)
170  from = g_entities;
171  else
172  from++;
173 
174  for (; from < &g_entities[level.num_entities]; from++) {
175  if (!from->inuse)
176  continue;
177  s = *(char**)((byte*)from + fieldofs);
178  if (!s)
179  continue;
180  if (!Q_stricmp(s, match))
181  return from;
182  }
183 
184  return NULL;
185 }

219  return NULL;
220  }
221

230 "activator" should be set to the entity that initiated the firing.

351 float vectoyaw(const vec3_t vec)
352 {
353  float yaw;
354 
355  if (vec[YAW] == 0 && vec[PITCH] == 0) {
356  yaw = 0;
357  } else {
358  if (vec[PITCH]) {
359  yaw = (atan2(vec[YAW], vec[PITCH]) * 180 / M_PI);
360  } else if (vec[YAW] > 0) {
361  yaw = 90;
362  } else {
363  yaw = 270;

366  yaw += 360;
367  }
368  }

382 /*
383 =================
384 G_Spawn
385 
386 Either finds a free entity, or allocates a new one.
387 
388  The slots from 0 to MAX_CLIENTS-1 are always reserved for clients, and will
389 never be used by anything else.
390 
391 Try to avoid reusing an entity that was recently freed, because it
392 can cause the client to think the entity morphed into something else
393 instead of being removed and recreated, which can cause interpolated
394 angles and bad trails.
395 =================
396 */
397 gentity_t* G_Spawn(void)
398 {
399  int i, force;
400  gentity_t* e;
401 
402  e = NULL; // shut up warning
403  for (force = 0; force < 2; force++) {
404  // if we go through all entities and can't find one to free,
405  // override the normal minimum times before use
406  e = &g_entities[MAX_CLIENTS];
407  for (i = MAX_CLIENTS; i < level.num_entities; i++, e++) {
408  if (e->inuse) {
409  continue;
410  }
411 
412  // the first couple seconds of server time can involve a lot of
413  // freeing and allocating, so relax the replacement policy
414  if (!force && e->freetime > level.startTime + 2000 && level.time - e->freetime < 1000) {
415  continue;
416  }
417 
418  // reuse this slot
419  G_InitGentity(e);
420  return e;
421  }
422  if (level.num_entities < ENTITYNUM_MAX_NORMAL) {
423  break;
424  }
425  }
426  if (level.num_entities == ENTITYNUM_MAX_NORMAL) {
427  for (i = 0; i < MAX_GENTITIES; i++) {
428  G_Printf("%4i: %s\n", i, g_entities[i].classname);
429  }
430  G_Error("G_Spawn: no free entities");
431  }
432 
433  // open up a new slot
434  level.num_entities++;
435 
436  // let the server system know that there are more entities
437  trap_LocateGameData(level.gentities, level.num_entities, sizeof(gentity_t),
438  &level.clients[0].ps, sizeof(level.clients[0]));
439 
440  G_InitGentity(e);
441  return e;
442 }
