107 static void Preferences_Event( void* ptr, int notification ) {
108  if( notification != QM_ACTIVATED ) {
109  return;
110  }
111 
112  switch( ((menucommon_s*)ptr)->id ) {
113  case ID_CROSSHAIR:
114  trap_Cvar_SetValue( "cg_drawCrosshair", s_preferences.crosshair.curvalue );
115  break;
116 
117  case ID_SIMPLEITEMS:
118  trap_Cvar_SetValue( "cg_simpleItems", s_preferences.simpleitems.curvalue );
119  break;
120 
121  case ID_HIGHQUALITYSKY:
122  trap_Cvar_SetValue( "r_fastsky", !s_preferences.highqualitysky.curvalue );
123  break;
124 
125  case ID_EJECTINGBRASS:
126  if ( s_preferences.brass.curvalue )
127  trap_Cvar_Reset( "cg_brassTime" );
128  else
129  trap_Cvar_SetValue( "cg_brassTime", 0 );
130  break;
131 
132  case ID_WALLMARKS:
133  trap_Cvar_SetValue( "cg_marks", s_preferences.wallmarks.curvalue );
134  break;
135 
136  case ID_DYNAMICLIGHTS:
137  trap_Cvar_SetValue( "r_dynamiclight", s_preferences.dynamiclights.curvalue );
138  break;
139 
140  case ID_IDENTIFYTARGET:
141  trap_Cvar_SetValue( "cg_drawCrosshairNames", s_preferences.identifytarget.curvalue );
142  break;
143 
144  case ID_SYNCEVERYFRAME:
145  trap_Cvar_SetValue( "r_finish", s_preferences.synceveryframe.curvalue );
146  break;
147 
148  case ID_FORCEMODEL:
149  trap_Cvar_SetValue( "cg_forcemodel", s_preferences.forcemodel.curvalue );
150  break;
151 
152  case ID_DRAWTEAMOVERLAY:
153  trap_Cvar_SetValue( "cg_drawTeamOverlay", s_preferences.drawteamoverlay.curvalue );
154  break;
155 
156  case ID_ALLOWDOWNLOAD:
157  trap_Cvar_SetValue( "cl_allowDownload", s_preferences.allowdownload.curvalue );
158  trap_Cvar_SetValue( "sv_allowDownload", s_preferences.allowdownload.curvalue );
159  break;
160 
161  case ID_BACK:
162  UI_PopMenu();
163  break;
164  }
165 }

173 static void Crosshair_Draw( void *self ) {
174  menulist_s *s;
175  float *color;
176  int x, y;
177  int style;
178  qboolean focus;
179 
180  s = (menulist_s *)self;
181  x = s->generic.x;
182  y = s->generic.y;
183 
184  style = UI_SMALLFONT;
185  focus = (s->generic.parent->cursor == s->generic.menuPosition);
186 
187  if ( s->generic.flags & QMF_GRAYED )
188  color = text_color_disabled;
189  else if ( focus )
190  {
191  color = text_color_highlight;
192  style |= UI_PULSE;
193  }
194  else if ( s->generic.flags & QMF_BLINK )
195  {
196  color = text_color_highlight;
197  style |= UI_BLINK;
198  }
199  else
200  color = text_color_normal;
201 
202  if ( focus )
203  {
204  // draw cursor
205  UI_FillRect( s->generic.left, s->generic.top, s->generic.right-s->generic.left+1, s->generic.bottom-s->generic.top+1, listbar_color );
206  UI_DrawChar( x, y, 13, UI_CENTER|UI_BLINK|UI_SMALLFONT, color);
207  }
208 
209  UI_DrawString( x - SMALLCHAR_WIDTH, y, s->generic.name, style|UI_RIGHT, color );
210  if( !s->curvalue ) {
211  return;
212  }
213  UI_DrawHandlePic( x + SMALLCHAR_WIDTH, y - 4, 24, 24, s_preferences.crosshairShader[s->curvalue] );
214 }
