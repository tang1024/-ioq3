60 void Log_Open(char *filename)
61 {
62  char *ospath;
63  if (!LibVarValue("log", "0")) return;
64  if (!filename || !strlen(filename))
65  {
66  botimport.Print(PRT_MESSAGE, "openlog <filename>\n");
67  return;
68  } //end if
69  if (logfile.fp)
70  {
71  botimport.Print(PRT_ERROR, "log file %s is already opened\n", logfile.filename);
72  return;
73  } //end if
74  ospath = FS_BuildOSPath(Cvar_VariableString("fs_homepath"), Cvar_VariableString("fs_game"), filename);
75  logfile.fp = fopen(ospath, "wb");
76  if (!logfile.fp)
77  {
78  botimport.Print(PRT_ERROR, "can't open the log file %s\n", filename);
79  return;
80  } //end if
81  Q_strncpyz(logfile.filename, filename, MAX_LOGFILENAMESIZE);
82  botimport.Print(PRT_MESSAGE, "Opened log %s\n", logfile.filename);
83 } //end of the function Log_Create
