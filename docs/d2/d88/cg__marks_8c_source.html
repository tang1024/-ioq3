68 void CG_FreeMarkPoly(markPoly_t* le)
69 {
70  if (!le->prevMark || !le->nextMark) {
71  CG_Error("CG_FreeLocalEntity: not active");
72  }
73 
74  // remove from the doubly linked active list
75  le->prevMark->nextMark = le->nextMark;
76  le->nextMark->prevMark = le->prevMark;
77 
78  // the free list is only singly linked
79  le->nextMark = cg_freeMarkPolys;
80  cg_freeMarkPolys = le;
81 }

83 /*
84 ===================
85 CG_AllocMark
86 
87 Will allways succeed, even if it requires freeing an old active mark
88 ===================
89 */
90 markPoly_t* CG_AllocMark(void)
91 {
92  markPoly_t* le;
93  int time;
94 
95  if (!cg_freeMarkPolys) {
96  // no free entities, so free the one at the end of the chain
97  // remove the oldest active entity
98  time = cg_activeMarkPolys.prevMark->time;
99  while (cg_activeMarkPolys.prevMark && time == cg_activeMarkPolys.prevMark->time) {
100  CG_FreeMarkPoly(cg_activeMarkPolys.prevMark);
101  }
102  }
103 
104  le = cg_freeMarkPolys;
105  cg_freeMarkPolys = cg_freeMarkPolys->nextMark;
106 
107  memset(le, 0, sizeof(*le));
108 
109  // link into the active list
110  le->nextMark = cg_activeMarkPolys.nextMark;
111  le->prevMark = &cg_activeMarkPolys;
112  cg_activeMarkPolys.nextMark->prevMark = le;
113  cg_activeMarkPolys.nextMark = le;
114  return le;
115 }

120 =================
121 CG_ImpactMark
122 
123 origin should be a point within a unit of the plane
124 dir should be the plane normal

132 
133 void CG_ImpactMark(qhandle_t markShader, const vec3_t origin, const vec3_t dir,
134  float orientation, float red, float green, float blue, float alpha,
135  qboolean alphaFade, float radius, qboolean temporary)
136 {
137  vec3_t axis[3];
138  float texCoordScale;
139  vec3_t originalPoints[4];
140  byte colors[4];
141  int i, j;
142  int numFragments;
143  markFragment_t markFragments[MAX_MARK_FRAGMENTS], * mf;
144  vec3_t markPoints[MAX_MARK_POINTS];
145  vec3_t projection;
146 
147  if (!cg_addMarks.integer) {
148  return;
149  }
150 
151  if (radius <= 0) {
152  CG_Error("CG_ImpactMark called with <= 0 radius");
153  }
154 
155  //if ( markTotal >= MAX_MARK_POLYS ) {
156  // return;
157  //}
158 
159  // create the texture axis
160  VectorNormalize2(dir, axis[0]);
161  PerpendicularVector(axis[1], axis[0]);
162  RotatePointAroundVector(axis[2], axis[0], axis[1], orientation);
163  CrossProduct(axis[0], axis[2], axis[1]);
164 
165  texCoordScale = 0.5 * 1.0 / radius;
166 
167  // create the full polygon
168  for (i = 0; i < 3; i++) {
169  originalPoints[0][i] = origin[i] - radius * axis[1][i] - radius * axis[2][i];
170  originalPoints[1][i] = origin[i] + radius * axis[1][i] - radius * axis[2][i];
171  originalPoints[2][i] = origin[i] + radius * axis[1][i] + radius * axis[2][i];
172  originalPoints[3][i] = origin[i] - radius * axis[1][i] + radius * axis[2][i];
173  }
174 
175  // get the fragments
176  VectorScale(dir, -20, projection);
177  numFragments = trap_CM_MarkFragments(4, (void*)originalPoints,
178  projection, MAX_MARK_POINTS, markPoints[0],
179  MAX_MARK_FRAGMENTS, markFragments);
180 
181  colors[0] = red * 255;
182  colors[1] = green * 255;
183  colors[2] = blue * 255;
184  colors[3] = alpha * 255;
185 
186  for (i = 0, mf = markFragments; i < numFragments; i++, mf++) {
187  polyVert_t* v;
188  polyVert_t verts[MAX_VERTS_ON_POLY];
189  markPoly_t* mark;
190 
191  // we have an upper limit on the complexity of polygons
192  // that we store persistantly
193  if (mf->numPoints > MAX_VERTS_ON_POLY) {
194  mf->numPoints = MAX_VERTS_ON_POLY;
195  }
196  for (j = 0, v = verts; j < mf->numPoints; j++, v++) {
197  vec3_t delta;
198 
199  VectorCopy(markPoints[mf->firstPoint + j], v->xyz);
200 
201  VectorSubtract(v->xyz, origin, delta);
202  v->st[0] = 0.5 + DotProduct(delta, axis[1]) * texCoordScale;
203  v->st[1] = 0.5 + DotProduct(delta, axis[2]) * texCoordScale;
204  *(int*)v->modulate = *(int*)colors;
205  }
206 
207  // if it is a temporary (shadow) mark, add it immediately and forget about it
208  if (temporary) {
209  trap_R_AddPolyToScene(markShader, mf->numPoints, verts);
210  continue;
211  }
212 
213  // otherwise save it persistantly
214  mark = CG_AllocMark();
215  mark->time = cg.time;
216  mark->alphaFade = alphaFade;
217  mark->markShader = markShader;
218  mark->poly.numVerts = mf->numPoints;
219  mark->color[0] = red;
220  mark->color[1] = green;
221  mark->color[2] = blue;
222  mark->color[3] = alpha;
223  memcpy(mark->verts, verts, mf->numPoints * sizeof(verts[0]));
224  markTotal++;
225  }
226 }

234 #define MARK_TOTAL_TIME 10000
235 #define MARK_FADE_TIME 1000
236 
237 void CG_AddMarks(void)
238 {
239  int j;
240  markPoly_t* mp, * next;
241  int t;
242  int fade;
243 
244  if (!cg_addMarks.integer) {
245  return;
246  }
247 
248  mp = cg_activeMarkPolys.nextMark;
249  for (; mp != &cg_activeMarkPolys; mp = next) {
250  // grab next now, so if the local entity is freed we
251  // still have it
252  next = mp->nextMark;
253 
254  // see if it is time to completely remove it
255  if (cg.time > mp->time + MARK_TOTAL_TIME) {
256  CG_FreeMarkPoly(mp);
257  continue;
258  }
259 
260  // fade out the energy bursts
261  if (mp->markShader == cgs.media.energyMarkShader) {
262 
263  fade = 450 - 450 * ((cg.time - mp->time) / 3000.0);
264  if (fade < 255) {
265  if (fade < 0) {
266  fade = 0;
267  }
268  if (mp->verts[0].modulate[0] != 0) {
269  for (j = 0; j < mp->poly.numVerts; j++) {
270  mp->verts[j].modulate[0] = mp->color[0] * fade;
271  mp->verts[j].modulate[1] = mp->color[1] * fade;
272  mp->verts[j].modulate[2] = mp->color[2] * fade;
273  }
274  }
275  }
276  }
277 
278  // fade all marks out with time
279  t = mp->time + MARK_TOTAL_TIME - cg.time;
280  if (t < MARK_FADE_TIME) {
281  fade = 255 * t / MARK_FADE_TIME;
282  if (mp->alphaFade) {
283  for (j = 0; j < mp->poly.numVerts; j++) {
284  mp->verts[j].modulate[3] = fade;
285  }
286  } else {
287  for (j = 0; j < mp->poly.numVerts; j++) {
288  mp->verts[j].modulate[0] = mp->color[0] * fade;
289  mp->verts[j].modulate[1] = mp->color[1] * fade;
290  mp->verts[j].modulate[2] = mp->color[2] * fade;
291  }
292  }
293  }
294 
295 
296  trap_R_AddPolyToScene(mp->markShader, mp->poly.numVerts, mp->verts);
297  }
298 }
