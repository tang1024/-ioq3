46 // "multiply" 3x4 matrices, these are assumed to be the top 3 rows
47 // of a 4x4 matrix with the last row = (0 0 0 1)
48 static void Matrix34Multiply( const float *a, const float *b, float *out ) {
49  out[ 0] = a[0] * b[0] + a[1] * b[4] + a[ 2] * b[ 8];
50  out[ 1] = a[0] * b[1] + a[1] * b[5] + a[ 2] * b[ 9];
51  out[ 2] = a[0] * b[2] + a[1] * b[6] + a[ 2] * b[10];
52  out[ 3] = a[0] * b[3] + a[1] * b[7] + a[ 2] * b[11] + a[ 3];
53  out[ 4] = a[4] * b[0] + a[5] * b[4] + a[ 6] * b[ 8];
54  out[ 5] = a[4] * b[1] + a[5] * b[5] + a[ 6] * b[ 9];
55  out[ 6] = a[4] * b[2] + a[5] * b[6] + a[ 6] * b[10];
56  out[ 7] = a[4] * b[3] + a[5] * b[7] + a[ 6] * b[11] + a[ 7];
57  out[ 8] = a[8] * b[0] + a[9] * b[4] + a[10] * b[ 8];
58  out[ 9] = a[8] * b[1] + a[9] * b[5] + a[10] * b[ 9];
59  out[10] = a[8] * b[2] + a[9] * b[6] + a[10] * b[10];
60  out[11] = a[8] * b[3] + a[9] * b[7] + a[10] * b[11] + a[11];
61 }
62 static void JointToMatrix( const quat_t rot, const vec3_t scale, const vec3_t trans,
63  float *mat ) {
64  float xx = 2.0f * rot[0] * rot[0];
65  float yy = 2.0f * rot[1] * rot[1];
66  float zz = 2.0f * rot[2] * rot[2];
67  float xy = 2.0f * rot[0] * rot[1];
68  float xz = 2.0f * rot[0] * rot[2];
69  float yz = 2.0f * rot[1] * rot[2];
70  float wx = 2.0f * rot[3] * rot[0];
71  float wy = 2.0f * rot[3] * rot[1];
72  float wz = 2.0f * rot[3] * rot[2];
73 
74  mat[ 0] = scale[0] * (1.0f - (yy + zz));
75  mat[ 1] = scale[0] * (xy - wz);
76  mat[ 2] = scale[0] * (xz + wy);
77  mat[ 3] = trans[0];
78  mat[ 4] = scale[1] * (xy + wz);
79  mat[ 5] = scale[1] * (1.0f - (xx + zz));
80  mat[ 6] = scale[1] * (yz - wx);
81  mat[ 7] = trans[1];
82  mat[ 8] = scale[2] * (xz - wy);
83  mat[ 9] = scale[2] * (yz + wx);
84  mat[10] = scale[2] * (1.0f - (xx + yy));
85  mat[11] = trans[2];
86 }
87 static void Matrix34Invert( const float *inMat, float *outMat ) {
88  vec3_t trans;
89  float invSqrLen, *v;
90 
91  outMat[ 0] = inMat[ 0]; outMat[ 1] = inMat[ 4]; outMat[ 2] = inMat[ 8];
92  outMat[ 4] = inMat[ 1]; outMat[ 5] = inMat[ 5]; outMat[ 6] = inMat[ 9];
93  outMat[ 8] = inMat[ 2]; outMat[ 9] = inMat[ 6]; outMat[10] = inMat[10];
94 
95  v = outMat + 0; invSqrLen = 1.0f / DotProduct(v, v); VectorScale(v, invSqrLen, v);
96  v = outMat + 4; invSqrLen = 1.0f / DotProduct(v, v); VectorScale(v, invSqrLen, v);
97  v = outMat + 8; invSqrLen = 1.0f / DotProduct(v, v); VectorScale(v, invSqrLen, v);
98 
99  trans[0] = inMat[ 3];
100  trans[1] = inMat[ 7];
101  trans[2] = inMat[11];
102 
103  outMat[ 3] = -DotProduct(outMat + 0, trans);
104  outMat[ 7] = -DotProduct(outMat + 4, trans);
105  outMat[11] = -DotProduct(outMat + 8, trans);
106 }
107 static void QuatSlerp(const quat_t from, const quat_t _to, float fraction, quat_t out) {
108  float angle, cosAngle, sinAngle, backlerp, lerp;
109  quat_t to;
110 
111  // cos() of angle
112  cosAngle = from[0] * _to[0] + from[1] * _to[1] + from[2] * _to[2] + from[3] * _to[3];
113 
114  // negative handling is needed for taking shortest path (required for model joints)
115  if ( cosAngle < 0.0f ) {
116  cosAngle = -cosAngle;
117  to[0] = - _to[0];
118  to[1] = - _to[1];
119  to[2] = - _to[2];
120  to[3] = - _to[3];
121  } else {
122  QuatCopy( _to, to );
123  }
124 
125  if ( cosAngle < 0.999999f ) {
126  // spherical lerp (slerp)
127  angle = acosf( cosAngle );
128  sinAngle = sinf( angle );
129  backlerp = sinf( ( 1.0f - fraction ) * angle ) / sinAngle;
130  lerp = sinf( fraction * angle ) / sinAngle;
131  } else {
132  // linear lerp
133  backlerp = 1.0f - fraction;
134  lerp = fraction;
135  }
136 
137  out[0] = from[0] * backlerp + to[0] * lerp;
138  out[1] = from[1] * backlerp + to[1] * lerp;
139  out[2] = from[2] * backlerp + to[2] * lerp;
140  out[3] = from[3] * backlerp + to[3] * lerp;
141 }
142 static vec_t QuatNormalize2( const quat_t v, quat_t out) {
143  float length, ilength;
144 
145  length = v[0]*v[0] + v[1]*v[1] + v[2]*v[2] + v[3]*v[3];
146 
147  if (length) {
148  /* writing it this way allows gcc to recognize that rsqrt can be used */
149  ilength = 1/(float)sqrt (length);
150  /* sqrt(length) = length * (1 / sqrt(length)) */
151  length *= ilength;
152  out[0] = v[0]*ilength;
153  out[1] = v[1]*ilength;
154  out[2] = v[2]*ilength;
155  out[3] = v[3]*ilength;
156  } else {
157  out[0] = out[1] = out[2] = 0;
158  out[3] = -1;
159  }
160 
161  return length;
162 }

1343 static void ComputePoseMats( iqmData_t *data, int frame, int oldframe,
1344  float backlerp, float *poseMats ) {
1345  iqmTransform_t relativeJoints[IQM_MAX_JOINTS];
1346  iqmTransform_t *relativeJoint;
1347  const iqmTransform_t *pose;
1348  const iqmTransform_t *oldpose;
1349  const int *jointParent;
1350  const float *invBindMat;
1351  float *poseMat, lerp;
1352  int i;
1353 
1354  relativeJoint = relativeJoints;
1355 
1356  // copy or lerp animation frame pose
1357  if ( oldframe == frame ) {
1358  pose = &data->poses[frame * data->num_poses];
1359  for ( i = 0; i < data->num_poses; i++, pose++, relativeJoint++ ) {
1360  VectorCopy( pose->translate, relativeJoint->translate );
1361  QuatCopy( pose->rotate, relativeJoint->rotate );
1362  VectorCopy( pose->scale, relativeJoint->scale );
1363  }
1364  } else {
1365  lerp = 1.0f - backlerp;
1366  pose = &data->poses[frame * data->num_poses];
1367  oldpose = &data->poses[oldframe * data->num_poses];
1368  for ( i = 0; i < data->num_poses; i++, oldpose++, pose++, relativeJoint++ ) {
1369  relativeJoint->translate[0] = oldpose->translate[0] * backlerp + pose->translate[0] * lerp;
1370  relativeJoint->translate[1] = oldpose->translate[1] * backlerp + pose->translate[1] * lerp;
1371  relativeJoint->translate[2] = oldpose->translate[2] * backlerp + pose->translate[2] * lerp;
1372 
1373  relativeJoint->scale[0] = oldpose->scale[0] * backlerp + pose->scale[0] * lerp;
1374  relativeJoint->scale[1] = oldpose->scale[1] * backlerp + pose->scale[1] * lerp;
1375  relativeJoint->scale[2] = oldpose->scale[2] * backlerp + pose->scale[2] * lerp;
1376 
1377  QuatSlerp( oldpose->rotate, pose->rotate, lerp, relativeJoint->rotate );
1378  }
1379  }
1380 
1381  // multiply by inverse of bind pose and parent 'pose mat' (bind pose transform matrix)
1382  relativeJoint = relativeJoints;
1383  jointParent = data->jointParents;
1384  invBindMat = data->invBindJoints;
1385  poseMat = poseMats;
1386  for ( i = 0; i < data->num_poses; i++, relativeJoint++, jointParent++, invBindMat += 12, poseMat += 12 ) {
1387  float mat1[12], mat2[12];
1388 
1389  JointToMatrix( relativeJoint->rotate, relativeJoint->scale, relativeJoint->translate, mat1 );
1390 
1391  if ( *jointParent >= 0 ) {
1392  Matrix34Multiply( &data->bindJoints[(*jointParent)*12], mat1, mat2 );
1393  Matrix34Multiply( mat2, invBindMat, mat1 );
1394  Matrix34Multiply( &poseMats[(*jointParent)*12], mat1, poseMat );
1395  } else {
1396  Matrix34Multiply( mat1, invBindMat, poseMat );
1397  }
1398  }
1399 }
