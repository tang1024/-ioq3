295 void RB_ProjectionShadowDeform( void ) {

302  float d;

306 
307  ground[0] = backEnd.or.axis[0][2];
308  ground[1] = backEnd.or.axis[1][2];
309  ground[2] = backEnd.or.axis[2][2];
310 
311  groundDist = backEnd.or.origin[2] - backEnd.currentEntity->e.shadowPlane;
312 
313  VectorCopy( backEnd.currentEntity->modelLightDir, lightDir );
314  d = DotProduct( lightDir, ground );
315  // don't let the shadows get too long or go negative
316  if ( d < 0.5 ) {
317  VectorMA( lightDir, (0.5 - d), ground, lightDir );
318  d = DotProduct( lightDir, ground );
319  }
320  d = 1.0 / d;
321 
322  light[0] = lightDir[0] * d;
323  light[1] = lightDir[1] * d;
324  light[2] = lightDir[2] * d;
325 
326  for ( i = 0; i < tess.numVertexes; i++, xyz += 4 ) {
327  h = DotProduct( xyz, ground ) + groundDist;
328 
329  xyz[0] -= light[0] * h;
330  xyz[1] -= light[1] * h;
331  xyz[2] -= light[2] * h;
332  }
333 }
