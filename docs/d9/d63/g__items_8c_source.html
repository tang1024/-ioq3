25 /*
26 
27  Items are any object that a player can touch to gain some effect.
28 
29  Pickup will return the number of seconds until they should respawn.
30 
31  all items should pop when dropped in lava or slime
32 
33  Respawnable items don't actually go away when picked up, they are
34  just made invisible and untouchable. This allows them to ride
35  movers and respawn appropriately.
36 */

40 #define RESPAWN_HEALTH 35
41 #define RESPAWN_AMMO 40
42 #define RESPAWN_HOLDABLE 60
43 #define RESPAWN_MEGAHEALTH 35//120
44 #define RESPAWN_POWERUP 120

49 int Pickup_Powerup(gentity_t* ent, gentity_t* other)
50 {
51  int quantity;
52  int i;
53  gclient_t* client;
54 
55  if (!other->client->ps.powerups[ent->item->giTag]) {
56  // round timing to seconds to make multiple powerup timers
57  // count in sync
58  other->client->ps.powerups[ent->item->giTag] =
59  level.time - (level.time % 1000);
60  }
61 
62  if (ent->count) {
63  quantity = ent->count;
64  } else {
65  quantity = ent->item->quantity;
66  }
67 
68  other->client->ps.powerups[ent->item->giTag] += quantity * 1000;
69 
70  // give any nearby players a "denied" anti-reward
71  for (i = 0; i < level.maxclients; i++) {
72  vec3_t delta;
73  float len;
74  vec3_t forward;
75  trace_t tr;
76 
77  client = &level.clients[i];
78  if (client == other->client) {
79  continue;
80  }
81  if (client->pers.connected == CON_DISCONNECTED) {
82  continue;
83  }
84  if (client->ps.stats[STAT_HEALTH] <= 0) {
85  continue;
86  }
87 
88  // if same team in team game, no sound
89  // cannot use OnSameTeam as it expects to g_entities, not clients
90  if (g_gametype.integer >= GT_TEAM && other->client->sess.sessionTeam == client->sess.sessionTeam) {
91  continue;
92  }
93 
94  // if too far away, no sound
95  VectorSubtract(ent->s.pos.trBase, client->ps.origin, delta);
96  len = VectorNormalize(delta);
97  if (len > 192) {
98  continue;
99  }
100 
101  // if not facing, no sound
102  AngleVectors(client->ps.viewangles, forward, NULL, NULL);
103  if (DotProduct(delta, forward) < 0.4) {
104  continue;
105  }
106 
107  // if not line of sight, no sound
108  trap_Trace(&tr, client->ps.origin, NULL, NULL, ent->s.pos.trBase, ENTITYNUM_NONE, CONTENTS_SOLID);
109  if (tr.fraction != 1.0) {
110  continue;
111  }
112 
113  // anti-reward
114  client->ps.persistant[PERS_PLAYEREVENTS] ^= PLAYEREVENT_DENIEDREWARD;
115  }
116  return RESPAWN_POWERUP;
117 }

238 int Pickup_Weapon(gentity_t* ent, gentity_t* other)
239 {
240  int quantity;
241 
242  if (ent->count < 0) {
243  quantity = 0; // None for you, sir!
244  } else {
245  if (ent->count) {
246  quantity = ent->count;
247  } else {
248  quantity = ent->item->quantity;
249  }
250 
251  // dropped items and teamplay weapons always have full ammo
252  if (!(ent->flags & FL_DROPPED_ITEM) && g_gametype.integer != GT_TEAM) {
253  // respawning rules
254  // drop the quantity if the already have over the minimum
255  if (other->client->ps.ammo[ent->item->giTag] < quantity) {
256  quantity = quantity - other->client->ps.ammo[ent->item->giTag];
257  } else {
258  quantity = 1; // only add a single shot
259  }
260  }
261  }
262 
263  // add the weapon
264  other->client->ps.stats[STAT_WEAPONS] |= (1 << ent->item->giTag);
265 
266  Add_Ammo(other, ent->item->giTag, quantity);
267 
268  if (ent->item->giTag == WP_GRAPPLING_HOOK)
269  other->client->ps.ammo[ent->item->giTag] = -1; // unlimited ammo
270 
271  // team deathmatch has slow weapon respawns
272  if (g_gametype.integer == GT_TEAM) {
273  return g_weaponTeamRespawn.integer;
274  }
275 
276  return g_weaponRespawn.integer;
277 }

282 int Pickup_Health(gentity_t* ent, gentity_t* other)
283 {
284  int max;
285  int quantity;
286 
287  // small and mega healths will go over the max
288 #ifdef MISSIONPACK
289  if (bg_itemlist[other->client->ps.stats[STAT_PERSISTANT_POWERUP]].giTag == PW_GUARD) {
290  max = other->client->ps.stats[STAT_MAX_HEALTH];
291  } else
292 #endif
293  if (ent->item->quantity != 5 && ent->item->quantity != 100) {
294  max = other->client->ps.stats[STAT_MAX_HEALTH];
295  } else {
296  max = other->client->ps.stats[STAT_MAX_HEALTH] * 2;
297  }
298 
299  if (ent->count) {
300  quantity = ent->count;
301  } else {
302  quantity = ent->item->quantity;
303  }
304 
305  other->health += quantity;
306 
307  if (other->health > max) {
308  other->health = max;
309  }
310  other->client->ps.stats[STAT_HEALTH] = other->health;
311 
312  if (ent->item->quantity == 100) { // mega health respawns slow
313  return RESPAWN_MEGAHEALTH;
314  }
315 
316  return RESPAWN_HEALTH;
317 }

321 int Pickup_Armor(gentity_t* ent, gentity_t* other)
322 {
323 #ifdef MISSIONPACK
324  int upperBound;
325 
326  other->client->ps.stats[STAT_ARMOR] += ent->item->quantity;
327 
328  if (other->client && bg_itemlist[other->client->ps.stats[STAT_PERSISTANT_POWERUP]].giTag == PW_GUARD) {
329  upperBound = other->client->ps.stats[STAT_MAX_HEALTH];
330  } else {
331  upperBound = other->client->ps.stats[STAT_MAX_HEALTH] * 2;
332  }
333 
334  if (other->client->ps.stats[STAT_ARMOR] > upperBound) {
335  other->client->ps.stats[STAT_ARMOR] = upperBound;
336  }
337 #else
338  other->client->ps.stats[STAT_ARMOR] += ent->item->quantity;
339  if (other->client->ps.stats[STAT_ARMOR] > other->client->ps.stats[STAT_MAX_HEALTH] * 2) {
340  other->client->ps.stats[STAT_ARMOR] = other->client->ps.stats[STAT_MAX_HEALTH] * 2;
341  }
342 #endif
343 
344  return RESPAWN_ARMOR;
345 }

349 /*
350 ===============
351 RespawnItem
352 ===============
353 */
354 void RespawnItem(gentity_t* ent)
355 {
356  if (!ent) {
357  return;
358  }
359 
360  // randomly select from teamed entities
361  if (ent->team) {
362  gentity_t* master;
363  int count;
364  int choice;
365 
366  if (!ent->teammaster) {
367  G_Error("RespawnItem: bad teammaster");
368  }
369  master = ent->teammaster;
370 
371  for (count = 0, ent = master; ent; ent = ent->teamchain, count++)
372  ;
373 
374  choice = rand() % count;
375 
376  for (count = 0, ent = master; ent && count < choice; ent = ent->teamchain, count++)
377  ;
378  }
379 
380  if (!ent) {
381  return;
382  }
383 
384  ent->r.contents = CONTENTS_TRIGGER;
385  ent->s.eFlags &= ~EF_NODRAW;
386  ent->r.svFlags &= ~SVF_NOCLIENT;
387  trap_LinkEntity(ent);
388 
389  if (ent->item->giType == IT_POWERUP) {
390  // play powerup spawn sound to all clients
391  gentity_t* te;
392 
393  // if the powerup respawn sound should Not be global
394  if (ent->speed) {
395  te = G_TempEntity(ent->s.pos.trBase, EV_GENERAL_SOUND);
396  } else {
397  te = G_TempEntity(ent->s.pos.trBase, EV_GLOBAL_SOUND);
398  }
399  te->s.eventParm = G_SoundIndex("sound/items/poweruprespawn.wav");
400  te->r.svFlags |= SVF_BROADCAST;
401  }
402 
403  if (ent->item->giType == IT_HOLDABLE && ent->item->giTag == HI_KAMIKAZE) {
404  // play powerup spawn sound to all clients
405  gentity_t* te;
406 
407  // if the powerup respawn sound should Not be global
408  if (ent->speed) {
409  te = G_TempEntity(ent->s.pos.trBase, EV_GENERAL_SOUND);
410  } else {
411  te = G_TempEntity(ent->s.pos.trBase, EV_GLOBAL_SOUND);
412  }
413  te->s.eventParm = G_SoundIndex("sound/items/kamikazerespawn.wav");
414  te->r.svFlags |= SVF_BROADCAST;
415  }
416 
417  // play the normal respawn sound only to nearby clients
418  G_AddEvent(ent, EV_ITEM_RESPAWN, 0);
419 
420  ent->nextthink = 0;
421 }

424 /*
425 ===============
426 Touch_Item
427 ===============
428 */
429 void Touch_Item(gentity_t* ent, gentity_t* other, trace_t* trace)
430 {
431  int respawn;
432  qboolean predict;
433 
434  if (!other->client)
435  return;
436  if (other->health < 1)
437  return; // dead people can't pickup
438 
439  // the same pickup rules are used for client side and server side
440  if (!BG_CanItemBeGrabbed(g_gametype.integer, &ent->s, &other->client->ps)) {
441  return;
442  }
443 
444  G_LogPrintf("Item: %i %s\n", other->s.number, ent->item->classname);
445 
446  predict = other->client->pers.predictItemPickup;
447 
448  // call the item-specific pickup function
449  switch (ent->item->giType) {
450  case IT_WEAPON:
451  respawn = Pickup_Weapon(ent, other);
452  // predict = qfalse;
453  break;
454  case IT_AMMO:
455  respawn = Pickup_Ammo(ent, other);
456  // predict = qfalse;
457  break;
458  case IT_ARMOR:
459  respawn = Pickup_Armor(ent, other);
460  break;
461  case IT_HEALTH:
462  respawn = Pickup_Health(ent, other);
463  break;
464  case IT_POWERUP:
465  respawn = Pickup_Powerup(ent, other);
466  predict = qfalse;
467  break;
468 #ifdef MISSIONPACK
469  case IT_PERSISTANT_POWERUP:
470  respawn = Pickup_PersistantPowerup(ent, other);
471  break;
472 #endif
473  case IT_TEAM:
474  respawn = Pickup_Team(ent, other);
475  break;
476  case IT_HOLDABLE:
477  respawn = Pickup_Holdable(ent, other);
478  break;
479  default:
480  return;
481  }
482 
483  if (!respawn) {
484  return;
485  }
486 
487  // play the normal pickup sound
488  if (predict) {
489  G_AddPredictableEvent(other, EV_ITEM_PICKUP, ent->s.modelindex);
490  } else {
491  G_AddEvent(other, EV_ITEM_PICKUP, ent->s.modelindex);
492  }
493 
494  // powerup pickups are global broadcasts
495  if (ent->item->giType == IT_POWERUP || ent->item->giType == IT_TEAM) {
496  // if we want the global sound to play
497  if (!ent->speed) {
498  gentity_t* te;
499 
500  te = G_TempEntity(ent->s.pos.trBase, EV_GLOBAL_ITEM_PICKUP);
501  te->s.eventParm = ent->s.modelindex;
502  te->r.svFlags |= SVF_BROADCAST;
503  } else {
504  gentity_t* te;
505 
506  te = G_TempEntity(ent->s.pos.trBase, EV_GLOBAL_ITEM_PICKUP);
507  te->s.eventParm = ent->s.modelindex;
508  // only send this temp entity to a single client
509  te->r.svFlags |= SVF_SINGLECLIENT;
510  te->r.singleClient = other->s.number;
511  }
512  }
513 
514  // fire item targets
515  G_UseTargets(ent, other);
516 
517  // wait of -1 will not respawn
518  if (ent->wait == -1) {
519  ent->r.svFlags |= SVF_NOCLIENT;
520  ent->s.eFlags |= EF_NODRAW;
521  ent->r.contents = 0;
522  ent->unlinkAfterEvent = qtrue;
523  return;
524  }
525 
526  // non zero wait overrides respawn time
527  if (ent->wait) {
528  respawn = ent->wait;
529  }
530 
531  // random can be used to vary the respawn time
532  if (ent->random) {
533  respawn += crandom() * ent->random;
534  if (respawn < 1) {
535  respawn = 1;
536  }
537  }
538 
539  // dropped items will not respawn
540  if (ent->flags & FL_DROPPED_ITEM) {
541  ent->freeAfterEvent = qtrue;
542  }
543 
544  // picked up items still stay around, they just don't
545  // draw anything. This allows respawnable items
546  // to be placed on movers.
547  ent->r.svFlags |= SVF_NOCLIENT;
548  ent->s.eFlags |= EF_NODRAW;
549  ent->r.contents = 0;
550 
551  // ZOID
552  // A negative respawn times means to never respawn this item (but don't
553  // delete it). This is used by items that are respawned by third party
554  // events such as ctf flags
555  if (respawn <= 0) {
556  ent->nextthink = 0;
557  ent->think = 0;
558  } else {
559  ent->nextthink = level.time + respawn * 1000;
560  ent->think = RespawnItem;
561  }
562  trap_LinkEntity(ent);
563 }

568 /*
569 ================
570 LaunchItem
571 
572 Spawns an item and tosses it forward
573 ================
574 */
575 gentity_t* LaunchItem(gitem_t* item, vec3_t origin, vec3_t velocity)
576 {
577  gentity_t* dropped;
578 
579  dropped = G_Spawn();
580 
581  dropped->s.eType = ET_ITEM;
582  dropped->s.modelindex = item - bg_itemlist; // store item number in modelindex
583  dropped->s.modelindex2 = 1; // This is non-zero is it's a dropped item
584 
585  dropped->classname = item->classname;
586  dropped->item = item;
587  VectorSet(dropped->r.mins, -ITEM_RADIUS, -ITEM_RADIUS, -ITEM_RADIUS);
588  VectorSet(dropped->r.maxs, ITEM_RADIUS, ITEM_RADIUS, ITEM_RADIUS);
589  dropped->r.contents = CONTENTS_TRIGGER;
590 
591  dropped->touch = Touch_Item;
592 
593  G_SetOrigin(dropped, origin);
594  dropped->s.pos.trType = TR_GRAVITY;
595  dropped->s.pos.trTime = level.time;
596  VectorCopy(velocity, dropped->s.pos.trDelta);
597 
598  dropped->s.eFlags |= EF_BOUNCE_HALF;
599 #ifdef MISSIONPACK
600  if ((g_gametype.integer == GT_CTF || g_gametype.integer == GT_1FCTF) && item->giType == IT_TEAM) { // Special case for CTF flags
601 #else
602  if (g_gametype.integer == GT_CTF && item->giType == IT_TEAM) { // Special case for CTF flags
603 #endif
604  dropped->think = Team_DroppedFlagThink;
605  dropped->nextthink = level.time + 30000;
606  Team_CheckDroppedItem(dropped);
607  } else { // auto-remove after 30 seconds
608  dropped->think = G_FreeEntity;
609  dropped->nextthink = level.time + 30000;
610  }
611 
612  dropped->flags = FL_DROPPED_ITEM;
613 
614  trap_LinkEntity(dropped);
615 
616  return dropped;
617 }

621 Drop_Item

624 ================

626 gentity_t* Drop_Item(gentity_t * ent, gitem_t * item, float angle)
627 {
628  vec3_t velocity;
629  vec3_t angles;
630 
631  VectorCopy(ent->s.apos.trBase, angles);
632  angles[YAW] += angle;
633  angles[PITCH] = 0; // always forward
634 
635  AngleVectors(angles, velocity, NULL, NULL);
636  VectorScale(velocity, 150, velocity);
637  velocity[2] += 200 + crandom() * 50;
638 
639  return LaunchItem(item, ent->s.pos.trBase, velocity);
640 }

657 /*
658 ================
659 FinishSpawningItem
660 
661 Traces down to find where an item should rest, instead of letting them
662 free fall from their spawn points
663 ================
664 */
665 void FinishSpawningItem(gentity_t * ent)
666 {
667  trace_t tr;
668  vec3_t dest;
669 
670  VectorSet(ent->r.mins, -ITEM_RADIUS, -ITEM_RADIUS, -ITEM_RADIUS);
671  VectorSet(ent->r.maxs, ITEM_RADIUS, ITEM_RADIUS, ITEM_RADIUS);
672 
673  ent->s.eType = ET_ITEM;
674  ent->s.modelindex = ent->item - bg_itemlist; // store item number in modelindex
675  ent->s.modelindex2 = 0; // zero indicates this isn't a dropped item
676 
677  ent->r.contents = CONTENTS_TRIGGER;
678  ent->touch = Touch_Item;
679  // using an item causes it to respawn
680  ent->use = Use_Item;
681 
682  if (ent->spawnflags & 1) {
683  // suspended
684  G_SetOrigin(ent, ent->s.origin);
685  } else {
686  // drop to floor
687  VectorSet(dest, ent->s.origin[0], ent->s.origin[1], ent->s.origin[2] - 4096);
688  trap_Trace(&tr, ent->s.origin, ent->r.mins, ent->r.maxs, dest, ent->s.number, MASK_SOLID);
689  if (tr.startsolid) {
690  G_Printf("FinishSpawningItem: %s startsolid at %s\n", ent->classname, vtos(ent->s.origin));
691  G_FreeEntity(ent);
692  return;
693  }
694 
695  // allow to ride movers
696  ent->s.groundEntityNum = tr.entityNum;
697 
698  G_SetOrigin(ent, tr.endpos);
699  }
700 
701  // team slaves and targeted items aren't present at start
702  if ((ent->flags & FL_TEAMSLAVE) || ent->targetname) {
703  ent->s.eFlags |= EF_NODRAW;
704  ent->r.contents = 0;
705  return;
706  }
707 
708  // powerups don't spawn in for a while
709  if (ent->item->giType == IT_POWERUP) {
710  float respawn;
711 
712  respawn = 45 + crandom() * 15;
713  ent->s.eFlags |= EF_NODRAW;

928 {
929  vec3_t velocity;
930  float dot;
931  int hitTime;
932 
933  // reflect the velocity on the trace plane
934  hitTime = level.previousTime + (level.time - level.previousTime) * trace->fraction;
935  BG_EvaluateTrajectoryDelta(&ent->s.pos, hitTime, velocity);
936  dot = DotProduct(velocity, trace->plane.normal);
937  VectorMA(velocity, -2 * dot, trace->plane.normal, ent->s.pos.trDelta);
938 
939  // cut the velocity to keep from bouncing forever
940  VectorScale(ent->s.pos.trDelta, ent->physicsBounce, ent->s.pos.trDelta);
941 
942  // check for stop
943  if (trace->plane.normal[2] > 0 && ent->s.pos.trDelta[2] < 40) {
944  trace->endpos[2] += 1.0; // make sure it is off ground
945  SnapVector(trace->endpos);
946  G_SetOrigin(ent, trace->endpos);
947  ent->s.groundEntityNum = trace->entityNum;
948  return;
949  }
950 
951  VectorAdd(ent->r.currentOrigin, trace->plane.normal, ent->r.currentOrigin);
952  VectorCopy(ent->r.currentOrigin, ent->s.pos.trBase);
953  ent->s.pos.trTime = level.time;
