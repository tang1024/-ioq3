272 PlaneFromPoints
273 
274 Returns false if the triangle is degenrate.
275 The normal will point out of the clock for clockwise ordered points
276 =====================
277 */
278 qboolean PlaneFromPoints(vec4_t plane, const vec3_t a, const vec3_t b, const vec3_t c)
279 {
280  vec3_t d1, d2;
281 
282  VectorSubtract(b, a, d1);
283  VectorSubtract(c, a, d2);
284  CrossProduct(d2, d1, plane);
285  if (VectorNormalize(plane) == 0) {
286  return qfalse;
287  }
288 
289  plane[3] = DotProduct(a, plane);
290  return qtrue;
291 }

294 ===============
295 RotatePointAroundVector
296 
297 This is not implemented very well...
298 ===============

300 void RotatePointAroundVector(vec3_t dst, const vec3_t dir, const vec3_t point,
301  float degrees)
302 {
303  float m[3][3];
304  float im[3][3];
305  float zrot[3][3];
306  float tmpmat[3][3];
307  float rot[3][3];
308  int i;
309  vec3_t vr, vup, vf;
310  float rad;
311 
312  vf[0] = dir[0];
313  vf[1] = dir[1];
314  vf[2] = dir[2];
315 
316  PerpendicularVector(vr, dir);
317  CrossProduct(vr, vf, vup);
318 
319  m[0][0] = vr[0];
320  m[1][0] = vr[1];
321  m[2][0] = vr[2];
322 
323  m[0][1] = vup[0];
324  m[1][1] = vup[1];
325  m[2][1] = vup[2];
326 
327  m[0][2] = vf[0];
328  m[1][2] = vf[1];
329  m[2][2] = vf[2];
330 
331  memcpy(im, m, sizeof(im));
332 
333  im[0][1] = m[1][0];
334  im[0][2] = m[2][0];
335  im[1][0] = m[0][1];
336  im[1][2] = m[2][1];
337  im[2][0] = m[0][2];
338  im[2][1] = m[1][2];
339 
340  memset(zrot, 0, sizeof(zrot));
341  zrot[0][0] = zrot[1][1] = zrot[2][2] = 1.0F;
342 
343  rad = DEG2RAD(degrees);
344  zrot[0][0] = cos(rad);
345  zrot[0][1] = sin(rad);
346  zrot[1][0] = -sin(rad);
347  zrot[1][1] = cos(rad);
348 
349  MatrixMultiply(m, zrot, tmpmat);
350  MatrixMultiply(tmpmat, im, rot);
351 
352  for (i = 0; i < 3; i++) {
353  dst[i] = rot[i][0] * point[0] + rot[i][1] * point[1] + rot[i][2] * point[2];
354  }
355 }

382 void vectoangles(const vec3_t value1, vec3_t angles)
383 {
384  float forward;
385  float yaw, pitch;
386 
387  if (value1[1] == 0 && value1[0] == 0) {
388  yaw = 0;
389  if (value1[2] > 0) {
390  pitch = 90;
391  } else {
392  pitch = 270;
393  }
394  } else {
395  if (value1[0]) {
396  yaw = (atan2(value1[1], value1[0]) * 180 / M_PI);
397  } else if (value1[1] > 0) {
398  yaw = 90;
399  } else {
400  yaw = 270;
401  }
402  if (yaw < 0) {
403  yaw += 360;
404  }
405 
406  forward = sqrt(value1[0] * value1[0] + value1[1] * value1[1]);
407  pitch = (atan2(value1[2], forward) * 180 / M_PI);
408  if (pitch < 0) {
409  pitch += 360;
410  }
411  }
412 
413  angles[PITCH] = -pitch;
414  angles[YAW] = yaw;
415  angles[ROLL] = 0;
416 }

453 void ProjectPointOnPlane(vec3_t dst, const vec3_t p, const vec3_t normal)
454 {
455  float d;
456  vec3_t n;
457  float inv_denom;
458 
459  inv_denom = DotProduct(normal, normal);
460 #ifndef Q3_VM
461  assert(Q_fabs(inv_denom) != 0.0f); // zero vectors get here
462 #endif
463  inv_denom = 1.0f / inv_denom;
464 
465  d = DotProduct(normal, p) * inv_denom;
466 
467  n[0] = normal[0] * inv_denom;
468  n[1] = normal[1] * inv_denom;
469  n[2] = normal[2] * inv_denom;
470 
471  dst[0] = p[0] - d * n[0];
472  dst[1] = p[1] - d * n[1];
473  dst[2] = p[2] - d * n[2];
474 }
