200 */
201 float BotGetTime(bot_match_t* match)
202 {
203  bot_match_t timematch;
204  char timestring[MAX_MESSAGE_SIZE];
205  float t;
206 
207  //if the matched string has a time
208  if (match->subtype & ST_TIME) {
209  //get the time string
210  trap_BotMatchVariable(match, TIME, timestring, MAX_MESSAGE_SIZE);
211  //match it to find out if the time is in seconds or minutes
212  if (trap_BotFindMatch(timestring, &timematch, MTCONTEXT_TIME)) {
213  if (timematch.type == MSG_FOREVER) {
214  t = 99999999.0f;
215  } else if (timematch.type == MSG_FORAWHILE) {
216  t = 10 * 60; // 10 minutes
217  } else if (timematch.type == MSG_FORALONGTIME) {
218  t = 30 * 60; // 30 minutes
219  } else {
220  trap_BotMatchVariable(&timematch, TIME, timestring, MAX_MESSAGE_SIZE);
221  if (timematch.type == MSG_MINUTES) t = atof(timestring) * 60;
222  else if (timematch.type == MSG_SECONDS) t = atof(timestring);
223  else t = 0;
224  }
225  //if there's a valid time
226  if (t > 0) return FloatTime() + t;
227  }
228  }
229  return 0;
230 }
