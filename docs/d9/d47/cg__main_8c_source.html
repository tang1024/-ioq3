498 The server says this item is used on this level

501 static void CG_RegisterItemSounds(int itemNum)
502 {
503  gitem_t* item;
504  char data[MAX_QPATH];
505  char* s, * start;
506  int len;
507 
508  item = &bg_itemlist[itemNum];
509 
510  if (item->pickup_sound) {
511  trap_S_RegisterSound(item->pickup_sound, qfalse);
512  }
513 
514  // parse the space separated precache string for other media
515  s = item->sounds;
516  if (!s || !s[0])
517  return;
518 
519  while (*s) {
520  start = s;
521  while (*s && *s != ' ') {
522  s++;
523  }
524 
525  len = s - start;
526  if (len >= MAX_QPATH || len < 5) {
527  CG_Error("PrecacheItem: %s has bad precache string",
528  item->classname);
529  return;
530  }
531  memcpy(data, start, len);
532  data[len] = 0;
533  if (*s) {
534  s++;
535  }
536 
537  if (!strcmp(data + len - 3, "wav")) {
538  trap_S_RegisterSound(data, qfalse);
539  }
540  }
541 }
