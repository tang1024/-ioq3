223 /*
224 ================
225 return a hash value for the sfx name
226 ================
227 */
228 static long S_HashSFXName(const char *name) {
229  int i;
230  long hash;
231  char letter;
232 
233  hash = 0;
234  i = 0;
235  while (name[i] != '\0') {
236  letter = tolower(name[i]);
237  if (letter =='.') break; // don't include extension
238  if (letter =='\\') letter = '/'; // damn path names
239  hash+=(long)(letter)*(i+119);
240  i++;
241  }
242  hash &= (LOOP_HASH-1);
243  return hash;
244 }

246 /*
247 ==================
248 S_FindName
249 
250 Will allocate a new sfx if it isn't found
251 ==================
252 */
253 static sfx_t *S_FindName( const char *name ) {
254  int i;
255  int hash;
256 
257  sfx_t *sfx;
258 
259  if (!name) {
260  Com_Error(ERR_FATAL, "Sound name is NULL");
261  }
262 
263  if (!name[0]) {
264  Com_Printf( S_COLOR_YELLOW "WARNING: Sound name is empty\n" );
265  return NULL;
266  }
267 
268  if (strlen(name) >= MAX_QPATH) {
269  Com_Printf( S_COLOR_YELLOW "WARNING: Sound name is too long: %s\n", name );
270  return NULL;
271  }
272 
273  if (name[0] == '*') {
274  Com_Printf( S_COLOR_YELLOW "WARNING: Tried to load player sound directly: %s\n", name );
275  return NULL;
276  }
277 
278  hash = S_HashSFXName(name);
279 
280  sfx = sfxHash[hash];
281  // see if already loaded
282  while (sfx) {
283  if (!Q_stricmp(sfx->soundName, name) ) {
284  return sfx;
285  }
286  sfx = sfx->next;
287  }
288 
289  // find a free sfx
290  for (i=0 ; i < s_numSfx ; i++) {
291  if (!s_knownSfx[i].soundName[0]) {
292  break;
293  }
294  }
295 
296  if (i == s_numSfx) {
297  if (s_numSfx == MAX_SFX) {
298  Com_Error (ERR_FATAL, "S_FindName: out of sfx_t");
299  }
300  s_numSfx++;
301  }
302 
303  sfx = &s_knownSfx[i];
304  Com_Memset (sfx, 0, sizeof(*sfx));
305  strcpy (sfx->soundName, name);
306 
307  sfx->next = sfxHash[hash];
308  sfxHash[hash] = sfx;
309 
310  return sfx;
311 }

417 /*
418 =================
419 S_SpatializeOrigin
420 
421 Used for spatializing s_channels
422 =================
423 */
424 void S_SpatializeOrigin (vec3_t origin, int master_vol, int *left_vol, int *right_vol)
425 {
426  vec_t dot;
427  vec_t dist;
428  vec_t lscale, rscale, scale;
429  vec3_t source_vec;
430  vec3_t vec;
431 
432  const float dist_mult = SOUND_ATTENUATE;
433 
434  // calculate stereo separation and distance attenuation
435  VectorSubtract(origin, listener_origin, source_vec);
436 
437  dist = VectorNormalize(source_vec);
438  dist -= SOUND_FULLVOLUME;
439  if (dist < 0)
440  dist = 0; // close enough to be at full volume
441  dist *= dist_mult; // different attenuation levels
442 
443  VectorRotate( source_vec, listener_axis, vec );
444 
445  dot = -vec[1];
446 
447  if (dma.channels == 1)
448  { // no attenuation = no spatialization
449  rscale = 1.0;
450  lscale = 1.0;
451  }
452  else
453  {
454  rscale = 0.5 * (1.0 + dot);
455  lscale = 0.5 * (1.0 - dot);
456  if ( rscale < 0 ) {
457  rscale = 0;
458  }
459  if ( lscale < 0 ) {
460  lscale = 0;
461  }
462  }
463 
464  // add in distance effect
465  scale = (1.0 - dist) * rscale;
466  *right_vol = (master_vol * scale);
467  if (*right_vol < 0)
468  *right_vol = 0;
469 
470  scale = (1.0 - dist) * lscale;
471  *left_vol = (master_vol * scale);
472  if (*left_vol < 0)
473  *left_vol = 0;
474 }

480 /*
481 =================
482 S_Base_HearingThroughEntity
483 
484 Also see S_AL_HearingThroughEntity
485 =================
486 */
487 static qboolean S_Base_HearingThroughEntity( int entityNum, vec3_t origin )
488 {
489  float distanceSq;
490  vec3_t sorigin;
491 
492  if (origin)
493  VectorCopy(origin, sorigin);
494  else
495  VectorCopy(loopSounds[entityNum].origin, sorigin);
496 
497  if( listener_number == entityNum )
498  {
499  // This is an outrageous hack to detect
500  // whether or not the player is rendering in third person or not. We can't
501  // ask the renderer because the renderer has no notion of entities and we
502  // can't ask cgame since that would involve changing the API and hence mod
503  // compatibility. I don't think there is any way around this, but I'll leave
504  // the FIXME just in case anyone has a bright idea.
505  distanceSq = DistanceSquared(
506  sorigin,
507  listener_origin );
508 
509  if( distanceSq > THIRD_PERSON_THRESHOLD_SQ )
510  return qfalse; //we're the player, but third person
511  else
512  return qtrue; //we're the player
513  }
514  else
515  return qfalse; //not the player
516 }

518 /*
519 ====================
520 S_Base_StartSoundEx
521 
522 Validates the parms and ques the sound up
523 if origin is NULL, the sound will be dynamically sourced from the entity
524 Entchannel 0 will never override a playing sound
525 ====================
526 */
527 static void S_Base_StartSoundEx( vec3_t origin, int entityNum, int entchannel, sfxHandle_t sfxHandle, qboolean localSound ) {
528  channel_t *ch;
529  sfx_t *sfx;
530  int i, oldest, chosen, time;
531  int inplay, allowed;
532  qboolean fullVolume;
533 
534  if ( !s_soundStarted || s_soundMuted ) {
535  return;
536  }
537 
538  if ( !origin && ( entityNum < 0 || entityNum >= MAX_GENTITIES ) ) {
539  Com_Error( ERR_DROP, "S_StartSound: bad entitynum %i", entityNum );
540  }
541 
542  if ( sfxHandle < 0 || sfxHandle >= s_numSfx ) {
543  Com_Printf( S_COLOR_YELLOW "S_StartSound: handle %i out of range\n", sfxHandle );
544  return;
545  }
546 
547  sfx = &s_knownSfx[ sfxHandle ];
548 
549  if (sfx->inMemory == qfalse) {
550  S_memoryLoad(sfx);
551  }
552 
553  if ( s_show->integer == 1 ) {
554  Com_Printf( "%i : %s\n", s_paintedtime, sfx->soundName );
555  }
556 
557  time = Com_Milliseconds();
558 
559 // Com_Printf("playing %s\n", sfx->soundName);
560  // pick a channel to play on
561 
562  allowed = 4;
563  if (entityNum == listener_number) {
564  allowed = 8;
565  }
566 
567  fullVolume = qfalse;
568  if (localSound || S_Base_HearingThroughEntity(entityNum, origin)) {
569  fullVolume = qtrue;
570  }
571 
572  ch = s_channels;
573  inplay = 0;
574  for ( i = 0; i < MAX_CHANNELS ; i++, ch++ ) {
575  if (ch->entnum == entityNum && ch->thesfx == sfx) {
576  if (time - ch->allocTime < 50) {
577 // if (Cvar_VariableValue( "cg_showmiss" )) {
578 // Com_Printf("double sound start\n");
579 // }
580  return;
581  }
582  inplay++;
583  }
584  }
585 
586  if (inplay>allowed) {
587  return;
588  }
589 
590  sfx->lastTimeUsed = time;
591 
592  ch = S_ChannelMalloc(); // entityNum, entchannel);
593  if (!ch) {
594  ch = s_channels;
595 
596  oldest = sfx->lastTimeUsed;
597  chosen = -1;
598  for ( i = 0 ; i < MAX_CHANNELS ; i++, ch++ ) {
599  if (ch->entnum != listener_number && ch->entnum == entityNum && ch->allocTime<oldest && ch->entchannel != CHAN_ANNOUNCER) {
600  oldest = ch->allocTime;
601  chosen = i;
602  }
603  }
604  if (chosen == -1) {
605  ch = s_channels;
606  for ( i = 0 ; i < MAX_CHANNELS ; i++, ch++ ) {
607  if (ch->entnum != listener_number && ch->allocTime<oldest && ch->entchannel != CHAN_ANNOUNCER) {
608  oldest = ch->allocTime;
609  chosen = i;
610  }
611  }
612  if (chosen == -1) {
613  ch = s_channels;
614  if (ch->entnum == listener_number) {
615  for ( i = 0 ; i < MAX_CHANNELS ; i++, ch++ ) {
616  if (ch->allocTime<oldest) {
617  oldest = ch->allocTime;
618  chosen = i;
619  }
620  }
621  }
622  if (chosen == -1) {
623  Com_Printf("dropping sound\n");
624  return;
625  }
626  }
627  }
628  ch = &s_channels[chosen];
629  ch->allocTime = sfx->lastTimeUsed;
630  }
631 
632  if (origin) {
633  VectorCopy (origin, ch->origin);
634  ch->fixed_origin = qtrue;
635  } else {
636  ch->fixed_origin = qfalse;
637  }
638 
639  ch->master_vol = 127;
640  ch->entnum = entityNum;
641  ch->thesfx = sfx;
642  ch->startSample = START_SAMPLE_IMMEDIATE;
643  ch->entchannel = entchannel;
644  ch->leftvol = ch->master_vol; // these will get calced at next spatialize
645  ch->rightvol = ch->master_vol; // unless the game isn't running
646  ch->doppler = qfalse;
647  ch->fullVolume = fullVolume;
648 }
