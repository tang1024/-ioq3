175 /*
176 ==================
177 PM_Friction
178 
179 Handles both ground friction and water friction
180 ==================
181 */
182 static void PM_Friction(void)
183 {
184  vec3_t vec;
185  float* vel;
186  float speed, newspeed, control;
187  float drop;
188 
189  vel = pm->ps->velocity;
190 
191  VectorCopy(vel, vec);
192  if (pml.walking) {
193  vec[2] = 0; // ignore slope movement
194  }
195 
196  speed = VectorLength(vec);
197  if (speed < 1) {
198  vel[0] = 0;
199  vel[1] = 0; // allow sinking underwater
200  // FIXME: still have z friction underwater?
201  return;
202  }
203 
204  drop = 0;
205 
206  // apply ground friction
207  if (pm->waterlevel <= 1) {
208  if (pml.walking && !(pml.groundTrace.surfaceFlags & SURF_SLICK)) {
209  // if getting knocked back, no friction
210  if (!(pm->ps->pm_flags & PMF_TIME_KNOCKBACK)) {
211  control = speed < pm_stopspeed ? pm_stopspeed : speed;
212  drop += control * pm_friction * pml.frametime;
213  }
214  }
215  }
216 
217  // apply water friction even if just wading
218  if (pm->waterlevel) {
219  drop += speed * pm_waterfriction * pm->waterlevel * pml.frametime;
220  }
221 
222  // apply flying friction
223  if (pm->ps->powerups[PW_FLIGHT]) {
224  drop += speed * pm_flightfriction * pml.frametime;
225  }
226 
227  if (pm->ps->pm_type == PM_SPECTATOR) {
228  drop += speed * pm_spectatorfriction * pml.frametime;
229  }
230 
231  // scale the velocity
232  newspeed = speed - drop;
233  if (newspeed < 0) {
234  newspeed = 0;
235  }
236  newspeed /= speed;
237 
238  vel[0] = vel[0] * newspeed;
239  vel[1] = vel[1] * newspeed;
240  vel[2] = vel[2] * newspeed;
241 }

293 /*
294 ============
295 PM_CmdScale
296 
297 Returns the scale factor to apply to cmd movements
298 This allows the clients to use axial -127 to 127 values for all directions
299 without getting a sqrt(2) distortion in speed.
300 ============
301 */
302 static float PM_CmdScale(usercmd_t* cmd)
303 {
304  int max;
305  float total;
306  float scale;
307 
308  max = abs(cmd->forwardmove);
309  if (abs(cmd->rightmove) > max) {
310  max = abs(cmd->rightmove);
311  }
312  if (abs(cmd->upmove) > max) {
313  max = abs(cmd->upmove);
314  }
315  if (!max) {
316  return 0;
317  }
318 
319  total = sqrt(cmd->forwardmove * cmd->forwardmove
320  + cmd->rightmove * cmd->rightmove + cmd->upmove * cmd->upmove);
321  scale = (float)pm->ps->speed * max / (127.0 * total);
322 
323  return scale;
324 }

940
