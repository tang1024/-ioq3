30 void daub4(float b[], unsigned long n, int isign)
31 {
32  float wksp[4097] = { 0.0f };
33 #define a(x) b[(x)-1] // numerical recipies so a[1] = b[0]
34 
35  unsigned long nh,nh1,i,j;
36 
37  if (n < 4) return;
38 
39  nh1=(nh=n >> 1)+1;
40  if (isign >= 0) {
41  for (i=1,j=1;j<=n-3;j+=2,i++) {
42  wksp[i] = C0*a(j)+C1*a(j+1)+C2*a(j+2)+C3*a(j+3);
43  wksp[i+nh] = C3*a(j)-C2*a(j+1)+C1*a(j+2)-C0*a(j+3);
44  }
45  wksp[i ] = C0*a(n-1)+C1*a(n)+C2*a(1)+C3*a(2);
46  wksp[i+nh] = C3*a(n-1)-C2*a(n)+C1*a(1)-C0*a(2);
47  } else {
48  wksp[1] = C2*a(nh)+C1*a(n)+C0*a(1)+C3*a(nh1);
49  wksp[2] = C3*a(nh)-C0*a(n)+C1*a(1)-C2*a(nh1);
50  for (i=1,j=3;i<nh;i++) {
51  wksp[j++] = C2*a(i)+C1*a(i+nh)+C0*a(i+1)+C3*a(i+nh1);
52  wksp[j++] = C3*a(i)-C0*a(i+nh)+C1*a(i+1)-C2*a(i+nh1);
53  }
54  }
55  for (i=1;i<=n;i++) {
56  a(i)=wksp[i];
57  }
58 #undef a
59 }

85 byte MuLawEncode(short s) {
86  unsigned long adjusted;
87  byte sign, exponent, mantissa;
88 
89  sign = (s<0)?0:0x80;
90 
91  if (s<0) s=-s;
92  adjusted = (long)s << (16-sizeof(short)*8);
93  adjusted += 128L + 4L;
94  if (adjusted > 32767) adjusted = 32767;
95  exponent = numBits[(adjusted>>7)&0xff] - 1;
96  mantissa = (adjusted>>(exponent+3))&0xf;
97  return ~(sign | (exponent<<4) | mantissa);
98 }

100 short MuLawDecode(byte uLaw) {
101  signed long adjusted;
102  byte exponent, mantissa;
103 
104  uLaw = ~uLaw;
105  exponent = (uLaw>>4) & 0x7;
106  mantissa = (uLaw&0xf) + 16;
107  adjusted = (mantissa << (exponent +3)) - 128 - 4;
108 
109  return (uLaw & 0x80)? adjusted : -adjusted;
110 }
