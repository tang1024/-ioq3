44 // the trigger was just activated
45 // ent->activator should be set to the activator so it can be held through a delay
46 // so wait for the delay time before firing
47 void multi_trigger(gentity_t* ent, gentity_t* activator)
48 {
49  ent->activator = activator;
50  if (ent->nextthink) {
51  return; // can't retrigger until the wait is over
52  }
53 
54  if (activator->client) {
55  if ((ent->spawnflags & 1) &&
56  activator->client->sess.sessionTeam != TEAM_RED) {
57  return;
58  }
59  if ((ent->spawnflags & 2) &&
60  activator->client->sess.sessionTeam != TEAM_BLUE) {
61  return;
62  }
63  }
64 
65  G_UseTargets(ent, ent->activator);
66 
67  if (ent->wait > 0) {
68  ent->think = multi_wait;
69  ent->nextthink = level.time + (ent->wait + ent->random * crandom()) * 1000;
70  } else {
71  // we can't just remove (self) here, because this is a touch function
72  // called while looping through area links...
73  ent->touch = 0;
74  ent->nextthink = level.time + FRAMETIME;
75  ent->think = G_FreeEntity;
76  }
77 }

164 AimAtTarget
165 
166 Calculate origin2 so the target apogee will be hit
167 =================
168 */
169 void AimAtTarget(gentity_t* self)
170 {
171  gentity_t* ent;
172  vec3_t origin;
173  float height, gravity, time, forward;
174  float dist;
175 
176  VectorAdd(self->r.absmin, self->r.absmax, origin);
177  VectorScale(origin, 0.5, origin);
178 
179  ent = G_PickTarget(self->target);
180  if (!ent) {
181  G_FreeEntity(self);
182  return;
183  }
184 
185  height = ent->s.origin[2] - origin[2];
186  gravity = g_gravity.value;
187  time = sqrt(height / (.5 * gravity));
188  if (!time) {
189  G_FreeEntity(self);
190  return;
191  }
192 
193  // set s.origin2 to the push velocity
194  VectorSubtract(ent->s.origin, origin, self->s.origin2);
195  self->s.origin2[2] = 0;
196  dist = VectorNormalize(self->s.origin2);
197 
198  forward = dist / time;
199  VectorScale(self->s.origin2, forward, self->s.origin2);
200 
201  self->s.origin2[2] = time * gravity;
202 }
