274 itemconfig_t *LoadItemConfig(char *filename)
275 {
276  int max_iteminfo;
277  token_t token;
278  char path[MAX_QPATH];
279  source_t *source;
280  itemconfig_t *ic;
281  iteminfo_t *ii;
282 
283  max_iteminfo = (int) LibVarValue("max_iteminfo", "256");
284  if (max_iteminfo < 0)
285  {
286  botimport.Print(PRT_ERROR, "max_iteminfo = %d\n", max_iteminfo);
287  max_iteminfo = 256;
288  LibVarSet( "max_iteminfo", "256" );
289  }
290 
291  Q_strncpyz(path, filename, sizeof(path));
292  PC_SetBaseFolder(BOTFILESBASEFOLDER);
293  source = LoadSourceFile( path );
294  if( !source ) {
295  botimport.Print( PRT_ERROR, "counldn't load %s\n", path );
296  return NULL;
297  } //end if
298  //initialize item config
299  ic = (itemconfig_t *) GetClearedHunkMemory(sizeof(itemconfig_t) +
300  max_iteminfo * sizeof(iteminfo_t));
301  ic->iteminfo = (iteminfo_t *) ((char *) ic + sizeof(itemconfig_t));
302  ic->numiteminfo = 0;
303  //parse the item config file
304  while(PC_ReadToken(source, &token))
305  {
306  if (!strcmp(token.string, "iteminfo"))
307  {
308  if (ic->numiteminfo >= max_iteminfo)
309  {
310  SourceError(source, "more than %d item info defined", max_iteminfo);
311  FreeMemory(ic);
312  FreeSource(source);
313  return NULL;
314  } //end if
315  ii = &ic->iteminfo[ic->numiteminfo];
316  Com_Memset(ii, 0, sizeof(iteminfo_t));
317  if (!PC_ExpectTokenType(source, TT_STRING, 0, &token))
318  {
319  FreeMemory(ic);
320  FreeSource(source);
321  return NULL;
322  } //end if
323  StripDoubleQuotes(token.string);
324  Q_strncpyz(ii->classname, token.string, sizeof(ii->classname));
325  if (!ReadStructure(source, &iteminfo_struct, (char *) ii))
326  {
327  FreeMemory(ic);
328  FreeSource(source);
329  return NULL;
330  } //end if
331  ii->number = ic->numiteminfo;
332  ic->numiteminfo++;
333  } //end if
334  else
335  {
336  SourceError(source, "unknown definition %s", token.string);
337  FreeMemory(ic);
338  FreeSource(source);
339  return NULL;
340  } //end else
341  } //end while
342  FreeSource(source);
343  //
344  if (!ic->numiteminfo) botimport.Print(PRT_WARNING, "no item info loaded\n");
345  botimport.Print(PRT_MESSAGE, "loaded %s\n", path);
346  return ic;
347 } //end of the function LoadItemConfig

544 void BotInitLevelItems(void)
545 {

547  char classname[MAX_EPAIRKEY];
548  vec3_t origin, end;
549  int ent, goalareanum;

551  levelitem_t *li;
552  bsp_trace_t trace;
553 
554  //initialize the map locations and camp spots
555  BotInitInfoEntities();
556 
557  //initialize the level item heap
558  InitLevelItemHeap();
559  levelitems = NULL;
560  numlevelitems = 0;
561  //
562  ic = itemconfig;
563  if (!ic) return;
564 
565  //if there's no AAS file loaded
566  if (!AAS_Loaded()) return;
567 
568  //validate the modelindexes of the item info
569  for (i = 0; i < ic->numiteminfo; i++)
570  {
571  if (!ic->iteminfo[i].modelindex)
572  {
573  Log_Write("item %s has modelindex 0", ic->iteminfo[i].classname);
574  } //end if
575  } //end for
576 
577  for (ent = AAS_NextBSPEntity(0); ent; ent = AAS_NextBSPEntity(ent))
578  {
579  if (!AAS_ValueForBSPEpairKey(ent, "classname", classname, MAX_EPAIRKEY)) continue;
580  //
581  spawnflags = 0;
582  AAS_IntForBSPEpairKey(ent, "spawnflags", &spawnflags);
583  //
584  for (i = 0; i < ic->numiteminfo; i++)
585  {
586  if (!strcmp(classname, ic->iteminfo[i].classname)) break;
587  } //end for
588  if (i >= ic->numiteminfo)
589  {
590  Log_Write("entity %s unknown item\r\n", classname);
591  continue;
592  } //end if
593  //get the origin of the item
594  if (!AAS_VectorForBSPEpairKey(ent, "origin", origin))
595  {
596  botimport.Print(PRT_ERROR, "item %s without origin\n", classname);
597  continue;
598  } //end else
599  //
600  goalareanum = 0;
601  //if it is a floating item
602  if (spawnflags & 1)
603  {
604  //if the item is not floating in water
605  if (!(AAS_PointContents(origin) & CONTENTS_WATER))
606  {
607  VectorCopy(origin, end);
608  end[2] -= 32;
609  trace = AAS_Trace(origin, ic->iteminfo[i].mins, ic->iteminfo[i].maxs, end, -1, CONTENTS_SOLID|CONTENTS_PLAYERCLIP);
610  //if the item not near the ground
611  if (trace.fraction >= 1)
612  {
613  //if the item is not reachable from a jumppad
614  goalareanum = AAS_BestReachableFromJumpPadArea(origin, ic->iteminfo[i].mins, ic->iteminfo[i].maxs);
615  Log_Write("item %s reachable from jumppad area %d\r\n", ic->iteminfo[i].classname, goalareanum);
616  //botimport.Print(PRT_MESSAGE, "item %s reachable from jumppad area %d\r\n", ic->iteminfo[i].classname, goalareanum);
617  if (!goalareanum) continue;
618  } //end if
619  } //end if
620  } //end if
621 
622  li = AllocLevelItem();
623  if (!li) return;
624  //
625  li->number = ++numlevelitems;
626  li->timeout = 0;
627  li->entitynum = 0;
628  //
629  li->flags = 0;
630  AAS_IntForBSPEpairKey(ent, "notfree", &value);
631  if (value) li->flags |= IFL_NOTFREE;
632  AAS_IntForBSPEpairKey(ent, "notteam", &value);
633  if (value) li->flags |= IFL_NOTTEAM;
634  AAS_IntForBSPEpairKey(ent, "notsingle", &value);
635  if (value) li->flags |= IFL_NOTSINGLE;
636  AAS_IntForBSPEpairKey(ent, "notbot", &value);
637  if (value) li->flags |= IFL_NOTBOT;
638  if (!strcmp(classname, "item_botroam"))
639  {
640  li->flags |= IFL_ROAM;
641  AAS_FloatForBSPEpairKey(ent, "weight", &li->weight);
642  } //end if
643  //if not a stationary item
644  if (!(spawnflags & 1))
645  {
646  if (!AAS_DropToFloor(origin, ic->iteminfo[i].mins, ic->iteminfo[i].maxs))
647  {
648  botimport.Print(PRT_MESSAGE, "%s in solid at (%1.1f %1.1f %1.1f)\n",
649  classname, origin[0], origin[1], origin[2]);
650  } //end if
651  } //end if
652  //item info of the level item
653  li->iteminfo = i;
654  //origin of the item
655  VectorCopy(origin, li->origin);
656  //
657  if (goalareanum)
658  {
659  li->goalareanum = goalareanum;
660  VectorCopy(origin, li->goalorigin);
661  } //end if
662  else
663  {
664  //get the item goal area and goal origin
665  li->goalareanum = AAS_BestReachableArea(origin,
666  ic->iteminfo[i].mins, ic->iteminfo[i].maxs,
667  li->goalorigin);
668  if (!li->goalareanum)
669  {
670  botimport.Print(PRT_MESSAGE, "%s not reachable for bots at (%1.1f %1.1f %1.1f)\n",
671  classname, origin[0], origin[1], origin[2]);
672  } //end if
673  } //end else
674  //
675  AddLevelItemToList(li);
676  } //end for
677  botimport.Print(PRT_MESSAGE, "found %d level items\n", numlevelitems);
678 } //end of the function BotInitLevelItems
