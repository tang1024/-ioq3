74 
75 Lengyel, Eric. “Computing Tangent Space Basis Vectors for an Arbitrary Mesh”. Terathon Software 3D Graphics Library, 2001. http://www.terathon.com/code/tangent.html

78 void R_CalcTexDirs(vec3_t sdir, vec3_t tdir, const vec3_t v1, const vec3_t v2,
79  const vec3_t v3, const vec2_t w1, const vec2_t w2, const vec2_t w3)
80 {
81  float x1, x2, y1, y2, z1, z2;
82  float s1, s2, t1, t2, r;
83 
84  x1 = v2[0] - v1[0];
85  x2 = v3[0] - v1[0];
86  y1 = v2[1] - v1[1];
87  y2 = v3[1] - v1[1];
88  z1 = v2[2] - v1[2];

90 
91  s1 = w2[0] - w1[0];
92  s2 = w3[0] - w1[0];
93  t1 = w2[1] - w1[1];
94  t2 = w3[1] - w1[1];
95 
96  r = s1 * t2 - s2 * t1;
97  if (r) r = 1.0f / r;
98 
99  VectorSet(sdir, (t2 * x1 - t1 * x2) * r, (t2 * y1 - t1 * y2) * r, (t2 * z1 - t1 * z2) * r);
100  VectorSet(tdir, (s1 * x2 - s2 * x1) * r, (s1 * y2 - s2 * y1) * r, (s1 * z2 - s2 * z1) * r);

104 =============
105 R_CalcTangentSpace
106 
107 Lengyel, Eric. “Computing Tangent Space Basis Vectors for an Arbitrary Mesh”. Terathon Software 3D Graphics Library, 2001. http://www.terathon.com/code/tangent.html
108 =============
109 */
110 vec_t R_CalcTangentSpace(vec3_t tangent, vec3_t bitangent, const vec3_t normal, const vec3_t sdir, const vec3_t tdir)
111 {
112  vec3_t n_cross_t;
113  vec_t n_dot_t, handedness;
114 
115  // Gram-Schmidt orthogonalize
116  n_dot_t = DotProduct(normal, sdir);
117  VectorMA(sdir, -n_dot_t, normal, tangent);
118  VectorNormalize(tangent);
119 
120  // Calculate handedness
121  CrossProduct(normal, sdir, n_cross_t);
122  handedness = (DotProduct(n_cross_t, tdir) < 0.0f) ? -1.0f : 1.0f;
123 
124  // Calculate orthogonal bitangent, if necessary
125  if (bitangent)
126  CrossProduct(normal, tangent, bitangent);
127 
128  return handedness;
129 }

131 qboolean R_CalcTangentVectors(srfVert_t * dv[3])
132 {
133  int i;
134  float bb, s, t;
135  vec3_t bary;
136 
137 
138  /* calculate barycentric basis for the triangle */
139  bb = (dv[1]->st[0] - dv[0]->st[0]) * (dv[2]->st[1] - dv[0]->st[1]) - (dv[2]->st[0] - dv[0]->st[0]) * (dv[1]->st[1] - dv[0]->st[1]);
140  if(fabs(bb) < 0.00000001f)
141  return qfalse;
142 
143  /* do each vertex */
144  for(i = 0; i < 3; i++)
145  {
146  vec4_t tangent;
147  vec3_t normal, bitangent, nxt;
148 
149  // calculate s tangent vector
150  s = dv[i]->st[0] + 10.0f;
151  t = dv[i]->st[1];
152  bary[0] = ((dv[1]->st[0] - s) * (dv[2]->st[1] - t) - (dv[2]->st[0] - s) * (dv[1]->st[1] - t)) / bb;
153  bary[1] = ((dv[2]->st[0] - s) * (dv[0]->st[1] - t) - (dv[0]->st[0] - s) * (dv[2]->st[1] - t)) / bb;
154  bary[2] = ((dv[0]->st[0] - s) * (dv[1]->st[1] - t) - (dv[1]->st[0] - s) * (dv[0]->st[1] - t)) / bb;
155 
156  tangent[0] = bary[0] * dv[0]->xyz[0] + bary[1] * dv[1]->xyz[0] + bary[2] * dv[2]->xyz[0];
157  tangent[1] = bary[0] * dv[0]->xyz[1] + bary[1] * dv[1]->xyz[1] + bary[2] * dv[2]->xyz[1];
158  tangent[2] = bary[0] * dv[0]->xyz[2] + bary[1] * dv[1]->xyz[2] + bary[2] * dv[2]->xyz[2];
159 
160  VectorSubtract(tangent, dv[i]->xyz, tangent);
161  VectorNormalize(tangent);
162 
163  // calculate t tangent vector
164  s = dv[i]->st[0];
165  t = dv[i]->st[1] + 10.0f;
166  bary[0] = ((dv[1]->st[0] - s) * (dv[2]->st[1] - t) - (dv[2]->st[0] - s) * (dv[1]->st[1] - t)) / bb;
167  bary[1] = ((dv[2]->st[0] - s) * (dv[0]->st[1] - t) - (dv[0]->st[0] - s) * (dv[2]->st[1] - t)) / bb;
168  bary[2] = ((dv[0]->st[0] - s) * (dv[1]->st[1] - t) - (dv[1]->st[0] - s) * (dv[0]->st[1] - t)) / bb;
169 
170  bitangent[0] = bary[0] * dv[0]->xyz[0] + bary[1] * dv[1]->xyz[0] + bary[2] * dv[2]->xyz[0];
171  bitangent[1] = bary[0] * dv[0]->xyz[1] + bary[1] * dv[1]->xyz[1] + bary[2] * dv[2]->xyz[1];
172  bitangent[2] = bary[0] * dv[0]->xyz[2] + bary[1] * dv[1]->xyz[2] + bary[2] * dv[2]->xyz[2];
173 
174  VectorSubtract(bitangent, dv[i]->xyz, bitangent);
175  VectorNormalize(bitangent);
176 
177  // store bitangent handedness
178  R_VaoUnpackNormal(normal, dv[i]->normal);
179  CrossProduct(normal, tangent, nxt);
180  tangent[3] = (DotProduct(nxt, bitangent) < 0.0f) ? -1.0f : 1.0f;
181 
182  R_VaoPackTangent(dv[i]->tangent, tangent);
183 
184  // debug code
185  //% Sys_FPrintf( SYS_VRB, "%d S: (%f %f %f) T: (%f %f %f)\n", i,
186  //% stv[ i ][ 0 ], stv[ i ][ 1 ], stv[ i ][ 2 ], ttv[ i ][ 0 ], ttv[ i ][ 1 ], ttv[ i ][ 2 ] );
187  }
188 
189  return qtrue;
190 }

678 =================
679 */
680 void R_SetupFrustum (viewParms_t *dest, float xmin, float xmax, float ymax, float zProj, float zFar, float stereoSep)
681 {

683  float oppleg, adjleg, length;
684  int i;
685

687  {
688  // symmetric case can be simplified
689  VectorCopy(dest->or.origin, ofsorigin);
690 
691  length = sqrt(xmax * xmax + zProj * zProj);
692  oppleg = xmax / length;
693  adjleg = zProj / length;

699  VectorMA(dest->frustum[1].normal, -adjleg, dest->or.axis[1], dest->frustum[1].normal);
700  }
701  else
702  {
703  // In stereo rendering, due to the modification of the projection matrix, dest->or.origin is not the
704  // actual origin that we're rendering so offset the tip of the view pyramid.
705  VectorMA(dest->or.origin, stereoSep, dest->or.axis[1], ofsorigin);
706 
707  oppleg = xmax + stereoSep;
708  length = sqrt(oppleg * oppleg + zProj * zProj);
709  VectorScale(dest->or.axis[0], oppleg / length, dest->frustum[0].normal);
710  VectorMA(dest->frustum[0].normal, zProj / length, dest->or.axis[1], dest->frustum[0].normal);

713  length = sqrt(oppleg * oppleg + zProj * zProj);

740 
741  dest->frustum[4].type = PLANE_NON_AXIAL;
742  dest->frustum[4].dist = DotProduct (farpoint, dest->frustum[4].normal);

745  }

748 /*
749 ===============
750 R_SetupProjection
751 ===============
752 */
753 void R_SetupProjection(viewParms_t *dest, float zProj, float zFar, qboolean computeFrustum)
754 {
755  float xmin, xmax, ymin, ymax;
756  float width, height, stereoSep = r_stereoSeparation->value;
757 
758  /*
759  * offset the view origin of the viewer for stereo rendering
760  * by setting the projection matrix appropriately.
761  */
762 
763  if(stereoSep != 0)
764  {
765  if(dest->stereoFrame == STEREO_LEFT)
766  stereoSep = zProj / stereoSep;
767  else if(dest->stereoFrame == STEREO_RIGHT)
768  stereoSep = zProj / -stereoSep;
769  else
770  stereoSep = 0;
771  }
772 
773  ymax = zProj * tan(dest->fovY * M_PI / 360.0f);
774  ymin = -ymax;
775 
776  xmax = zProj * tan(dest->fovX * M_PI / 360.0f);
777  xmin = -xmax;
778 
779  width = xmax - xmin;
780  height = ymax - ymin;
781 
782  dest->projectionMatrix[0] = 2 * zProj / width;
783  dest->projectionMatrix[4] = 0;
784  dest->projectionMatrix[8] = (xmax + xmin + 2 * stereoSep) / width;
785  dest->projectionMatrix[12] = 2 * zProj * stereoSep / width;
786 
787  dest->projectionMatrix[1] = 0;
788  dest->projectionMatrix[5] = 2 * zProj / height;
789  dest->projectionMatrix[9] = ( ymax + ymin ) / height; // normally 0
790  dest->projectionMatrix[13] = 0;
791 
792  dest->projectionMatrix[3] = 0;
793  dest->projectionMatrix[7] = 0;
794  dest->projectionMatrix[11] = -1;
795  dest->projectionMatrix[15] = 0;
796 
797  // Now that we have all the data for the projection matrix we can also setup the view frustum.
798  if(computeFrustum)
799  R_SetupFrustum(dest, xmin, xmax, ymax, zProj, zFar, stereoSep);
800 }

803 ===============
804 R_SetupProjectionZ
805 
806 Sets the z-component transformation part in the projection matrix
807 ===============
808 */
809 void R_SetupProjectionZ(viewParms_t *dest)
810 {
811  float zNear, zFar, depth;
812 
813  zNear = r_znear->value;
814  zFar = dest->zFar;
815 
816  depth = zFar - zNear;
817 
818  dest->projectionMatrix[2] = 0;
819  dest->projectionMatrix[6] = 0;
820  dest->projectionMatrix[10] = -( zFar + zNear ) / depth;
821  dest->projectionMatrix[14] = -2 * zFar * zNear / depth;
822 
823  if (dest->isPortal)
824  {
825  float plane[4];
826  float plane2[4];
827  vec4_t q, c;
828 
829  // transform portal plane into camera space
830  plane[0] = dest->portalPlane.normal[0];

832  plane[2] = dest->portalPlane.normal[2];

836  plane2[1] = DotProduct (dest->or.axis[2], plane);
837  plane2[2] = -DotProduct (dest->or.axis[0], plane);
838  plane2[3] = DotProduct (plane, dest->or.origin) - plane[3];

841  // Terathon Software 3D Graphics Library, 2004. http://www.terathon.com/code/oblique.html

843  q[1] = (SGN(plane2[1]) + dest->projectionMatrix[9]) / dest->projectionMatrix[5];
844  q[2] = -1.0f;
845  q[3] = (1.0f + dest->projectionMatrix[10]) / dest->projectionMatrix[14];
846 
847  VectorScale4(plane2, 2.0f / DotProduct4(plane2, q), c);
848 
849  dest->projectionMatrix[2] = c[0];
850  dest->projectionMatrix[6] = c[1];
851  dest->projectionMatrix[10] = c[2] + 1.0f;
852  dest->projectionMatrix[14] = c[3];
853 
854  }

1222  int j;

1224

1229  if ( clip[j] >= clip[3] )
1230  {
1231  pointFlags |= (1 << (j*2));
1232  }
1233  else if ( clip[j] <= -clip[3] )
1234  {
1235  pointFlags |= ( 1 << (j*2+1));
1236  }
1237  }
1238  pointAnd &= pointFlags;
1239  pointOr |= pointFlags;
1240  }
1241 
1242  // trivially reject
1243  if ( pointAnd )
1244  {
1245  return qtrue;
1246  }
1247 
1248  // determine if this surface is backfaced and also determine the distance
1249  // to the nearest vertex so we can cull based on portal range. Culling
1250  // based on vertex distance isn't 100% correct (we should be checking for
1251  // range to the surface), but it's good enough for the types of portals
1252  // we have in the game right now.
1253  numTriangles = tess.numIndexes / 3;
1254

1256  {

1260

1262

1264  if ( len < shortest )
1265  {
1266  shortest = len;
1267  }
1268 
1269  R_VaoUnpackNormal(tNormal, tess.normal[tess.indexes[i]]);
1270

1272  {
1273  numTriangles--;
1274  }
1275  }
1276  if ( !numTriangles )
1277  {
1278  return qtrue;
1279  }
1280 
1281  // mirrors can early out at this point, since we don't do a fade over distance
1282  // with them (although we could)
1283  if ( IsMirror( drawSurf, entityNum ) )
1284  {
1285  return qfalse;
1286  }
1287 
1288  if ( shortest > (tess.shader->portalRange*tess.shader->portalRange) )
1289  {
1290  return qtrue;
1291  }
1292

1294 }

1408 static ID_INLINE void R_Radix( int byte, int size, drawSurf_t *source, drawSurf_t *dest )
1409 {
1410  int count[ 256 ] = { 0 };
1411  int index[ 256 ];
1412  int i;
1413  unsigned char *sortKey = NULL;
1414  unsigned char *end = NULL;
1415 
1416  sortKey = ( (unsigned char *)&source[ 0 ].sort ) + byte;
1417  end = sortKey + ( size * sizeof( drawSurf_t ) );
1418  for( ; sortKey < end; sortKey += sizeof( drawSurf_t ) )
1419  ++count[ *sortKey ];
1420 
1421  index[ 0 ] = 0;
1422 
1423  for( i = 1; i < 256; ++i )
1424  index[ i ] = index[ i - 1 ] + count[ i - 1 ];
1425 
1426  sortKey = ( (unsigned char *)&source[ 0 ].sort ) + byte;
1427  for( i = 0; i < size; ++i, sortKey += sizeof( drawSurf_t ) )
1428  dest[ index[ *sortKey ]++ ] = source[ i ];
1429 }
