91 const char *VM_ValueToSymbol( vm_t *vm, int value ) {
92  vmSymbol_t *sym;
93  static char text[MAX_TOKEN_CHARS];
94 
95  sym = vm->symbols;
96  if ( !sym ) {
97  return "NO SYMBOLS";
98  }
99 
100  // find the symbol
101  while ( sym->next && sym->next->symValue <= value ) {
102  sym = sym->next;
103  }
104 
105  if ( value == sym->symValue ) {
106  return sym->symName;
107  }
108 
109  Com_sprintf( text, sizeof( text ), "%s+%i", sym->symName, value - sym->symValue );
110 
111  return text;
112 }

186 /*
187 ===============
188 ParseHex
189 ===============
190 */
191 int ParseHex( const char *text ) {
192  int value;
193  int c;
194 
195  value = 0;
196  while ( ( c = *text++ ) != 0 ) {
197  if ( c >= '0' && c <= '9' ) {
198  value = value * 16 + c - '0';
199  continue;
200  }
201  if ( c >= 'a' && c <= 'f' ) {
202  value = value * 16 + 10 + c - 'a';
203  continue;
204  }
205  if ( c >= 'A' && c <= 'F' ) {
206  value = value * 16 + 10 + c - 'A';
207  continue;
208  }
209  }
210 
211  return value;
212 }

214 /*
215 ===============
216 VM_LoadSymbols
217 ===============
218 */
219 void VM_LoadSymbols( vm_t *vm ) {
220  union {
221  char *c;
222  void *v;
223  } mapfile;
224  char *text_p, *token;
225  char name[MAX_QPATH];
226  char symbols[MAX_QPATH];
227  vmSymbol_t **prev, *sym;
228  int count;
229  int value;
230  int chars;
231  int segment;
232  int numInstructions;
233 
234  // don't load symbols if not developer
235  if ( !com_developer->integer ) {
236  return;
237  }
238 
239  COM_StripExtension(vm->name, name, sizeof(name));
240  Com_sprintf( symbols, sizeof( symbols ), "vm/%s.map", name );
241  FS_ReadFile( symbols, &mapfile.v );
242  if ( !mapfile.c ) {
243  Com_Printf( "Couldn't load symbol file: %s\n", symbols );
244  return;
245  }
246 
247  numInstructions = vm->instructionCount;
248 
249  // parse the symbols
250  text_p = mapfile.c;
251  prev = &vm->symbols;
252  count = 0;
253 
254  while ( 1 ) {
255  token = COM_Parse( &text_p );
256  if ( !token[0] ) {
257  break;
258  }
259  segment = ParseHex( token );
260  if ( segment ) {
261  COM_Parse( &text_p );
262  COM_Parse( &text_p );
263  continue; // only load code segment values
264  }
265 
266  token = COM_Parse( &text_p );
267  if ( !token[0] ) {
268  Com_Printf( "WARNING: incomplete line at end of file\n" );
269  break;
270  }
271  value = ParseHex( token );
272 
273  token = COM_Parse( &text_p );
274  if ( !token[0] ) {
275  Com_Printf( "WARNING: incomplete line at end of file\n" );
276  break;
277  }
278  chars = strlen( token );
279  sym = Hunk_Alloc( sizeof( *sym ) + chars, h_high );
280  *prev = sym;
281  prev = &sym->next;
282  sym->next = NULL;
283 
284  // convert value from an instruction number to a code offset
285  if ( value >= 0 && value < numInstructions ) {
286  value = vm->instructionPointers[value];
287  }
288 
289  sym->symValue = value;
290  Q_strncpyz( sym->symName, token, chars + 1 );
291 
292  count++;
293  }
294 
295  vm->numSymbols = count;
296  Com_Printf( "%i symbols parsed from %s\n", count, symbols );
297  FS_FreeFile( mapfile.v );
298 }

300 /*
301 ============
302 VM_DllSyscall
303 
304 Dlls will call this directly
305 
306  rcg010206 The horror; the horror.
307 
308  The syscall mechanism relies on stack manipulation to get its args.
309  This is likely due to C's inability to pass "..." parameters to
310  a function in one clean chunk. On PowerPC Linux, these parameters
311  are not necessarily passed on the stack, so while (&arg[0] == arg)
312  is true, (&arg[1] == 2nd function parameter) is not necessarily
313  accurate, as arg's value might have been stored to the stack or
314  other piece of scratch memory to give it a valid address, but the
315  next parameter might still be sitting in a register.
316 
317  Quake's syscall system also assumes that the stack grows downward,
318  and that any needed types can be squeezed, safely, into a signed int.
319 
320  This hack below copies all needed values for an argument to a
321  array in memory, so that Quake can get the correct values. This can
322  also be used on systems where the stack grows upwards, as the
323  presumably standard and safe stdargs.h macros are used.
324 
325  As for having enough space in a signed int for your datatypes, well,
326  it might be better to wait for DOOM 3 before you start porting. :)
327 
328  The original code, while probably still inherently dangerous, seems
329  to work well enough for the platforms it already works on. Rather
330  than add the performance hit for those platforms, the original code
331  is still in use there.
332 
333  For speed, we just grab 15 arguments, and don't worry about exactly
334  how many the syscall actually needs; the extra is thrown away.
335 
336 ============
337 */
338 intptr_t QDECL VM_DllSyscall( intptr_t arg, ... ) {
339 #if !id386 || defined __clang__
340  // rcg010206 - see commentary above
341  intptr_t args[MAX_VMSYSCALL_ARGS];
342  int i;
343  va_list ap;
344 
345  args[0] = arg;
346 
347  va_start(ap, arg);
348  for (i = 1; i < ARRAY_LEN (args); i++)
349  args[i] = va_arg(ap, intptr_t);
350  va_end(ap);
351 
352  return currentVM->systemCall( args );
353 #else // original id code
354  return currentVM->systemCall( &arg );
355 #endif
356 }

361 VM_LoadQVM
362 
363 Load a .qvm file
364 =================
365 */
366 vmHeader_t *VM_LoadQVM( vm_t *vm, qboolean alloc, qboolean unpure)
367 {
368  int dataLength;
369  int i;
370  char filename[MAX_QPATH];
371  union {
372  vmHeader_t *h;
373  void *v;
374  } header;
375 
376  // load the image
377  Com_sprintf( filename, sizeof(filename), "vm/%s.qvm", vm->name );
378  Com_Printf( "Loading vm file %s...\n", filename );
379 
380  FS_ReadFileDir(filename, vm->searchPath, unpure, &header.v);
381 
382  if ( !header.h ) {
383  Com_Printf( "Failed.\n" );
384  VM_Free( vm );
385 
386  Com_Printf(S_COLOR_YELLOW "Warning: Couldn't open VM file %s\n", filename);
387 
388  return NULL;
389  }
390 
391  // show where the qvm was loaded from
392  FS_Which(filename, vm->searchPath);
393 
394  if( LittleLong( header.h->vmMagic ) == VM_MAGIC_VER2 ) {
395  Com_Printf( "...which has vmMagic VM_MAGIC_VER2\n" );
396 
397  // byte swap the header
398  for ( i = 0 ; i < sizeof( vmHeader_t ) / 4 ; i++ ) {
399  ((int *)header.h)[i] = LittleLong( ((int *)header.h)[i] );
400  }
401 
402  // validate
403  if ( header.h->jtrgLength < 0
404  || header.h->bssLength < 0
405  || header.h->dataLength < 0
406  || header.h->litLength < 0
407  || header.h->codeLength <= 0 )
408  {
409  VM_Free(vm);
410  FS_FreeFile(header.v);
411 
412  Com_Printf(S_COLOR_YELLOW "Warning: %s has bad header\n", filename);
413  return NULL;
414  }
415  } else if( LittleLong( header.h->vmMagic ) == VM_MAGIC ) {
416  // byte swap the header
417  // sizeof( vmHeader_t ) - sizeof( int ) is the 1.32b vm header size
418  for ( i = 0 ; i < ( sizeof( vmHeader_t ) - sizeof( int ) ) / 4 ; i++ ) {
419  ((int *)header.h)[i] = LittleLong( ((int *)header.h)[i] );
420  }
421 
422  // validate
423  if ( header.h->bssLength < 0
424  || header.h->dataLength < 0
425  || header.h->litLength < 0
426  || header.h->codeLength <= 0 )
427  {
428  VM_Free(vm);
429  FS_FreeFile(header.v);
430 
431  Com_Printf(S_COLOR_YELLOW "Warning: %s has bad header\n", filename);
432  return NULL;
433  }
434  } else {
435  VM_Free( vm );
436  FS_FreeFile(header.v);
437 
438  Com_Printf(S_COLOR_YELLOW "Warning: %s does not have a recognisable "
439  "magic number in its header\n", filename);
440  return NULL;
441  }
442 
443  // round up to next power of 2 so all data operations can
444  // be mask protected
445  dataLength = header.h->dataLength + header.h->litLength +
446  header.h->bssLength;
447  for ( i = 0 ; dataLength > ( 1 << i ) ; i++ ) {
448  }
449  dataLength = 1 << i;
450 
451  if(alloc)
452  {
453  // allocate zero filled space for initialized and uninitialized data
454  // leave some space beyond data mask so we can secure all mask operations
455  vm->dataAlloc = dataLength + 4;
456  vm->dataBase = Hunk_Alloc(vm->dataAlloc, h_high);
457  vm->dataMask = dataLength - 1;
458  }
459  else
460  {
461  // clear the data, but make sure we're not clearing more than allocated
462  if(vm->dataAlloc != dataLength + 4)
463  {
464  VM_Free(vm);
465  FS_FreeFile(header.v);
466 
467  Com_Printf(S_COLOR_YELLOW "Warning: Data region size of %s not matching after "
468  "VM_Restart()\n", filename);
469  return NULL;
470  }
471 
472  Com_Memset(vm->dataBase, 0, vm->dataAlloc);
473  }
474 
475  // copy the intialized data
476  Com_Memcpy( vm->dataBase, (byte *)header.h + header.h->dataOffset,
477  header.h->dataLength + header.h->litLength );
478 
479  // byte swap the longs
480  for ( i = 0 ; i < header.h->dataLength ; i += 4 ) {
481  *(int *)(vm->dataBase + i) = LittleLong( *(int *)(vm->dataBase + i ) );
482  }
483 
484  if(header.h->vmMagic == VM_MAGIC_VER2)
485  {
486  int previousNumJumpTableTargets = vm->numJumpTableTargets;
487 
488  header.h->jtrgLength &= ~0x03;
489 
490  vm->numJumpTableTargets = header.h->jtrgLength >> 2;
491  Com_Printf("Loading %d jump table targets\n", vm->numJumpTableTargets);
492 
493  if(alloc)
494  {
495  vm->jumpTableTargets = Hunk_Alloc(header.h->jtrgLength, h_high);
496  }
497  else
498  {
499  if(vm->numJumpTableTargets != previousNumJumpTableTargets)
500  {
501  VM_Free(vm);
502  FS_FreeFile(header.v);
503 
504  Com_Printf(S_COLOR_YELLOW "Warning: Jump table size of %s not matching after "
505  "VM_Restart()\n", filename);
506  return NULL;
507  }
508 
509  Com_Memset(vm->jumpTableTargets, 0, header.h->jtrgLength);
510  }
511 
512  Com_Memcpy(vm->jumpTableTargets, (byte *) header.h + header.h->dataOffset +
513  header.h->dataLength + header.h->litLength, header.h->jtrgLength);
514 
515  // byte swap the longs
516  for ( i = 0 ; i < header.h->jtrgLength ; i += 4 ) {
517  *(int *)(vm->jumpTableTargets + i) = LittleLong( *(int *)(vm->jumpTableTargets + i ) );
518  }
519  }
520 
521  return header.h;
522 }

568 /*
569 ================
570 VM_Create
571 
572 If image ends in .qvm it will be interpreted, otherwise
573 it will attempt to load as a system dll
574 ================
575 */
576 vm_t *VM_Create( const char *module, intptr_t (*systemCalls)(intptr_t *),
577  vmInterpret_t interpret ) {
578  vm_t *vm;
579  vmHeader_t *header;
580  int i, remaining, retval;
581  char filename[MAX_OSPATH];
582  void *startSearch = NULL;
583 
584  if ( !module || !module[0] || !systemCalls ) {
585  Com_Error( ERR_FATAL, "VM_Create: bad parms" );
586  }
587 
588  remaining = Hunk_MemoryRemaining();
589 
590  // see if we already have the VM
591  for ( i = 0 ; i < MAX_VM ; i++ ) {
592  if (!Q_stricmp(vmTable[i].name, module)) {
593  vm = &vmTable[i];
594  return vm;
595  }
596  }
597 
598  // find a free vm
599  for ( i = 0 ; i < MAX_VM ; i++ ) {
600  if ( !vmTable[i].name[0] ) {
601  break;
602  }
603  }
604 
605  if ( i == MAX_VM ) {
606  Com_Error( ERR_FATAL, "VM_Create: no free vm_t" );
607  }
608 
609  vm = &vmTable[i];
610 
611  Q_strncpyz(vm->name, module, sizeof(vm->name));
612 
613  do
614  {
615  retval = FS_FindVM(&startSearch, filename, sizeof(filename), module, (interpret == VMI_NATIVE));
616 
617  if(retval == VMI_NATIVE)
618  {
619  Com_Printf("Try loading dll file %s\n", filename);
620 
621  vm->dllHandle = Sys_LoadGameDll(filename, &vm->entryPoint, VM_DllSyscall);
622 
623  if(vm->dllHandle)
624  {
625  vm->systemCall = systemCalls;
626  return vm;
627  }
628 
629  Com_Printf("Failed loading dll, trying next\n");
630  }
631  else if(retval == VMI_COMPILED)
632  {
633  vm->searchPath = startSearch;
634  if((header = VM_LoadQVM(vm, qtrue, qfalse)))
635  break;
636 
637  // VM_Free overwrites the name on failed load
638  Q_strncpyz(vm->name, module, sizeof(vm->name));
639  }
640  } while(retval >= 0);
641 
642  if(retval < 0)
643  return NULL;
644 
645  vm->systemCall = systemCalls;
646 
647  // allocate space for the jump targets, which will be filled in by the compile/prep functions
648  vm->instructionCount = header->instructionCount;
649  vm->instructionPointers = Hunk_Alloc(vm->instructionCount * sizeof(*vm->instructionPointers), h_high);
650 
651  // copy or compile the instructions
652  vm->codeLength = header->codeLength;
653 
654  vm->compiled = qfalse;
655 
656 #ifdef NO_VM_COMPILED
657  if(interpret >= VMI_COMPILED) {
658  Com_Printf("Architecture doesn't have a bytecode compiler, using interpreter\n");
659  interpret = VMI_BYTECODE;
660  }
661 #else
662  if(interpret != VMI_BYTECODE)
663  {
664  vm->compiled = qtrue;
665  VM_Compile( vm, header );
666  }
667 #endif
668  // VM_Compile may have reset vm->compiled if compilation failed
669  if (!vm->compiled)
670  {
671  VM_PrepareInterpreter( vm, header );
672  }
673 
674  // free the original file
675  FS_FreeFile( header );
676 
677  // load the map file
678  VM_LoadSymbols( vm );
679 
680  // the stack is implicitly at the end of the image
681  vm->programStack = vm->dataMask + 1;
682  vm->stackBottom = vm->programStack - PROGRAM_STACK_SIZE;
683 
684  Com_Printf("%s loaded in %d bytes on the hunk\n", module, remaining - Hunk_MemoryRemaining());
685 
686  return vm;
687 }

785 VM_Call
786 
787 
788 Upon a system call, the stack will look like:
