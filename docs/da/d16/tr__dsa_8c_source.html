38 void GL_BindNullTextures(void)
39 {
40  int i;
41 
42  if (glRefConfig.directStateAccess)
43  {
44  for (i = 0; i < NUM_TEXTURE_BUNDLES; i++)
45  {
46  qglBindMultiTextureEXT(GL_TEXTURE0 + i, GL_TEXTURE_2D, 0);
47  glDsaState.textures[i] = 0;
48  }
49  }
50  else
51  {
52  for (i = 0; i < NUM_TEXTURE_BUNDLES; i++)
53  {
54  qglActiveTexture(GL_TEXTURE0 + i);
55  qglBindTexture(GL_TEXTURE_2D, 0);
56  glDsaState.textures[i] = 0;
57  }
58 
59  qglActiveTexture(GL_TEXTURE0);
60  glDsaState.texunit = GL_TEXTURE0;
61  }
62 }

64 int GL_BindMultiTexture(GLenum texunit, GLenum target, GLuint texture)
65 {
66  GLuint tmu = texunit - GL_TEXTURE0;
67 
68  if (glDsaState.textures[tmu] == texture)
69  return 0;
70 
71  if (target >= GL_TEXTURE_CUBE_MAP_POSITIVE_X && target <= GL_TEXTURE_CUBE_MAP_NEGATIVE_Z)
72  target = GL_TEXTURE_CUBE_MAP;
73 
74  qglBindMultiTextureEXT(texunit, target, texture);
75  glDsaState.textures[tmu] = texture;
76  return 1;
77 }

216 void GL_BindFramebuffer(GLenum target, GLuint framebuffer)
217 {
218  switch (target)
219  {
220  case GL_FRAMEBUFFER:
221  if (framebuffer != glDsaState.drawFramebuffer || framebuffer != glDsaState.readFramebuffer)
222  {
223  qglBindFramebuffer(target, framebuffer);
224  glDsaState.drawFramebuffer = glDsaState.readFramebuffer = framebuffer;
225  }
226  break;
227 
228  case GL_DRAW_FRAMEBUFFER:
229  if (framebuffer != glDsaState.drawFramebuffer)
230  {
231  qglBindFramebuffer(target, framebuffer);
232  glDsaState.drawFramebuffer = framebuffer;
233  }
234  break;
235 
236  case GL_READ_FRAMEBUFFER:
237  if (framebuffer != glDsaState.readFramebuffer)
238  {
239  qglBindFramebuffer(target, framebuffer);
240  glDsaState.readFramebuffer = framebuffer;
241  }
242  break;
243  }
244 }
