97 /*
98 ================
99 G_InitSessionData
100 
101 Called on a first-time connect
102 ================
103 */
104 void G_InitSessionData(gclient_t* client, char* userinfo)
105 {
106  clientSession_t* sess;
107  const char* value;
108 
109  sess = &client->sess;
110 
111  // check for team preference, mainly for bots
112  value = Info_ValueForKey(userinfo, "teampref");
113 
114  // check for human's team preference set by start server menu
115  if (!value[0] && g_localTeamPref.string[0] && client->pers.localClient) {
116  value = g_localTeamPref.string;
117 
118  // clear team so it's only used once
119  trap_Cvar_Set("g_localTeamPref", "");
120  }
121 
122  // initial team determination
123  if (g_gametype.integer >= GT_TEAM) {
124  // always spawn as spectator in team games
125  sess->sessionTeam = TEAM_SPECTATOR;
126  sess->spectatorState = SPECTATOR_FREE;
127 
128  if (value[0] || g_teamAutoJoin.integer) {
129  SetTeam(&g_entities[client - level.clients], value);
130  }
131  } else {
132  if (value[0] == 's') {
133  // a willing spectator, not a waiting-in-line
134  sess->sessionTeam = TEAM_SPECTATOR;
135  } else {
136  switch (g_gametype.integer) {
137  default:
138  case GT_FFA:
139  case GT_SINGLE_PLAYER:
140  if (g_maxGameClients.integer > 0 &&
141  level.numNonSpectatorClients >= g_maxGameClients.integer) {
142  sess->sessionTeam = TEAM_SPECTATOR;
143  } else {
144  sess->sessionTeam = TEAM_FREE;
145  }
146  break;
147  case GT_TOURNAMENT:
148  // if the game is full, go into a waiting mode
149  if (level.numNonSpectatorClients >= 2) {
150  sess->sessionTeam = TEAM_SPECTATOR;
151  } else {
152  sess->sessionTeam = TEAM_FREE;
153  }
154  break;
155  }
156  }
157 
158  sess->spectatorState = SPECTATOR_FREE;
159  }
160 
161  AddTournamentQueue(client);
162 
163  G_WriteClientSessionData(client);
164 }
