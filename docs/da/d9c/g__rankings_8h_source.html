27 /*
28 ==============================================================================
29 
30 Key digits:
31  10^9: report type
32  1 = normal
33  2 = developer-only
34  10^8: stat type
35  0 = match stat
36  1 = single player stat
37  2 = duel stat
38  10^7: data type
39  0 = string
40  1 = uint32
41  10^6: calculation
42  0 = use raw value
43  1 = add to total
44  2 = average
45  3 = max
46  4 = min
47  10^5
48  10^4: category
49  00 = general
50  01 = session
51  02 = weapon
52  03 = ammo
53  04 = health
54  05 = armor
55  06 = powerup
56  07 = holdable
57  08 = hazard
58  09 = reward
59  10 = teammate
60  11 = ctf
61  10^3:
62  10^2: sub-category
63  10^1:
64  10^0: ordinal
65 
66 ==============================================================================
67 */
