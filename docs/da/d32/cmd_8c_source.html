83 /*
84 ============
85 Cbuf_AddText
86 
87 Adds command text at the end of the buffer, does NOT add a final \n
88 ============
89 */
90 void Cbuf_AddText( const char *text ) {
91  int l;
92 
93  l = strlen (text);
94 
95  if (cmd_text.cursize + l >= cmd_text.maxsize)
96  {
97  Com_Printf ("Cbuf_AddText: overflow\n");
98  return;
99  }
100  Com_Memcpy(&cmd_text.data[cmd_text.cursize], text, l);
101  cmd_text.cursize += l;
102 }

105 /*
106 ============
107 Cbuf_InsertText
108 
109 Adds command text immediately after the current command
110 Adds a \n to the text
111 ============
112 */
113 void Cbuf_InsertText( const char *text ) {
114  int len;
115  int i;
116 
117  len = strlen( text ) + 1;
118  if ( len + cmd_text.cursize > cmd_text.maxsize ) {
119  Com_Printf( "Cbuf_InsertText overflowed\n" );
120  return;
121  }
122 
123  // move the existing command text
124  for ( i = cmd_text.cursize - 1 ; i >= 0 ; i-- ) {
125  cmd_text.data[ i + len ] = cmd_text.data[ i ];
126  }
127 
128  // copy the new text in
129  Com_Memcpy( cmd_text.data, text, len - 1 );
130 
131  // add a \n
132  cmd_text.data[ len - 1 ] = '\n';
133 
134  cmd_text.cursize += len;
135 }

138 /*
139 ============
140 Cbuf_ExecuteText
141 ============
142 */
143 void Cbuf_ExecuteText (int exec_when, const char *text)
144 {
145  switch (exec_when)
146  {
147  case EXEC_NOW:
148  if (text && strlen(text) > 0) {
149  Com_DPrintf(S_COLOR_YELLOW "EXEC_NOW %s\n", text);
150  Cmd_ExecuteString (text);
151  } else {
152  Cbuf_Execute();
153  Com_DPrintf(S_COLOR_YELLOW "EXEC_NOW %s\n", cmd_text.data);
154  }
155  break;
156  case EXEC_INSERT:
157  Cbuf_InsertText (text);
158  break;
159  case EXEC_APPEND:
160  Cbuf_AddText (text);
161  break;
162  default:
163  Com_Error (ERR_FATAL, "Cbuf_ExecuteText: bad exec_when");
164  }
165 }

167 /*
168 ============
169 Cbuf_Execute
170 ============
171 */
172 void Cbuf_Execute (void)
173 {
174  int i;
175  char *text;
176  char line[MAX_CMD_LINE];
177  int quotes;
178 
179  // This will keep // style comments all on one line by not breaking on
180  // a semicolon. It will keep /* ... */ style comments all on one line by not
181  // breaking it for semicolon or newline.
182  qboolean in_star_comment = qfalse;
183  qboolean in_slash_comment = qfalse;
184  while (cmd_text.cursize)
185  {
186  if ( cmd_wait > 0 ) {
187  // skip out while text still remains in buffer, leaving it
188  // for next frame
189  cmd_wait--;
190  break;
191  }
192 
193  // find a \n or ; line break or comment: // or /* */
194  text = (char *)cmd_text.data;
195 
196  quotes = 0;
197  for (i=0 ; i< cmd_text.cursize ; i++)
198  {
199  if (text[i] == '"')
200  quotes++;
201 
202  if ( !(quotes&1)) {
203  if (i < cmd_text.cursize - 1) {
204  if (! in_star_comment && text[i] == '/' && text[i+1] == '/')
205  in_slash_comment = qtrue;
206  else if (! in_slash_comment && text[i] == '/' && text[i+1] == '*')
207  in_star_comment = qtrue;
208  else if (in_star_comment && text[i] == '*' && text[i+1] == '/') {
209  in_star_comment = qfalse;
210  // If we are in a star comment, then the part after it is valid
211  // Note: This will cause it to NUL out the terminating '/'
212  // but ExecuteString doesn't require it anyway.
213  i++;
214  break;
215  }
216  }
217  if (! in_slash_comment && ! in_star_comment && text[i] == ';')
218  break;
219  }
220  if (! in_star_comment && (text[i] == '\n' || text[i] == '\r')) {
221  in_slash_comment = qfalse;
222  break;
223  }
224  }
225 
226  if( i >= (MAX_CMD_LINE - 1)) {
227  i = MAX_CMD_LINE - 1;
228  }
229 
230  Com_Memcpy (line, text, i);
231  line[i] = 0;
232 
233 // delete the text from the command buffer and move remaining commands down
234 // this is necessary because commands (exec) can insert data at the
235 // beginning of the text buffer
236 
237  if (i == cmd_text.cursize)
238  cmd_text.cursize = 0;
239  else
240  {
241  i++;
242  cmd_text.cursize -= i;
243  memmove (text, text+i, cmd_text.cursize);
244  }
245 
246 // execute the command line
247 
248  Cmd_ExecuteString (line);
249  }
250 }

288  return;
289  }
290  if (!quiet)
291  Com_Printf ("execing %s\n", filename);
292 
293  Cbuf_InsertText (f.c);
294 
295  FS_FreeFile (f.v);
296 }

487 /*
488 ============
489 Cmd_TokenizeString
490 
491 Parses the given string into command line tokens.
492 The text is copied to a separate buffer and 0 characters
493 are inserted in the appropriate place, The argv array
494 will point into this temporary buffer.
495 ============
496 */
497 // NOTE TTimo define that to track tokenization issues
498 //#define TKN_DBG
499 static void Cmd_TokenizeString2( const char *text_in, qboolean ignoreQuotes ) {
500  const char *text;
501  char *textOut;
502 
503 #ifdef TKN_DBG
504  // FIXME TTimo blunt hook to try to find the tokenization of userinfo
505  Com_DPrintf("Cmd_TokenizeString: %s\n", text_in);
506 #endif
507 
508  // clear previous args
509  cmd_argc = 0;
510 
511  if ( !text_in ) {
512  return;
513  }
514 
515  Q_strncpyz( cmd_cmd, text_in, sizeof(cmd_cmd) );
516 
517  text = text_in;
518  textOut = cmd_tokenized;
519 
520  while ( 1 ) {
521  if ( cmd_argc == MAX_STRING_TOKENS ) {
522  return; // this is usually something malicious
523  }
524 
525  while ( 1 ) {
526  // skip whitespace
527  while ( *text && *text <= ' ' ) {
528  text++;
529  }
530  if ( !*text ) {
531  return; // all tokens parsed
532  }
533 
534  // skip // comments
535  if ( text[0] == '/' && text[1] == '/' ) {
536  return; // all tokens parsed
537  }
538 
539  // skip /* */ comments
540  if ( text[0] == '/' && text[1] =='*' ) {
541  while ( *text && ( text[0] != '*' || text[1] != '/' ) ) {
542  text++;
543  }
544  if ( !*text ) {
545  return; // all tokens parsed
546  }
547  text += 2;
548  } else {
549  break; // we are ready to parse a token
550  }
551  }
552 
553  // handle quoted strings
554  // NOTE TTimo this doesn't handle \" escaping
555  if ( !ignoreQuotes && *text == '"' ) {
556  cmd_argv[cmd_argc] = textOut;
557  cmd_argc++;
558  text++;
559  while ( *text && *text != '"' ) {
560  *textOut++ = *text++;
561  }
562  *textOut++ = 0;
563  if ( !*text ) {
564  return; // all tokens parsed
565  }
566  text++;
567  continue;
568  }
569 
570  // regular token
571  cmd_argv[cmd_argc] = textOut;
572  cmd_argc++;
573 
574  // skip until whitespace, quote, or command
575  while ( *text > ' ' ) {
576  if ( !ignoreQuotes && text[0] == '"' ) {
577  break;
578  }
579 
580  if ( text[0] == '/' && text[1] == '/' ) {
581  break;
582  }
583 
584  // skip /* */ comments
585  if ( text[0] == '/' && text[1] =='*' ) {
586  break;
587  }
588 
589  *textOut++ = *text++;
590  }
591 
592  *textOut++ = 0;
593 
594  if ( !*text ) {
595  return; // all tokens parsed
596  }
597  }
598 
599 }

755 /*
756 ============
757 Cmd_ExecuteString
758 
759 A complete command line has been parsed, so try to execute it
760 ============
761 */
762 void Cmd_ExecuteString( const char *text ) {
763  cmd_function_t *cmd, **prev;
764 
765  // execute the command line
766  Cmd_TokenizeString( text );
767  if ( !Cmd_Argc() ) {
768  return; // no tokens
769  }
770 
771  // check registered command functions
772  for ( prev = &cmd_functions ; *prev ; prev = &cmd->next ) {
773  cmd = *prev;
774  if ( !Q_stricmp( cmd_argv[0],cmd->name ) ) {
775  // rearrange the links so that the command will be
776  // near the head of the list next time it is used
777  *prev = cmd->next;
778  cmd->next = cmd_functions;
779  cmd_functions = cmd;
780 
781  // perform the action
782  if ( !cmd->function ) {
783  // let the cgame or game handle it
784  break;
785  } else {
786  cmd->function ();
787  }
788  return;
789  }
790  }
791 
792  // check cvars
793  if ( Cvar_Command() ) {
794  return;
795  }
796 
797  // check client game commands
798  if ( com_cl_running && com_cl_running->integer && CL_GameCommand() ) {
799  return;
800  }
801 
802  // check server game commands
803  if ( com_sv_running && com_sv_running->integer && SV_GameCommand() ) {
804  return;
805  }
806 
807  // check ui commands
808  if ( com_cl_running && com_cl_running->integer && UI_GameCommand() ) {
809  return;
810  }
811 
812  // send it as a server command if we are connected
813  // this will usually result in a chat message
814  CL_ForwardCommandToServer ( text );
815 }
