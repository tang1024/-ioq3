177 void CL_cURL_Cleanup(void)
178 {
179  if(clc.downloadCURLM) {
180  CURLMcode result;
181 
182  if(clc.downloadCURL) {
183  result = qcurl_multi_remove_handle(clc.downloadCURLM,
184  clc.downloadCURL);
185  if(result != CURLM_OK) {
186  Com_DPrintf("qcurl_multi_remove_handle failed: %s\n", qcurl_multi_strerror(result));
187  }
188  qcurl_easy_cleanup(clc.downloadCURL);
189  }
190  result = qcurl_multi_cleanup(clc.downloadCURLM);
191  if(result != CURLM_OK) {
192  Com_DPrintf("CL_cURL_Cleanup: qcurl_multi_cleanup failed: %s\n", qcurl_multi_strerror(result));
193  }
194  clc.downloadCURLM = NULL;
195  clc.downloadCURL = NULL;
196  }
197  else if(clc.downloadCURL) {
198  qcurl_easy_cleanup(clc.downloadCURL);
199  clc.downloadCURL = NULL;
200  }
201 }

220 CURLcode qcurl_easy_setopt_warn(CURL *curl, CURLoption option, ...)
221 {
222  CURLcode result;
223 
224  va_list argp;
225  va_start(argp, option);
226 
227  if(option < CURLOPTTYPE_OBJECTPOINT) {
228  long longValue = va_arg(argp, long);
229  result = qcurl_easy_setopt(curl, option, longValue);
230  } else if(option < CURLOPTTYPE_OFF_T) {
231  void *pointerValue = va_arg(argp, void *);
232  result = qcurl_easy_setopt(curl, option, pointerValue);
233  } else {
234  curl_off_t offsetValue = va_arg(argp, curl_off_t);
235  result = qcurl_easy_setopt(curl, option, offsetValue);
236  }
237 
238  if(result != CURLE_OK) {
239  Com_DPrintf("qcurl_easy_setopt failed: %s\n", qcurl_easy_strerror(result));
240  }
241  va_end(argp);
242 
243  return result;
244 }

246 void CL_cURL_BeginDownload( const char *localName, const char *remoteURL )
247 {
248  CURLMcode result;
249 
250  clc.cURLUsed = qtrue;
251  Com_Printf("URL: %s\n", remoteURL);
252  Com_DPrintf("***** CL_cURL_BeginDownload *****\n"
253  "Localname: %s\n"
254  "RemoteURL: %s\n"
255  "****************************\n", localName, remoteURL);
256  CL_cURL_Cleanup();
257  Q_strncpyz(clc.downloadURL, remoteURL, sizeof(clc.downloadURL));
258  Q_strncpyz(clc.downloadName, localName, sizeof(clc.downloadName));
259  Com_sprintf(clc.downloadTempName, sizeof(clc.downloadTempName),
260  "%s.tmp", localName);
261 
262  // Set so UI gets access to it
263  Cvar_Set("cl_downloadName", localName);
264  Cvar_Set("cl_downloadSize", "0");
265  Cvar_Set("cl_downloadCount", "0");
266  Cvar_SetValue("cl_downloadTime", cls.realtime);
267 
268  clc.downloadBlock = 0; // Starting new file
269  clc.downloadCount = 0;
270 
271  clc.downloadCURL = qcurl_easy_init();
272  if(!clc.downloadCURL) {
273  Com_Error(ERR_DROP, "CL_cURL_BeginDownload: qcurl_easy_init() "
274  "failed");
275  return;
276  }
277  clc.download = FS_SV_FOpenFileWrite(clc.downloadTempName);
278  if(!clc.download) {
279  Com_Error(ERR_DROP, "CL_cURL_BeginDownload: failed to open "
280  "%s for writing", clc.downloadTempName);
281  return;
282  }
283 
284  if(com_developer->integer)
285  qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_VERBOSE, 1);
286  qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_URL, clc.downloadURL);
287  qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_TRANSFERTEXT, 0);
288  qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_REFERER, va("ioQ3://%s",
289  NET_AdrToString(clc.serverAddress)));
290  qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_USERAGENT, va("%s %s",
291  Q3_VERSION, qcurl_version()));
292  qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_WRITEFUNCTION,
293  CL_cURL_CallbackWrite);
294  qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_WRITEDATA, &clc.download);
295  qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_NOPROGRESS, 0);
296  qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_PROGRESSFUNCTION,
297  CL_cURL_CallbackProgress);
298  qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_PROGRESSDATA, NULL);
299  qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_FAILONERROR, 1);
300  qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_FOLLOWLOCATION, 1);
301  qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_MAXREDIRS, 5);
302  qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_PROTOCOLS,
303  CURLPROTO_HTTP | CURLPROTO_HTTPS | CURLPROTO_FTP | CURLPROTO_FTPS);
304  qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_BUFFERSIZE, CURL_MAX_READ_SIZE);
305  clc.downloadCURLM = qcurl_multi_init();
306  if(!clc.downloadCURLM) {
307  qcurl_easy_cleanup(clc.downloadCURL);
308  clc.downloadCURL = NULL;
309  Com_Error(ERR_DROP, "CL_cURL_BeginDownload: qcurl_multi_init() "
310  "failed");
311  return;
312  }
313  result = qcurl_multi_add_handle(clc.downloadCURLM, clc.downloadCURL);
314  if(result != CURLM_OK) {
315  qcurl_easy_cleanup(clc.downloadCURL);
316  clc.downloadCURL = NULL;
317  Com_Error(ERR_DROP,"CL_cURL_BeginDownload: qcurl_multi_add_handle() failed: %s", qcurl_multi_strerror(result));
318  return;
319  }
320 
321  if(!(clc.sv_allowDownload & DLF_NO_DISCONNECT) &&
322  !clc.cURLDisconnected) {
323 
324  CL_AddReliableCommand("disconnect", qtrue);
325  CL_WritePacket();
326  CL_WritePacket();
327  CL_WritePacket();
328  clc.cURLDisconnected = qtrue;
329  }
330 }

332 void CL_cURL_PerformDownload(void)
333 {
334  CURLMcode res;
335  CURLMsg *msg;
336  int c;
337  int i = 0;
338 
339  res = qcurl_multi_perform(clc.downloadCURLM, &c);
340  while(res == CURLM_CALL_MULTI_PERFORM && i < 100) {
341  res = qcurl_multi_perform(clc.downloadCURLM, &c);
342  i++;
343  }
344  if(res == CURLM_CALL_MULTI_PERFORM)
345  return;
346  msg = qcurl_multi_info_read(clc.downloadCURLM, &c);
347  if(msg == NULL) {
348  return;
349  }
350  FS_FCloseFile(clc.download);
351  if(msg->msg == CURLMSG_DONE && msg->data.result == CURLE_OK) {
352  FS_SV_Rename(clc.downloadTempName, clc.downloadName, qfalse);
353  clc.downloadRestart = qtrue;
354  }
355  else {
356  long code;
357 
358  qcurl_easy_getinfo(msg->easy_handle, CURLINFO_RESPONSE_CODE,
359  &code);
360  Com_Error(ERR_DROP, "Download Error: %s Code: %ld URL: %s",
361  qcurl_easy_strerror(msg->data.result),
362  code, clc.downloadURL);
363  }
364 
365  CL_NextDownload();
366 }
