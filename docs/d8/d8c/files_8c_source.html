384 static long FS_HashFileName(const char* fname, int hashSize)
385 {
386  int i;
387  long hash;
388  char letter;
389 
390  hash = 0;
391  i = 0;
392  while (fname[i] != '\0') {
393  letter = tolower(fname[i]);
394  if (letter == '.') break; // don't include extension
395  if (letter == '\\') letter = '/'; // damn path names
396  if (letter == PATH_SEP) letter = '/'; // damn path names
397  hash += (long)(letter) * (i + 119);
398  i++;
399  }
400  hash = (hash ^ (hash >> 10) ^ (hash >> 20));
401  hash &= (hashSize - 1);
402  return hash;
403 }
