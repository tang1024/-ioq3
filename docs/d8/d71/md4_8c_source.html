134 static void mdfour_tail(byte *in, int n)
135 {

161 static void mdfour_update(struct mdfour *md, byte *in, int n)
162 {
163  uint32_t M[16];
164 
165  m = md;
166 
167  if (n == 0) mdfour_tail(in, n);
168 
169  while (n >= 64) {
170  copy64(M, in);
171  mdfour64(M);
172  in += 64;
173  n -= 64;
174  m->totalN += 64;
175  }
176 
177  mdfour_tail(in, n);
178 }

189 static void mdfour(byte *out, byte *in, int n)
190 {
