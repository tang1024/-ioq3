79 =================
80 */
81 void UI_PushMenu( menuframework_s *menu )
82 {
83  int i;
84  menucommon_s* item;
85 
86  // avoid stacking menus invoked by hotkeys
87  for (i=0 ; i<uis.menusp ; i++)
88  {
89  if (uis.stack[i] == menu)
90  {
91  uis.menusp = i;
92  break;
93  }
94  }
95 
96  if (i == uis.menusp)
97  {
98  if (uis.menusp >= MAX_MENUDEPTH)
99  trap_Error("UI_PushMenu: menu stack overflow");
100 
101  uis.stack[uis.menusp++] = menu;
102  }
103 
104  uis.activemenu = menu;
105 
106  // default cursor position
107  menu->cursor = 0;
108  menu->cursor_prev = 0;
109 
110  m_entersound = qtrue;
111 
112  trap_Key_SetCatcher( KEYCATCH_UI );
113 
114  // force first available item to have focus
115  for (i=0; i<menu->nitems; i++)
116  {
117  item = (menucommon_s *)menu->items[i];
118  if (!(item->flags & (QMF_GRAYED|QMF_MOUSEONLY|QMF_INACTIVE)))
119  {
120  menu->cursor_prev = -1;
121  Menu_SetCursor( menu, i );
122  break;
123  }
124  }
125 
126  uis.firstdraw = qtrue;
127 }

338 static void UI_DrawBannerString2( int x, int y, const char* str, vec4_t color )
339 {
340  const char* s;
341  unsigned char ch;
342  float ax;
343  float ay;
344  float aw;
345  float ah;
346  float frow;
347  float fcol;
348  float fwidth;
349  float fheight;
350 
351  // draw the colored text
352  trap_R_SetColor( color );
353 
354  ax = x * uis.xscale + uis.bias;
355  ay = y * uis.yscale;
356 
357  s = str;
358  while ( *s )
359  {
360  ch = *s & 127;
361  if ( ch == ' ' ) {
362  ax += ((float)PROPB_SPACE_WIDTH + (float)PROPB_GAP_WIDTH)* uis.xscale;
363  }
364  else if ( ch >= 'A' && ch <= 'Z' ) {
365  ch -= 'A';
366  fcol = (float)propMapB[ch][0] / 256.0f;
367  frow = (float)propMapB[ch][1] / 256.0f;

370  aw = (float)propMapB[ch][2] * uis.xscale;
371  ah = (float)PROPB_HEIGHT * uis.yscale;

373  ax += (aw + (float)PROPB_GAP_WIDTH * uis.xscale);
374  }
375  s++;
376  }
377 
378  trap_R_SetColor( NULL );

572 UI_DrawProportionalString_Wrapped

575 void UI_DrawProportionalString_AutoWrapped( int x, int y, int xmax, int ystep, const char* str, int style, vec4_t color ) {
576  int width;
577  char *s1,*s2,*s3;
578  char c_bcp;
579  char buf[1024];
580  float sizeScale;
581 
582  if (!str || str[0]=='\0')
583  return;
584 
585  sizeScale = UI_ProportionalSizeScale( style );
586 
587  Q_strncpyz(buf, str, sizeof(buf));
588  s1 = s2 = s3 = buf;
589 
590  while (1) {
591  do {
592  s3++;
593  } while (*s3!=' ' && *s3!='\0');
594  c_bcp = *s3;
595  *s3 = '\0';
596  width = UI_ProportionalStringWidth(s1) * sizeScale;
597  *s3 = c_bcp;
598  if (width > xmax) {
599  if (s1==s2)
600  {
601  // fuck, don't have a clean cut, we'll overflow
602  s2 = s3;
603  }
604  *s2 = '\0';
605  UI_DrawProportionalString(x, y, s1, style, color);
606  y += ystep;
607  if (c_bcp == '\0')
608  {
609  // that was the last word
610  // we could start a new loop, but that wouldn't be much use
611  // even if the word is too long, we would overflow it (see above)
612  // so just print it now if needed
613  s2++;
614  if (*s2 != '\0') // if we are printing an overflowing line we have s2 == s3
615  UI_DrawProportionalString(x, y, s2, style, color);
616  break;
617  }
618  s2++;
619  s1 = s2;
620  s3 = s2;
621  }
622  else
623  {
624  s2 = s3;
625  if (c_bcp == '\0') // we reached the end
626  {
627  UI_DrawProportionalString(x, y, s1, style, color);
628  break;
629  }
630  }
631  }
632 }

876 void UI_MouseEvent( int dx, int dy )
877 {

879  int bias;
880  menucommon_s* m;
881 
882  if (!uis.activemenu)
883  return;
884 
885  // convert X bias to 640 coords
886  bias = uis.bias / uis.xscale;
887 
888  // update mouse screen position
889  uis.cursorx += dx;
890  if (uis.cursorx < -bias)
891  uis.cursorx = -bias;

899  uis.cursory = SCREEN_HEIGHT;

902  for (i=0; i<uis.activemenu->nitems; i++)
903  {
904  m = (menucommon_s*)uis.activemenu->items[i];
905 
906  if (m->flags & (QMF_GRAYED|QMF_INACTIVE))
907  continue;
908 
909  if ((uis.cursorx < m->left) ||
910  (uis.cursorx > m->right) ||
911  (uis.cursory < m->top) ||
912  (uis.cursory > m->bottom))
913  {
914  // cursor out of item bounds
915  continue;
916  }
917 
918  // set focus to item at cursor
919  if (uis.activemenu->cursor != i)
920  {
921  Menu_SetCursor( uis.activemenu, i );
922  ((menucommon_s*)(uis.activemenu->items[uis.activemenu->cursor_prev]))->flags &= ~QMF_HASMOUSEFOCUS;
923 
924  if ( !(((menucommon_s*)(uis.activemenu->items[uis.activemenu->cursor]))->flags & QMF_SILENT ) ) {
925  trap_S_StartLocalSound( menu_move_sound, CHAN_LOCAL_SOUND );
926  }
927  }
928 
929  ((menucommon_s*)(uis.activemenu->items[uis.activemenu->cursor]))->flags |= QMF_HASMOUSEFOCUS;
930  return;
931  }
932 
933  if (uis.activemenu->nitems > 0) {
934  // out of any region
935  ((menucommon_s*)(uis.activemenu->items[uis.activemenu->cursor]))->flags &= ~QMF_HASMOUSEFOCUS;
936  }
