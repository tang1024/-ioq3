27 /*
28 ===============
29 G_DamageFeedback
30 
31 Called just before a snapshot is sent to the given player.
32 Totals up all damage and generates both the player_state_t
33 damage values to that client for pain blends and kicks, and
34 global pain sound events for all clients.
35 ===============
36 */
37 void P_DamageFeedback(gentity_t* player)
38 {
39  gclient_t* client;
40  float count;
41  vec3_t angles;
42 
43  client = player->client;
44  if (client->ps.pm_type == PM_DEAD) {
45  return;
46  }
47 
48  // total points of damage shot at the player this frame
49  count = client->damage_blood + client->damage_armor;
50  if (count == 0) {
51  return; // didn't take any damage
52  }
53 
54  if (count > 255) {
55  count = 255;
56  }
57 
58  // send the information to the client
59 
60  // world damage (falling, slime, etc) uses a special code
61  // to make the blend blob centered instead of positional
62  if (client->damage_fromWorld) {
63  client->ps.damagePitch = 255;
64  client->ps.damageYaw = 255;
65 
66  client->damage_fromWorld = qfalse;
67  } else {
68  vectoangles(client->damage_from, angles);
69  client->ps.damagePitch = angles[PITCH] / 360.0 * 256;
70  client->ps.damageYaw = angles[YAW] / 360.0 * 256;
71  }
72 
73  // play an appropriate pain sound
74  if ((level.time > player->pain_debounce_time) && !(player->flags & FL_GODMODE)) {
75  player->pain_debounce_time = level.time + 700;
76  G_AddEvent(player, EV_PAIN, player->health);
77  client->ps.damageEvent++;
78  }
79 
80 
81  client->ps.damageCount = count;
82 
83  //
84  // clear totals
85  //
86  client->damage_blood = 0;
87  client->damage_armor = 0;
88  client->damage_knockback = 0;
89 }

93 /*
94 =============
95 P_WorldEffects
96 
97 Check for lava / slime contents and drowning
98 =============
99 */
100 void P_WorldEffects(gentity_t* ent)
101 {
102  qboolean envirosuit;
103  int waterlevel;
104 
105  if (ent->client->noclip) {
106  ent->client->airOutTime = level.time + 12000; // don't need air
107  return;
108  }
109 
110  waterlevel = ent->waterlevel;
111 
112  envirosuit = ent->client->ps.powerups[PW_BATTLESUIT] > level.time;
113 
114  //
115  // check for drowning
116  //
117  if (waterlevel == 3) {
118  // envirosuit give air
119  if (envirosuit) {
120  ent->client->airOutTime = level.time + 10000;
121  }
122 
123  // if out of air, start drowning
124  if (ent->client->airOutTime < level.time) {
125  // drown!
126  ent->client->airOutTime += 1000;
127  if (ent->health > 0) {
128  // take more damage the longer underwater
129  ent->damage += 2;
130  if (ent->damage > 15)
131  ent->damage = 15;
132 
133  // don't play a normal pain sound
134  ent->pain_debounce_time = level.time + 200;
135 
136  G_Damage(ent, NULL, NULL, NULL, NULL,
137  ent->damage, DAMAGE_NO_ARMOR, MOD_WATER);
138  }
139  }
140  } else {
141  ent->client->airOutTime = level.time + 12000;
142  ent->damage = 2;
143  }
144 
145  //
146  // check for sizzle damage (move to pmove?)
147  //
148  if (waterlevel &&
149  (ent->watertype & (CONTENTS_LAVA | CONTENTS_SLIME))) {
150  if (ent->health > 0
151  && ent->pain_debounce_time <= level.time) {
152 
153  if (envirosuit) {
154  G_AddEvent(ent, EV_POWERUP_BATTLESUIT, 0);
155  } else {
156  if (ent->watertype & CONTENTS_LAVA) {
157  G_Damage(ent, NULL, NULL, NULL, NULL,
158  30 * waterlevel, 0, MOD_LAVA);
159  }
160 
161  if (ent->watertype & CONTENTS_SLIME) {
162  G_Damage(ent, NULL, NULL, NULL, NULL,
163  10 * waterlevel, 0, MOD_SLIME);
164  }
165  }
166  }
167  }

231 /*
232 ============
233 G_TouchTriggers
234 
235 Find all trigger entities that ent's current position touches.
236 Spectators will only interact with teleporters.
237 ============
238 */
239 void G_TouchTriggers(gentity_t* ent)
240 {
241  int i, num;
242  int touch[MAX_GENTITIES];
243  gentity_t* hit;
244  trace_t trace;
245  vec3_t mins, maxs;
246  static vec3_t range = { 40, 40, 52 };
247 
248  if (!ent->client) {
249  return;
250  }
251 
252  // dead clients don't activate triggers!
253  if (ent->client->ps.stats[STAT_HEALTH] <= 0) {
254  return;
255  }
256 
257  VectorSubtract(ent->client->ps.origin, range, mins);
258  VectorAdd(ent->client->ps.origin, range, maxs);
259 
260  num = trap_EntitiesInBox(mins, maxs, touch, MAX_GENTITIES);
261 
262  // can't use ent->absmin, because that has a one unit pad
263  VectorAdd(ent->client->ps.origin, ent->r.mins, mins);
264  VectorAdd(ent->client->ps.origin, ent->r.maxs, maxs);
265 
266  for (i = 0; i < num; i++) {
267  hit = &g_entities[touch[i]];
268 
269  if (!hit->touch && !ent->touch) {
270  continue;
271  }
272  if (!(hit->r.contents & CONTENTS_TRIGGER)) {
273  continue;
274  }
275 
276  // ignore most entities if a spectator
277  if (ent->client->sess.sessionTeam == TEAM_SPECTATOR) {
278  if (hit->s.eType != ET_TELEPORT_TRIGGER &&
279  // this is ugly but adding a new ET_? type will
280  // most likely cause network incompatibilities
281  hit->touch != Touch_DoorTrigger) {
282  continue;
283  }
284  }
285 
286  // use separate code for determining if an item is picked up
287  // so you don't have to actually contact its bounding box
288  if (hit->s.eType == ET_ITEM) {
289  if (!BG_PlayerTouchesItem(&ent->client->ps, &hit->s, level.time)) {
290  continue;
291  }
292  } else {
293  if (!trap_EntityContact(mins, maxs, hit)) {
294  continue;
295  }
296  }
297 
298  memset(&trace, 0, sizeof(trace));
299 
300  if (hit->touch) {
301  hit->touch(hit, ent, &trace);
302  }
303 
304  if ((ent->r.svFlags & SVF_BOT) && (ent->touch)) {
305  ent->touch(ent, hit, &trace);
306  }
307  }
308 
309  // if we didn't touch a jump pad this pmove frame
310  if (ent->client->ps.jumppad_frame != ent->client->ps.pmove_framecount) {
311  ent->client->ps.jumppad_frame = 0;
312  ent->client->ps.jumppad_ent = 0;
313  }
314 }

344  pm.cmd = *ucmd;
345  pm.tracemask = MASK_PLAYERSOLID & ~CONTENTS_BODY; // spectators can fly through bodies
346  pm.trace = trap_Trace;
347  pm.pointcontents = trap_PointContents;
348 
349  // perform a pmove
350  Pmove(&pm);
351  // save results of pmove
352  VectorCopy(client->ps.origin, ent->s.origin);
353 
354  G_TouchTriggers(ent);
355  trap_UnlinkEntity(ent);
356  }
357 
358  client->oldbuttons = client->buttons;
359  client->buttons = ucmd->buttons;
360 
361  // attack button cycles through spectators
362  if ((client->buttons & BUTTON_ATTACK) && !(client->oldbuttons & BUTTON_ATTACK)) {
363  Cmd_FollowCycle_f(ent, 1);

375 {

380  client->inactivityWarning = qfalse;
381  } else if (client->pers.cmd.forwardmove ||

383  client->pers.cmd.upmove ||
384  (client->pers.cmd.buttons & BUTTON_ATTACK)) {

386  client->inactivityWarning = qfalse;
387  } else if (!client->pers.localClient) {

389  trap_DropClient(client - level.clients, "Dropped due to inactivity");

391  }

393  client->inactivityWarning = qtrue;
394  trap_SendServerCommand(client - level.clients, "cp \"Ten seconds until inactivity drop!\n\"");
395  }
396  }
397  return qtrue;

402 ClientTimerActions
403 
404 Actions that happen once a second
405 ==================
406 */
407 void ClientTimerActions(gentity_t* ent, int msec)
408 {
409  gclient_t* client;
410 #ifdef MISSIONPACK
411  int maxHealth;
412 #endif
413 
414  client = ent->client;
415  client->timeResidual += msec;
416 
417  while (client->timeResidual >= 1000) {
418  client->timeResidual -= 1000;
419 
420  // regenerate
421 #ifdef MISSIONPACK
422  if (bg_itemlist[client->ps.stats[STAT_PERSISTANT_POWERUP]].giTag == PW_GUARD) {
423  maxHealth = client->ps.stats[STAT_MAX_HEALTH] / 2;
424  } else if (client->ps.powerups[PW_REGEN]) {
425  maxHealth = client->ps.stats[STAT_MAX_HEALTH];
426  } else {
427  maxHealth = 0;
428  }
429  if (maxHealth) {
430  if (ent->health < maxHealth) {
431  ent->health += 15;
432  if (ent->health > maxHealth * 1.1) {
433  ent->health = maxHealth * 1.1;
434  }
435  G_AddEvent(ent, EV_POWERUP_REGEN, 0);
436  } else if (ent->health < maxHealth * 2) {
437  ent->health += 5;
438  if (ent->health > maxHealth * 2) {
439  ent->health = maxHealth * 2;
440  }
441  G_AddEvent(ent, EV_POWERUP_REGEN, 0);
442  }
443 #else
444  if (client->ps.powerups[PW_REGEN]) {
445  if (ent->health < client->ps.stats[STAT_MAX_HEALTH]) {
446  ent->health += 15;
447  if (ent->health > client->ps.stats[STAT_MAX_HEALTH] * 1.1) {
448  ent->health = client->ps.stats[STAT_MAX_HEALTH] * 1.1;
449  }
450  G_AddEvent(ent, EV_POWERUP_REGEN, 0);
451  } else if (ent->health < client->ps.stats[STAT_MAX_HEALTH] * 2) {
452  ent->health += 5;
453  if (ent->health > client->ps.stats[STAT_MAX_HEALTH] * 2) {
454  ent->health = client->ps.stats[STAT_MAX_HEALTH] * 2;
455  }
456  G_AddEvent(ent, EV_POWERUP_REGEN, 0);
457  }
458 #endif
459  } else {
460  // count down health when over max
461  if (ent->health > client->ps.stats[STAT_MAX_HEALTH]) {
462  ent->health--;
463  }
464  }
465 
466  // count down armor when over max
467  if (client->ps.stats[STAT_ARMOR] > client->ps.stats[STAT_MAX_HEALTH]) {
468  client->ps.stats[STAT_ARMOR]--;
469  }
470  }
471 #ifdef MISSIONPACK
472  if (bg_itemlist[client->ps.stats[STAT_PERSISTANT_POWERUP]].giTag == PW_AMMOREGEN) {
473  int w, max, inc, t, i;
474  int weapList[] = { WP_MACHINEGUN,WP_SHOTGUN,WP_GRENADE_LAUNCHER,WP_ROCKET_LAUNCHER,WP_LIGHTNING,WP_RAILGUN,WP_PLASMAGUN,WP_BFG,WP_NAILGUN,WP_PROX_LAUNCHER,WP_CHAINGUN };
475  int weapCount = ARRAY_LEN(weapList);
476  //
477  for (i = 0; i < weapCount; i++) {
478  w = weapList[i];
479 
480  switch (w) {
481  case WP_MACHINEGUN: max = 50; inc = 4; t = 1000; break;
482  case WP_SHOTGUN: max = 10; inc = 1; t = 1500; break;
483  case WP_GRENADE_LAUNCHER: max = 10; inc = 1; t = 2000; break;
484  case WP_ROCKET_LAUNCHER: max = 10; inc = 1; t = 1750; break;
485  case WP_LIGHTNING: max = 50; inc = 5; t = 1500; break;
486  case WP_RAILGUN: max = 10; inc = 1; t = 1750; break;
487  case WP_PLASMAGUN: max = 50; inc = 5; t = 1500; break;
488  case WP_BFG: max = 10; inc = 1; t = 4000; break;
489  case WP_NAILGUN: max = 10; inc = 1; t = 1250; break;
490  case WP_PROX_LAUNCHER: max = 5; inc = 1; t = 2000; break;
491  case WP_CHAINGUN: max = 100; inc = 5; t = 1000; break;
492  default: max = 0; inc = 0; t = 1000; break;
493  }
494  client->ammoTimes[w] += msec;
495  if (client->ps.ammo[w] >= max) {
496  client->ammoTimes[w] = 0;
497  }
498  if (client->ammoTimes[w] >= t) {
499  while (client->ammoTimes[w] >= t)
500  client->ammoTimes[w] -= t;
501  client->ps.ammo[w] += inc;
502  if (client->ps.ammo[w] > max) {
503  client->ps.ammo[w] = max;
504  }
505  }
506  }
507  }
508 #endif
509 }

637 
638  break;
639 
640 #ifdef MISSIONPACK
641  case EV_USE_ITEM3: // kamikaze
642  // make sure the invulnerability is off
643  ent->client->invulnerabilityTime = 0;
644  // start the kamikze
