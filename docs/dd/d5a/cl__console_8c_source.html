173 /*
174 ================
175 Con_Dump_f
176 
177 Save the console contents out to a file
178 ================
179 */
180 void Con_Dump_f (void)
181 {
182  int l, x, i;
183  short *line;
184  fileHandle_t f;
185  int bufferlen;
186  char *buffer;
187  char filename[MAX_QPATH];
188 
189  if (Cmd_Argc() != 2)
190  {
191  Com_Printf ("usage: condump <filename>\n");
192  return;
193  }
194 
195  Q_strncpyz( filename, Cmd_Argv( 1 ), sizeof( filename ) );
196  COM_DefaultExtension( filename, sizeof( filename ), ".txt" );
197 
198  if (!COM_CompareExtension(filename, ".txt"))
199  {
200  Com_Printf("Con_Dump_f: Only the \".txt\" extension is supported by this command!\n");
201  return;
202  }
203 
204  f = FS_FOpenFileWrite( filename );
205  if (!f)
206  {
207  Com_Printf ("ERROR: couldn't open %s.\n", filename);
208  return;
209  }
210 
211  Com_Printf ("Dumped console text to %s.\n", filename );
212 
213  // skip empty lines
214  for (l = con.current - con.totallines + 1 ; l <= con.current ; l++)
215  {
216  line = con.text + (l%con.totallines)*con.linewidth;
217  for (x=0 ; x<con.linewidth ; x++)
218  if ((line[x] & 0xff) != ' ')
219  break;
220  if (x != con.linewidth)
221  break;
222  }
223 
224 #ifdef _WIN32
225  bufferlen = con.linewidth + 3 * sizeof ( char );
226 #else
227  bufferlen = con.linewidth + 2 * sizeof ( char );
228 #endif
229 
230  buffer = Hunk_AllocateTempMemory( bufferlen );
231 
232  // write the remaining lines
233  buffer[bufferlen-1] = 0;
234  for ( ; l <= con.current ; l++)
235  {
236  line = con.text + (l%con.totallines)*con.linewidth;
237  for(i=0; i<con.linewidth; i++)
238  buffer[i] = line[i] & 0xff;
239  for (x=con.linewidth-1 ; x>=0 ; x--)
240  {
241  if (buffer[x] == ' ')
242  buffer[x] = 0;
243  else
244  break;
245  }
246 #ifdef _WIN32
247  Q_strcat(buffer, bufferlen, "\r\n");
248 #else
249  Q_strcat(buffer, bufferlen, "\n");
250 #endif
251  FS_Write(buffer, strlen(buffer), f);
252  }
253 
254  Hunk_FreeTempMemory( buffer );
255  FS_FCloseFile( f );
256 }

274 /*
275 ================
276 Con_CheckResize
277 
278 If the line width has changed, reformat the buffer.
279 ================
280 */
281 void Con_CheckResize (void)
282 {
283  int i, j, width, oldwidth, oldtotallines, numlines, numchars;
284  short tbuf[CON_TEXTSIZE];
285 
286  width = (SCREEN_WIDTH / SMALLCHAR_WIDTH) - 2;
287 
288  if (width == con.linewidth)
289  return;
290 
291  if (width < 1) // video hasn't been initialized yet
292  {
293  width = DEFAULT_CONSOLE_WIDTH;
294  con.linewidth = width;
295  con.totallines = CON_TEXTSIZE / con.linewidth;
296  for(i=0; i<CON_TEXTSIZE; i++)
297 
298  con.text[i] = (ColorIndex(COLOR_WHITE)<<8) | ' ';
299  }
300  else
301  {
302  oldwidth = con.linewidth;
303  con.linewidth = width;
304  oldtotallines = con.totallines;
305  con.totallines = CON_TEXTSIZE / con.linewidth;
306  numlines = oldtotallines;
307 
308  if (con.totallines < numlines)
309  numlines = con.totallines;
310 
311  numchars = oldwidth;
312 
313  if (con.linewidth < numchars)
314  numchars = con.linewidth;
315 
316  Com_Memcpy (tbuf, con.text, CON_TEXTSIZE * sizeof(short));
317  for(i=0; i<CON_TEXTSIZE; i++)
318 
319  con.text[i] = (ColorIndex(COLOR_WHITE)<<8) | ' ';
320 
321 
322  for (i=0 ; i<numlines ; i++)
323  {
324  for (j=0 ; j<numchars ; j++)
325  {
326  con.text[(con.totallines - 1 - i) * con.linewidth + j] =
327  tbuf[((con.current - i + oldtotallines) %
328  oldtotallines) * oldwidth + j];
329  }
330  }
331 
332  Con_ClearNotify ();
333  }
334 
335  con.current = con.totallines - 1;
336  con.display = con.current;
337 }

400 ===============
401 Con_Linefeed
402 ===============
403 */
404 void Con_Linefeed (qboolean skipnotify)
405 {
406  int i;
407 
408  // mark time for transparent overlay
409  if (con.current >= 0)
410  {
411  if (skipnotify)
412  con.times[con.current % NUM_CON_TIMES] = 0;
413  else
414  con.times[con.current % NUM_CON_TIMES] = cls.realtime;
415  }
416 
417  con.x = 0;
418  if (con.display == con.current)
419  con.display++;
420  con.current++;
421  for(i=0; i<con.linewidth; i++)
422  con.text[(con.current%con.totallines)*con.linewidth+i] = (ColorIndex(COLOR_WHITE)<<8) | ' ';
423 }

425 /*
426 ================
427 CL_ConsolePrint
428 
429 Handles cursor positioning, line wrapping, etc
430 All console printing must go through this in order to be logged to disk
431 If no console is visible, the text will appear at the top of the game window
432 ================
433 */
434 void CL_ConsolePrint( char *txt ) {
435  int y, l;
436  unsigned char c;
437  unsigned short color;
438  qboolean skipnotify = qfalse; // NERVE - SMF
439  int prev; // NERVE - SMF
440 
441  // TTimo - prefix for text that shows up in console but not in notify
442  // backported from RTCW
443  if ( !Q_strncmp( txt, "[skipnotify]", 12 ) ) {
444  skipnotify = qtrue;
445  txt += 12;
446  }
447 
448  // for some demos we don't want to ever show anything on the console
449  if ( cl_noprint && cl_noprint->integer ) {
450  return;
451  }
452 
453  if (!con.initialized) {
454  con.color[0] =
455  con.color[1] =
456  con.color[2] =
457  con.color[3] = 1.0f;
458  con.linewidth = -1;
459  Con_CheckResize ();
460  con.initialized = qtrue;
461  }
462 
463  color = ColorIndex(COLOR_WHITE);
464 
465  while ( (c = *((unsigned char *) txt)) != 0 ) {
466  if ( Q_IsColorString( txt ) ) {
467  color = ColorIndex( *(txt+1) );
468  txt += 2;
469  continue;
470  }
471 
472  // count word length
473  for (l=0 ; l< con.linewidth ; l++) {
474  if ( txt[l] <= ' ') {
475  break;
476  }
477 
478  }
479 
480  // word wrap
481  if (l != con.linewidth && (con.x + l >= con.linewidth) ) {
482  Con_Linefeed(skipnotify);
483 
484  }
485 
486  txt++;
487 
488  switch (c)
489  {
490  case '\n':
491  Con_Linefeed (skipnotify);
492  break;
493  case '\r':
494  con.x = 0;
495  break;
496  default: // display character and advance
497  y = con.current % con.totallines;
498  con.text[y*con.linewidth+con.x] = (color << 8) | c;
499  con.x++;
500  if(con.x >= con.linewidth)
501  Con_Linefeed(skipnotify);
502  break;
503  }
504  }
505 
506 
507  // mark time for transparent overlay
508  if (con.current >= 0) {
509  // NERVE - SMF
510  if ( skipnotify ) {
511  prev = con.current % NUM_CON_TIMES - 1;
512  if ( prev < 0 )
513  prev = NUM_CON_TIMES - 1;
514  con.times[prev] = 0;
515  }
516  else
517  // -NERVE - SMF
518  con.times[con.current % NUM_CON_TIMES] = cls.realtime;
519  }
520 }

557 /*
558 ================
559 Con_DrawNotify
560 
561 Draws the last few lines of output transparently over the game top
562 ================
563 */
564 void Con_DrawNotify (void)
565 {
566  int x, v;
567  short *text;
568  int i;
569  int time;
570  int skip;
571  int currentColor;
572 
573  currentColor = 7;
574  re.SetColor( g_color_table[currentColor] );
575 
576  v = 0;
577  for (i= con.current-NUM_CON_TIMES+1 ; i<=con.current ; i++)
578  {
579  if (i < 0)
580  continue;
581  time = con.times[i % NUM_CON_TIMES];
582  if (time == 0)
583  continue;
584  time = cls.realtime - time;
585  if (time > con_notifytime->value*1000)
586  continue;
587  text = con.text + (i % con.totallines)*con.linewidth;
588 
589  if (cl.snap.ps.pm_type != PM_INTERMISSION && Key_GetCatcher( ) & (KEYCATCH_UI | KEYCATCH_CGAME) ) {
590  continue;
591  }
592 
593  for (x = 0 ; x < con.linewidth ; x++) {
594  if ( ( text[x] & 0xff ) == ' ' ) {
595  continue;
596  }
597  if ( ColorIndexForNumber( text[x]>>8 ) != currentColor ) {
598  currentColor = ColorIndexForNumber( text[x]>>8 );
599  re.SetColor( g_color_table[currentColor] );
600  }
601  SCR_DrawSmallChar( cl_conXOffset->integer + con.xadjust + (x+1)*SMALLCHAR_WIDTH, v, text[x] & 0xff );
602  }
603 
604  v += SMALLCHAR_HEIGHT;
605  }
606 
607  re.SetColor( NULL );
608 
609  if (Key_GetCatcher( ) & (KEYCATCH_UI | KEYCATCH_CGAME) ) {
610  return;
611  }
612 
613  // draw the chat line
614  if ( Key_GetCatcher( ) & KEYCATCH_MESSAGE )
615  {
616  if (chat_team)
617  {
618  SCR_DrawBigString (8, v, "say_team:", 1.0f, qfalse );
619  skip = 10;
620  }
621  else
622  {
623  SCR_DrawBigString (8, v, "say:", 1.0f, qfalse );
624  skip = 5;
625  }
626 
627  Field_BigDraw( &chatField, skip * BIGCHAR_WIDTH, v,
628  SCREEN_WIDTH - ( skip + 1 ) * BIGCHAR_WIDTH, qtrue, qtrue );
629  }
630 
631 }

633 /*
634 ================
635 Con_DrawSolidConsole
636 
637 Draws the console with the solid background
638 ================
639 */
640 void Con_DrawSolidConsole( float frac ) {
641  int i, x, y;
642  int rows;
643  short *text;
644  int row;
645  int lines;
646 // qhandle_t conShader;
647  int currentColor;
648  vec4_t color;
649 
650  lines = cls.glconfig.vidHeight * frac;
651  if (lines <= 0)
652  return;
653 
654  if (lines > cls.glconfig.vidHeight )
655  lines = cls.glconfig.vidHeight;
656 
657  // on wide screens, we will center the text
658  con.xadjust = 0;
659  SCR_AdjustFrom640( &con.xadjust, NULL, NULL, NULL );
660 
661  // draw the background
662  y = frac * SCREEN_HEIGHT;
663  if ( y < 1 ) {
664  y = 0;
665  }
666  else {
667  SCR_DrawPic( 0, 0, SCREEN_WIDTH, y, cls.consoleShader );
668  }
669 
670  color[0] = 1;
671  color[1] = 0;
672  color[2] = 0;
673  color[3] = 1;
674  SCR_FillRect( 0, y, SCREEN_WIDTH, 2, color );
675 
676 
677  // draw the version number
678 
679  re.SetColor( g_color_table[ColorIndex(COLOR_RED)] );
680 
681  i = strlen( Q3_VERSION );
682 
683  for (x=0 ; x<i ; x++) {
684  SCR_DrawSmallChar( cls.glconfig.vidWidth - ( i - x + 1 ) * SMALLCHAR_WIDTH,
685  lines - SMALLCHAR_HEIGHT, Q3_VERSION[x] );
686  }
687 
688 
689  // draw the text
690  con.vislines = lines;
691  rows = (lines-SMALLCHAR_HEIGHT)/SMALLCHAR_HEIGHT; // rows of text to draw
692 
693  y = lines - (SMALLCHAR_HEIGHT*3);
694 
695  // draw from the bottom up
696  if (con.display != con.current)
697  {
698  // draw arrows to show the buffer is backscrolled
699  re.SetColor( g_color_table[ColorIndex(COLOR_RED)] );
700  for (x=0 ; x<con.linewidth ; x+=4)
701  SCR_DrawSmallChar( con.xadjust + (x+1)*SMALLCHAR_WIDTH, y, '^' );
702  y -= SMALLCHAR_HEIGHT;
703  rows--;
704  }
705 
706  row = con.display;
707 
708  if ( con.x == 0 ) {
709  row--;
710  }
711 
712  currentColor = 7;
713  re.SetColor( g_color_table[currentColor] );
714 
715  for (i=0 ; i<rows ; i++, y -= SMALLCHAR_HEIGHT, row--)
716  {
717  if (row < 0)
718  break;
719  if (con.current - row >= con.totallines) {
720  // past scrollback wrap point
721  continue;
722  }
723 
724  text = con.text + (row % con.totallines)*con.linewidth;
725 
726  for (x=0 ; x<con.linewidth ; x++) {
727  if ( ( text[x] & 0xff ) == ' ' ) {
728  continue;
729  }
730 
731  if ( ColorIndexForNumber( text[x]>>8 ) != currentColor ) {
732  currentColor = ColorIndexForNumber( text[x]>>8 );
733  re.SetColor( g_color_table[currentColor] );
734  }
735  SCR_DrawSmallChar( con.xadjust + (x+1)*SMALLCHAR_WIDTH, y, text[x] & 0xff );
736  }
737  }
738 
739  // draw the input prompt, user text, and cursor if desired
740  Con_DrawInput ();
741 
742  re.SetColor( NULL );
743 }

776 /*
777 ==================
778 Con_RunConsole
779 
780 Scroll it up or down
781 ==================
782 */
783 void Con_RunConsole (void) {
784  // decide on the destination height of the console
785  if ( Key_GetCatcher( ) & KEYCATCH_CONSOLE )
786  con.finalFrac = 0.5; // half screen
787  else
788  con.finalFrac = 0; // none visible
789 
790  // scroll towards the destination height
791  if (con.finalFrac < con.displayFrac)
792  {
793  con.displayFrac -= con_conspeed->value*cls.realFrametime*0.001;
794  if (con.finalFrac > con.displayFrac)
795  con.displayFrac = con.finalFrac;
796
