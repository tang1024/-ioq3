28 /*
29 ======================
30 CG_PositionEntityOnTag
31 
32 Modifies the entities position and axis by the given
33 tag location
34 ======================
35 */
36 void CG_PositionEntityOnTag( refEntity_t *entity, const refEntity_t *parent,
37  qhandle_t parentModel, char *tagName ) {
38  int i;
39  orientation_t lerped;
40 
41  // lerp the tag
42  trap_R_LerpTag( &lerped, parentModel, parent->oldframe, parent->frame,
43  1.0 - parent->backlerp, tagName );
44 
45  // FIXME: allow origin offsets along tag?
46  VectorCopy( parent->origin, entity->origin );
47  for ( i = 0 ; i < 3 ; i++ ) {
48  VectorMA( entity->origin, lerped.origin[i], parent->axis[i], entity->origin );
49  }
50 
51  // had to cast away the const to avoid compiler problems...
52  MatrixMultiply( lerped.axis, ((refEntity_t *)parent)->axis, entity->axis );
53  entity->backlerp = parent->backlerp;
54 }

79  VectorMA( entity->origin, lerped.origin[i], parent->axis[i], entity->origin );

81 
82  // had to cast away the const to avoid compiler problems...
83  MatrixMultiply( entity->axis, lerped.axis, tempAxis );
84  MatrixMultiply( tempAxis, ((refEntity_t *)parent)->axis, entity->axis );
85 }

222 /*
223 ==================
224 CG_Item
225 ==================
226 */
227 static void CG_Item( centity_t *cent ) {
228  refEntity_t ent;
229  entityState_t *es;
230  gitem_t *item;
231  int msec;
232  float frac;
233  float scale;
234  weaponInfo_t *wi;
235 
236  es = &cent->currentState;
237  if ( es->modelindex >= bg_numItems ) {
238  CG_Error( "Bad item index %i on entity", es->modelindex );
239  }
240 
241  // if set to invisible, skip
242  if ( !es->modelindex || ( es->eFlags & EF_NODRAW ) ) {
243  return;
244  }
245 
246  item = &bg_itemlist[ es->modelindex ];
247  if ( cg_simpleItems.integer && item->giType != IT_TEAM ) {
248  memset( &ent, 0, sizeof( ent ) );
249  ent.reType = RT_SPRITE;
250  VectorCopy( cent->lerpOrigin, ent.origin );
251  ent.radius = 14;
252  ent.customShader = cg_items[es->modelindex].icon;
253  ent.shaderRGBA[0] = 255;
254  ent.shaderRGBA[1] = 255;
255  ent.shaderRGBA[2] = 255;
256  ent.shaderRGBA[3] = 255;
257  trap_R_AddRefEntityToScene(&ent);
258  return;
259  }
260 
261  // items bob up and down continuously
262  scale = 0.005 + cent->currentState.number * 0.00001;
263  cent->lerpOrigin[2] += 4 + cos( ( cg.time + 1000 ) * scale ) * 4;
264 
265  memset (&ent, 0, sizeof(ent));
266 
267  // autorotate at one of two speeds
268  if ( item->giType == IT_HEALTH ) {
269  VectorCopy( cg.autoAnglesFast, cent->lerpAngles );
270  AxisCopy( cg.autoAxisFast, ent.axis );
271  } else {
272  VectorCopy( cg.autoAngles, cent->lerpAngles );
273  AxisCopy( cg.autoAxis, ent.axis );
274  }
275 
276  wi = NULL;
277  // the weapons have their origin where they attatch to player
278  // models, so we need to offset them or they will rotate
279  // eccentricly
280  if ( item->giType == IT_WEAPON ) {
281  wi = &cg_weapons[item->giTag];
282  cent->lerpOrigin[0] -=
283  wi->weaponMidpoint[0] * ent.axis[0][0] +
284  wi->weaponMidpoint[1] * ent.axis[1][0] +
285  wi->weaponMidpoint[2] * ent.axis[2][0];
286  cent->lerpOrigin[1] -=
287  wi->weaponMidpoint[0] * ent.axis[0][1] +
288  wi->weaponMidpoint[1] * ent.axis[1][1] +
289  wi->weaponMidpoint[2] * ent.axis[2][1];
290  cent->lerpOrigin[2] -=
291  wi->weaponMidpoint[0] * ent.axis[0][2] +
292  wi->weaponMidpoint[1] * ent.axis[1][2] +
293  wi->weaponMidpoint[2] * ent.axis[2][2];
294 
295  cent->lerpOrigin[2] += 8; // an extra height boost
296  }
297 
298  if( item->giType == IT_WEAPON && item->giTag == WP_RAILGUN ) {
299  clientInfo_t *ci = &cgs.clientinfo[cg.snap->ps.clientNum];
300  Byte4Copy( ci->c1RGBA, ent.shaderRGBA );
301  }
302 
303  ent.hModel = cg_items[es->modelindex].models[0];
304 
305  VectorCopy( cent->lerpOrigin, ent.origin);
306  VectorCopy( cent->lerpOrigin, ent.oldorigin);
307 
308  ent.nonNormalizedAxes = qfalse;
309 
310  // if just respawned, slowly scale up
311  msec = cg.time - cent->miscTime;
312  if ( msec >= 0 && msec < ITEM_SCALEUP_TIME ) {
313  frac = (float)msec / ITEM_SCALEUP_TIME;
314  VectorScale( ent.axis[0], frac, ent.axis[0] );
315  VectorScale( ent.axis[1], frac, ent.axis[1] );
316  VectorScale( ent.axis[2], frac, ent.axis[2] );
317  ent.nonNormalizedAxes = qtrue;
318  } else {
319  frac = 1.0;
320  }
321 
322  // items without glow textures need to keep a minimum light value
323  // so they are always visible
324  if ( ( item->giType == IT_WEAPON ) ||
325  ( item->giType == IT_ARMOR ) ) {
326  ent.renderfx |= RF_MINLIGHT;
327  }
328 
329  // increase the size of the weapons when they are presented as items
330  if ( item->giType == IT_WEAPON ) {
331  VectorScale( ent.axis[0], 1.5, ent.axis[0] );
332  VectorScale( ent.axis[1], 1.5, ent.axis[1] );
333  VectorScale( ent.axis[2], 1.5, ent.axis[2] );
334  ent.nonNormalizedAxes = qtrue;
335 #ifdef MISSIONPACK
336  trap_S_AddLoopingSound( cent->currentState.number, cent->lerpOrigin, vec3_origin, cgs.media.weaponHoverSound );
337 #endif
338  }
339 
340 #ifdef MISSIONPACK
341  if ( item->giType == IT_HOLDABLE && item->giTag == HI_KAMIKAZE ) {
342  VectorScale( ent.axis[0], 2, ent.axis[0] );
343  VectorScale( ent.axis[1], 2, ent.axis[1] );
344  VectorScale( ent.axis[2], 2, ent.axis[2] );
345  ent.nonNormalizedAxes = qtrue;
346  }
347 #endif
348 
349  // add to refresh list
350  trap_R_AddRefEntityToScene(&ent);
351 
352  if ( item->giType == IT_WEAPON && wi && wi->barrelModel ) {
353  refEntity_t barrel;
354  vec3_t angles;
355 
356  memset( &barrel, 0, sizeof( barrel ) );
357 
358  barrel.hModel = wi->barrelModel;
359 
360  VectorCopy( ent.lightingOrigin, barrel.lightingOrigin );
361  barrel.shadowPlane = ent.shadowPlane;
362  barrel.renderfx = ent.renderfx;
363 
364  angles[YAW] = 0;
365  angles[PITCH] = 0;
366  angles[ROLL] = 0;
367  AnglesToAxis( angles, barrel.axis );
368 
369  CG_PositionRotatedEntityOnTag( &barrel, &ent, wi->weaponModel, "tag_barrel" );
370 
371  barrel.nonNormalizedAxes = ent.nonNormalizedAxes;
372 
373  trap_R_AddRefEntityToScene( &barrel );
374  }
375 
376  // accompanying rings / spheres for powerups
377  if ( !cg_simpleItems.integer )
378  {
379  vec3_t spinAngles;
380 
381  VectorClear( spinAngles );
382 
383  if ( item->giType == IT_HEALTH || item->giType == IT_POWERUP )
384  {
385  if ( ( ent.hModel = cg_items[es->modelindex].models[1] ) != 0 )
386  {
387  if ( item->giType == IT_POWERUP )
388  {
389  ent.origin[2] += 12;
390  spinAngles[1] = ( cg.time & 1023 ) * 360 / -1024.0f;
391  }
392  AnglesToAxis( spinAngles, ent.axis );
393 
394  // scale up if respawning
395  if ( frac != 1.0 ) {
396  VectorScale( ent.axis[0], frac, ent.axis[0] );
397  VectorScale( ent.axis[1], frac, ent.axis[1] );
398  VectorScale( ent.axis[2], frac, ent.axis[2] );
399  ent.nonNormalizedAxes = qtrue;
400  }
401  trap_R_AddRefEntityToScene( &ent );
402  }
403  }
404  }
405 }

414 static void CG_Missile( centity_t *cent ) {
415  refEntity_t ent;
416  entityState_t *s1;
417  const weaponInfo_t *weapon;
418 // int col;
419 
420  s1 = &cent->currentState;
421  if ( s1->weapon >= WP_NUM_WEAPONS ) {
422  s1->weapon = 0;
423  }
424  weapon = &cg_weapons[s1->weapon];
425 
426  // calculate the axis
427  VectorCopy( s1->angles, cent->lerpAngles);
428 
429  // add trails
430  if ( weapon->missileTrailFunc )
431  {
432  weapon->missileTrailFunc( cent, weapon );
433  }
434 /*
435  if ( cent->currentState.modelindex == TEAM_RED ) {
436  col = 1;
437  }
438  else if ( cent->currentState.modelindex == TEAM_BLUE ) {
439  col = 2;
440  }
441  else {
442  col = 0;
443  }
444 
445  // add dynamic light
446  if ( weapon->missileDlight ) {
447  trap_R_AddLightToScene(cent->lerpOrigin, weapon->missileDlight,
448  weapon->missileDlightColor[col][0], weapon->missileDlightColor[col][1], weapon->missileDlightColor[col][2] );
449  }
450 */
451  // add dynamic light
452  if ( weapon->missileDlight ) {
453  trap_R_AddLightToScene(cent->lerpOrigin, weapon->missileDlight,
454  weapon->missileDlightColor[0], weapon->missileDlightColor[1], weapon->missileDlightColor[2] );
455  }
456 
457  // add missile sound
458  if ( weapon->missileSound ) {
459  vec3_t velocity;
460 
461  BG_EvaluateTrajectoryDelta( &cent->currentState.pos, cg.time, velocity );
462 
463  trap_S_AddLoopingSound( cent->currentState.number, cent->lerpOrigin, velocity, weapon->missileSound );
464  }
465 
466  // create the render entity
467  memset (&ent, 0, sizeof(ent));
468  VectorCopy( cent->lerpOrigin, ent.origin);
469  VectorCopy( cent->lerpOrigin, ent.oldorigin);
470 
471  if ( cent->currentState.weapon == WP_PLASMAGUN ) {
472  ent.reType = RT_SPRITE;
473  ent.radius = 16;
474  ent.rotation = 0;
475  ent.customShader = cgs.media.plasmaBallShader;
476  trap_R_AddRefEntityToScene( &ent );
477  return;
478  }
479 
480  // flicker between two skins
481  ent.skinNum = cg.clientFrame & 1;
482  ent.hModel = weapon->missileModel;
483  ent.renderfx = weapon->missileRenderfx | RF_NOSHADOW;
484 
485 #ifdef MISSIONPACK
486  if ( cent->currentState.weapon == WP_PROX_LAUNCHER ) {
487  if (s1->generic1 == TEAM_BLUE) {
488  ent.hModel = cgs.media.blueProxMine;
489  }
490  }
491 #endif
492 
493  // convert direction of travel into axis
494  if ( VectorNormalize2( s1->pos.trDelta, ent.axis[0] ) == 0 ) {
495  ent.axis[0][2] = 1;
496  }
497 
498  // spin as it moves
499  if ( s1->pos.trType != TR_STATIONARY ) {
500  RotateAroundDirection( ent.axis, cg.time / 4 );
501  } else {
502 #ifdef MISSIONPACK
503  if ( s1->weapon == WP_PROX_LAUNCHER ) {
504  AnglesToAxis( cent->lerpAngles, ent.axis );
505  }
506  else
507 #endif
508  {
509  RotateAroundDirection( ent.axis, s1->time );
510  }
511  }
512 
513  // add to refresh list, possibly with quad glow
514  CG_AddRefEntityWithPowerups( &ent, s1, TEAM_FREE );
515 }

713  centity_t *cent;
714  vec3_t oldOrigin, origin, deltaOrigin;
715  vec3_t oldAngles, angles, deltaAngles;
716  vec3_t matrix[3], transpose[3];
717  vec3_t org, org2, move2;
718 
719  if ( moverNum <= 0 || moverNum >= ENTITYNUM_MAX_NORMAL ) {
720  VectorCopy( in, out );
721  VectorCopy(angles_in, angles_out);
722  return;
723  }
724 
725  cent = &cg_entities[ moverNum ];
726  if ( cent->currentState.eType != ET_MOVER ) {
727  VectorCopy( in, out );
728  VectorCopy(angles_in, angles_out);
729  return;
730  }
731 
732  BG_EvaluateTrajectory( &cent->currentState.pos, fromTime, oldOrigin );
733  BG_EvaluateTrajectory( &cent->currentState.apos, fromTime, oldAngles );
734 
735  BG_EvaluateTrajectory( &cent->currentState.pos, toTime, origin );
736  BG_EvaluateTrajectory( &cent->currentState.apos, toTime, angles );
737 
738  VectorSubtract( origin, oldOrigin, deltaOrigin );
739  VectorSubtract( angles, oldAngles, deltaAngles );
740 
741  // origin change when on a rotating object
742  CG_CreateRotationMatrix( deltaAngles, transpose );
743  CG_TransposeMatrix( transpose, matrix );
744  VectorSubtract( in, oldOrigin, org );
745  VectorCopy( org, org2 );
746  CG_RotatePoint( org2, matrix );
747  VectorSubtract( org2, org, move2 );
748  VectorAdd( deltaOrigin, move2, deltaOrigin );
749 
750  VectorAdd( in, deltaOrigin, out );
751  VectorAdd( angles_in, deltaAngles, angles_out );

760 static void CG_InterpolateEntityPosition( centity_t *cent ) {
761  vec3_t current, next;
762  float f;
763 
764  // it would be an internal error to find an entity that interpolates without
765  // a snapshot ahead of the current one
766  if ( cg.nextSnap == NULL ) {
767  CG_Error( "CG_InterpoateEntityPosition: cg.nextSnap == NULL" );
768  }
769 
770  f = cg.frameInterpolation;
771 
772  // this will linearize a sine or parabolic curve, but it is important
773  // to not extrapolate player positions if more recent data is available
774  BG_EvaluateTrajectory( &cent->currentState.pos, cg.snap->serverTime, current );
775  BG_EvaluateTrajectory( &cent->nextState.pos, cg.nextSnap->serverTime, next );
776 
777  cent->lerpOrigin[0] = current[0] + f * ( next[0] - current[0] );
778  cent->lerpOrigin[1] = current[1] + f * ( next[1] - current[1] );
779  cent->lerpOrigin[2] = current[2] + f * ( next[2] - current[2] );
780 
781  BG_EvaluateTrajectory( &cent->currentState.apos, cg.snap->serverTime, current );
782  BG_EvaluateTrajectory( &cent->nextState.apos, cg.nextSnap->serverTime, next );
783 
784  cent->lerpAngles[0] = LerpAngle( current[0], next[0], f );
785  cent->lerpAngles[1] = LerpAngle( current[1], next[1], f );
786  cent->lerpAngles[2] = LerpAngle( current[2], next[2], f );
787 
788 }

796 static void CG_CalcEntityLerpPositions( centity_t *cent ) {
797 
798  // if this player does not want to see extrapolated players
799  if ( !cg_smoothClients.integer ) {

815  cent->currentState.number < MAX_CLIENTS) {
816  CG_InterpolateEntityPosition( cent );

821  BG_EvaluateTrajectory( &cent->currentState.pos, cg.time, cent->lerpOrigin );
822  BG_EvaluateTrajectory( &cent->currentState.apos, cg.time, cent->lerpAngles );

826  if ( cent != &cg.predictedPlayerEntity ) {
