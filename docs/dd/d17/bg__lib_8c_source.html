99 void
100 qsort(a, n, es, cmp)
101 void* a;
102 size_t n, es;
103 cmp_t* cmp;
104 {
105  char* pa, * pb, * pc, * pd, * pl, * pm, * pn;
106  int d, r, swaptype, swap_cnt;
107 
108 loop: SWAPINIT(a, es);
109  swap_cnt = 0;
110  if (n < 7) {
111  for (pm = (char*)a + es; pm < (char*)a + n * es; pm += es)
112  for (pl = pm; pl > (char*)a&& cmp(pl - es, pl) > 0;
113  pl -= es)
114  swap(pl, pl - es);
115  return;
116  }
117  pm = (char*)a + (n / 2) * es;
118  if (n > 7) {
119  pl = a;
120  pn = (char*)a + (n - 1) * es;
121  if (n > 40) {
122  d = (n / 8) * es;
123  pl = med3(pl, pl + d, pl + 2 * d, cmp);
124  pm = med3(pm - d, pm, pm + d, cmp);
125  pn = med3(pn - 2 * d, pn - d, pn, cmp);
126  }
127  pm = med3(pl, pm, pn, cmp);
128  }
129  swap(a, pm);
130  pa = pb = (char*)a + es;
131 
132  pc = pd = (char*)a + (n - 1) * es;
133  for (;;) {
134  while (pb <= pc && (r = cmp(pb, a)) <= 0) {
135  if (r == 0) {
136  swap_cnt = 1;
137  swap(pa, pb);
138  pa += es;
139  }
140  pb += es;
141  }
142  while (pb <= pc && (r = cmp(pc, a)) >= 0) {
143  if (r == 0) {
144  swap_cnt = 1;
145  swap(pc, pd);
146  pd -= es;
147  }
148  pc -= es;
149  }
150  if (pb > pc)
151  break;
152  swap(pb, pc);
153  swap_cnt = 1;
154  pb += es;
155  pc -= es;
156  }
157  if (swap_cnt == 0) { /* Switch to insertion sort */
158  for (pm = (char*)a + es; pm < (char*)a + n * es; pm += es)
159  for (pl = pm; pl > (char*)a&& cmp(pl - es, pl) > 0;
160  pl -= es)
161  swap(pl, pl - es);
162  return;
163  }
164 
165  pn = (char*)a + n * es;
166  r = MIN(pa - (char*)a, pb - pa);
167  vecswap(a, pb - r, r);
168  r = MIN(pd - pc, pn - pd - es);
169  vecswap(pb, pn - r, r);
170  if ((r = pb - pa) > es)
171  qsort(a, r / es, es, cmp);
172  if ((r = pd - pc) > es) {
173  /* Iterate rather than recurse to save stack space */
174  a = pn - r;
175  n = r / es;
176  goto loop;
177  }
178  /* qsort(pn - r, r / es, es, cmp);*/
179 }

365

375 
376  do {

979 Handles inf[inity], nan (ignoring case), hexadecimals, and decimals

984 in the nptr string that was not used in the conversion
985 ==============

988 {

990  qboolean neg = qfalse;
991

993  while (isspace(*nptr))
994  nptr++;

997  if (Q_stricmpn(nptr, "nan", 3) == 0) {
998  floatint_t nan;
999 
1000  if (endptr)
1001  *endptr = (char*)&nptr[3];
1002

1004  // or decimal) which is then put in the mantissa
1005  // this can be used to generate signalling or quiet NaNs, for
1006  // example (though I doubt it'll ever be used)
1007  // note that nan(0) is infinity!
1008  if (nptr[3] == '(') {
1009  char* end;
1010  int mantissa = strtol(&nptr[4], &end, 0);
1011  if (*end == ')') {
1012  nan.ui = 0x7f800000 | (mantissa & 0x7fffff);
1013  if (endptr)
1014  *endptr = &end[1];
1015  return nan.f;
1016  }
1017  }
1018  nan.ui = 0x7fffffff;
1019  return nan.f;

1021  if (Q_stricmpn(nptr, "inf", 3) == 0) {
1022  floatint_t inf;

1024  if (endptr == NULL)
1025  return inf.f;

1027  *endptr = (char*)&nptr[8];
1028  else
1029  *endptr = (char*)&nptr[3];

1034  // sign
1035  if (*nptr == '-') {
1036  nptr++;

1038  } else if (*nptr == '+')
1039  nptr++;

1061  nptr++;

1063  // I don't expect the float accuracy to hold out for
1064  // very long but since we need to know the length of
1065  // the string anyway we keep on going regardless

1068  res += place * (*nptr++ - '0');
1069  else if (*nptr >= 'A' && *nptr <= 'F')
1070  res += place * (10 + *nptr++ - 'A');
1071  else if (*nptr >= 'a' && *nptr <= 'f')
1072  res += place * (10 + *nptr++ - 'a');
1073  else
1074  break;
1075  }
1076  if (end < nptr)
1077  end = nptr;
1078  }

1080  // by a power of two
1081  // exponents are only valid if we encountered at least one

1904  signvalue = '+';

1911 #endif

1914

1918  */

1926

1929  fracpart -= powN(10, max);
1930  }

1942  if (iplace == 20) iplace--;
1943  iconvert[iplace] = 0;
1944 
1945  /* Convert fractional part */
1946  do {
1947  fconvert[fplace++] =

1949  fracpart = (fracpart / 10);
1950  } while (fracpart && (fplace < 20));
1951  if (fplace == 20) fplace--;
1952  fconvert[fplace] = 0;
1953

1957  if (zpadlen < 0)
1958  zpadlen = 0;
1959  if (padlen < 0)
1960  padlen = 0;

1963

1967  --padlen;
1968  signvalue = 0;
1969  }

1971  total += dopr_outch(buffer, currlen, maxlen, '0');
1972  --padlen;
1973  }
1974  }
1975  while (padlen > 0) {
1976  total += dopr_outch(buffer, currlen, maxlen, ' ');
1977  --padlen;
1978  }
1979  if (signvalue)
1980  total += dopr_outch(buffer, currlen, maxlen, signvalue);
1981

1983  total += dopr_outch(buffer, currlen, maxlen, iconvert[--iplace]);

1986  * Decimal point. This should probably use locale to find the correct
1987  * char to print out.
1988  */

1993  total += dopr_outch(buffer, currlen, maxlen, '0');
1994

1996  total += dopr_outch(buffer, currlen, maxlen, fconvert[--fplace]);
1997  }
1998 
1999  while (padlen < 0) {
2000  total += dopr_outch(buffer, currlen, maxlen, ' ');
2001  ++padlen;
2002  }
2003
