64 /*
65 ===============
66 G_ParseInfos
67 ===============
68 */
69 int G_ParseInfos(char* buf, int max, char* infos[])
70 {
71  char* token;
72  int count;
73  char key[MAX_TOKEN_CHARS];
74  char info[MAX_INFO_STRING];
75 
76  count = 0;
77 
78  while (1) {
79  token = COM_Parse(&buf);
80  if (!token[0]) {
81  break;
82  }
83  if (strcmp(token, "{")) {
84  Com_Printf("Missing { in info file\n");
85  break;
86  }
87 
88  if (count == max) {
89  Com_Printf("Max infos exceeded\n");
90  break;
91  }
92 
93  info[0] = '\0';
94  while (1) {
95  token = COM_ParseExt(&buf, qtrue);
96  if (!token[0]) {
97  Com_Printf("Unexpected end of info file\n");
98  break;
99  }
100  if (!strcmp(token, "}")) {
101  break;
102  }
103  Q_strncpyz(key, token, sizeof(key));
104 
105  token = COM_ParseExt(&buf, qfalse);
106  if (!token[0]) {
107  strcpy(token, "<NULL>");
108  }
109  Info_SetValueForKey(info, key, token);
110  }
111  //NOTE: extra space for arena number
112  infos[count] = G_Alloc(strlen(info) + strlen("\\num\\") + strlen(va("%d", MAX_ARENAS)) + 1);
113  if (infos[count]) {
114  strcpy(infos[count], info);
115  count++;
116  }
117  }
118  return count;
119 }

126 static void G_LoadArenasFromFile(char* filename)
127 {
128  int len;
129  fileHandle_t f;
130  char buf[MAX_ARENAS_TEXT];
131 
132  len = trap_FS_FOpenFile(filename, &f, FS_READ);
133  if (!f) {
134  trap_Print(va(S_COLOR_RED "file not found: %s\n", filename));
135  return;
136  }
137  if (len >= MAX_ARENAS_TEXT) {
138  trap_FS_FCloseFile(f);
139  trap_Print(va(S_COLOR_RED "file too large: %s is %i, max allowed is %i\n", filename, len, MAX_ARENAS_TEXT));
140  return;
141  }
142 
143  trap_FS_Read(buf, len, f);
144  buf[len] = 0;
145  trap_FS_FCloseFile(f);
146 
147  g_numArenas += G_ParseInfos(buf, MAX_ARENAS - g_numArenas, &g_arenaInfos[g_numArenas]);
148 }

269 /*
270 ===============
271 G_SelectRandomBotInfo
272 
273 Get random least used bot info on team or whole server if team is -1.
274 ===============
275 */
276 int G_SelectRandomBotInfo(int team)
277 {
278  int selection[MAX_BOTS];
279  int n, num;
280  int count, bestCount;
281  char* value;
282 
283  // don't add duplicate bots to the server if there are less bots than bot types
284  if (team != -1 && G_CountBotPlayersByName(NULL, -1) < g_numBots) {
285  team = -1;
286  }
287 
288  num = 0;
289  bestCount = MAX_CLIENTS;
290  for (n = 0; n < g_numBots; n++) {
291  value = Info_ValueForKey(g_botInfos[n], "funname");
292  if (!value[0]) {
293  value = Info_ValueForKey(g_botInfos[n], "name");
294  }
295  //
296  count = G_CountBotPlayersByName(value, team);
297 
298  if (count < bestCount) {
299  bestCount = count;
300  num = 0;
301  }
302 
303  if (count == bestCount) {
304  selection[num++] = n;
305 
306  if (num == MAX_BOTS) {
307  break;
308  }
309  }
310  }
311 
312  if (num > 0) {
313  num = random() * (num - 1);
314  return selection[num];
315  }
316 
317  return -1;
318 }

420 /*
421 ===============
422 G_CheckMinimumPlayers
423 ===============
424 */
425 void G_CheckMinimumPlayers(void)
426 {
427  int minplayers;
428  int humanplayers, botplayers;
429  static int checkminimumplayers_time;
430 
431  if (level.intermissiontime) return;
432  //only check once each 10 seconds
433  if (checkminimumplayers_time > level.time - 10000) {
434  return;
435  }
436  checkminimumplayers_time = level.time;
437  trap_Cvar_Update(&bot_minplayers);
438  minplayers = bot_minplayers.integer;
439  if (minplayers <= 0) return;
440 
441  if (g_gametype.integer >= GT_TEAM) {
442  if (minplayers >= g_maxclients.integer / 2) {
443  minplayers = (g_maxclients.integer / 2) - 1;
444  }
445 
446  humanplayers = G_CountHumanPlayers(TEAM_RED);
447  botplayers = G_CountBotPlayers(TEAM_RED);
448  //
449  if (humanplayers + botplayers < minplayers) {
450  G_AddRandomBot(TEAM_RED);
451  } else if (humanplayers + botplayers > minplayers&& botplayers) {
452  G_RemoveRandomBot(TEAM_RED);
453  }
454  //
455  humanplayers = G_CountHumanPlayers(TEAM_BLUE);
456  botplayers = G_CountBotPlayers(TEAM_BLUE);
457  //
458  if (humanplayers + botplayers < minplayers) {
459  G_AddRandomBot(TEAM_BLUE);
460  } else if (humanplayers + botplayers > minplayers&& botplayers) {
461  G_RemoveRandomBot(TEAM_BLUE);
462  }
463  } else if (g_gametype.integer == GT_TOURNAMENT) {
464  if (minplayers >= g_maxclients.integer) {
465  minplayers = g_maxclients.integer - 1;
466  }
467  humanplayers = G_CountHumanPlayers(-1);
468  botplayers = G_CountBotPlayers(-1);
469  //
470  if (humanplayers + botplayers < minplayers) {
471  G_AddRandomBot(TEAM_FREE);
472  } else if (humanplayers + botplayers > minplayers&& botplayers) {
473  // try to remove spectators first
474  if (!G_RemoveRandomBot(TEAM_SPECTATOR)) {
475  // just remove the bot that is playing
476  G_RemoveRandomBot(-1);
477  }
478  }
479  } else if (g_gametype.integer == GT_FFA) {
480  if (minplayers >= g_maxclients.integer) {
481  minplayers = g_maxclients.integer - 1;
482  }
483  humanplayers = G_CountHumanPlayers(TEAM_FREE);
484  botplayers = G_CountBotPlayers(TEAM_FREE);
485  //
486  if (humanplayers + botplayers < minplayers) {
487  G_AddRandomBot(TEAM_FREE);
488  } else if (humanplayers + botplayers > minplayers&& botplayers) {
489  G_RemoveRandomBot(TEAM_FREE);
490  }
491  }
492 }

591 /*
592 ===============
593 G_AddBot
594 ===============
595 */
596 static void G_AddBot(const char* name, float skill, const char* team, int delay, char* altname)
597 {
598  int clientNum;
599  int teamNum;
600  int botinfoNum;
601  char* botinfo;
602  char* key;
603  char* s;
604  char* botname;
605  char* model;
606  char* headmodel;
607  char userinfo[MAX_INFO_STRING];
608 
609  // have the server allocate a client slot
610  clientNum = trap_BotAllocateClient();
611  if (clientNum == -1) {
612  G_Printf(S_COLOR_RED "Unable to add bot. All player slots are in use.\n");
613  G_Printf(S_COLOR_RED "Start server with more 'open' slots (or check setting of sv_maxclients cvar).\n");
614  return;
615  }
616 
617  // set default team
618  if (!team || !*team) {
619  if (g_gametype.integer >= GT_TEAM) {
620  if (PickTeam(clientNum) == TEAM_RED) {
621  team = "red";
622  } else {
623  team = "blue";
624  }
625  } else {
626  team = "free";
627  }
628  }
629 
630  // get the botinfo from bots.txt
631  if (Q_stricmp(name, "random") == 0) {
632  if (Q_stricmp(team, "red") == 0 || Q_stricmp(team, "r") == 0) {
633  teamNum = TEAM_RED;
634  } else if (Q_stricmp(team, "blue") == 0 || Q_stricmp(team, "b") == 0) {
635  teamNum = TEAM_BLUE;
636  } else if (!Q_stricmp(team, "spectator") || !Q_stricmp(team, "s")) {
637  teamNum = TEAM_SPECTATOR;
638  } else {
639  teamNum = TEAM_FREE;
640  }
641 
642  botinfoNum = G_SelectRandomBotInfo(teamNum);
643 
644  if (botinfoNum < 0) {
645  G_Printf(S_COLOR_RED "Error: Cannot add random bot, no bot info available.\n");
646  trap_BotFreeClient(clientNum);
647  return;
648  }
649 
650  botinfo = G_GetBotInfoByNumber(botinfoNum);
651  } else {
652  botinfo = G_GetBotInfoByName(name);
653  }
654 
655  if (!botinfo) {
656  G_Printf(S_COLOR_RED "Error: Bot '%s' not defined\n", name);
657  trap_BotFreeClient(clientNum);
658  return;
659  }
660 
661  // create the bot's userinfo
662  userinfo[0] = '\0';
663 
664  botname = Info_ValueForKey(botinfo, "funname");
665  if (!botname[0]) {
666  botname = Info_ValueForKey(botinfo, "name");
667  }
668  // check for an alternative name
669  if (altname && altname[0]) {
670  botname = altname;
671  }
672  Info_SetValueForKey(userinfo, "name", botname);
673  Info_SetValueForKey(userinfo, "rate", "25000");
674  Info_SetValueForKey(userinfo, "snaps", "20");
675  Info_SetValueForKey(userinfo, "skill", va("%.2f", skill));
676  Info_SetValueForKey(userinfo, "teampref", team);
677 
678  if (skill >= 1 && skill < 2) {
679  Info_SetValueForKey(userinfo, "handicap", "50");
680  } else if (skill >= 2 && skill < 3) {
681  Info_SetValueForKey(userinfo, "handicap", "70");
682  } else if (skill >= 3 && skill < 4) {
683  Info_SetValueForKey(userinfo, "handicap", "90");
684  }
685 
686  key = "model";
687  model = Info_ValueForKey(botinfo, key);
688  if (!*model) {
689  model = "visor/default";
690  }
691  Info_SetValueForKey(userinfo, key, model);
692  key = "team_model";
693  Info_SetValueForKey(userinfo, key, model);
694 
695  key = "headmodel";
696  headmodel = Info_ValueForKey(botinfo, key);
697  if (!*headmodel) {
698  headmodel = model;
699  }
700  Info_SetValueForKey(userinfo, key, headmodel);
701  key = "team_headmodel";
702  Info_SetValueForKey(userinfo, key, headmodel);
703 
704  key = "gender";
705  s = Info_ValueForKey(botinfo, key);
706  if (!*s) {
707  s = "male";
708  }
709  Info_SetValueForKey(userinfo, "sex", s);
710 
711  key = "color1";
712  s = Info_ValueForKey(botinfo, key);
713  if (!*s) {
714  s = "4";
715  }
716  Info_SetValueForKey(userinfo, key, s);
717 
718  key = "color2";
719  s = Info_ValueForKey(botinfo, key);
720  if (!*s) {
721  s = "5";
722  }
723  Info_SetValueForKey(userinfo, key, s);
724 
725  s = Info_ValueForKey(botinfo, "aifile");
726  if (!*s) {
727  trap_Print(S_COLOR_RED "Error: bot has no aifile specified\n");
728  trap_BotFreeClient(clientNum);
729  return;
730  }
731  Info_SetValueForKey(userinfo, "characterfile", s);
732 
733  // don't send tinfo to bots, they don't parse it
734  Info_SetValueForKey(userinfo, "teamoverlay", "0");
735 
736  // register the userinfo
737  trap_SetUserinfo(clientNum, userinfo);
738 
739  // have it connect to the game as a normal client
740  if (ClientConnect(clientNum, qtrue, qtrue)) {
741  return;
742  }
743 
744  if (delay == 0) {
745  ClientBegin(clientNum);
746  return;
747  }
748 
749  AddBotToSpawnQueue(clientNum, delay);
750 }
