45 float LibVarStringValue(const char *string)
46 {
47  int dotfound = 0;
48  float value = 0;
49 
50  while(*string)
51  {
52  if (*string < '0' || *string > '9')
53  {
54  if (dotfound || *string != '.')
55  {
56  return 0;
57  } //end if
58  else
59  {
60  dotfound = 10;
61  string++;
62  } //end if
63  } //end if
64  if (dotfound)
65  {
66  value = value + (float) (*string - '0') / (float) dotfound;
67  dotfound *= 10;
68  } //end if
69  else
70  {
71  value = value * 10.0 + (float) (*string - '0');
72  } //end else
73  string++;
74  } //end while
75  return value;
76 } //end of the function LibVarStringValue
