184 /*
185 =================
186 S_AL_BufferFind
187 
188 Find a sound effect if loaded, set up a handle otherwise
189 =================
190 */
191 static sfxHandle_t S_AL_BufferFind(const char *filename)
192 {
193  // Look it up in the table
194  sfxHandle_t sfx = -1;
195  int i;
196 
197  if ( !filename ) {
198  Com_Error( ERR_FATAL, "Sound name is NULL" );
199  }
200 
201  if ( !filename[0] ) {
202  Com_Printf( S_COLOR_YELLOW "WARNING: Sound name is empty\n" );
203  return 0;
204  }
205 
206  if ( strlen( filename ) >= MAX_QPATH ) {
207  Com_Printf( S_COLOR_YELLOW "WARNING: Sound name is too long: %s\n", filename );
208  return 0;
209  }
210 
211  if ( filename[0] == '*' ) {
212  Com_Printf( S_COLOR_YELLOW "WARNING: Tried to load player sound directly: %s\n", filename );
213  return 0;
214  }
215 
216  for(i = 0; i < numSfx; i++)
217  {
218  if(!Q_stricmp(knownSfx[i].filename, filename))
219  {
220  sfx = i;
221  break;
222  }
223  }
224 
225  // Not found in table?
226  if(sfx == -1)
227  {
228  alSfx_t *ptr;
229 
230  sfx = S_AL_BufferFindFree();
231 
232  // Clear and copy the filename over
233  ptr = &knownSfx[sfx];
234  memset(ptr, 0, sizeof(*ptr));
235  ptr->masterLoopSrc = -1;
236  strcpy(ptr->filename, filename);
237  }
238 
239  // Return the handle
240  return sfx;
241 }

316 =================
317 S_AL_GenBuffers
318 =================
319 */
320 static qboolean S_AL_GenBuffers(ALsizei numBuffers, ALuint *buffers, const char *name)
321 {
322  ALenum error;
323 
324  S_AL_ClearError( qfalse );
325  qalGenBuffers( numBuffers, buffers );
326  error = qalGetError();
327 
328  // If we ran out of buffers, start evicting the least recently used sounds
329  while( error == AL_INVALID_VALUE )
330  {
331  if( !S_AL_BufferEvict( ) )
332  {
333  Com_Printf( S_COLOR_RED "ERROR: Out of audio buffers\n");
334  return qfalse;
335  }
336 
337  // Try again
338  S_AL_ClearError( qfalse );
339  qalGenBuffers( numBuffers, buffers );
340  error = qalGetError();
341  }
342 
343  if( error != AL_NO_ERROR )
344  {
345  Com_Printf( S_COLOR_RED "ERROR: Can't create a sound buffer for %s - %s\n",
346  name, S_AL_ErrorMsg(error));
347  return qfalse;
348  }
349 
350  return qtrue;
351 }

353 /*
354 =================
355 S_AL_BufferLoad
356 =================
357 */
358 static void S_AL_BufferLoad(sfxHandle_t sfx, qboolean cache)
359 {
360  ALenum error;
361  ALuint format;
362 
363  void *data;
364  snd_info_t info;
365  alSfx_t *curSfx = &knownSfx[sfx];
366 
367  // Nothing?
368  if(curSfx->filename[0] == '\0')
369  return;
370 
371  // Already done?
372  if((curSfx->inMemory) || (curSfx->isDefault) || (!cache && curSfx->isDefaultChecked))
373  return;
374 
375  // Try to load
376  data = S_CodecLoad(curSfx->filename, &info);
377  if(!data)
378  {
379  S_AL_BufferUseDefault(sfx);
380  return;
381  }
382 
383  curSfx->isDefaultChecked = qtrue;
384 
385  if (!cache)
386  {
387  // Don't create AL cache
388  Hunk_FreeTempMemory(data);
389  return;
390  }
391 
392  format = S_AL_Format(info.width, info.channels);
393 
394  // Create a buffer
395  if (!S_AL_GenBuffers(1, &curSfx->buffer, curSfx->filename))
396  {
397  S_AL_BufferUseDefault(sfx);
398  Hunk_FreeTempMemory(data);
399  return;
400  }
401 
402  // Fill the buffer
403  if( info.size == 0 )
404  {
405  // We have no data to buffer, so buffer silence
406  byte dummyData[ 2 ] = { 0 };
407 
408  qalBufferData(curSfx->buffer, AL_FORMAT_MONO16, (void *)dummyData, 2, 22050);
409  }
410  else
411  qalBufferData(curSfx->buffer, format, data, info.size, info.rate);
412 
413  error = qalGetError();
414 
415  // If we ran out of memory, start evicting the least recently used sounds
416  while(error == AL_OUT_OF_MEMORY)
417  {
418  if( !S_AL_BufferEvict( ) )
419  {
420  qalDeleteBuffers(1, &curSfx->buffer);
421  S_AL_BufferUseDefault(sfx);
422  Hunk_FreeTempMemory(data);
423  Com_Printf( S_COLOR_RED "ERROR: Out of memory loading %s\n", curSfx->filename);
424  return;
425  }
426 
427  // Try load it again
428  qalBufferData(curSfx->buffer, format, data, info.size, info.rate);
429  error = qalGetError();
430  }
431 
432  // Some other error condition
433  if(error != AL_NO_ERROR)
434  {
435  qalDeleteBuffers(1, &curSfx->buffer);
436  S_AL_BufferUseDefault(sfx);
437  Hunk_FreeTempMemory(data);
438  Com_Printf( S_COLOR_RED "ERROR: Can't fill sound buffer for %s - %s\n",
439  curSfx->filename, S_AL_ErrorMsg(error));
440  return;
441  }
442 
443  curSfx->info = info;
444 
445  // Free the memory
446  Hunk_FreeTempMemory(data);
447 
448  // Woo!
449  curSfx->inMemory = qtrue;
450 }

643 /*
644 =================
645 S_AL_ScaleGain
646 Adapt the gain if necessary to get a quicker fadeout when the source is too far away.
647 =================
648 */
649 
650 static void S_AL_ScaleGain(src_t *chksrc, vec3_t origin)
651 {
652  float distance;
653 
654  if(!chksrc->local)
655  distance = Distance(origin, lastListenerOrigin);
656 
657  // If we exceed a certain distance, scale the gain linearly until the sound
658  // vanishes into nothingness.
659  if(!chksrc->local && (distance -= s_alMaxDistance->value) > 0)
660  {
661  float scaleFactor;
662 
663  if(distance >= s_alGraceDistance->value)
664  scaleFactor = 0.0f;
665  else
666  scaleFactor = 1.0f - distance / s_alGraceDistance->value;
667 
668  scaleFactor *= chksrc->curGain;
669 
670  if(chksrc->scaleGain != scaleFactor)
671  {
672  chksrc->scaleGain = scaleFactor;
673  S_AL_Gain(chksrc->alSource, chksrc->scaleGain);
674  }
675  }
676  else if(chksrc->scaleGain != chksrc->curGain)
677  {
678  chksrc->scaleGain = chksrc->curGain;
679  S_AL_Gain(chksrc->alSource, chksrc->scaleGain);
680  }
681 }

872 /*
873 =================
874 S_AL_NewLoopMaster
875 Remove given source as loop master if it is the master and hand off master status to another source in this case.
876 =================
877 */
878 
879 static void S_AL_NewLoopMaster(src_t *rmSource, qboolean iskilled)
880 {
881  int index;
882  src_t *curSource = NULL;
883  alSfx_t *curSfx;
884 
885  curSfx = &knownSfx[rmSource->sfx];
886 
887  if(rmSource->isPlaying)
888  curSfx->loopActiveCnt--;
889  if(iskilled)
890  curSfx->loopCnt--;
891 
892  if(curSfx->loopCnt)
893  {
894  if(rmSource->priority == SRCPRI_ENTITY)
895  {
896  if(!iskilled && rmSource->isPlaying)
897  {
898  // only sync ambient loops...
899  // It makes more sense to have sounds for weapons/projectiles unsynced
900  S_AL_SaveLoopPos(rmSource, rmSource->alSource);
901  }
902  }
903  else if(curSfx->masterLoopSrc != -1 &&
904  rmSource == &srcList[curSfx->masterLoopSrc])
905  {
906  int firstInactive = -1;
907 
908  // Only if rmSource was the master and if there are still playing loops for
909  // this sound will we need to find a new master.
910 
911  if(iskilled || curSfx->loopActiveCnt)
912  {
913  for(index = 0; index < srcCount; index++)
914  {
915  curSource = &srcList[index];
916 
917  if(curSource->sfx == rmSource->sfx && curSource != rmSource &&
918  curSource->isActive && curSource->isLooping && curSource->priority == SRCPRI_AMBIENT)
919  {
920  if(curSource->isPlaying)
921  {
922  curSfx->masterLoopSrc = index;
923  break;
924  }
925  else if(firstInactive < 0)
926  firstInactive = index;
927  }
928  }
929  }
930 
931  if(!curSfx->loopActiveCnt)
932  {
933  if(firstInactive < 0)
934  {
935  if(iskilled)
936  {
937  curSfx->masterLoopSrc = -1;
938  return;
939  }
940  else
941  curSource = rmSource;
942  }
943  else
944  curSource = &srcList[firstInactive];
945 
946  if(rmSource->isPlaying)
947  {
948  // this was the last not stopped source, save last sample position + time
949  S_AL_SaveLoopPos(curSource, rmSource->alSource);
950  }
951  else
952  {
953  // second case: all loops using this sound have stopped due to listener being of of range,
954  // and now the inactive master gets deleted. Just move over the soundpos settings to the
955  // new master.
956  curSource->lastTimePos = rmSource->lastTimePos;
957  curSource->lastSampleTime = rmSource->lastSampleTime;
958  }
959  }
960  }
961  }
962  else
963  curSfx->masterLoopSrc = -1;
964 }

1029 {

1032  int weakest = -1;

1041  {

1050  {
1051  empty = i;

1058  (curSource->priority < weakest_pri ||

1060  {

1066  weakest_gain = curSource->scaleGain;
1067  weakest = i;
1068  }
1069  }
1070  else

1076  curSource->priority < weakest_pri ||
1077  curSource->lastUsedTime < weakest_time)
1078  {
1079  // Sources currently not playing of course have lowest priority
1080  // also try to always keep at least one loop master for every loop sound
1081  weakest_pri = curSource->priority;
1082  weakest_time = curSource->lastUsedTime;
1083  weakest_numloops = knownSfx[curSource->sfx].loopCnt;

1085  }
1086  }

1092  // Is it an exact match, and not on channel 0?

1095  S_AL_SrcKill(i);

1102  empty = weakest;
1103

1833  }
1834

1840  }
1841 }
