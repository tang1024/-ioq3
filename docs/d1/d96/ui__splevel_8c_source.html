129  char *skin;

134  if ( skin ) {

140 
141  Com_sprintf(iconName, iconNameMaxSize, "models/players/%s/icon_%s.tga", model, skin );

143  if( !trap_R_RegisterShaderNoMip( iconName ) && Q_stricmp( skin, "default" ) != 0 ) {
144  Com_sprintf(iconName, iconNameMaxSize, "models/players/%s/icon_default.tga", model );

162 /*
163 =================
164 UI_SPLevelMenu_SetBots
165 =================
166 */
167 static void UI_SPLevelMenu_SetBots( void ) {
168  char *p;
169  char *bot;
170  char *botInfo;
171  char bots[MAX_INFO_STRING];
172 
173  levelMenuInfo.numBots = 0;
174  if ( selectedArenaSet > currentSet ) {
175  return;
176  }
177 
178  Q_strncpyz( bots, Info_ValueForKey( levelMenuInfo.selectedArenaInfo, "bots" ), sizeof(bots) );
179 
180  p = &bots[0];
181  while( *p && levelMenuInfo.numBots < 7 ) {
182  //skip spaces
183  while( *p && *p == ' ' ) {
184  p++;
185  }
186  if( !*p ) {
187  break;
188  }
189 
190  // mark start of bot name
191  bot = p;
192 
193  // skip until space of null
194  while( *p && *p != ' ' ) {
195  p++;
196  }
197  if( *p ) {
198  *p++ = 0;
199  }
200 
201  botInfo = UI_GetBotInfoByName( bot );
202  if(!botInfo)
203  {
204  botInfo = UI_GetBotInfoByNumber( levelMenuInfo.numBots );
205  }
206 
207  if( botInfo ) {
208  levelMenuInfo.botPics[levelMenuInfo.numBots] = PlayerIconHandle( Info_ValueForKey( botInfo, "model" ) );
209  Q_strncpyz( levelMenuInfo.botNames[levelMenuInfo.numBots], Info_ValueForKey( botInfo, "name" ), 10 );
210  }
211  else {
212  levelMenuInfo.botPics[levelMenuInfo.numBots] = 0;
213  Q_strncpyz( levelMenuInfo.botNames[levelMenuInfo.numBots], bot, 10 );
214  }
215  Q_CleanStr( levelMenuInfo.botNames[levelMenuInfo.numBots] );
216  levelMenuInfo.numBots++;
217  }
218 }

289  levelMenuInfo.levelPicNames[2][0] = 0;
290  levelMenuInfo.levelPicNames[3][0] = 0;

292  levelMenuInfo.item_maps[2].shader = 0;
293  levelMenuInfo.item_maps[3].shader = 0;
294  }
295  else if( selectedArenaSet == finalTier ) {
296  arenaInfo = UI_GetSpecialArenaInfo( "final" );
297  level = atoi( Info_ValueForKey( arenaInfo, "num" ) );
298  UI_SPLevelMenu_SetMenuArena( 0, level, arenaInfo );
299  levelMenuInfo.selectedArenaInfo = arenaInfo;
300 
301  levelMenuInfo.item_maps[0].generic.x = 256;
302  Bitmap_Init( &levelMenuInfo.item_maps[0] );
303  levelMenuInfo.item_maps[0].generic.bottom += 32;
304  levelMenuInfo.numMaps = 1;
305 
306  levelMenuInfo.item_maps[1].generic.flags |= QMF_INACTIVE;
307  levelMenuInfo.item_maps[2].generic.flags |= QMF_INACTIVE;

310  levelMenuInfo.levelPicNames[2][0] = 0;

317  levelMenuInfo.item_maps[0].generic.x = 46;
318  Bitmap_Init( &levelMenuInfo.item_maps[0] );
319  levelMenuInfo.item_maps[0].generic.bottom += 18;
320  levelMenuInfo.numMaps = 4;
321 
322  for ( n = 0; n < 4; n++ ) {
323  level = selectedArenaSet * ARENAS_PER_TIER + n;
324  arenaInfo = UI_GetArenaInfoByNumber( level );
325  UI_SPLevelMenu_SetMenuArena( n, level, arenaInfo );
326  }
327 
328  if( selectedArena != -1 ) {
329  levelMenuInfo.selectedArenaInfo = UI_GetArenaInfoByNumber( selectedArenaSet * ARENAS_PER_TIER + selectedArena );
330  }
331  }
332 
333  // enable/disable arrows when they are valid/invalid
334  if ( selectedArenaSet == minTier ) {
335  levelMenuInfo.item_leftarrow.generic.flags |= ( QMF_INACTIVE | QMF_HIDDEN );
336  }
337  else {
338  levelMenuInfo.item_leftarrow.generic.flags &= ~( QMF_INACTIVE | QMF_HIDDEN );
339  }
340 
341  if ( selectedArenaSet == maxTier ) {
342  levelMenuInfo.item_rightarrow.generic.flags |= ( QMF_INACTIVE | QMF_HIDDEN );
343  }
344  else {
345  levelMenuInfo.item_rightarrow.generic.flags &= ~( QMF_INACTIVE | QMF_HIDDEN );
346  }
347 
348  UI_SPLevelMenu_SetBots();
349 }

637  }
638  UI_DrawHandlePic( x-1, y-1, 130, 130 - 14, levelMenuInfo.levelSelectedPic );
