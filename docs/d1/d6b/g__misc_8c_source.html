83 void TeleportPlayer(gentity_t* player, vec3_t origin, vec3_t angles)
84 {
85  gentity_t* tent;
86  qboolean noAngles;
87 
88  noAngles = (angles[0] > 999999.0);
89  // use temp events at source and destination to prevent the effect
90  // from getting dropped by a second player event
91  if (player->client->sess.sessionTeam != TEAM_SPECTATOR) {
92  tent = G_TempEntity(player->client->ps.origin, EV_PLAYER_TELEPORT_OUT);
93  tent->s.clientNum = player->s.clientNum;
94 
95  tent = G_TempEntity(origin, EV_PLAYER_TELEPORT_IN);
96  tent->s.clientNum = player->s.clientNum;
97  }
98 
99  // unlink to make sure it can't possibly interfere with G_KillBox
100  trap_UnlinkEntity(player);
101 
102  VectorCopy(origin, player->client->ps.origin);
103  player->client->ps.origin[2] += 1;
104  if (!noAngles) {
105  // spit the player out
106  AngleVectors(angles, player->client->ps.velocity, NULL, NULL);
107  VectorScale(player->client->ps.velocity, 400, player->client->ps.velocity);
108  player->client->ps.pm_time = 160; // hold time
109  player->client->ps.pm_flags |= PMF_TIME_KNOCKBACK;
110  // set angles
111  SetClientViewAngle(player, angles);
112  }
113  // toggle the teleport bit so the client knows to not lerp
114  player->client->ps.eFlags ^= EF_TELEPORT_BIT;
115  // kill anything at the destination
116  if (player->client->sess.sessionTeam != TEAM_SPECTATOR) {
117  G_KillBox(player);
118  }
119 
120  // save results of pmove
121  BG_PlayerStateToEntityState(&player->client->ps, &player->s, qtrue);
122 
123  // use the precise origin for linking
124  VectorCopy(player->client->ps.origin, player->r.currentOrigin);
125 
126  if (player->client->sess.sessionTeam != TEAM_SPECTATOR) {
127  trap_LinkEntity(player);
128  }
129 }

165 void locateCamera(gentity_t* ent)
166 {
167  vec3_t dir;
168  gentity_t* target;
169  gentity_t* owner;
170 
171  owner = G_PickTarget(ent->target);
172  if (!owner) {
173  G_Printf("Couldn't find target for misc_partal_surface\n");
174  G_FreeEntity(ent);
175  return;
176  }
177  ent->r.ownerNum = owner->s.number;
178 
179  // frame holds the rotate speed
180  if (owner->spawnflags & 1) {
181  ent->s.frame = 25;
182  } else if (owner->spawnflags & 2) {
183  ent->s.frame = 75;
184  }
185 
186  // swing camera ?
187  if (owner->spawnflags & 4) {
188  // set to 0 for no rotation at all
189  ent->s.powerups = 0;
190  } else {
191  ent->s.powerups = 1;
192  }
193 
194  // clientNum holds the rotate offset
195  ent->s.clientNum = owner->s.clientNum;
196 
197  VectorCopy(owner->s.origin, ent->s.origin2);
198 
199  // see if the portal_camera has a target
200  target = G_PickTarget(owner->target);
201  if (target) {
202  VectorSubtract(target->s.origin, owner->s.origin, dir);
203  VectorNormalize(dir);
204  } else {
205  G_SetMovedir(owner->s.angles, dir);
206  }
207 
208  ent->s.eventParm = DirToByte(dir);
209 }

257 void Use_Shooter(gentity_t* ent, gentity_t* other, gentity_t* activator)
258 {
259  vec3_t dir;
260  float deg;
261  vec3_t up, right;
262 
263  // see if we have a target
264  if (ent->enemy) {
265  VectorSubtract(ent->enemy->r.currentOrigin, ent->s.origin, dir);
266  VectorNormalize(dir);
267  } else {
268  VectorCopy(ent->movedir, dir);
269  }
270 
271  // randomize a bit
272  PerpendicularVector(up, dir);
273  CrossProduct(up, dir, right);
274 
275  deg = crandom() * ent->random;
276  VectorMA(dir, deg, up, dir);
277 
278  deg = crandom() * ent->random;
279  VectorMA(dir, deg, right, dir);
280 
281  VectorNormalize(dir);
282 
283  switch (ent->s.weapon) {
284  case WP_GRENADE_LAUNCHER:
285  fire_grenade(ent, ent->s.origin, dir);
286  break;
287  case WP_ROCKET_LAUNCHER:
288  fire_rocket(ent, ent->s.origin, dir);
289  break;
290  case WP_PLASMAGUN:
291  fire_plasma(ent, ent->s.origin, dir);
292  break;
293  }
294 
295  G_AddEvent(ent, EV_FIRE_WEAPON, 0);
296 }

401 static void PortalTouch(gentity_t* self, gentity_t* other, trace_t* trace)
402 {
403  gentity_t* destination;
404 
405  // see if we will even let other try to use it
406  if (other->health <= 0) {
407  return;
408  }
409  if (!other->client) {
410  return;
411  }
412  // if( other->client->ps.persistant[PERS_TEAM] != self->spawnflags ) {
413  // return;
414  // }
415 
416  if (other->client->ps.powerups[PW_NEUTRALFLAG]) { // only happens in One Flag CTF
417  Drop_Item(other, BG_FindItemForPowerup(PW_NEUTRALFLAG), 0);
418  other->client->ps.powerups[PW_NEUTRALFLAG] = 0;
419  } else if (other->client->ps.powerups[PW_REDFLAG]) { // only happens in standard CTF
420  Drop_Item(other, BG_FindItemForPowerup(PW_REDFLAG), 0);
421  other->client->ps.powerups[PW_REDFLAG] = 0;
422  } else if (other->client->ps.powerups[PW_BLUEFLAG]) { // only happens in standard CTF
423  Drop_Item(other, BG_FindItemForPowerup(PW_BLUEFLAG), 0);
424  other->client->ps.powerups[PW_BLUEFLAG] = 0;
425  }
426 
427  // find the destination
428  destination = NULL;
429  while ((destination = G_Find(destination, FOFS(classname), "hi_portal destination")) != NULL) {
430  if (destination->count == self->count) {
431  break;
432  }
433  }
434 
435  // if there is not one, die!
436  if (!destination) {
437  if (self->pos1[0] || self->pos1[1] || self->pos1[2]) {
438  TeleportPlayer(other, self->pos1, self->s.angles);
439  }
440  G_Damage(other, other, other, NULL, NULL, 100000, DAMAGE_NO_PROTECTION, MOD_TELEFRAG);
441  return;
442  }
443 
444  TeleportPlayer(other, destination->s.pos.trBase, destination->s.angles);
445 }
