401 void S_Play_f( void ) {
402  int i;
403  int c;
404  sfxHandle_t h;
405 
406  if( !si.RegisterSound || !si.StartLocalSound ) {
407  return;
408  }
409 
410  c = Cmd_Argc();
411 
412  if( c < 2 ) {
413  Com_Printf ("Usage: play <sound filename> [sound filename] [sound filename] ...\n");
414  return;
415  }
416 
417  for( i = 1; i < c; i++ ) {
418  h = si.RegisterSound( Cmd_Argv(i), qfalse );
419 
420  if( h ) {
421  si.StartLocalSound( h, CHAN_LOCAL_SOUND );
422  }
423  }

467 /*
468 =================
469 S_Init
470 =================
471 */
472 void S_Init( void )
473 {
474  cvar_t *cv;
475  qboolean started = qfalse;
476 
477  Com_Printf( "------ Initializing Sound ------\n" );
478 
479  s_volume = Cvar_Get( "s_volume", "0.8", CVAR_ARCHIVE );
480  s_musicVolume = Cvar_Get( "s_musicvolume", "0.25", CVAR_ARCHIVE );
481  s_muted = Cvar_Get("s_muted", "0", CVAR_ROM);
482  s_doppler = Cvar_Get( "s_doppler", "1", CVAR_ARCHIVE );
483  s_backend = Cvar_Get( "s_backend", "", CVAR_ROM );
484  s_muteWhenMinimized = Cvar_Get( "s_muteWhenMinimized", "0", CVAR_ARCHIVE );
485  s_muteWhenUnfocused = Cvar_Get( "s_muteWhenUnfocused", "0", CVAR_ARCHIVE );
486 
487  cv = Cvar_Get( "s_initsound", "1", 0 );
488  if( !cv->integer ) {
489  Com_Printf( "Sound disabled.\n" );
490  } else {
491 
492  S_CodecInit( );
493 
494  Cmd_AddCommand( "play", S_Play_f );
495  Cmd_AddCommand( "music", S_Music_f );
496  Cmd_AddCommand( "stopmusic", S_StopMusic_f );
497  Cmd_AddCommand( "s_list", S_SoundList );
498  Cmd_AddCommand( "s_stop", S_StopAllSounds );
499  Cmd_AddCommand( "s_info", S_SoundInfo );
500 
501  cv = Cvar_Get( "s_useOpenAL", "1", CVAR_ARCHIVE | CVAR_LATCH );
502  if( cv->integer ) {
503  //OpenAL
504  started = S_AL_Init( &si );
505  Cvar_Set( "s_backend", "OpenAL" );
506  }
507 
508  if( !started ) {
509  started = S_Base_Init( &si );
510  Cvar_Set( "s_backend", "base" );
511  }
512 
513  if( started ) {
514  if( !S_ValidSoundInterface( &si ) ) {
515  Com_Error( ERR_FATAL, "Sound interface invalid" );
516  }
517 
518  S_SoundInfo( );
519  Com_Printf( "Sound initialization successful.\n" );
520  } else {
521  Com_Printf( "Sound initialization failed.\n" );
522  }
523  }
524 
525  Com_Printf( "--------------------------------\n");
526 }
