93 static void UI_AddBotsMenu_FightEvent( void* ptr, int event ) {
94  const char *team;
95  int skill;
96 
97  if (event != QM_ACTIVATED) {
98  return;
99  }
100 
101  team = addBotsMenuInfo.team.itemnames[addBotsMenuInfo.team.curvalue];
102  skill = addBotsMenuInfo.skill.curvalue + 1;
103 
104  trap_Cmd_ExecuteText( EXEC_APPEND, va("addbot %s %i %s %i\n",
105  addBotsMenuInfo.botnames[addBotsMenuInfo.selectedBotNum], skill, team, addBotsMenuInfo.delay) );
106 
107  addBotsMenuInfo.delay += 1500;
108 }

196 static int QDECL UI_AddBotsMenu_SortCompare( const void *arg1, const void *arg2 ) {
197  int num1, num2;
198  const char *info1, *info2;
199  const char *name1, *name2;
200 
201  num1 = *(int *)arg1;
202  num2 = *(int *)arg2;
203 
204  info1 = UI_GetBotInfoByNumber( num1 );
205  info2 = UI_GetBotInfoByNumber( num2 );
206 
207  name1 = Info_ValueForKey( info1, "name" );
208  name2 = Info_ValueForKey( info2, "name" );
209 
210  return Q_stricmp( name1, name2 );
211 }

213 static void UI_AddBotsMenu_GetSortedBotNums( void ) {
214  int n;
215 
216  // initialize the array
217  for( n = 0; n < addBotsMenuInfo.numBots; n++ ) {
218  addBotsMenuInfo.sortedBotNums[n] = n;
219  }
220 
221  qsort( addBotsMenuInfo.sortedBotNums, addBotsMenuInfo.numBots, sizeof(addBotsMenuInfo.sortedBotNums[0]), UI_AddBotsMenu_SortCompare );
222 }
