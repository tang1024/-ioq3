31 /*
32 ====================
33 R_RegisterMD3
34 ====================
35 */
36 qhandle_t R_RegisterMD3(const char *name, model_t *mod)
37 {
38  union {
39  unsigned *u;
40  void *v;
41  } buf;
42  int size;
43  int lod;
44  int ident;
45  qboolean loaded = qfalse;
46  int numLoaded;
47  char filename[MAX_QPATH], namebuf[MAX_QPATH+20];
48  char *fext, defex[] = "md3";
49 
50  numLoaded = 0;
51 
52  strcpy(filename, name);
53 
54  fext = strchr(filename, '.');
55  if(!fext)
56  fext = defex;
57  else
58  {
59  *fext = '\0';
60  fext++;
61  }
62 
63  for (lod = MD3_MAX_LODS - 1 ; lod >= 0 ; lod--)
64  {
65  if(lod)
66  Com_sprintf(namebuf, sizeof(namebuf), "%s_%d.%s", filename, lod, fext);
67  else
68  Com_sprintf(namebuf, sizeof(namebuf), "%s.%s", filename, fext);
69 
70  size = ri.FS_ReadFile( namebuf, &buf.v );
71  if(!buf.u)
72  continue;
73 
74  ident = LittleLong(* (unsigned *) buf.u);
75  if (ident == MD3_IDENT)
76  loaded = R_LoadMD3(mod, lod, buf.u, size, name);
77  else
78  ri.Printf(PRINT_WARNING,"R_RegisterMD3: unknown fileid for %s\n", name);
79 
80  ri.FS_FreeFile(buf.v);
81 
82  if(loaded)
83  {
84  mod->numLods++;
85  numLoaded++;
86  }
87  else
88  break;
89  }
90 
91  if(numLoaded)
92  {
93  // duplicate into higher lod spots that weren't
94  // loaded, in case the user changes r_lodbias on the fly
95  for(lod--; lod >= 0; lod--)
96  {
97  mod->numLods++;
98  mod->mdv[lod] = mod->mdv[lod + 1];
99  }
100 
101  return mod->index;
102  }
103 
104 #ifdef _DEBUG
105  ri.Printf(PRINT_WARNING,"R_RegisterMD3: couldn't load %s\n", name);
106 #endif
107 
108  mod->type = MOD_BAD;
109  return 0;
110 }

145  }

241 /*
242 ====================
243 RE_RegisterModel
244 
245 Loads in a model for the given name
246 
247 Zero will be returned if the model fails to load.
248 An entry will be retained for failed models as an
249 optimization to prevent disk rescanning if they are
250 asked for again.
251 ====================
252 */
253 qhandle_t RE_RegisterModel( const char *name ) {
254  model_t *mod;
255  qhandle_t hModel;
256  qboolean orgNameFailed = qfalse;
257  int orgLoader = -1;
258  int i;
259  char localName[ MAX_QPATH ];
260  const char *ext;
261  char altName[ MAX_QPATH ];
262 
263  if ( !name || !name[0] ) {
264  ri.Printf( PRINT_ALL, "RE_RegisterModel: NULL name\n" );
265  return 0;
266  }
267 
268  if ( strlen( name ) >= MAX_QPATH ) {
269  ri.Printf( PRINT_ALL, "Model name exceeds MAX_QPATH\n" );
270  return 0;
271  }
272 
273  //
274  // search the currently loaded models
275  //
276  for ( hModel = 1 ; hModel < tr.numModels; hModel++ ) {
277  mod = tr.models[hModel];
278  if ( !strcmp( mod->name, name ) ) {
279  if( mod->type == MOD_BAD ) {
280  return 0;
281  }
282  return hModel;
283  }
284  }
285 
286  // allocate a new model_t
287 
288  if ( ( mod = R_AllocModel() ) == NULL ) {
289  ri.Printf( PRINT_WARNING, "RE_RegisterModel: R_AllocModel() failed for '%s'\n", name);
290  return 0;
291  }
292 
293  // only set the name after the model has been successfully loaded
294  Q_strncpyz( mod->name, name, sizeof( mod->name ) );
295 
296 
297  R_IssuePendingRenderCommands();
298 
299  mod->type = MOD_BAD;
300  mod->numLods = 0;
301 
302  //
303  // load the files
304  //
305  Q_strncpyz( localName, name, MAX_QPATH );
306 
307  ext = COM_GetExtension( localName );
308 
309  if( *ext )
310  {
311  // Look for the correct loader and use it
312  for( i = 0; i < numModelLoaders; i++ )
313  {
314  if( !Q_stricmp( ext, modelLoaders[ i ].ext ) )
315  {
316  // Load
317  hModel = modelLoaders[ i ].ModelLoader( localName, mod );
318  break;
319  }
320  }
321 
322  // A loader was found
323  if( i < numModelLoaders )
324  {
325  if( !hModel )
326  {
327  // Loader failed, most likely because the file isn't there;
328  // try again without the extension
329  orgNameFailed = qtrue;
330  orgLoader = i;
331  COM_StripExtension( name, localName, MAX_QPATH );
332  }
333  else
334  {
335  // Something loaded
336  return mod->index;
337  }
338  }
339  }
340 
341  // Try and find a suitable match using all
342  // the model formats supported
343  for( i = 0; i < numModelLoaders; i++ )
344  {
345  if (i == orgLoader)
346  continue;
347 
348  Com_sprintf( altName, sizeof (altName), "%s.%s", localName, modelLoaders[ i ].ext );
349 
350  // Load
351  hModel = modelLoaders[ i ].ModelLoader( altName, mod );
352 
353  if( hModel )
354  {
355  if( orgNameFailed )
356  {
357  ri.Printf( PRINT_DEVELOPER, "WARNING: %s not present, using %s instead\n",
358  name, altName );
359  }
360 
361  break;
362  }
363  }
364 
365  return hModel;
366 }

373 static qboolean R_LoadMD3(model_t * mod, int lod, void *buffer, int bufferSize, const char *modName)
374 {
375  int f, i, j;
376 
377  md3Header_t *md3Model;
378  md3Frame_t *md3Frame;
379  md3Surface_t *md3Surf;
380  md3Shader_t *md3Shader;
381  md3Triangle_t *md3Tri;
382  md3St_t *md3st;
383  md3XyzNormal_t *md3xyz;
384  md3Tag_t *md3Tag;
385 
386  mdvModel_t *mdvModel;
387  mdvFrame_t *frame;
388  mdvSurface_t *surf;//, *surface;
389  int *shaderIndex;
390  glIndex_t *tri;
391  mdvVertex_t *v;
392  mdvSt_t *st;
393  mdvTag_t *tag;
394  mdvTagName_t *tagName;
395 
396  int version;
397  int size;
398 
399  md3Model = (md3Header_t *) buffer;
400 
401  version = LittleLong(md3Model->version);
402  if(version != MD3_VERSION)
403  {
404  ri.Printf(PRINT_WARNING, "R_LoadMD3: %s has wrong version (%i should be %i)\n", modName, version, MD3_VERSION);
405  return qfalse;
406  }
407 
408  mod->type = MOD_MESH;
409  size = LittleLong(md3Model->ofsEnd);
410  mod->dataSize += size;
411  mdvModel = mod->mdv[lod] = ri.Hunk_Alloc(sizeof(mdvModel_t), h_low);
412 
413 // Com_Memcpy(mod->md3[lod], buffer, LittleLong(md3Model->ofsEnd));
414 
415  LL(md3Model->ident);
416  LL(md3Model->version);
417  LL(md3Model->numFrames);
418  LL(md3Model->numTags);
419  LL(md3Model->numSurfaces);
420  LL(md3Model->ofsFrames);
421  LL(md3Model->ofsTags);
422  LL(md3Model->ofsSurfaces);
423  LL(md3Model->ofsEnd);
424 
425  if(md3Model->numFrames < 1)
426  {
427  ri.Printf(PRINT_WARNING, "R_LoadMD3: %s has no frames\n", modName);
428  return qfalse;
429  }
430 
431  // swap all the frames
432  mdvModel->numFrames = md3Model->numFrames;
433  mdvModel->frames = frame = ri.Hunk_Alloc(sizeof(*frame) * md3Model->numFrames, h_low);
434 
435  md3Frame = (md3Frame_t *) ((byte *) md3Model + md3Model->ofsFrames);
436  for(i = 0; i < md3Model->numFrames; i++, frame++, md3Frame++)
437  {
438  frame->radius = LittleFloat(md3Frame->radius);
439  for(j = 0; j < 3; j++)
440  {
441  frame->bounds[0][j] = LittleFloat(md3Frame->bounds[0][j]);
442  frame->bounds[1][j] = LittleFloat(md3Frame->bounds[1][j]);
443  frame->localOrigin[j] = LittleFloat(md3Frame->localOrigin[j]);
444  }
445  }
446 
447  // swap all the tags
448  mdvModel->numTags = md3Model->numTags;
449  mdvModel->tags = tag = ri.Hunk_Alloc(sizeof(*tag) * (md3Model->numTags * md3Model->numFrames), h_low);
450 
451  md3Tag = (md3Tag_t *) ((byte *) md3Model + md3Model->ofsTags);
452  for(i = 0; i < md3Model->numTags * md3Model->numFrames; i++, tag++, md3Tag++)
453  {
454  for(j = 0; j < 3; j++)
455  {
456  tag->origin[j] = LittleFloat(md3Tag->origin[j]);
457  tag->axis[0][j] = LittleFloat(md3Tag->axis[0][j]);
458  tag->axis[1][j] = LittleFloat(md3Tag->axis[1][j]);
459  tag->axis[2][j] = LittleFloat(md3Tag->axis[2][j]);
460  }
461  }
462 
463 
464  mdvModel->tagNames = tagName = ri.Hunk_Alloc(sizeof(*tagName) * (md3Model->numTags), h_low);
465 
466  md3Tag = (md3Tag_t *) ((byte *) md3Model + md3Model->ofsTags);
467  for(i = 0; i < md3Model->numTags; i++, tagName++, md3Tag++)
468  {
469  Q_strncpyz(tagName->name, md3Tag->name, sizeof(tagName->name));
470  }
471 
472  // swap all the surfaces
473  mdvModel->numSurfaces = md3Model->numSurfaces;
474  mdvModel->surfaces = surf = ri.Hunk_Alloc(sizeof(*surf) * md3Model->numSurfaces, h_low);
475 
476  md3Surf = (md3Surface_t *) ((byte *) md3Model + md3Model->ofsSurfaces);
477  for(i = 0; i < md3Model->numSurfaces; i++)
478  {
479  LL(md3Surf->ident);
480  LL(md3Surf->flags);
481  LL(md3Surf->numFrames);
482  LL(md3Surf->numShaders);
483  LL(md3Surf->numTriangles);
484  LL(md3Surf->ofsTriangles);
485  LL(md3Surf->numVerts);
486  LL(md3Surf->ofsShaders);
487  LL(md3Surf->ofsSt);
488  LL(md3Surf->ofsXyzNormals);
489  LL(md3Surf->ofsEnd);
490 
491  if(md3Surf->numVerts >= SHADER_MAX_VERTEXES)
492  {
493  ri.Printf(PRINT_WARNING, "R_LoadMD3: %s has more than %i verts on %s (%i).\n",
494  modName, SHADER_MAX_VERTEXES - 1, md3Surf->name[0] ? md3Surf->name : "a surface",
495  md3Surf->numVerts );
496  return qfalse;
497  }
498  if(md3Surf->numTriangles * 3 >= SHADER_MAX_INDEXES)
499  {
500  ri.Printf(PRINT_WARNING, "R_LoadMD3: %s has more than %i triangles on %s (%i).\n",
501  modName, ( SHADER_MAX_INDEXES / 3 ) - 1, md3Surf->name[0] ? md3Surf->name : "a surface",
502  md3Surf->numTriangles );
503  return qfalse;
504  }
505 
506  // change to surface identifier
507  surf->surfaceType = SF_MDV;
508 
509  // give pointer to model for Tess_SurfaceMDX
510  surf->model = mdvModel;
511 
512  // copy surface name
513  Q_strncpyz(surf->name, md3Surf->name, sizeof(surf->name));
514 
515  // lowercase the surface name so skin compares are faster
516  Q_strlwr(surf->name);
517 
518  // strip off a trailing _1 or _2
519  // this is a crutch for q3data being a mess
520  j = strlen(surf->name);
521  if(j > 2 && surf->name[j - 2] == '_')
522  {
523  surf->name[j - 2] = 0;
524  }
525 
526  // register the shaders
527  surf->numShaderIndexes = md3Surf->numShaders;
528  surf->shaderIndexes = shaderIndex = ri.Hunk_Alloc(sizeof(*shaderIndex) * md3Surf->numShaders, h_low);
529 
530  md3Shader = (md3Shader_t *) ((byte *) md3Surf + md3Surf->ofsShaders);
531  for(j = 0; j < md3Surf->numShaders; j++, shaderIndex++, md3Shader++)
532  {
533  shader_t *sh;
534 
535  sh = R_FindShader(md3Shader->name, LIGHTMAP_NONE, qtrue);
536  if(sh->defaultShader)
537  {
538  *shaderIndex = 0;
539  }
540  else
541  {
542  *shaderIndex = sh->index;
543  }
544  }
545 
546  // swap all the triangles
547  surf->numIndexes = md3Surf->numTriangles * 3;
548  surf->indexes = tri = ri.Hunk_Alloc(sizeof(*tri) * 3 * md3Surf->numTriangles, h_low);
549 
550  md3Tri = (md3Triangle_t *) ((byte *) md3Surf + md3Surf->ofsTriangles);
551  for(j = 0; j < md3Surf->numTriangles; j++, tri += 3, md3Tri++)
552  {
553  tri[0] = LittleLong(md3Tri->indexes[0]);
554  tri[1] = LittleLong(md3Tri->indexes[1]);
555  tri[2] = LittleLong(md3Tri->indexes[2]);
556  }
557 
558  // swap all the XyzNormals
559  surf->numVerts = md3Surf->numVerts;
560  surf->verts = v = ri.Hunk_Alloc(sizeof(*v) * (md3Surf->numVerts * md3Surf->numFrames), h_low);
561 
562  md3xyz = (md3XyzNormal_t *) ((byte *) md3Surf + md3Surf->ofsXyzNormals);
563  for(j = 0; j < md3Surf->numVerts * md3Surf->numFrames; j++, md3xyz++, v++)
564  {
565  unsigned lat, lng;
566  unsigned short normal;
567  vec3_t fNormal;
568 
569  v->xyz[0] = LittleShort(md3xyz->xyz[0]) * MD3_XYZ_SCALE;
570  v->xyz[1] = LittleShort(md3xyz->xyz[1]) * MD3_XYZ_SCALE;
571  v->xyz[2] = LittleShort(md3xyz->xyz[2]) * MD3_XYZ_SCALE;
572 
573  normal = LittleShort(md3xyz->normal);
574 
575  lat = ( normal >> 8 ) & 0xff;
576  lng = ( normal & 0xff );
577  lat *= (FUNCTABLE_SIZE/256);
578  lng *= (FUNCTABLE_SIZE/256);
579 
580  // decode X as cos( lat ) * sin( long )
581  // decode Y as sin( lat ) * sin( long )
582  // decode Z as cos( long )
583 
584  fNormal[0] = tr.sinTable[(lat+(FUNCTABLE_SIZE/4))&FUNCTABLE_MASK] * tr.sinTable[lng];
585  fNormal[1] = tr.sinTable[lat] * tr.sinTable[lng];
586  fNormal[2] = tr.sinTable[(lng+(FUNCTABLE_SIZE/4))&FUNCTABLE_MASK];
587 
588  R_VaoPackNormal(v->normal, fNormal);
589  }
590 
591  // swap all the ST
592  surf->st = st = ri.Hunk_Alloc(sizeof(*st) * md3Surf->numVerts, h_low);
593 
594  md3st = (md3St_t *) ((byte *) md3Surf + md3Surf->ofsSt);
595  for(j = 0; j < md3Surf->numVerts; j++, md3st++, st++)
596  {
597  st->st[0] = LittleFloat(md3st->st[0]);
598  st->st[1] = LittleFloat(md3st->st[1]);
599  }
600 
601  // calc tangent spaces
602  {
603  vec3_t *sdirs = ri.Malloc(sizeof(*sdirs) * surf->numVerts * mdvModel->numFrames);
604  vec3_t *tdirs = ri.Malloc(sizeof(*tdirs) * surf->numVerts * mdvModel->numFrames);
605 
606  for(j = 0, v = surf->verts; j < (surf->numVerts * mdvModel->numFrames); j++, v++)
607  {
608  VectorClear(sdirs[j]);
609  VectorClear(tdirs[j]);
610  }
611 
612  for(f = 0; f < mdvModel->numFrames; f++)
613  {
614  for(j = 0, tri = surf->indexes; j < surf->numIndexes; j += 3, tri += 3)
615  {
616  vec3_t sdir, tdir;
617  const float *v0, *v1, *v2, *t0, *t1, *t2;
618  glIndex_t index0, index1, index2;
619 
620  index0 = surf->numVerts * f + tri[0];
621  index1 = surf->numVerts * f + tri[1];
622  index2 = surf->numVerts * f + tri[2];
623 
624  v0 = surf->verts[index0].xyz;
625  v1 = surf->verts[index1].xyz;
626  v2 = surf->verts[index2].xyz;
627 
628  t0 = surf->st[tri[0]].st;
629  t1 = surf->st[tri[1]].st;
630  t2 = surf->st[tri[2]].st;
631 
632  R_CalcTexDirs(sdir, tdir, v0, v1, v2, t0, t1, t2);
633 
634  VectorAdd(sdir, sdirs[index0], sdirs[index0]);
635  VectorAdd(sdir, sdirs[index1], sdirs[index1]);
636  VectorAdd(sdir, sdirs[index2], sdirs[index2]);
637  VectorAdd(tdir, tdirs[index0], tdirs[index0]);
638  VectorAdd(tdir, tdirs[index1], tdirs[index1]);
639  VectorAdd(tdir, tdirs[index2], tdirs[index2]);
640  }
641  }
642 
643  for(j = 0, v = surf->verts; j < (surf->numVerts * mdvModel->numFrames); j++, v++)
644  {
645  vec3_t normal;
646  vec4_t tangent;
647 
648  VectorNormalize(sdirs[j]);
649  VectorNormalize(tdirs[j]);
650 
651  R_VaoUnpackNormal(normal, v->normal);
652 
653  tangent[3] = R_CalcTangentSpace(tangent, NULL, normal, sdirs[j], tdirs[j]);
654 
655  R_VaoPackTangent(v->tangent, tangent);
656  }
657 
658  ri.Free(sdirs);
659  ri.Free(tdirs);
660  }
661 
662  // find the next surface
663  md3Surf = (md3Surface_t *) ((byte *) md3Surf + md3Surf->ofsEnd);
664  surf++;
665  }
666 
667  {
668  srfVaoMdvMesh_t *vaoSurf;
669 
670  mdvModel->numVaoSurfaces = mdvModel->numSurfaces;
671  mdvModel->vaoSurfaces = ri.Hunk_Alloc(sizeof(*mdvModel->vaoSurfaces) * mdvModel->numSurfaces, h_low);
672 
673  vaoSurf = mdvModel->vaoSurfaces;
674  surf = mdvModel->surfaces;
675  for (i = 0; i < mdvModel->numSurfaces; i++, vaoSurf++, surf++)
676  {
677  uint32_t offset_xyz, offset_st, offset_normal, offset_tangent;
678  uint32_t stride_xyz, stride_st, stride_normal, stride_tangent;
679  uint32_t dataSize, dataOfs;
680  uint8_t *data;
681 
682  if (mdvModel->numFrames > 1)
683  {
684  // vertex animation, store texcoords first, then position/normal/tangents
685  offset_st = 0;
686  offset_xyz = surf->numVerts * sizeof(vec2_t);
687  offset_normal = offset_xyz + sizeof(vec3_t);
688  offset_tangent = offset_normal + sizeof(int16_t) * 4;
689  stride_st = sizeof(vec2_t);
690  stride_xyz = sizeof(vec3_t) + sizeof(int16_t) * 4;
691  stride_xyz += sizeof(int16_t) * 4;
692  stride_normal = stride_tangent = stride_xyz;
693 
694  dataSize = offset_xyz + surf->numVerts * mdvModel->numFrames * stride_xyz;
695  }
696  else
697  {
698  // no animation, interleave everything
699  offset_xyz = 0;
700  offset_st = offset_xyz + sizeof(vec3_t);
701  offset_normal = offset_st + sizeof(vec2_t);
702  offset_tangent = offset_normal + sizeof(int16_t) * 4;
703  stride_xyz = offset_tangent + sizeof(int16_t) * 4;
704  stride_st = stride_normal = stride_tangent = stride_xyz;
705

707  }

711  dataOfs = 0;
712

718  dataOfs += sizeof(st->st);
719  }

723  {
724  // xyz
725  memcpy(data + dataOfs, &v->xyz, sizeof(vec3_t));
726  dataOfs += sizeof(vec3_t);
727

729  memcpy(data + dataOfs, &v->normal, sizeof(int16_t) * 4);
730  dataOfs += sizeof(int16_t) * 4;
731 
732  // tangent
733  memcpy(data + dataOfs, &v->tangent, sizeof(int16_t) * 4);
734  dataOfs += sizeof(int16_t) * 4;
735  }
736  }
737  else
738  {
739  v = surf->verts;
740  st = surf->st;
741  for ( j = 0; j < surf->numVerts; j++, v++, st++ )
742  {
743  // xyz
744  memcpy(data + dataOfs, &v->xyz, sizeof(vec3_t));
745  dataOfs += sizeof(v->xyz);
746 
747  // st
748  memcpy(data + dataOfs, &st->st, sizeof(vec2_t));
749  dataOfs += sizeof(st->st);
750

752  memcpy(data + dataOfs, &v->normal, sizeof(int16_t) * 4);

768

819 /*
820 =================
821 R_LoadMDR
822 =================
823 */
824 static qboolean R_LoadMDR( model_t *mod, void *buffer, int filesize, const char *mod_name )
825 {
826  int i, j, k, l;
827  mdrHeader_t *pinmodel, *mdr;
828  mdrFrame_t *frame;
829  mdrLOD_t *lod, *curlod;
830  mdrSurface_t *surf, *cursurf;
831  mdrTriangle_t *tri, *curtri;
832  mdrVertex_t *v, *curv;
833  mdrWeight_t *weight, *curweight;
834  mdrTag_t *tag, *curtag;
835  int size;
836  shader_t *sh;
837 
838  pinmodel = (mdrHeader_t *)buffer;
839 
840  pinmodel->version = LittleLong(pinmodel->version);
841  if (pinmodel->version != MDR_VERSION)
842  {
843  ri.Printf(PRINT_WARNING, "R_LoadMDR: %s has wrong version (%i should be %i)\n", mod_name, pinmodel->version, MDR_VERSION);
844  return qfalse;
845  }
846 
847  size = LittleLong(pinmodel->ofsEnd);
848 
849  if(size > filesize)
850  {
851  ri.Printf(PRINT_WARNING, "R_LoadMDR: Header of %s is broken. Wrong filesize declared!\n", mod_name);
852  return qfalse;
853  }
854 
855  mod->type = MOD_MDR;
856 
857  LL(pinmodel->numFrames);
858  LL(pinmodel->numBones);
859  LL(pinmodel->ofsFrames);
860 
861  // This is a model that uses some type of compressed Bones. We don't want to uncompress every bone for each rendered frame
862  // over and over again, we'll uncompress it in this function already, so we must adjust the size of the target mdr.
863  if(pinmodel->ofsFrames < 0)
864  {
865  // mdrFrame_t is larger than mdrCompFrame_t:
866  size += pinmodel->numFrames * sizeof(frame->name);
867  // now add enough space for the uncompressed bones.
868  size += pinmodel->numFrames * pinmodel->numBones * ((sizeof(mdrBone_t) - sizeof(mdrCompBone_t)));
869  }
870 
871  // simple bounds check
872  if(pinmodel->numBones < 0 ||
873  sizeof(*mdr) + pinmodel->numFrames * (sizeof(*frame) + (pinmodel->numBones - 1) * sizeof(*frame->bones)) > size)
874  {
875  ri.Printf(PRINT_WARNING, "R_LoadMDR: %s has broken structure.\n", mod_name);
876  return qfalse;
877  }
878 
879  mod->dataSize += size;
880  mod->modelData = mdr = ri.Hunk_Alloc( size, h_low );
881 
882  // Copy all the values over from the file and fix endian issues in the process, if necessary.
883 
884  mdr->ident = LittleLong(pinmodel->ident);
885  mdr->version = pinmodel->version; // Don't need to swap byte order on this one, we already did above.
886  Q_strncpyz(mdr->name, pinmodel->name, sizeof(mdr->name));
887  mdr->numFrames = pinmodel->numFrames;
888  mdr->numBones = pinmodel->numBones;
889  mdr->numLODs = LittleLong(pinmodel->numLODs);
890  mdr->numTags = LittleLong(pinmodel->numTags);
891  // We don't care about the other offset values, we'll generate them ourselves while loading.
892 
893  mod->numLods = mdr->numLODs;
894 
895  if ( mdr->numFrames < 1 )
896  {
897  ri.Printf(PRINT_WARNING, "R_LoadMDR: %s has no frames\n", mod_name);
898  return qfalse;
899  }
900 
901  /* The first frame will be put into the first free space after the header */
902  frame = (mdrFrame_t *)(mdr + 1);
903  mdr->ofsFrames = (int)((byte *) frame - (byte *) mdr);
904 
905  if (pinmodel->ofsFrames < 0)
906  {
907  mdrCompFrame_t *cframe;
908 
909  // compressed model...
910  cframe = (mdrCompFrame_t *)((byte *) pinmodel - pinmodel->ofsFrames);
911 
912  for(i = 0; i < mdr->numFrames; i++)
913  {
914  for(j = 0; j < 3; j++)
915  {
916  frame->bounds[0][j] = LittleFloat(cframe->bounds[0][j]);
917  frame->bounds[1][j] = LittleFloat(cframe->bounds[1][j]);
918  frame->localOrigin[j] = LittleFloat(cframe->localOrigin[j]);
919  }
920 
921  frame->radius = LittleFloat(cframe->radius);
922  frame->name[0] = '\0'; // No name supplied in the compressed version.
923 
924  for(j = 0; j < mdr->numBones; j++)
925  {
926  for(k = 0; k < (sizeof(cframe->bones[j].Comp) / 2); k++)
927  {
928  // Do swapping for the uncompressing functions. They seem to use shorts
929  // values only, so I assume this will work. Never tested it on other
930  // platforms, though.
931 
932  ((unsigned short *)(cframe->bones[j].Comp))[k] =
933  LittleShort( ((unsigned short *)(cframe->bones[j].Comp))[k] );
934  }
935 
936  /* Now do the actual uncompressing */
937  MC_UnCompress(frame->bones[j].matrix, cframe->bones[j].Comp);
938  }
939 
940  // Next Frame...
941  cframe = (mdrCompFrame_t *) &cframe->bones[j];
942  frame = (mdrFrame_t *) &frame->bones[j];
943  }
944  }
945  else
946  {
947  mdrFrame_t *curframe;
948 
949  // uncompressed model...
950  //
951 
952  curframe = (mdrFrame_t *)((byte *) pinmodel + pinmodel->ofsFrames);
953 
954  // swap all the frames
955  for ( i = 0 ; i < mdr->numFrames ; i++)
956  {
957  for(j = 0; j < 3; j++)
958  {
959  frame->bounds[0][j] = LittleFloat(curframe->bounds[0][j]);
960  frame->bounds[1][j] = LittleFloat(curframe->bounds[1][j]);
961  frame->localOrigin[j] = LittleFloat(curframe->localOrigin[j]);
962  }
963 
964  frame->radius = LittleFloat(curframe->radius);
965  Q_strncpyz(frame->name, curframe->name, sizeof(frame->name));
966 
967  for (j = 0; j < (int) (mdr->numBones * sizeof(mdrBone_t) / 4); j++)
968  {
969  ((float *)frame->bones)[j] = LittleFloat( ((float *)curframe->bones)[j] );
970  }
971 
972  curframe = (mdrFrame_t *) &curframe->bones[mdr->numBones];
973  frame = (mdrFrame_t *) &frame->bones[mdr->numBones];
974  }
975  }
976 
977  // frame should now point to the first free address after all frames.
978  lod = (mdrLOD_t *) frame;
979  mdr->ofsLODs = (int) ((byte *) lod - (byte *)mdr);
980 
981  curlod = (mdrLOD_t *)((byte *) pinmodel + LittleLong(pinmodel->ofsLODs));
982 
983  // swap all the LOD's
984  for ( l = 0 ; l < mdr->numLODs ; l++)
985  {
986  // simple bounds check
987  if((byte *) (lod + 1) > (byte *) mdr + size)
988  {
989  ri.Printf(PRINT_WARNING, "R_LoadMDR: %s has broken structure.\n", mod_name);
990  return qfalse;
991  }
992 
993  lod->numSurfaces = LittleLong(curlod->numSurfaces);
994 
995  // swap all the surfaces
996  surf = (mdrSurface_t *) (lod + 1);
997  lod->ofsSurfaces = (int)((byte *) surf - (byte *) lod);
998  cursurf = (mdrSurface_t *) ((byte *)curlod + LittleLong(curlod->ofsSurfaces));
999 
1000  for ( i = 0 ; i < lod->numSurfaces ; i++)
1001  {
1002  // simple bounds check
1003  if((byte *) (surf + 1) > (byte *) mdr + size)
1004  {
1005  ri.Printf(PRINT_WARNING, "R_LoadMDR: %s has broken structure.\n", mod_name);
1006  return qfalse;
1007  }
1008 
1009  // first do some copying stuff
1010 
1011  surf->ident = SF_MDR;
1012  Q_strncpyz(surf->name, cursurf->name, sizeof(surf->name));
1013  Q_strncpyz(surf->shader, cursurf->shader, sizeof(surf->shader));
1014 
1015  surf->ofsHeader = (byte *) mdr - (byte *) surf;
1016 
1017  surf->numVerts = LittleLong(cursurf->numVerts);
1018  surf->numTriangles = LittleLong(cursurf->numTriangles);
1019  // numBoneReferences and BoneReferences generally seem to be unused
1020 
1021  // now do the checks that may fail.
1022  if ( surf->numVerts >= SHADER_MAX_VERTEXES )
1023  {
1024  ri.Printf(PRINT_WARNING, "R_LoadMDR: %s has more than %i verts on %s (%i).\n",
1025  mod_name, SHADER_MAX_VERTEXES - 1, surf->name[0] ? surf->name : "a surface",
1026  surf->numVerts );
1027  return qfalse;
1028  }
1029  if ( surf->numTriangles*3 >= SHADER_MAX_INDEXES )
1030  {
1031  ri.Printf(PRINT_WARNING, "R_LoadMDR: %s has more than %i triangles on %s (%i).\n",
1032  mod_name, ( SHADER_MAX_INDEXES / 3 ) - 1, surf->name[0] ? surf->name : "a surface",
1033  surf->numTriangles );
1034  return qfalse;
1035  }
1036  // lowercase the surface name so skin compares are faster
1037  Q_strlwr( surf->name );
1038 
1039  // register the shaders
1040  sh = R_FindShader(surf->shader, LIGHTMAP_NONE, qtrue);
1041  if ( sh->defaultShader ) {
1042  surf->shaderIndex = 0;
1043  } else {
1044  surf->shaderIndex = sh->index;
1045  }
1046 
1047  // now copy the vertexes.
1048  v = (mdrVertex_t *) (surf + 1);
1049  surf->ofsVerts = (int)((byte *) v - (byte *) surf);
1050  curv = (mdrVertex_t *) ((byte *)cursurf + LittleLong(cursurf->ofsVerts));
1051 
1052  for(j = 0; j < surf->numVerts; j++)
1053  {
1054  LL(curv->numWeights);
1055 
1056  // simple bounds check
1057  if(curv->numWeights < 0 || (byte *) (v + 1) + (curv->numWeights - 1) * sizeof(*weight) > (byte *) mdr + size)
1058  {
1059  ri.Printf(PRINT_WARNING, "R_LoadMDR: %s has broken structure.\n", mod_name);
1060  return qfalse;
1061  }
1062 
1063  v->normal[0] = LittleFloat(curv->normal[0]);
1064  v->normal[1] = LittleFloat(curv->normal[1]);
1065  v->normal[2] = LittleFloat(curv->normal[2]);
1066 
1067  v->texCoords[0] = LittleFloat(curv->texCoords[0]);
1068  v->texCoords[1] = LittleFloat(curv->texCoords[1]);
1069 
1070  v->numWeights = curv->numWeights;
1071  weight = &v->weights[0];
1072  curweight = &curv->weights[0];
1073 
1074  // Now copy all the weights
1075  for(k = 0; k < v->numWeights; k++)
1076  {
1077  weight->boneIndex = LittleLong(curweight->boneIndex);
1078  weight->boneWeight = LittleFloat(curweight->boneWeight);
1079 
1080  weight->offset[0] = LittleFloat(curweight->offset[0]);
1081  weight->offset[1] = LittleFloat(curweight->offset[1]);
1082  weight->offset[2] = LittleFloat(curweight->offset[2]);
1083 
1084  weight++;
1085  curweight++;
1086  }
1087 
1088  v = (mdrVertex_t *) weight;
1089  curv = (mdrVertex_t *) curweight;
1090  }
1091 
1092  // we know the offset to the triangles now:
1093  tri = (mdrTriangle_t *) v;
1094  surf->ofsTriangles = (int)((byte *) tri - (byte *) surf);
1095  curtri = (mdrTriangle_t *)((byte *) cursurf + LittleLong(cursurf->ofsTriangles));
1096 
1097  // simple bounds check
1098  if(surf->numTriangles < 0 || (byte *) (tri + surf->numTriangles) > (byte *) mdr + size)
1099  {
1100  ri.Printf(PRINT_WARNING, "R_LoadMDR: %s has broken structure.\n", mod_name);
1101  return qfalse;
1102  }
1103 
1104  for(j = 0; j < surf->numTriangles; j++)
1105  {
1106  tri->indexes[0] = LittleLong(curtri->indexes[0]);
1107  tri->indexes[1] = LittleLong(curtri->indexes[1]);
1108  tri->indexes[2] = LittleLong(curtri->indexes[2]);
1109 
1110  tri++;
1111  curtri++;
1112  }
1113 
1114  // tri now points to the end of the surface.
1115  surf->ofsEnd = (byte *) tri - (byte *) surf;
1116  surf = (mdrSurface_t *) tri;
1117 
1118  // find the next surface.
1119  cursurf = (mdrSurface_t *) ((byte *) cursurf + LittleLong(cursurf->ofsEnd));
1120  }
1121 
1122  // surf points to the next lod now.
1123  lod->ofsEnd = (int)((byte *) surf - (byte *) lod);
1124  lod = (mdrLOD_t *) surf;
1125 
1126  // find the next LOD.
1127  curlod = (mdrLOD_t *)((byte *) curlod + LittleLong(curlod->ofsEnd));
1128  }
1129 
1130  // lod points to the first tag now, so update the offset too.
1131  tag = (mdrTag_t *) lod;
1132  mdr->ofsTags = (int)((byte *) tag - (byte *) mdr);
1133  curtag = (mdrTag_t *) ((byte *)pinmodel + LittleLong(pinmodel->ofsTags));
1134 
1135  // simple bounds check
1136  if(mdr->numTags < 0 || (byte *) (tag + mdr->numTags) > (byte *) mdr + size)
1137  {
1138  ri.Printf(PRINT_WARNING, "R_LoadMDR: %s has broken structure.\n", mod_name);
1139  return qfalse;
1140  }
1141 
1142  for (i = 0 ; i < mdr->numTags ; i++)
1143  {
1144  tag->boneIndex = LittleLong(curtag->boneIndex);
1145  Q_strncpyz(tag->name, curtag->name, sizeof(tag->name));
1146 
1147  tag++;
1148  curtag++;
1149  }
1150 
1151  // And finally we know the real offset to the end.
1152  mdr->ofsEnd = (int)((byte *) tag - (byte *) mdr);
1153 
1154  // phew! we're done.
1155 
1156  return qtrue;
1157 }
